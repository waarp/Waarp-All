package org.waarp.common.logging;

/**
 * The log level that {@link WaarpLogger} can log at.
 */
public enum WaarpLogLevel {
  /**
   * 'TRACE' log level.
   */
  TRACE,
  /**
   * 'DEBUG' log level.
   */
  DEBUG,
  /**
   * 'INFO' log level.
   */
  INFO,
  /**
   * 'WARN' log level.
   */
  WARN,
  /**
   * 'ERROR' log level.
   */
  ERROR,
  /**
   * 'NONE' log level.
   */
  NONE
}
