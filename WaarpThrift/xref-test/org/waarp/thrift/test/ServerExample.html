22  import org.apache.thrift.server.TNonblockingServer;
23  import org.apache.thrift.server.TServer;
24  import org.apache.thrift.server.TThreadPoolServer;
25  import org.apache.thrift.transport.TNonblockingServerSocket;
26  import org.apache.thrift.transport.TNonblockingServerTransport;
27  import org.apache.thrift.transport.TServerSocket;
28  import org.apache.thrift.transport.TServerTransport;
29  import org.apache.thrift.transport.TTransportException;
30  import org.waarp.thrift.r66.R66Service;
31  
32  import java.util.concurrent.ExecutorService;
33  import java.util.concurrent.Executors;
34  import java.util.concurrent.locks.Lock;
35  import java.util.concurrent.locks.ReentrantLock;
36  
37  /**
38   *
39   */
40  public class ServerExample implements Runnable {
41    /*
42     * Benchmark<br> Blocking: JDK6-32 13640/s JDK6-64 7004 JKD7-64 7385<br> NonBlocking: JDK6-32 17125/s JKD7-64
43     * 7923<br> Blocking: 5Threads JKD7-64 10436 JDK7-32 6324<br> NonBlocking: 5Threads JKD7-64 12820 JDK7-32
44     * 4958<br>
45     *
46     * New tests: JDK7-32 5Threads: in BinaryProtocol<br> NonBlocking TThreadedSelectorServer: 8908/29561<br>
47     * NonBlocking TNonblockingServer: 11285/34148<br> Blocking TThreadPoolServer: 17021/52714<br>
48     *
49     * Same in TJSONProtocol<br> Blocking TThreadPoolServer: ?/?<br>
50     *
51     * Same in TCompactProtocol<br> Blocking TThreadPoolServer: ?/49309<br>
52     */
53    private static boolean serverMode = true;
54    private static int PORT = 7911;
55    private static TServer server;
56    private static Thread runThread;
57    private static ExecutorService service;
58    private static final Lock lock = new ReentrantLock();
59  
60    public static void main(String[] args) throws InterruptedException {
61      startServer(true, PORT);
62    }
63  
64    public static void startServer(boolean mode, int port)
65        throws InterruptedException {
66      serverMode = mode;
67      PORT = port;
68      runThread = new Thread(new ServerExample());
69      runThread.setDaemon(true);
70      service = Executors.newSingleThreadScheduledExecutor();
71      service.execute(runThread);
72      Thread.sleep(100);
73      lock.lock();
74      lock.unlock();
75    }
76  
77    public static void stopServer() {
78      if (server != null) {
79        server.setShouldStop(true);
80        server.stop();
81      }
82      runThread.interrupt();
83      service.shutdownNow();
84    }
85  
86    @Override
87    public void run() {
88      lock.lock();
89      try {
90        TServerTransport serverTransport = null;
91        if (serverMode) {
92          serverTransport = new TServerSocket(PORT);
93        } else {
94          serverTransport = new TNonblockingServerSocket(PORT);
95        }
96        final R66Service.Processor<R66ServiceImpl> processor =
97            new R66Service.Processor<R66ServiceImpl>(new R66ServiceImpl());
98        server = null;
99        if (serverMode) {
100         server = new TThreadPoolServer(
101             new TThreadPoolServer.Args(serverTransport).processor(processor));
102       } else {
103         /*
104          * server = new TThreadedSelectorServer( new TThreadedSelectorServer.Args((TNonblockingServerTransport)
105          * serverTransport) .processor(processor));
106          */
107         server = new TNonblockingServer(new TNonblockingServer.Args(
108             (TNonblockingServerTransport) serverTransport).processor(
109             processor));
110       }
111       System.out.println("Starting server on port " + PORT);
112       lock.unlock();
113       server.serve();
114     } catch (final TTransportException e) {
115       e.printStackTrace();
116       lock.unlock();
117     }
118   }
119 }
