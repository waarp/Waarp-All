34  public class R66ServiceImpl implements R66Service.Iface {
35  
36    @Override
37    public R66Result transferRequestQuery(R66Request request) throws TException {
38      final R66Result result =
39          new R66Result(request.getMode(), ErrorCode.CompleteOk, "Test only");
40      result.setFromuid(request.getFromuid());
41      result.setDestuid(request.getDestuid());
42      result.setFile("Target_" + request.getFile());
43      result.setRule(request.getRule());
44      result.setTid(System.currentTimeMillis());
45      result.setGloballaststep(2);
46      result.setGlobalstep(2);
47      result.setStep(1);
48      result.setRank(1023);
49      result.setRetrievemode(true);
50      result.setIsmoved(false);
51      result.setOriginalfilename(request.getFile());
52      result.setModetransfer(4);
53      result.setStart("2009-08-13 12:26:43.209");
54      result.setStart("2009-08-13 12:26:46.079");
55      return result;
56    }
57  
58    @Override
59    public R66Result infoTransferQuery(R66Request request) throws TException {
60      final R66Result result =
61          new R66Result(request.getMode(), ErrorCode.CompleteOk,
62                        "Test Info only");
63      result.setDestuid(request.getDestuid());
64      result.setTid(request.getTid());
65      result.setAction(request.getAction());
66      result.setGloballaststep(2);
67      result.setGlobalstep(2);
68      result.setStep(1);
69      result.setRank(1023);
70      result.setRetrievemode(true);
71      result.setIsmoved(false);
72      result.setModetransfer(4);
73      result.setStart("2009-08-13 12:26:43.209");
74      result.setStart("2009-08-13 12:26:46.079");
75      return result;
76    }
77

81      return true;
82    }
