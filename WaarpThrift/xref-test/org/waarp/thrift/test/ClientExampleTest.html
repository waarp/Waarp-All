279   @Override
280   public void run() {
281     TTransport transport = null;
282     try {
283       if (isBlocking) {
284         transport = new TSocket("localhost", PORT);
285       } else {
286         transport = new TFramedTransport(new TSocket("localhost", PORT));
287       }
288       final TProtocol protocol = new TBinaryProtocol(transport);
289       final R66Service.Client client =
290           new R66Service.Client.Factory().getClient(protocol);
291       transport.open();
292       final R66Request request = new R66Request(RequestMode.SYNCTRANSFER);
293       request.setFromuid("myclient");
294       request.setDestuid("mypartner");
295       request.setRule("myruletouse");
296       request.setFile("pathtomyfile.txt");
297       request.setInfo("my info send on the wire");
298       request.setMd5(true);
299 
300       System.out.println("REQUEST1THREAD: " + request);
301       R66Result result = client.transferRequestQuery(request);
302       System.out.println("RESULT1THREAD: " + result);
303 
304       final long start = System.currentTimeMillis();
305       for (int i = 0; i < tries; i++) {
306         result = client.transferRequestQuery(request);
307         System.out.print('.');
308       }
309       final long end = System.currentTimeMillis();
310       System.out.println();
311       System.out.println(
312           "Delay: " + (end - start) + " : " + (tries * 1000) / (end - start));
313     } catch (final TTransportException e) {
314       e.printStackTrace();
315     } catch (final TException e) {
316       e.printStackTrace();
317     } finally {
318       try {
319         Thread.sleep(500);
320       } catch (final InterruptedException e) {//NOSONAR
321         e.printStackTrace();
322       }
323       if (transport != null) {
324         transport.close();
325       }
326     }
327   }
