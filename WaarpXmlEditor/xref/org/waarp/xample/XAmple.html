87  /**
88   * A simple application demonstrates usage of the Swing-based XML editing
89   * component
90   * {@link FXDoubleView}. The component generates
91   * document-specific GUI for each particular
92   * type of the XML document, specified by the XML Schema. A User-Manual
93   * document, which is included into
94   * XAmple XML Editor distribution package, provides detailed description of how
95   * to customize XML Schema
96   * documents with the editor-specific features, such as custom field editors,
97   * additional labeling,
98   * node-specific messages, etc.
99   * <p>
100  * </p>
101  * The application creares an instance of the {@link FFileDialog}
102  * and registers it
103  * with the {@link FXDoubleView} under "FileDialog" id, so it
104  * can be used as a global field
105  * editor. (Actually there is no particular sense in doing it since the dialog
106  * is not initialized with any
107  * specific data. It is just an example).
108  *
109  * @author Felix Golubov
110  * @version 2.0
111  */

314         r.height = screenSize.height;
315       }

317         r.x = screenSize.width - r.width;
318       }
319       if (r.y + r.height > screenSize.height) {
320         r.y = screenSize.height - r.height;

323       frame.validate();
324     } catch (final Exception ex) {
325       frame.pack();
326       final Dimension d = frame.getSize();
327       if (d.height > screenSize.height) {
328         d.height = screenSize.height;
329       }
330       if (d.width > screenSize.width) {
331         d.width = screenSize.width;
332       }
333       frame.setLocation((screenSize.width - d.width) / 2,
334                         (screenSize.height - d.height) / 2);
335     }
336     frame.setVisible(true);

339   static void assignDefaultFont() {
340     final Properties props = new Properties();
341     InputStream in = null;
342     String fontName;
343     int fontSize = 12;
344     try {
345       final File file = new File(FILE_GUI);
346       if (!file.exists()) {
347         return;
348       }
349       in = new FileInputStream(file);
350       props.load(in);
351       fontName = props.getProperty(FONT_NAME);
352       if (fontName == null || fontName.length() == 0) {
353         return;
354       }
355       try {
356         fontSize = Integer.parseInt(props.getProperty(FONT_SIZE, "12"));
357       } catch (final Exception ignore) {
358         // nothing
359       }
360     } catch (final Exception ex) {
361       SysErrLogger.FAKE_LOGGER.syserr(ex);
362       return;
363     } finally {
364       FileUtils.close(in);
365     }
366     final Font font = new Font(fontName, Font.PLAIN, fontSize);
367     final Enumeration<Object> keys = UIManager.getDefaults().keys();
368     while (keys.hasMoreElements()) {
369       final Object key = keys.nextElement();
370       final Object value = UIManager.get(key);
371       if (value instanceof FontUIResource) {
372         UIManager.put(key, font);
373       }
374     }
375   }

431   boolean saveXML() {
432     final Document doc = model.getDocument();
433     if (doc == null) {
434       return false;
435     }
436     fileDlg.setFileFilter(xmlFilter);
437     if (fileDlg.showSaveDialog(this) == JFileChooser.APPROVE_OPTION) {
438       File file = fileDlg.getSelectedFile();
439       if (file == null) {
440         return false;
441       }
442       if (file.getName().indexOf('.') < 0) {
443         file = new File(file.getAbsolutePath() + ".xml");
444       }
445       FileOutputStream out = null;
446       Writer writer = null;
447       try {
448         file.createNewFile();//NOSONAR
449         out = new FileOutputStream(file);
450         final OutputFormat format =
451             new OutputFormat(doc, WaarpStringUtils.UTF_8, true);
452         writer = new OutputStreamWriter(out, WaarpStringUtils.UTF8);
453         final XMLSerializer serial = new XMLSerializer(writer, format);
454         serial.asDOMSerializer();
455         serial.serialize(doc);
456         model.setDocumentChanged(false);
457         xmlFileName = file.getName();
458         showXMLTitle();
459         final History childHistory = history.getFirstChild();
460         if (childHistory != null) {
461           childHistory.put(file.getAbsolutePath());
462           btnOpenXML.setItems(childHistory.items);
463         }
464         return true;
465       } catch (final Exception ex) {
466         dblView.showErrorMessage("Error: " + ex.getMessage());
467         return false;
468       } finally {
469         FileUtils.close(writer);
470         FileUtils.close(out);
471       }
472     } else {
473       return false;
474     }
475   }

518       setTitle("XAmple");

523   boolean loadXMLDocument(final File file) {
524     final History childHistory = history.getFirstChild();
525     try {
526       final URL url = file.toURI().toURL();
527       final List<?> lostElements =
528           model.openDocument(model.getSchemaURL(), url);
529       if (lostElements != null) {
530         final StringBuilder sb = new StringBuilder(
531             "Error: The source XML Configuration document is invalid.\n" +
532             "The following elements have not been loaded:");
533         for (int i = 0; i < lostElements.size(); i++) {
534           sb.append('\n');
535           final int k = sb.length();
536           final Node element = (Node) lostElements.get(i);
537           sb.append(element.getNodeName());
538           Node node = element.getParentNode();
539           while (node != null && !(node instanceof Document)) {
540             sb.insert(k, node.getNodeName() + '/');
541             node = node.getParentNode();
542           }
543         }
544         dblView.showErrorMessage(sb.toString());
545       }
546       setXMLEnabled(true);
547       xmlFileName = file.getName();
548       showXMLTitle();
549       childHistory.put(file.getAbsolutePath());
550       btnOpenXML.setItems(childHistory.items);
551       dblView.showInfoMessage(
552           "XML Configuration Document: " + file.getAbsolutePath());
553       return true;
554     } catch (final Exception ex) {
555       childHistory.remove(file.getAbsolutePath());
556       btnOpenXML.setItems(childHistory.items);
557       // dblView.clear()
558       setXMLEnabled(false);
559       dblView.showErrorMessage("Error: " + ex.getMessage());
560       JOptionPane
561           .showMessageDialog(this, "Can't load the XML Configuration Document",
562                              "Error", JOptionPane.ERROR_MESSAGE);
563       showXSDTitle();
564       return false;
565     }
566   }

585   @Override
586   public void actionPerformed(final ActionEvent e) {
587     final Object source = e.getSource();
588     if (source == btnOpenXSD || source == mOpenXSD) {
589       openXSD();
590     } else if (source == btnOpenXML || source == mOpenXML) {
591       openXML();
592     } else if (source == btnNewXML || source == mNewXML) {
593       newXML();
594     } else if (source == btnSaveXML || source == mSaveXML) {
595       saveXML();
596     } else if (source == btnHorizSplit) {
597       dblView.setOrientation(JSplitPane.HORIZONTAL_SPLIT);
598     } else if (source == btnVertSplit) {
599       dblView.setOrientation(JSplitPane.VERTICAL_SPLIT);
600     } else if (source == btnSync) {
601       dblView.setSyncSelectNodes(btnSync.getModel().isSelected());
602     } else if (source instanceof JRadioButtonMenuItem) {
603       updateLookAndFeel(((JRadioButtonMenuItem) source).getText());
604     } else if (source == mAbout) {
605       showDlgAbout();
606     } else if (source == mExit) {
607       exit();
608     }
609   }

611   void updateLookAndFeel(final String lfName) {
612     final LookAndFeel lf = UIManager.getLookAndFeel();
613     if (lf != null && lf.getName().equals(lfName)) {
614       return;
615     }
616     final UIManager.LookAndFeelInfo[] lfi =
617         UIManager.getInstalledLookAndFeels();
618     for (final LookAndFeelInfo element : lfi) {
619       if (element.getName().equals(lfName)) {
620         try {
621           // dblView.stopEditing()
622           UIManager.setLookAndFeel(element.getClassName());
623           SwingUtilities.updateComponentTreeUI(this);
624           final FileFilter currFilter = fileDlg.getFileFilter();
625           fileDlg.setAcceptAllFileFilterUsed(
626               fileDlg.getAcceptAllFileFilter() == null);
627           SwingUtilities.updateComponentTreeUI(fileDlg);
628           fileDlg.setAcceptAllFileFilterUsed(true);
629           fileDlg.setFileFilter(currFilter);
630           SwingUtilities.updateComponentTreeUI(fFileDialog);
631           return;
632         } catch (final Exception ignore) {
633           // nothing
634         }
635       }
636     }
637   }

678   void newXML() {
679     if (!confirmation()) {
680       return;
681     }
682     try {
683       if (model.getSchemaURL() == null) {
684         throw new FXModelException("No XML Configuration Schema loaded");
685       }
686       final NewDocumentDialog dlg = new NewDocumentDialog(this, model);
687       dlg.pack();
688       dlg.setLocationRelativeTo(dblView);
689       dlg.setVisible(true);
690       if (!dlg.isStatusOK()) {
691         return;
692       }
693       model.newDocument(model.getSchemaURL(), dlg.getRootNamespace(),
694                         dlg.getRootElementName());
695       setXMLEnabled(true);
696     } catch (final Exception ex) {
697       SysErrLogger.FAKE_LOGGER.syserr(ex);
698       setXMLEnabled(false);
699       dblView.showErrorMessage("Error: " + ex.getMessage());
700       JOptionPane
701           .showMessageDialog(this, "Can't open new XML Configuration Document",
702                              "Error", JOptionPane.ERROR_MESSAGE);
703     }
704     showXSDTitle();
705   }

725   class TheFileFilter extends FileFilter {
726     final String extension;
727     final String description;
728 
729     TheFileFilter(final String extension, final String description) {
730       this.extension = extension;
731       this.description = description;
732     }
733 
734     @Override
735     public boolean accept(final File f) {
736       if (f.isDirectory()) {
737         return true;
738       }
739       final String ext;
740       final String name = f.getName();
741       final int i = name.lastIndexOf('.');
742       if (i > 0 && i < name.length() - 1) {
743         ext = name.substring(i + 1).toLowerCase();
744       } else {
745         ext = "";
746       }
747       return ext.equals(extension);
748     }
749 
750     @Override
751     public String getDescription() {
752       return description;
753     }
754   }
