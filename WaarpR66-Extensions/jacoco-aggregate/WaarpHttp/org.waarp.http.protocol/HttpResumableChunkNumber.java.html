package org.waarp.http.protocol;

/**
 * Chunk Number during a Http Resumable operation
 */
public class HttpResumableChunkNumber {
  private final int number;

  public HttpResumableChunkNumber(final int number) {
    this.number = number;
  }

  @Override
  public final int hashCode() {
    return number;
  }

  @Override
  public final boolean equals(final Object obj) {
    return obj instanceof HttpResumableChunkNumber &&
           ((HttpResumableChunkNumber) obj).number == this.number;
  }

  @Override
  public String toString() {
    return Integer.toString(number);
  }
}
