148   public String getServerConfigFile() {
149     if (configFile != null) {
150       logger.warn("ConfigFile already set to {]", configFile);
151       return configFile.getAbsolutePath();
152     }
153     logger.warn("Build configFile");
154     ClassLoader classLoader =
155         S3ScenarioBaseLoopBenchmark.class.getClassLoader();
156     File file =
157         new File(classLoader.getResource(RESOURCES_SERVER_1_XML).getFile());
158     if (!file.exists()) {
159       SysErrLogger.FAKE_LOGGER.syserr(
160           "Cannot find in  " + file.getAbsolutePath());
161       fail("Cannot find " + file.getAbsolutePath());
162     }
163     String content = WaarpStringUtils.readFile(file.getAbsolutePath());
164     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getJdbcUrl());
165     String driver = getJDC().getDriverClassName();
166     String target = "notfound";
167     String jdbcUrl = getJDC().getJdbcUrl();
168     if (driver.equalsIgnoreCase("org.mariadb.jdbc.Driver")) {
169       target = "mariadb";
170     } else if (driver.equalsIgnoreCase("org.h2.Driver")) {
171       target = "h2";
172     } else if (driver.equalsIgnoreCase("oracle.jdbc.OracleDriver")) {
173       target = "oracle";
174       jdbcUrl = "jdbc:oracle:thin:@//localhost:1521/test";
175       SysErrLogger.FAKE_LOGGER.syserr(
176           jdbcUrl + " while should be something like " + jdbcUrl);
177       throw new UnsupportedOperationException(
178           "Unsupported Test for Oracle since wrong JDBC driver");
179     } else if (driver.equalsIgnoreCase("org.postgresql.Driver")) {
180       target = "postgresql";
181     } else if (DbModelMysql.MYSQL_DRIVER_JRE6.equalsIgnoreCase(driver) ||
182                DbModelMysql.MYSQL_DRIVER_JRE8.equalsIgnoreCase(driver)) {
183       target = "mysql";
184     } else {
185       SysErrLogger.FAKE_LOGGER.syserr("Cannot find driver for " + driver);
186     }
187     content = content.replace("XXXJDBCXXX", jdbcUrl);
188     content = content.replace("XXXDRIVERXXX", target);
189     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getDriverClassName());
190     SysErrLogger.FAKE_LOGGER.sysout(target);
191     File fileTo = new File(TMP_R66_CONFIG_R1);
192     fileTo.getParentFile().mkdirs();
193     FileWriter writer = null;
194     try {
195       writer = new FileWriter(fileTo);
196       writer.write(content);
197       writer.flush();
198     } catch (IOException e) {
199       e.printStackTrace();
200     } finally {
201       if (writer != null) {
202         FileUtils.close(writer);
203       }
204     }
205     configFile = fileTo;
206     logger.warn("Config file created at {}", fileTo);
207     File copy = new File(TMP_CONFIG_XML);
208     try {
209       FileUtils.copy(configFile, copy, false, false);
210       logger.warn("Copy from {} to {}", configFile, copy);
211     } catch (Reply550Exception e) {
212       e.printStackTrace();
213     }
214     logger.warn("Copy from {} to {} and return {}", configFile, copy,
215                 TMP_R66_CONFIG_R1);
216     return TMP_R66_CONFIG_R1;
217   }

325   public static void tearDownAfterClass(final Thread delegate)
326       throws Exception {
327     CloseableHttpClient httpClient = null;
328     int max = SystemPropertyUtil.get(IT_LONG_TEST, false)? 8000 : 60;
329     int totalTransfers = max;
330     int nb = 0;
331     int every10sec = 10;
332     final long startTime = System.currentTimeMillis();
333     try {
334       httpClient = HttpClientBuilder.create().setConnectionManagerShared(true)
335                                     .disableAutomaticRetries().build();
336       final HttpGet request =
337           new HttpGet("http://127.0.0.1:8098/v2/transfers?countOrder=true");
338       CloseableHttpResponse response = null;
339       try {
340         response = httpClient.execute(request);
341         if (200 == response.getStatusLine().getStatusCode()) {
342           final String content = EntityUtils.toString(response.getEntity());
343           final ObjectNode node = JsonHandler.getFromString(content);
344           if (node != null) {
345             final JsonNode number = node.findValue("totalResults");
346             final long newNb = number.asLong();
347             max += newNb;
348             totalTransfers = (int) newNb;
349             logger.warn("Found {} transfers", newNb);
350           }
351         }
352       } finally {
353         if (response != null) {
354           response.close();
355         }
356       }
357       while (nb < max) {
358         try {
359           delegate.run();
360           response = httpClient.execute(request);
361           if (200 != response.getStatusLine().getStatusCode()) {
362             break;
363           }
364           final String content = EntityUtils.toString(response.getEntity());
365           final ObjectNode node = JsonHandler.getFromString(content);
366           if (node != null) {
367             final JsonNode number = node.findValue("totalResults");
368             final int newNb = number.asInt();
369             if (newNb != nb || every10sec == 0) {
370               every10sec = 10;
371               nb = newNb;
372               logger.warn("Found {} transfers", nb);
373             }
374             every10sec--;
375           }
376         } finally {
377           if (response != null) {
378             response.close();
379           }
380         }
381         Thread.sleep(1000);
382       }
383     } catch (ExecuteException e) {
384       // ignore
385     } finally {
386       if (httpClient != null) {
387         httpClient.close();
388       }
389     }
390     final long stopTime = System.currentTimeMillis();
391     totalTransfers = nb - totalTransfers;
392     logger.warn("Duration {} for {} item so {} items/s",
393                 (stopTime - startTime) / 1000.0, totalTransfers,
394                 totalTransfers / ((stopTime - startTime) / 1000.0));
395     WaarpSystemUtil.stopLogger(true);
396     Configuration.configuration.setTimeoutCon(100);
397     for (int pid : PIDS) {
398       Processes.kill(pid, true);
399     }
400     tearDownAfterClassClient();
401     tearDownAfterClassMinimal();
402     tearDownAfterClassServer();
403     final File base = new File("/tmp/R66");
404     FileUtils.deleteRecursiveDir(base);
405   }

422   private void checkMemory() {

431   private int initBenchmark(final int gap)
432       throws IOException, OpenR66ProtocolNetworkException, Reply550Exception {
433     NUMBER_FILES = SystemPropertyUtil.get(IT_LONG_TEST, false)? 2000 : 10;
434     final int factor = 250 * 1024 * 2 / NUMBER_FILES;
435     Assume.assumeNotNull(networkTransaction);
436     Configuration.configuration.changeNetworkLimit(0, 0, 0, 0, 1000);
437     if (rulename.equalsIgnoreCase("loop")) {
438       final File baseDir = new File("/tmp/R66/" + PATH_COMMON + "/R1/test/");
439       baseDir.mkdirs();
440       final WaarpR66S3Client client =
441           new WaarpR66S3Client(ACCESS_KEY, SECRET_KEY, s3Url);
442       for (int i = 1; i <= NUMBER_FILES; i++) {
443         final int size = (10000 + i * factor) + gap;
444         final File fileOut = new File(baseDir, "hello" + size);
445         final File outHello = generateOutFile(fileOut.getAbsolutePath(), size);
446         client.createFile(BUCKET, FILEPATHSRC + size, outHello, null);
447       }
448     } else {
449       final File baseDir = new File("/tmp/R66/" + PATH_COMMON + "/R1/out/");
450       baseDir.mkdirs();
451       final File baseDir2 = new File("/tmp/R66/" + PATH_COMMON + "/R2/out/");
452       baseDir2.mkdirs();
453       for (int i = 1; i <= NUMBER_FILES; i++) {
454         final int size = (10000 + i * factor) + gap;
455         final File fileOut = new File(baseDir, "hello" + size);
456         generateOutFile(fileOut.getAbsolutePath(), size);
457         final File fileOut2 = new File(baseDir2, "hello" + size);
458         FileUtils.copy(fileOut, fileOut2, false, false);
459       }
460     }
461     logger.warn("End of file creations");
462     return factor;
463   }

465   private void runLoopInit(String ruleName, String serverName, int factor,
466                            int gap) {
467     try {
468       DbRule rule = new DbRule(ruleName);
469     } catch (WaarpDatabaseException e) {
470       e.printStackTrace();
471     }
472     R66Future[] futures = new R66Future[NUMBER_FILES];
473     for (int i = 1; i <= NUMBER_FILES; i++) {
474       int size = (10000 + i * factor) + gap;
475       final R66Future future = new R66Future(true);
476       futures[i - 1] = future;
477       /*
478       final TestTransferNoDb transaction =
479               new TestTransferNoDb(future, "server2-ssl", "hello" + size, "loop",
480                       "Test Loop Send", true, BLOCK_SIZE,
481                       DbConstantR66.ILLEGALVALUE,
482                       networkTransaction);
483       transaction.setNormalInfoAsWarn(false);
484       transaction.run();
485        */
486 
487       final SubmitTransfer transaction =
488           new SubmitTransfer(future, serverName, "hello" + size, ruleName,
489                              (s3Url == null? "" : s3Url) + " " + FILEPATHSRC +
490                              size + " Test Loop Send " + size, true, BLOCK_SIZE,
491                              DbConstantR66.ILLEGALVALUE, null);
492       TransferArgs.forceAnalyzeFollow(transaction);
493       transaction.setNormalInfoAsWarn(false);
494       transaction.run();
495     }
496     for (int i = 0; i < NUMBER_FILES; i++) {
497       futures[i].awaitOrInterruptible();
498       assertTrue(futures[i].isSuccess());
499     }
500   }

504       throws IOException, OpenR66ProtocolNetworkException, Reply550Exception {

518                 Configuration.configuration.getServerGlobalReadLimit() /

521                 1000000.0, (factor * (NUMBER_FILES / 2)), BLOCK_SIZE);
522     checkMemory();
523     logger.warn("End {}", Processes.getCurrentMethodName());
524   }
