40  /**
41   * HttpSessions Tester.
42   *
43   * @author <Authors name>
44   * @version 1.0
45   * @since <pre>nov. 23, 2019</pre>
46   */
47  public class HttpSessionsTest {
48    @Rule(order = Integer.MIN_VALUE)
49    public TestWatcher watchman = new TestWatcherJunit4();
50  
51    final int chunkNumber = 2;
52    final int chunkSize = 1024;
53    final long totalSize = 10000;
54    final int toalChunk =
55        (int) Math.ceil(((double) totalSize) / ((double) chunkSize));
56    final String identifier = "abcdef";
57    final String filename = "filename.txt";
58    final String relativePath = "relative/path/filename.txt";
59    HttpResumableInfo resumableInfo;
60    HttpAuthentDefaultExtended authent;
61  
62    @BeforeClass
63    public static void beforeClass() {
64      WaarpLoggerFactory.setDefaultFactoryIfNotSame(
65          new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
66      ResourceLeakDetector.setLevel(Level.PARANOID);
67    }
68  
69    @Before
70    public void before() throws Exception {
71      WaarpSystemUtil.setJunit(true);
72      resumableInfo =
73          new HttpResumableInfo(chunkNumber, chunkSize, totalSize, identifier,
74                                filename, relativePath);
75      authent = new HttpAuthentDefaultExtended();
76      authent.setUser("user");
77      authent.setKey("key".getBytes(WaarpStringUtils.UTF8));
78    }
79  
80    @After
81    public void after() throws Exception {
82    }
83  
84    /**
85     * Method: getInstance()
86     */
87    @Test
88    public void testGetInstance() throws Exception {
89      HttpSessions sessions = HttpSessions.getInstance();
90      assertNotNull(sessions);
91    }
92  
93    /**
94     * Method: getOrCreateSession(final HttpResumableInfo resumableInfo, final
95     * String rulename, final HttpAuthent authent)
96     */
97    @Test
98    public void testGetOrCreateSession() throws Exception {
99      HttpSessionAbstract session =
100         new HttpResumableSession(resumableInfo, "rule", "comment", authent);
101     HttpSessions sessions = HttpSessions.getInstance();
102     assertFalse(sessions.contains(resumableInfo));
103     HttpSessionAbstract session2 =
104         sessions.getOrCreateResumableSession(resumableInfo, "rule", "comment",
105                                              authent);
106     assertTrue(sessions.contains(resumableInfo));
107     assertFalse(session == session2);
108     assertTrue(sessions.removeSession(resumableInfo));
109     assertFalse(sessions.removeSession(resumableInfo));
110 
111     HttpResumableSession session3 =
112         sessions.getOrCreateResumableSession(resumableInfo, "rule", "comment",
113                                              authent);
114     assertTrue(sessions.contains(resumableInfo));
115     assertFalse(session == session2);
116     assertFalse(session3 == session2);
117     assertTrue(sessions.removeSession(session3));
118     assertFalse(sessions.removeSession(session3));
119   }
