  private static final long serialVersionUID = 3794468302790427511L;

  /**
   *
   */
  public OpenR66RunnerErrorException() {
  }

  /**
   * @param arg0
   * @param arg1
   */

    super(arg0, arg1);
  }

  /**
   * @param arg0
   */

    super(arg0);
  }

  /**
   * @param arg0
   */

    super(arg0);
  }

}
