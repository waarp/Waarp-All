59    @BeforeClass
60    public static void initialize() {
61      final File dir = new File(".");
62      final File directory = new File(dir.getParent(), "src/test/testdata");
63      System.err.println(directory.getAbsolutePath());
64      final File[] subFiles = directory.listFiles();
65      if (subFiles != null) {
66        for (final File file : subFiles) {
67          if (file.isFile()) {
68            files.add(file);
69          } else {
70            final File[] subsubFiles = file.listFiles();
71            if (subsubFiles != null) {
72              for (final File subfile : subsubFiles) {
73                if (subfile.isFile()) {
74                  files.add(subfile);
75                }
76              }
77            }
78          }
79        }
80      }
81      waarpZstdCodec = new WaarpZstdCodec();
82    }

85      start = System.currentTimeMillis();
86    }

94    public void destroy(final String type) {
95      end = System.currentTimeMillis();
96      try {
97        Thread.sleep(100);
98      } catch (final InterruptedException e) {
99        //
100     }
101     System.out.println(
102         type + " Compressed " + compressedSize + " Decompressed " +
103         decompressedSize + " (" + toWrite + ") Ratio " +
104         (float) decompressedSize / compressedSize * 100.0 + " factice(" +
105         (float) toWrite / compressedSize * 100.0 + ") in " +
106         (files.size() * 1000.0) / (end - start));
107     compressedSize = 0;
108     decompressedSize = 0;
109     toWrite = 0;
110     start = 0;
111     end = 0;
112   }

114   private static byte[] readFileToBytes(final File file) throws IOException {
115     final byte[] bytes = new byte[(int) file.length()];
116     FileInputStream fis = null;
117     try {
118       fis = new FileInputStream(file);
119       //read file into bytes[]
120       int read;
121       int len = 0;
122       while (len < bytes.length &&
123              (read = fis.read(bytes, len, bytes.length - len)) > 0) {
124         len += read;
125       }
126     } finally {
127       if (fis != null) {
128         fis.close();
129       }
130     }
131     return bytes;
132   }

134   public void test(final CompressorCodec compressor,
135                    final CompressorCodec decompressor, final String type)
136       throws IOException {
137     byte[] bytes = readFileToBytes(files.get(0));
138     byte[] compressed;
139     byte[] decompressed;
140     compressed = compress(compressor, bytes);
141     decompressed = decompress(decompressor, compressed);
142     Assert.assertArrayEquals(bytes, decompressed);
143     startup();
144     for (final File file : files) {
145       bytes = readFileToBytes(file);
146       for (int i = 0; i < RUN; i++) {
147         compressed = compress(compressor, bytes);
148         decompressed = decompress(decompressor, compressed);
149       }
150       Assert.assertArrayEquals(bytes, decompressed);
151     }
152     destroy(type);
153   }

211   byte[] compress(final CompressorCodec compressor, final byte[] bytes) {
212     toWrite += bytes.length;
213     final byte[] compressed = compressor.compress(bytes, bytes.length);
214     compressedSize += compressed.length;
215     return compressed;
216   }
217 
218   byte[] decompress(final CompressorCodec decompressor,
219                     final byte[] compressed) {
220     final byte[] decompressed =
221         decompressor.decompress(compressed, compressed.length);
222     decompressedSize += decompressed.length;
223     return decompressed;
224   }
