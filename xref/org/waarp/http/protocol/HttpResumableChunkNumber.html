21  package org.waarp.http.protocol;
22  
23  /**
24   * Chunk Number during a Http Resumable operation
25   */
26  public class HttpResumableChunkNumber {
27    private final int number;
28  
29    public HttpResumableChunkNumber(final int number) {
30      this.number = number;
31    }
32  
33    @Override
34    public final int hashCode() {
35      return number;
36    }
37  
38    @Override
39    public final boolean equals(final Object obj) {
40      return obj instanceof HttpResumableChunkNumber &&
41             ((HttpResumableChunkNumber) obj).number == this.number;
42    }
43  
44    @Override
45    public String toString() {
46      return Integer.toString(number);
47    }
48  }
