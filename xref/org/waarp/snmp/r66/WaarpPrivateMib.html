197   protected WaarpPrivateMib(final String sysdesc, final int port,
198                             final int smiPrivateCodeFinal,
199                             final int typeWaarpObject,
200                             final String scontactName,
201                             final String stextualName, final String saddress,
202                             final int iservice) {
203     textualSysDecr = sysdesc;
204     smiPrivateCode = smiPrivateCodeFinal;
205     smiTypeWaarp = typeWaarpObject;
206     contactName = scontactName;
207     textualName = stextualName;
208     address = saddress;
209     service = iservice;
210     srootOIDWaarp =
211         rootEnterpriseMib + "." + smiPrivateCode + '.' + smiTypeWaarp;
212     applicationProtocolBase = srootOIDWaarp + ".1.1.4.";
213     ggObjectId = new OID(srootOIDWaarp);
214     applicationProtocol = new OID(applicationProtocolBase + port);
215     rootOIDWaarp = new OID(srootOIDWaarp);
216     rootOIDWaarpInfo = new OID(srootOIDWaarp + ".1");
217     rootOIDWaarpGlobal = new OID(srootOIDWaarp + ".2");
218     rootOIDWaarpGlobalUptime = new OID(
219         rootOIDWaarpGlobal + "." + WaarpGlobalValuesIndex.applUptime.getOID() +
220         ".0");
221     rootOIDWaarpDetailed = new OID(srootOIDWaarp + ".3");
222     rootOIDWaarpError = new OID(srootOIDWaarp + ".4");
223     rootOIDWaarpNotif = new OID(srootOIDWaarp + ".5.1");
224     rootOIDWaarpNotifStartOrShutdown = new OID(srootOIDWaarp + ".5.1.1.1");

242   /**
243    * Unregister and Register again the SNMPv2MIB with System adapted to this
244    * Mib
245    *
246    * @throws DuplicateRegistrationException
247    */
248   protected void agentRegisterSystem() throws DuplicateRegistrationException {
249     // Since BaseAgent registers some mibs by default we need to unregister
250     // one before we register our own sysDescr. Normally you would
251     // override that method and register the mibs that you need
252 
253     agent.unregisterManagedObject(agent.getSnmpv2MIB());
254 
255     // Register a system description, use one from you product environment
256     // to test with
257     snmpv2 = new SNMPv2MIB(new OctetString(textualSysDecr), ggObjectId,
258                            new Integer32(service));
259     snmpv2.setContact(new OctetString(contactName));
260     snmpv2.setLocation(new OctetString(address));
261     snmpv2.setName(new OctetString(textualName));
262     snmpv2.registerMOs(agent.getServer(), null);
263     if (logger.isDebugEnabled()) {
264       logger.debug("SNMPV2: {}:{}:{}:{}:{}:{}:{}", snmpv2.getContact(),
265                    snmpv2.getDescr(), snmpv2.getLocation(), snmpv2.getName(),
266                    snmpv2.getObjectID(), snmpv2.getServices(),
267                    snmpv2.getUpTime());
268     }
269     // Save UpTime reference since used everywhere
270     upTime = snmpv2.getSysUpTime();
271   }

273   /**
274    * Register this MIB
275    *
276    * @throws DuplicateRegistrationException
277    */
278   protected void defaultAgentRegisterWaarpMib()
279       throws DuplicateRegistrationException {
280     // register Static info
281     rowInfo = new WaarpMORow(this, rootOIDWaarpInfo, WaarpDefinition,
282                              MibLevel.staticInfo.ordinal());
283     rowInfo.registerMOs(agent.getServer(), null);
284     // register General info
285     rowGlobal = new WaarpMORow(this, rootOIDWaarpGlobal, WaarpGlobalValues,
286                                MibLevel.globalInfo.ordinal());
287     WaarpMOScalar memoryScalar =
288         rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryTotal.ordinal()];
289     memoryScalar.setValue(new MemoryGauge32(MemoryType.TotalMemory));
290     memoryScalar =
291         rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryFree.ordinal()];
292     memoryScalar.setValue(new MemoryGauge32(MemoryType.FreeMemory));
293     memoryScalar =
294         rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryUsed.ordinal()];
295     memoryScalar.setValue(new MemoryGauge32(MemoryType.UsedMemory));
296     rowGlobal.registerMOs(agent.getServer(), null);
297     // setup UpTime to SysUpTime and change status
298     scalarUptime =
299         rowGlobal.getRow()[WaarpGlobalValuesIndex.applUptime.ordinal()];
300     scalarUptime.setValue(new WaarpUptime(upTime));
301     changeStatus(OperStatus.restarting);
302     changeStatus(OperStatus.up);
303     // register Detailed info
304     rowDetailed =
305         new WaarpMORow(this, rootOIDWaarpDetailed, WaarpDetailedValues,
306                        MibLevel.detailedInfo.ordinal());
307     rowDetailed.registerMOs(agent.getServer(), null);
308     // register Error info
309     rowError = new WaarpMORow(this, rootOIDWaarpError, WaarpErrorValues,
310                               MibLevel.errorInfo.ordinal());
311     rowError.registerMOs(agent.getServer(), null);
312   }

345   /**
346    * Change the status and the LastChange Timeticks
347    *
348    * @param status
349    */
350   public void changeStatus(final OperStatus status) {
351     final WaarpMOScalar statusScalar =
352         rowGlobal.getRow()[WaarpGlobalValuesIndex.applOperStatus.ordinal()];
353     final Integer32 var = (Integer32) statusScalar.getValue();
354     if (var.getValue() != status.status) {
355       var.setValue(status.status);
356       final WaarpMOScalar lastTimeScalar =
357           rowGlobal.getRow()[WaarpGlobalValuesIndex.applLastChange.ordinal()];
358       final TimeTicks time = (TimeTicks) lastTimeScalar.getValue();
359       time.setValue(upTime.get().getValue());
360     }
361   }
