104   private File[] getRuleFiles() {
105     final File ruleDir = new File(
106         Configuration.configuration.getBaseDirectory() +
107         Configuration.configuration.getConfigPath());
108     final List<File> res = new ArrayList<File>();
109     if (ruleDir.isDirectory()) {
110       res.addAll(
111           Arrays.asList(ruleDir.listFiles(new ExtensionFilter(EXT_RULE))));
112       res.addAll(
113           Arrays.asList(ruleDir.listFiles(new ExtensionFilter(EXT_RULES))));
114     }
115     return res.toArray(FILE_0_LENGTH);
116   }

128   @Override
129   public List<Rule> getAll() throws DAOConnectionException {
130     final List<Rule> res = new ArrayList<Rule>();
131 
132     final File[] files = getRuleFiles();
133     for (final File ruleFile : files) {
134       try {
135         final DocumentBuilderFactory dbf = getDocumentBuilderFactory();
136         final Document document = dbf.newDocumentBuilder().parse(ruleFile);
137         // Setup XPath query
138         final XPath xPath = XPathFactory.newInstance().newXPath();
139         final XPathExpression xpe = xPath.compile(XML_GET_ALL);
140         final NodeList listNode =
141             (NodeList) xpe.evaluate(document, XPathConstants.NODESET);
142         // Iterate through all found nodes
143 
144         for (int i = 0; i < listNode.getLength(); i++) {
145           final Node node = listNode.item(i);
146           final Rule rule = getFromNode(node);
147           res.add(rule);
148           dbR66RuleHashMap.put(rule.getName(), rule);
149         }
150       } catch (final SAXException e) {
151         throw new DAOConnectionException(e);
152       } catch (final XPathExpressionException e) {
153         throw new DAOConnectionException(e);
154       } catch (final ParserConfigurationException e) {
155         throw new DAOConnectionException(e);
156       } catch (final IOException e) {
157         throw new DAOConnectionException(e);
158       }
159     }
160     return res;
161   }

192   private Rule getFromNode(Node parent) throws DAOConnectionException {
193     final Rule res = new Rule();
194 
195     final NodeList children = parent.getChildNodes();
196     for (int j = 0; j < children.getLength(); j++) {
197       final Node node = children.item(j);
198       if (node.getNodeName().equals(RULENAME_FIELD)) {
199         res.setName(node.getTextContent());
200       } else if (node.getNodeName().equals(HOSTIDS_LIST)) {
201         res.setHostids(retrieveHostids(node.getTextContent()));
202       } else if (node.getNodeName().equals(MODE_FIELD)) {
203         res.setMode(Integer.parseInt(node.getTextContent()));
204       } else if (node.getNodeName().equals(SEND_PATH_FIELD)) {
205         res.setSendPath(node.getTextContent());
206       } else if (node.getNodeName().equals(RECV_PATH_FIELD)) {
207         res.setRecvPath(node.getTextContent());
208       } else if (node.getNodeName().equals(ARCH_PATH_FIELD)) {
209         res.setArchivePath(node.getTextContent());
210       } else if (node.getNodeName().equals(WORK_PATH_FIELD)) {
211         res.setWorkPath(node.getTextContent());
212       } else if (node.getNodeName().equals(RPRE_TASKS_FIELD)) {
213         res.setRPreTasks(retrieveTasks(node));
214       } else if (node.getNodeName().equals(RPOST_TASKS_FIELD)) {
215         res.setRPostTasks(retrieveTasks(node));
216       } else if (node.getNodeName().equals(RERROR_TASKS_FIELD)) {
217         res.setRErrorTasks(retrieveTasks(node));
218       } else if (node.getNodeName().equals(SPRE_TASKS_FIELD)) {
219         res.setSPreTasks(retrieveTasks(node));
220       } else if (node.getNodeName().equals(SPOST_TASKS_FIELD)) {
221         res.setSPostTasks(retrieveTasks(node));
222       } else if (node.getNodeName().equals(SERROR_TASKS_FIELD)) {
223         res.setSErrorTasks(retrieveTasks(node));
224       }
225     }
226     return res;
227   }

231   private List<String> retrieveHostids(String xml)
232       throws DAOConnectionException {
233     final ArrayList<String> res = new ArrayList<String>();
234     if (xml == null || xml.isEmpty()) {
235       return res;
236     }
237     Document document;
238     InputStream stream = null;
239     try {
240       stream = new ByteArrayInputStream(xml.getBytes(WaarpStringUtils.UTF8));
241       final DocumentBuilderFactory dbf = getDocumentBuilderFactory();
242       document = dbf.newDocumentBuilder().parse(stream);
243     } catch (final IOException e) {
244       throw new DAOConnectionException(e);
245     } catch (final ParserConfigurationException e) {
246       throw new DAOConnectionException(e);
247     } catch (final SAXException e) {
248       throw new DAOConnectionException(e);
249     } finally {
250       if (stream != null) {
251         FileUtils.close(stream);
252       }
253     }
254     document.getDocumentElement().normalize();
255 
256     final NodeList hostsList = document.getElementsByTagName(HOSTID_FIELD);
257     for (int i = 0; i < hostsList.getLength(); i++) {
258       res.add(hostsList.item(i).getTextContent());
259     }
260     return res;
261   }

263   public static final String TASK_NODE = "task";
264   public static final String TYPE_FIELD = "type";
265   public static final String PATH_FIELD = "path";
266   public static final String DELAY_FIELD = "delay";
267 
268   private List<RuleTask> retrieveTasks(Node src) throws DAOConnectionException {
269     final List<RuleTask> res = new ArrayList<RuleTask>();
270     final NodeList feed = src.getChildNodes();
271     for (int i = 0; i < feed.getLength(); i++) {
272       final Node mainnode = feed.item(i);
273       if (mainnode.getNodeType() == Node.ELEMENT_NODE) {
274         final Element e = (Element) mainnode;
275         final NodeList tasksList = e.getElementsByTagName(TASK_NODE);
276         for (int j = 0; j < tasksList.getLength(); j++) {
277           final Node taskNode = tasksList.item(j);
278           if (taskNode.getNodeType() == Node.ELEMENT_NODE) {
279             final Element task = (Element) taskNode;
280             final String type =
281                 task.getElementsByTagName(TYPE_FIELD).item(0).getTextContent();
282             final String path =
283                 task.getElementsByTagName(PATH_FIELD).item(0).getTextContent();
284             final int delay = Integer.parseInt(
285                 task.getElementsByTagName(DELAY_FIELD).item(0)
286                     .getTextContent());
287             res.add(new RuleTask(type, path, delay));
288           }
289         }
290       }
291     }
292     return res;
293   }
