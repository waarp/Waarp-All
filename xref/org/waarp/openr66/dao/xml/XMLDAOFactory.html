21  package org.waarp.openr66.dao.xml;
22  
23  import org.waarp.openr66.dao.BusinessDAO;
24  import org.waarp.openr66.dao.DAOFactory;
25  import org.waarp.openr66.dao.HostDAO;
26  import org.waarp.openr66.dao.LimitDAO;
27  import org.waarp.openr66.dao.MultipleMonitorDAO;
28  import org.waarp.openr66.dao.RuleDAO;
29  import org.waarp.openr66.dao.TransferDAO;
30  import org.waarp.openr66.dao.exception.DAOConnectionException;
31  import org.waarp.openr66.protocol.configuration.Configuration;
32  
33  public class XMLDAOFactory extends DAOFactory {
34  
35    private final String confDir = Configuration.configuration.getConfigPath();
36  
37    private final String businessFile = confDir + "/business.xml";
38    private final String hostFile = Configuration.configuration.getAuthFile();
39    private final String limitFile = confDir + "/limit.xml";
40    private final String ruleFile = confDir;
41    private final String transferFile =
42        Configuration.configuration.getArchivePath();
43  
44    public XMLDAOFactory() {
45      // Empty
46    }
47  
48    @Override
49    public BusinessDAO getBusinessDAO() throws DAOConnectionException {
50      return new XMLBusinessDAO(businessFile);
51    }
52  
53    @Override
54    public HostDAO getHostDAO() throws DAOConnectionException {
55      return new XMLHostDAO(hostFile);
56    }
57  
58    @Override
59    public LimitDAO getLimitDAO() throws DAOConnectionException {
60      return new XMLLimitDAO(limitFile);
61    }
62  
63    @Override
64    public MultipleMonitorDAO getMultipleMonitorDAO()
65        throws DAOConnectionException {
66      throw new DAOConnectionException(
67          "MultipleMonitor is not supported on XML DAO");
68    }
69  
70    @Override
71    public RuleDAO getRuleDAO() throws DAOConnectionException {
72      return new XMLRuleDAO(ruleFile);
73    }
74  
75    @Override
76    public TransferDAO getTransferDAO() throws DAOConnectionException {
77      return new XMLTransferDAO(transferFile);
78    }
79  }
