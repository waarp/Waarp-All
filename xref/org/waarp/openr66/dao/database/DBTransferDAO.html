192   @Override
193   public void delete(final Transfer transfer)
194       throws DAOConnectionException, DAONoDataException {
195     PreparedStatement stm = null;
196     final Object[] params = {
197         transfer.getId(), transfer.getRequester(), transfer.getRequested(),
198         transfer.getOwnerRequest()
199     };
200     try {
201       stm = connection.prepareStatement(getDeleteRequest());
202       setParameters(stm, params);
203       try {
204         executeAction(stm);
205       } catch (final SQLException e2) {
206         throw new DAONoDataException(e2);
207       }
208     } catch (final SQLException e) {
209       throw new DAOConnectionException(e);
210     } finally {
211       closeStatement(stm);
212     }
213   }

215   private String prepareFindQuery(final List<Filter> filters,
216                                   final Object[] params) {
217     final StringBuilder query = new StringBuilder(getGetAllRequest());
218     final Iterator<Filter> it = filters.listIterator();
219     if (it.hasNext()) {
220       query.append(WHERE);
221     }
222     String prefix = "";
223     int i = 0;
224     while (it.hasNext()) {
225       query.append(prefix);
226       final Filter filter = it.next();
227       query.append(filter.key).append(' ').append(filter.operand).append(" ?");
228       params[i] = filter.value;
229       i++;
230       prefix = " AND ";
231     }
232     return query.toString();
233   }

235   @Override
236   public List<Transfer> find(final List<Filter> filters, final int limit)
237       throws DAOConnectionException {
238     final ArrayList<Transfer> transfers = new ArrayList<Transfer>();
239     // Create the SQL query
240     final Object[] params = new Object[filters.size()];
241     final StringBuilder query =
242         new StringBuilder(prepareFindQuery(filters, params));
243     // Set LIMIT
244     if (limit > 0) {
245       query.append(LIMIT2).append(limit);
246     }
247     // Execute query
248     PreparedStatement stm = null;
249     ResultSet res = null;
250     try {
251       stm = connection.prepareStatement(query.toString());
252       setParameters(stm, params);
253       res = executeQuery(stm);
254       while (res.next()) {
255         transfers.add(getFromResultSet(res));
256       }
257     } catch (final SQLException e) {
258       throw new DAOConnectionException(e);
259     } finally {
260       closeResultSet(res);
261       closeStatement(stm);
262     }
263     return transfers;
264   }

302   public List<Transfer> find(final List<Filter> filters, final String column,
303                              final boolean ascend)
304       throws DAOConnectionException {
305     final ArrayList<Transfer> transfers = new ArrayList<Transfer>();
306     // Create the SQL query
307     final Object[] params = new Object[filters.size()];
308     final StringBuilder query =
309         new StringBuilder(prepareFindQuery(filters, params));
310     // Set ORDER BY
311     if (column != null && !column.isEmpty()) {
312       query.append(" ORDER BY ").append(column);
313       if (!ascend) {
314         query.append(" DESC");
315       }
316     }
317     // Execute query
318     PreparedStatement stm = null;
319     ResultSet res = null;
320     try {
321       stm = connection.prepareStatement(query.toString());
322       setParameters(stm, params);
323       logger.trace(stm.toString());
324       res = executeQuery(stm);
325       while (res.next()) {
326         transfers.add(getFromResultSet(res));
327       }
328     } catch (final SQLException e) {
329       throw new DAOConnectionException(e);
330     } finally {
331       closeResultSet(res);
332       closeStatement(stm);
333     }
334     return transfers;

437   @Override
438   public Transfer select(final long id, final String requester,
439                          final String requested, final String owner)
440       throws DAOConnectionException, DAONoDataException {
441     PreparedStatement stm = null;
442     ResultSet res = null;
443     final Object[] params = { id, requester, requested, owner };
444     try {
445       stm = connection.prepareStatement(getSelectRequest());
446       setParameters(stm, params);
447       res = executeQuery(stm);
448       if (res.next()) {
449         return getFromResultSet(res);
450       } else {
451         throw new DAONoDataException("No " + getClass().getName() + " found");
452       }
453     } catch (final SQLException e) {
454       throw new DAOConnectionException(e);
455     } finally {
456       closeResultSet(res);
457       closeStatement(stm);
458     }

463   @Override
464   public void insert(final Transfer transfer) throws DAOConnectionException {
465     if (transfer.getId() == ILLEGALVALUE) {
466       transfer.setId(getNextId());
467     }
468     logger.trace("TRACE ID {}", transfer.getId());
469     super.insert(transfer);
470   }

472   @Override
473   public Transfer getFromResultSet(final ResultSet set) throws SQLException {
474     return new Transfer(set.getLong(ID_FIELD), set.getString(ID_RULE_FIELD),
475                         set.getInt(TRANSFER_MODE_FIELD),
476                         set.getString(FILENAME_FIELD),
477                         set.getString(ORIGINAL_NAME_FIELD),
478                         set.getString(FILE_INFO_FIELD),
479                         set.getBoolean(IS_MOVED_FIELD),
480                         set.getInt(BLOCK_SIZE_FIELD),
481                         set.getBoolean(RETRIEVE_MODE_FIELD),
482                         set.getString(OWNER_REQUEST_FIELD),
483                         set.getString(REQUESTER_FIELD),
484                         set.getString(REQUESTED_FIELD),
485                         set.getString(TRANSFER_INFO_FIELD), Transfer.TASKSTEP
486                             .valueOf(set.getInt(GLOBAL_STEP_FIELD)),
487                         Transfer.TASKSTEP
488                             .valueOf(set.getInt(GLOBAL_LAST_STEP_FIELD)),
489                         set.getInt(STEP_FIELD),
490                         ErrorCode.getFromCode(set.getString(STEP_STATUS_FIELD)),
491                         ErrorCode.getFromCode(set.getString(INFO_STATUS_FIELD)),
492                         set.getInt(RANK_FIELD),
493                         set.getTimestamp(TRANSFER_START_FIELD),
494                         set.getTimestamp(TRANSFER_STOP_FIELD),
495                         UpdatedInfo.valueOf(set.getInt(UPDATED_INFO_FIELD)));
496   }
