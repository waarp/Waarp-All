67  /**
68   * Elasticsearch client for Waarp
69   */
70  public class ElasticsearchMonitoringExporterClientImpl
71      implements ElasticsearchMonitoringExporterClient {
72    private static final WaarpLogger logger = WaarpLoggerFactory.getLogger(
73        ElasticsearchMonitoringExporterClientImpl.class);
74  
75    protected final String index;
76    protected final RestClientBuilder builder;
77  
78    protected RestHighLevelClient client;
79  
80    /**
81     * Note that only one among (username/pwd, token, apikey) is allowed and
82     * will be taken into account.
83     *
84     * @param username username to connect to Elasticsearch if any (Basic
85     *     authentication) (nullable)
86     * @param pwd password to connect to Elasticsearch if any (Basic
87     *     authentication) (nullable)
88     * @param token access token (Bearer Token authorization
89     *     by Header) (nullable)
90     * @param apiKey API Key (Base64 of 'apiId:apiKey') (ApiKey authorization
91     *     by Header) (nullable)
92     * @param prefix as '/prefix' or null if none
93     * @param index as 'waarpr66monitor' as the index name within
94     *     Elasticsearch, including extra dynamic information
95     * @param compression True to compress REST exchanges between the client
96     *     and the Elasticsearch server
97     * @param httpHosts array of HttpHost
98     */
99    public ElasticsearchMonitoringExporterClientImpl(final String username,
100                                                    final String pwd,
101                                                    final String token,
102                                                    final String apiKey,
103                                                    final String prefix,
104                                                    final String index,
105                                                    final boolean compression,
106                                                    final HttpHost... httpHosts) {
107     this.index = index;
108     builder = RestClient.builder(httpHosts).setCompressionEnabled(compression);
109     if (ParametersChecker.isNotEmpty(prefix)) {
110       builder.setPathPrefix(prefix);
111     }
112     int headerLen = 1;
113     if (ParametersChecker.isNotEmpty(apiKey, token)) {
114       headerLen = 2;
115     }
116     final Header[] defaultHeaders = new Header[headerLen];
117     headerLen = 0;
118     if (ParametersChecker.isNotEmpty(token)) {
119       defaultHeaders[headerLen] =
120           new BasicHeader("Authorization", "Bearer " + token);
121       headerLen++;
122     } else if (ParametersChecker.isNotEmpty(apiKey)) {
123       defaultHeaders[headerLen] =
124           new BasicHeader("Authorization", "ApiKey " + apiKey);
125       headerLen++;
126     }
127     if (headerLen > 0) {
128       builder.setDefaultHeaders(defaultHeaders);
129     }
130     boolean tls = false;
131     for (final HttpHost httpHost : httpHosts) {
132       tls |= httpHost.getSchemeName().equalsIgnoreCase("https");
133     }
134     final SSLContext sslContext;
135     if (tls) {
136       try {
137         final SSLContextBuilder sslBuilder = SSLContexts.custom()
138                                                         .loadKeyMaterial(
139                                                             NetworkSslServerInitializer.getWaarpSecureKeyStore()
140                                                                                        .getKeyStore(),
141                                                             NetworkSslServerInitializer.getWaarpSecureKeyStore()
142                                                                                        .getKeyStorePassword())
143                                                         .loadTrustMaterial(
144                                                             NetworkSslServerInitializer.getWaarpSecureKeyStore()
145                                                                                        .getKeyTrustStore(),
146                                                             null);
147         sslContext = sslBuilder.build();
148       } catch (final NoSuchAlgorithmException | KeyStoreException | UnrecoverableKeyException | KeyManagementException e) {
149         logger.error(e.getMessage());
150         throw new IllegalArgumentException(e);
151       }
152     } else {
153       sslContext = null;
154     }
155     if (ParametersChecker.isNotEmpty(username, pwd)) {
156       final CredentialsProvider credentialsProvider =
157           new BasicCredentialsProvider();
158       credentialsProvider.setCredentials(AuthScope.ANY,
159                                          new UsernamePasswordCredentials(
160                                              username, pwd));
161       builder.setHttpClientConfigCallback(new HttpClientConfigCallback() {
162         @Override
163         public HttpAsyncClientBuilder customizeHttpClient(
164             final HttpAsyncClientBuilder httpClientBuilder) {
165           if (sslContext != null) {
166             return httpClientBuilder.setDefaultCredentialsProvider(
167                 credentialsProvider).setSSLContext(sslContext);
168           } else {
169             return httpClientBuilder.setDefaultCredentialsProvider(
170                 credentialsProvider);
171           }
172         }
173       });
174     }
175     logger.info("Elasticsearch client: user {} pwd {} token {} apikey {} " +
176                 "prefix {} index {}", username, pwd, token, apiKey, prefix,
177                 index);
178     client = new RestHighLevelClient(builder);
179   }
180 
181   @Override
182   public final boolean post(final ObjectNode monitoredTransfers,
183                             final DateTime start, final DateTime stop,
184                             final String serverId) {
185     if (client == null) {
186       client = new RestHighLevelClient(builder);
187     }
188     final String finalIndex = index.replace(ELASTIC_WAARPHOST, serverId)
189                                    .replaceAll(ELASTIC_DATETIME,
190                                                stop.toString(FORMAT_DATETIME))
191                                    .replaceAll(ELASTIC_DATEHOUR,
192                                                stop.toString(FORMAT_DATEHOUR))
193                                    .replaceAll(ELASTIC_DATE,
194                                                stop.toString(FORMAT_DATE))
195                                    .replaceAll(ELASTIC_YEAR_MONTH,
196                                                stop.toString(FORMAT_YEAR_MONTH))
197                                    .replaceAll(ELASTIC_YEAR,
198                                                stop.toString(FORMAT_YEAR))
199                                    .toLowerCase();
200     logger.debug("Will post to {}", finalIndex);
201     final BulkRequest bulkRequest = new BulkRequest(finalIndex);
202     final ArrayNode arrayNode = (ArrayNode) monitoredTransfers.get(RESULTS);
203     final Iterator<JsonNode> iterator = arrayNode.elements();
204     while (iterator.hasNext()) {
205       final ObjectNode node = (ObjectNode) iterator.next();
206       final IndexRequest indexRequest =
207           new IndexRequest().id(node.get(UNIQUE_ID).asText());
208       indexRequest.source(
209           JsonUtils.nodeToString(node).getBytes(StandardCharsets.UTF_8),
210           XContentType.JSON);
211       bulkRequest.add(indexRequest);
212     }
213     final BulkResponse bulkResponse;
214     try {
215       bulkResponse = client.bulk(bulkRequest, RequestOptions.DEFAULT);
216     } catch (final IOException e) {
217       logger.error(e.getMessage());
218       return false;
219     }
220     logger.debug("ES failure? {} {}", bulkResponse.hasFailures(),
221                  bulkResponse.status().getStatus());
222     if (logger.isDebugEnabled() && bulkResponse.hasFailures()) {
223       final Iterator<BulkItemResponse> iterator1 = bulkResponse.iterator();
224       while (iterator1.hasNext()) {
225         final BulkItemResponse response = iterator1.next();
226         logger.debug("ES item: {}", response.getFailureMessage());
227       }
228     }
229     return !bulkResponse.hasFailures();
230   }

232   @Override
233   public final void close() {
234     try {
235       client.close();
236     } catch (final IOException e) {
237       SysErrLogger.FAKE_LOGGER.ignoreLog(e);
238     }
239     client = null;
240   }
