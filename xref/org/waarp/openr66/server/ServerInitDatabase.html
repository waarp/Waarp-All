46   * Utility class to initiate the database for a server
47   */
48  public class ServerInitDatabase {
49    /**
50     * Internal Logger
51     */
52    static volatile WaarpLogger logger;
53  
54    protected static final String INFO_ARGS =
55        Messages.getString("ServerInitDatabase.Help");
56  
57    static String sxml;
58    static boolean database;
59    static boolean upgradeDb;
60    static String sbusiness;
61    static String salias;
62    static String sroles;
63    static String sdirconfig;
64    static String shostauth;
65    static String slimitconfig;
66  
67    private ServerInitDatabase() {
68    }
69  
70    protected static boolean getParams(final String[] args) {
71      if (logger == null) {
72        logger = WaarpLoggerFactory.getLogger(ServerInitDatabase.class);
73      }
74      if (args.length < 1) {
75        logger.error(INFO_ARGS);
76        return false;
77      }
78      sxml = args[0];
79      for (int i = 1; i < args.length; i++) {
80        if ("-initdb".equalsIgnoreCase(args[i])) {
81          database = true;
82          FileBasedConfiguration.autoupgrade = false;
83          upgradeDb = true;
84        } else if ("-upgradeDb".equalsIgnoreCase(args[i])) {
85          upgradeDb = true;
86        } else if ("-loadBusiness".equalsIgnoreCase(args[i])) {
87          i++;
88          sbusiness = args[i];
89        } else if ("-loadAlias".equalsIgnoreCase(args[i])) {
90          i++;
91          salias = args[i];
92        } else if ("-loadRoles".equalsIgnoreCase(args[i])) {
93          i++;
94          sroles = args[i];
95        } else if ("-dir".equalsIgnoreCase(args[i])) {
96          i++;
97          sdirconfig = args[i];
98        } else if ("-limit".equalsIgnoreCase(args[i])) {
99          i++;
100         slimitconfig = args[i];
101       } else if ("-auth".equalsIgnoreCase(args[i])) {
102         i++;
103         shostauth = args[i];
104       }
105     }
106     return true;
107   }

109   /**
110    * @param args as config_database file [rules_directory host_authent
111    *     limit_configuration]
112    */
113   public static void main(final String[] args) {
114     WaarpLoggerFactory
115         .setDefaultFactoryIfNotSame(new WaarpSlf4JLoggerFactory(null));
116     if (logger == null) {
117       logger = WaarpLoggerFactory.getLogger(ServerInitDatabase.class);
118     }
119     if (!getParams(args)) {
120       SysErrLogger.FAKE_LOGGER.sysout(INFO_ARGS);
121       if (admin != null) {
122         admin.close();
123       }
124       if (DetectionUtils.isJunit()) {
125         return;
126       }
127       ChannelUtils.stopLogger();
128       System.exit(2);//NOSONAR
129     }
130 
131     try {
132       if (!FileBasedConfiguration
133           .setConfigurationInitDatabase(Configuration.configuration, args[0],
134                                         database)) {
135         System.err
136             .format(Messages.getString("Configuration.NeedCorrectConfig"));
137         SysErrLogger.FAKE_LOGGER.syserr();
138         if (DetectionUtils.isJunit()) {
139           return;
140         }
141         ChannelUtils.stopLogger();
142         System.exit(2);//NOSONAR
143       }
144       if (database) {
145         // Init database
146         SysErrLogger.FAKE_LOGGER.sysoutFormat(
147             Messages.getString("ServerInitDatabase.Create.Start"));
148         SysErrLogger.FAKE_LOGGER.sysout();
149         initdb();
150         SysErrLogger.FAKE_LOGGER
151             .sysoutFormat(Messages.getString("ServerInitDatabase.Create.Done"));
152         SysErrLogger.FAKE_LOGGER.sysout();
153       }
154       if (upgradeDb) {
155         // try to upgrade DB schema
156         SysErrLogger.FAKE_LOGGER.sysoutFormat(
157             Messages.getString("ServerInitDatabase.Upgrade.Start"));
158         SysErrLogger.FAKE_LOGGER.sysout();
159         // TODO Split check for update and upgrade actions
160         if (!upgradedb()) {
161           SysErrLogger.FAKE_LOGGER.syserr(
162               Messages.getString("ServerInitDatabase.SchemaNotUptodate"));
163           SysErrLogger.FAKE_LOGGER.syserr();
164           if (DetectionUtils.isJunit()) {
165             return;
166           }
167           System.exit(1);//NOSONAR
168         }
169         SysErrLogger.FAKE_LOGGER.sysoutFormat(
170             Messages.getString("ServerInitDatabase.Upgrade.Done"));
171         SysErrLogger.FAKE_LOGGER.sysout();
172       }
173       // Try to load some element directly into database from first
174       // configuration file
175       FileBasedConfiguration
176           .setConfigurationServerFromXml(Configuration.configuration, args[0]);
177       if (sdirconfig != null) {
178         // load Rules
179         SysErrLogger.FAKE_LOGGER.sysoutFormat(
180             Messages.getString("ServerInitDatabase.LoadRule.Start"),
181             sdirconfig);
182         SysErrLogger.FAKE_LOGGER.sysout();
183         final File dirConfig = new File(sdirconfig);
184         if (dirConfig.isDirectory()) {
185           if (!loadRules(dirConfig)) {
186             SysErrLogger.FAKE_LOGGER.sysoutFormat(
187                 Messages.getString("ServerInitDatabase.LoadRule.Failed"));
188             if (DetectionUtils.isJunit()) {
189               return;
190             }
191             System.exit(1);//NOSONAR
192           }
193         } else {
194           System.err.format(
195               Messages.getString("ServerInitDatabase.LoadRule.NoDirectory"),
196               sdirconfig);
197           SysErrLogger.FAKE_LOGGER.syserr();
198           if (DetectionUtils.isJunit()) {
199             return;
200           }
201           System.exit(1);//NOSONAR
202         }
203         SysErrLogger.FAKE_LOGGER.sysoutFormat(
204             Messages.getString("ServerInitDatabase.LoadRule.Done"));
205         SysErrLogger.FAKE_LOGGER.sysout();
206       }
207       if (shostauth != null) {
208         // Load Host Authentications
209         System.out
210             .format(Messages.getString("ServerInitDatabase.LoadAuth.Start"),
211                     shostauth);
212         SysErrLogger.FAKE_LOGGER.sysout();
213         if (!loadHostAuth(shostauth)) {
214           System.err
215               .format(Messages.getString("ServerInitDatabase.LoadAuth.Failed"));
216           SysErrLogger.FAKE_LOGGER.syserr();
217           if (DetectionUtils.isJunit()) {
218             return;
219           }
220           System.exit(1);//NOSONAR
221         }
222         System.out
223             .format(Messages.getString("ServerInitDatabase.LoadAuth.Done"));
224         SysErrLogger.FAKE_LOGGER.sysout();
225       }
226       if (slimitconfig != null) {
227         // Load configuration
228         System.out
229             .format(Messages.getString("ServerInitDatabase.LoadLimit.Start"),
230                     slimitconfig);
231         SysErrLogger.FAKE_LOGGER.sysout();
232         if (!FileBasedConfiguration
233             .setConfigurationLoadLimitFromXml(Configuration.configuration,
234                                               slimitconfig)) {
235           System.err.format(
236               Messages.getString("ServerInitDatabase.LoadLimit.Failed"));
237           SysErrLogger.FAKE_LOGGER.syserr();
238           if (DetectionUtils.isJunit()) {
239             return;
240           }
241           System.exit(1);//NOSONAR
242         }
243         System.out
244             .format(Messages.getString("ServerInitDatabase.LoadLimit.Done"));
245         SysErrLogger.FAKE_LOGGER.sysout();
246       }
247       if (sbusiness != null || salias != null || sroles != null) {
248         if (sbusiness != null) {
249           sbusiness = getXMLFromFile(sbusiness);
250         }
251         if (salias != null) {
252           salias = getXMLFromFile(salias);
253         }
254         if (sroles != null) {
255           sroles = getXMLFromFile(sroles);
256         }
257         DbHostConfiguration hostConfiguration;
258         try {
259           hostConfiguration =
260               new DbHostConfiguration(Configuration.configuration.getHostId());
261           if (salias != null && !salias.isEmpty()) {
262             hostConfiguration.setAliases(salias);
263           }
264           if (sbusiness != null && !sbusiness.isEmpty()) {
265             hostConfiguration.setBusiness(sbusiness);
266           }
267           if (sroles != null && !sroles.isEmpty()) {
268             hostConfiguration.setRoles(sroles);
269           }
270           hostConfiguration.update();
271         } catch (final WaarpDatabaseException e) {
272           hostConfiguration =
273               new DbHostConfiguration(Configuration.configuration.getHostId(),
274                                       sbusiness, sroles, salias, "");
275           hostConfiguration.insert();
276         }
277       }
278       SysErrLogger.FAKE_LOGGER
279           .sysout(Messages.getString("ServerInitDatabase.LoadDone"));
280       SysErrLogger.FAKE_LOGGER.sysout();
281       if (DetectionUtils.isJunit()) {
282         return;
283       }
284       System.exit(0);//NOSONAR
285     } catch (final WaarpDatabaseException e) {
286       SysErrLogger.FAKE_LOGGER
287           .syserr(Messages.getString("ServerInitDatabase.ErrDatabase"));
288       SysErrLogger.FAKE_LOGGER.syserr();
289       if (DetectionUtils.isJunit()) {
290         return;
291       }
292       System.exit(3);//NOSONAR
293     } finally {
294       if (admin != null) {
295         admin.close();
296       }
297     }
298   }

300   private static String getXMLFromFile(final String path) {
301     String res = "";
302     final File file = new File(path);
303     if (file.canRead()) {
304       try {
305         final String value =
306             FileUtils.readFileToString(file, WaarpStringUtils.UTF8);
307         if (value != null && !value.trim().isEmpty()) {
308           res = value.replaceAll("\r|\n|  ", " ").trim();
309         }
310       } catch (final IOException e) {
311         SysErrLogger.FAKE_LOGGER.syserr(e);
312       }
313     }
314     return res;
315   }

317   public static void initdb() throws WaarpDatabaseNoConnectionException {
318     // Create tables: configuration, hosts, rules, runner, cptrunner
319     admin.getSession().getAdmin().getDbModel().createTables(admin.getSession());
320   }

322   /**
323    * @return True if the base is up to date, else False (need Upgrade)
324    */
325   public static boolean upgradedb() {
326     if (logger == null) {
327       logger = WaarpLoggerFactory.getLogger(ServerInitDatabase.class);
328     }
329     // Update tables: runner
330     boolean uptodate = true;
331     // Check if the database is up to date
332     final String version = DbHostConfiguration
333         .getVersionDb(Configuration.configuration.getHostId());
334     try {
335       if (version != null) {
336         uptodate = admin.getSession().getAdmin().getDbModel()
337                         .needUpgradeDb(admin.getSession(), version, true);
338       } else {
339         uptodate = admin.getSession().getAdmin().getDbModel()
340                         .needUpgradeDb(admin.getSession(), "1.1.0", true);
341       }
342       if (uptodate) {
343         logger.error(Messages.getString(
344             "ServerInitDatabase.SchemaNotUptodate")); //$NON-NLS-1$
345         return false;
346       } else {
347         logger.debug(Messages.getString(
348             "ServerInitDatabase.SchemaUptodate")); //$NON-NLS-1$
349       }
350     } catch (final WaarpDatabaseNoConnectionException e) {
351       logger
352           .error(Messages.getString("Database.CannotConnect"), e); //$NON-NLS-1$
353       return false;
354     }
355     return !uptodate;
356   }

358   public static boolean loadRules(final File dirConfig)
359       throws WaarpDatabaseException {
360     try {
361       RuleFileBasedConfiguration.importRules(dirConfig);
362     } catch (final OpenR66ProtocolSystemException e) {
363       SysErrLogger.FAKE_LOGGER.syserr(e);
364       return false;
365     }
366     return true;
367   }

369   public static boolean loadHostAuth(final String filename) {
370     return AuthenticationFileBasedConfiguration
371         .loadAuthentication(Configuration.configuration, filename);
372   }
