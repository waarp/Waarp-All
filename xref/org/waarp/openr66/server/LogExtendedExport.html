161   @Override
162   public void run() {
163     if (logger == null) {
164       logger = WaarpLoggerFactory.getLogger(LogExtendedExport.class);
165     }
166     if (!(statusdone || statuserror || statuspending || statustransfer)) {
167       logger.error("No action required");
168       future.setResult(new R66Result(
169           new OpenR66ProtocolNoDataException("No action required"), null, true,
170           ErrorCode.IncorrectCommand, null));
171       future.setFailure(future.getResult().getException());
172       return;
173     }
174 
175     final byte type = purgeLog? LocalPacketFactory.LOGPURGEPACKET :
176         LocalPacketFactory.LOGPACKET;
177     final LogJsonPacketandler/packet/json/LogJsonPacket.html#LogJsonPacket">LogJsonPacket node = new LogJsonPacket();
178     node.setClean(clean);
179     node.setPurge(purgeLog);
180     node.setStart(start);
181     node.setStop(stop);
182     node.setStartid(startid);
183     node.setStopid(stopid);
184     node.setRule(rule);
185     node.setRequest(request);
186     node.setStatuspending(statuspending);
187     node.setStatustransfer(statustransfer);
188     node.setStatuserror(statuserror);
189     node.setStatusdone(statusdone);
190 
191     final JsonCommandPacketr/packet/JsonCommandPacket.html#JsonCommandPacket">JsonCommandPacket valid = new JsonCommandPacket(node, type);
192     logger.debug("ExtendedLogCommand: " + valid.getRequest());
193     final R66Future66Future.html#R66Future">R66Future newFuture = new R66Future(true);
194     final LocalChannelReference localChannelReference =
195         AbstractTransfer.tryConnect(host, newFuture, networkTransaction);
196     if (localChannelReference == null) {
197       future.setResult(newFuture.getResult());
198       future.setFailure(future.getCause());
199       return;
200     }
201     localChannelReference.sessionNewState(R66FiniteDualStates.VALIDOTHER);
202     try {
203       ChannelUtils
204           .writeAbstractLocalPacket(localChannelReference, valid, false);
205     } catch (final OpenR66ProtocolPacketException e) {
206       logger.error("Bad Protocol", e);
207       localChannelReference.close();
208       host = null;
209       future.setResult(
210           new R66Result(e, null, true, ErrorCode.TransferError, null));
211       future.setFailure(e);
212       return;
213     }
214     host = null;
215     newFuture.awaitOrInterruptible();
216     logger.info(
217         "Request done with " + (newFuture.isSuccess()? "success" : "error"));
218     if (newFuture.isSuccess() && ruleDownload != null &&
219         !ruleDownload.isEmpty()) {
220       try {
221         importLog(newFuture);
222       } catch (final OpenR66ProtocolBusinessException e) {
223         localChannelReference.close();
224         return;
225       }
226     }
227     future.setFilesize(newFuture.getFilesize());
228     future.setRunner(newFuture.getRunner());
229     future.setResult(newFuture.getResult());
230     if (newFuture.isSuccess()) {
231       future.setSuccess();
232     } else {
233       if (newFuture.getCause() != null) {
234         future.setFailure(newFuture.getCause());
235       }
236       future.cancel();
237     }
238     localChannelReference.close();
239   }

241   public void importLog(final R66Future future)
242       throws OpenR66ProtocolBusinessException {
243     if (future.isSuccess()) {
244       final JsonCommandPacket packet =
245           (JsonCommandPacket) future.getResult().getOther();
246       if (packet != null) {
247         final LogResponseJsonPacket res =
248             (LogResponseJsonPacket) packet.getJsonRequest();
249         final String fileExported = res.getFilename();
250         // download logs
251         if (fileExported != null && !fileExported.isEmpty()) {
252           final String ruleToExport = ruleDownload;
253           final R66Future6Future.html#R66Future">R66Future futuretemp = new R66Future(true);
254           final DirectTransfer transfer =
255               new DirectTransfer(futuretemp, host.getHostid(), fileExported,
256                                  ruleToExport, "Get Exported Logs from " +
257                                                Configuration.configuration
258                                                    .getHostId(), false,
259                                  Configuration.configuration.getBlockSize(),
260                                  ILLEGALVALUE, networkTransaction);
261           transfer.run();
262           final File logsFile;
263           if (!futuretemp.isSuccess()) {
264             if (futuretemp.getCause() != null) {
265               throw new OpenR66ProtocolBusinessException(futuretemp.getCause());
266             }
267             throw new OpenR66ProtocolBusinessException(
268                 "Download of exported logs in error");
269           } else {
270             logsFile = futuretemp.getResult().getFile().getTrueFile();
271           }
272           if (tryimport) {
273             try {
274               DbTaskRunner.loadXml(logsFile);
275             } catch (final OpenR66ProtocolBusinessException e) {
276               logger.warn(
277                   "Cannot load the logs from " + logsFile.getAbsolutePath() +
278                   " since: " + e.getMessage());
279               throw new OpenR66ProtocolBusinessException(
280                   "Cannot load the logs from " + logsFile.getAbsolutePath(), e);
281             }
282           }
283         } else {
284           throw new OpenR66ProtocolBusinessException(
285               "Export log with no file result");
286         }
287       } else {
288         throw new OpenR66ProtocolBusinessException(
289             "Export log with no file result");
290       }
291     } else {
292       if (future.getCause() != null) {
293         throw new OpenR66ProtocolBusinessException(future.getCause());
294       }
295       throw new OpenR66ProtocolBusinessException("Export log in error");
296     }
297   }

315   protected static boolean getParams(final String[] args) {
316     if (logger == null) {
317       logger = WaarpLoggerFactory.getLogger(LogExtendedExport.class);
318     }
319     if (args.length < 1) {
320       logger.error(INFO_ARGS);
321       return false;
322     }
323     if (!FileBasedConfiguration
324         .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
325       logger.error(INFO_ARGS);
326       return false;
327     }
328     String ssstart = null;
329     String ssstop = null;
330     for (int i = 1; i < args.length; i++) {
331       if ("-purge".equalsIgnoreCase(args[i])) {
332         spurgeLog = true;
333       } else if ("-clean".equalsIgnoreCase(args[i])) {
334         sclean = true;
335       } else if ("-start".equalsIgnoreCase(args[i])) {
336         i++;
337         ssstart = args[i];
338       } else if ("-stop".equalsIgnoreCase(args[i])) {
339         i++;
340         ssstop = args[i];
341       } else if ("-startid".equalsIgnoreCase(args[i])) {
342         i++;
343         sstartid = args[i];
344       } else if ("-stopid".equalsIgnoreCase(args[i])) {
345         i++;
346         sstopid = args[i];
347       } else if ("-rule".equalsIgnoreCase(args[i])) {
348         i++;
349         srule = args[i];
350       } else if ("-request".equalsIgnoreCase(args[i])) {
351         i++;
352         srequest = args[i];
353       } else if ("-pending".equalsIgnoreCase(args[i])) {
354         sstatuspending = true;
355       } else if ("-transfer".equalsIgnoreCase(args[i])) {
356         sstatustransfer = true;
357       } else if ("-done".equalsIgnoreCase(args[i])) {
358         sstatusdone = true;
359       } else if ("-error".equalsIgnoreCase(args[i])) {
360         sstatuserror = true;
361       } else if ("-import".equalsIgnoreCase(args[i])) {
362         stryimport = true;
363       } else if ("-host".equalsIgnoreCase(args[i])) {
364         i++;
365         stohost = args[i];
366       } else if ("-ruleDownload".equalsIgnoreCase(args[i])) {
367         i++;
368         sruleDownload = args[i];
369       }
370     }
371     if (ssstart != null) {
372       final Timestamp tstart = WaarpStringUtils.fixDate(ssstart);
373       if (tstart != null) {
374         sstart = tstart;
375       }
376     }
377     if (ssstop != null) {
378       final Timestamp tstop = WaarpStringUtils.fixDate(ssstop, sstart);
379       if (tstop != null) {
380         sstop = tstop;
381       }
382     }
383     if (ssstart == null && ssstop == null) {
384       sstop = WaarpStringUtils.getTodayMidnight();
385     }
386     if (stohost == null) {
387       stryimport = false;
388     }
389     return true;
390   }

392   public static void main(final String[] args) {
393     WaarpLoggerFactory
394         .setDefaultFactoryIfNotSame(new WaarpSlf4JLoggerFactory(null));
395     if (logger == null) {
396       logger = WaarpLoggerFactory.getLogger(LogExtendedExport.class);
397     }
398     if (!getParams(args)) {
399       logger.error("Wrong initialization");
400       if (admin != null) {
401         admin.close();
402       }
403       System.exit(1);//NOSONAR
404     }
405     final long time1 = System.currentTimeMillis();
406     DbHostAuth dbhost = null;
407     if (stohost != null) {
408       try {
409         dbhost = new DbHostAuth(stohost);
410       } catch (final WaarpDatabaseException e) {
411         logger.error("Wrong initialization");
412         if (admin != null) {
413           admin.close();
414         }
415         System.exit(2);//NOSONAR
416       }
417     } else {
418       dbhost = Configuration.configuration.getHostSslAuth();
419       stohost = Configuration.configuration.getHostSslId();
420     }
421     final R66Futures/R66Future.html#R66Future">R66Future future = new R66Future(true);
422 
423     Configuration.configuration.pipelineInit();
424     final NetworkTransactionnsaction.html#NetworkTransaction">NetworkTransaction networkTransaction = new NetworkTransaction();
425     try {
426       final LogExtendedExport transaction =
427           new LogExtendedExport(future, sclean, spurgeLog, sstart, sstop,
428                                 sstartid, sstopid, srule, srequest,
429                                 sstatuspending, sstatustransfer, sstatusdone,
430                                 sstatuserror, networkTransaction, dbhost);
431       transaction.setDownloadTryImport(sruleDownload, stryimport);
432       transaction.run();
433       future.awaitOrInterruptible();
434       final long time2 = System.currentTimeMillis();
435       final long delay = time2 - time1;
436       final R66Result result = future.getResult();
437       if (future.isSuccess()) {
438         if (result.getCode() == ErrorCode.Warning) {
439           logger.warn("WARNED on file:     " + (result.getOther() != null?
440               ((JsonCommandPacket) result.getOther()).getRequest() :
441               "no file") + "     delay: " + delay);
442         } else {
443           logger.warn("SUCCESS on Final file:     " +
444                       (result.getOther() != null?
445                           ((JsonCommandPacket) result.getOther()).getRequest() :
446                           "no file") + "     delay: " + delay);
447         }
448       } else {
449         if (result.getCode() == ErrorCode.Warning) {
450           logger.warn("LogExtendedExport is     WARNED", future.getCause());
451         } else {
452           logger.error("LogExtendedExport in     FAILURE", future.getCause());
453         }
454         networkTransaction.closeAll();
455         System.exit(result.getCode().ordinal());//NOSONAR
456       }
457     } finally {
458       networkTransaction.closeAll();
459       System.exit(0);//NOSONAR
460     }
461   }
