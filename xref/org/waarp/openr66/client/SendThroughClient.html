52  /**
53   * Class for Send Through client
54   * <p>
55   * This class does not included the real file transfer since it is up to the
56   * business project to implement how
57   * to read new data to be sent to the remote host. If an error occurs, no
58   * transfer log is kept.
59   * <p>
60   * 1) Configuration must have been loaded<br>
61   * <br>
62   * 2) Pipeline and NetworkTransaction must have been initiated:<br>
63   * <tt>     Configuration.configuration.pipelineInit();</tt><br>
64   * <tt>     NetworkTransaction networkTransaction = new
65   * NetworkTransaction();</tt><br>
66   * <br>
67   * 3) Prepare the request of transfer:<br>
68   * <tt>     R66Future futureReq = new R66Future(true);</tt><br>
69   * <tt>     SendThroughClient transaction = new SendThroughClient(futureReq,...);</tt><br>
70   * <tt>     if (! transaction.initiateRequest()) { error }</tt><br>
71   * <br>
72   * 4) Once initiateRequest() gives true, you are ready to send the data in
73   * through mode like:<br>
74   * <tt>     byte[] data = readOrGetInSomeWayData();</tt><br>
75   * <tt>     DataBlock block = transaction.transformToDataBlock(data);</tt><br>
76   * <tt>     futureWrite = transaction.writeWhenPossible(block);</tt><br>
77   * <br>
78   * 5) Once you have finished, so this is the last block, you have to do the
79   * following:<br>
80   * If the last block is not empty:<br>
81   * <tt>     DataBlock block = transaction.transformToDataBlock(data);</tt><br>
82   * <tt>     block.setEOF(true);</tt><br>
83   * Or if the last block is empty:<br>
84   * <tt>     DataBlock block = transaction.transformToDataBlock(null);</tt><br>
85   * Then <br>
86   * <tt>     futureWrite = transaction.writeWhenPossible(block);</tt><br>
87   * <tt>     futureWrite.awaitUninterruptibly();</tt><br>
88   * <br>
89   * 6) If everything is in success:<br>
90   * <tt>     transaction.finalizeRequest();</tt><br>
91   * <br>
92   * And now wait for the transfer to finish:<br>
93   * <tt>     futureReq.awaitUninterruptibly();</tt><br>
94   * <tt>     R66Result result = futureReq.getResult();</tt><br>
95   * <br>
96   * 7) If there is the need to re-do, just re-execute the steps from 3 to 6.<br>
97   * Don't forget at the very end to finish the global structure (steps 3 to 6 no
98   * more executed):<br>
99   * <tt>     networkTransaction.closeAll();</tt><br>
100  * <br>
101  * 8) In case of errors during steps 4 or 5 (and only those), call the
102  * following:<br>
103  * <tr>
104  * transaction.transferInError(openR66Exception);
105  * </tr>
106  * <br>
107  * <br>
108  *
109  * @see TestSendThroughClient Class as example of usage in test part
110  */
111 public abstract class SendThroughClient extends AbstractTransfer {
112   protected final NetworkTransaction networkTransaction;
113   protected LocalChannelReference localChannelReference;
114   protected DbTaskRunner taskRunner;

144   /**
145    * Prior to call this method, the pipeline and NetworkTransaction must have
146    * been initialized. It is the
147    * responsibility of the caller to finish all network resources. Note that
148    * this is only the first part of the
149    * execution for this client.
150    *
151    * @return True if the initiate of the request is OK, else False
152    */
153   public boolean initiateRequest() {
154     if (logger == null) {
155       logger = WaarpLoggerFactory.getLogger(SendThroughClient.class);
156     }
157     DbRule rule;
158     try {
159       rule = new DbRule(transferArgs.getRulename());
160     } catch (final WaarpDatabaseException e) {
161       logger.error("Cannot get Rule: " + transferArgs.getRulename(), e);
162       future.setResult(
163           new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
164                         ErrorCode.Internal, null));
165       future.setFailure(e);
166       return false;
167     }
168     int mode = rule.getMode();
169     if (transferArgs.isMD5()) {
170       mode = RequestPacket.getModeMD5(mode);
171     }
172     final String sep =
173         PartnerConfiguration.getSeparator(transferArgs.getRemoteHost());
174     final RequestPacket request =
175         new RequestPacket(transferArgs.getRulename(), mode,
176                           transferArgs.getFilename(),
177                           transferArgs.getBlockSize(), 0, transferArgs.getId(),
178                           transferArgs.getTransferInfo(), -1, sep);
179     // Not isRecv since it is the requester, so send => isSender is true
180     final boolean isSender = true;
181     try {
182       try {
183         // no starttime since immediate
184         taskRunner = new DbTaskRunner(rule, isSender, request,
185                                       transferArgs.getRemoteHost(), null);
186       } catch (final WaarpDatabaseException e) {
187         logger.error("Cannot get task", e);
188         future.setResult(
189             new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
190                           ErrorCode.Internal, null));
191         future.setFailure(e);
192         return false;
193       }
194       final ClientRunner runner =
195           new ClientRunner(networkTransaction, taskRunner, future);
196       runner.setSendThroughMode();
197       OpenR66ProtocolNotYetConnectionException exc = null;
198       for (int i = 0; i < Configuration.RETRYNB; i++) {
199         try {
200           localChannelReference = runner.initRequest();
201           exc = null;
202           break;
203         } catch (final OpenR66RunnerErrorException e) {
204           logger.error("Cannot Transfer", e);
205           future.setResult(
206               new R66Result(e, null, true, ErrorCode.Internal, taskRunner));
207           future.setFailure(e);
208           return false;
209         } catch (final OpenR66ProtocolNoConnectionException e) {
210           logger.error("Cannot Connect", e);
211           future.setResult(
212               new R66Result(e, null, true, ErrorCode.ConnectionImpossible,
213                             taskRunner));
214           finalizeInErrorTransferRequest(runner, taskRunner,
215                                          ErrorCode.ConnectionImpossible);
216           future.setFailure(e);
217           return false;
218         } catch (final OpenR66ProtocolPacketException e) {
219           logger.error("Bad Protocol", e);
220           future.setResult(new R66Result(e, null, true, ErrorCode.TransferError,
221                                          taskRunner));
222           future.setFailure(e);
223           return false;
224         } catch (final OpenR66ProtocolNotYetConnectionException e) {
225           logger.debug("Not Yet Connected", e);
226           exc = e;
227         }
228       }
229       if (exc != null) {
230         taskRunner.setLocalChannelReference(new LocalChannelReference());
231         logger.error("Cannot Connect", exc);
232         future.setResult(
233             new R66Result(exc, null, true, ErrorCode.ConnectionImpossible,
234                           taskRunner));
235         future.setFailure(exc);
236         return false;
237       }
238       try {
239         localChannelReference.waitReadyForSendThrough();
240       } catch (final OpenR66Exception e) {
241         logger.error("Cannot Transfer", e);
242         future.setResult(
243             new R66Result(e, null, true, ErrorCode.Internal, taskRunner));
244         future.setFailure(e);
245         return false;
246       }
247       // now start the send from external data
248       return true;
249     } finally {
250       if (taskRunner != null && future.isFailed()) {
251         try {
252           taskRunner.delete();
253         } catch (final WaarpDatabaseException ignored) {
254           // nothing
255         }
256       }
257     }
258   }

260   /**
261    * Finalize the request
262    */
263   public void finalizeRequest() {
264     try {
265       try {
266         ChannelUtils.writeEndTransfer(localChannelReference);
267       } catch (final OpenR66ProtocolPacketException e) {
268         // An error occurs!
269         try {
270           localChannelReference.getSession().setFinalizeTransfer(false,
271                                                                  new R66Result(
272                                                                      e,
273                                                                      localChannelReference
274                                                                          .getSession(),
275                                                                      false,
276                                                                      ErrorCode.Internal,
277                                                                      taskRunner));
278         } catch (final OpenR66RunnerErrorException e1) {
279           transferInError(e1);
280           return;
281         } catch (final OpenR66ProtocolSystemException e1) {
282           transferInError(e1);
283           return;
284         }
285       }
286       localChannelReference.getFutureEndTransfer().awaitOrInterruptible();
287       logger.debug("Await future End Transfer done: " +
288                    localChannelReference.getFutureEndTransfer().isSuccess());
289       if (localChannelReference.getFutureEndTransfer().isSuccess()) {
290         // send a validation
291         localChannelReference.sessionNewState(R66FiniteDualStates.ENDREQUESTS);
292         final EndRequestPacket validPacket =
293             new EndRequestPacket(ErrorCode.CompleteOk.ordinal());
294         final R66Session session = localChannelReference.getSession();
295         if (session != null && session.getExtendedProtocol() &&
296             session.getBusinessObject() != null &&
297             session.getBusinessObject().getInfo(session) != null) {
298           validPacket.setOptional(session.getBusinessObject().getInfo(session));
299         }
300         try {
301           ChannelUtils
302               .writeAbstractLocalPacket(localChannelReference, validPacket,
303                                         true);
304         } catch (final OpenR66ProtocolPacketException ignored) {
305           // nothing
306         }
307         if (!localChannelReference.getFutureRequest().awaitOrInterruptible()) {
308           // valid it however
309           localChannelReference.validateRequest(
310               localChannelReference.getFutureEndTransfer().getResult());
311         }
312         if (taskRunner != null && taskRunner.isSelfRequested()) {
313           localChannelReference.close();
314         }
315       } else {
316         transferInError(null);
317       }
318     } finally {
319       if (taskRunner != null) {
320         if (future.isDone() && !future.isSuccess() || nolog ||
321             taskRunner.shallIgnoreSave()) {
322           try {
323             taskRunner.delete();
324           } catch (final WaarpDatabaseException ignored) {
325             // nothing
326           }
327         }
328       }
329     }
330   }

332   /**
333    * To be used in case of error after a correct initiate of the request
334    *
335    * @param e
336    */
337   public void transferInError(OpenR66Exception e) {
338     if (!localChannelReference.getFutureEndTransfer().getResult()
339                               .isAnswered()) {
340       final R66Result result =
341           new R66Result(e, localChannelReference.getSession(), true,
342                         ErrorCode.TransferError, taskRunner);
343       logger.error("Transfer in error", e);
344       localChannelReference.sessionNewState(R66FiniteDualStates.ERROR);
345       final ErrorPackethandler/packet/ErrorPacket.html#ErrorPacket">ErrorPacket error = new ErrorPacket("Transfer in error",
346                                                 ErrorCode.TransferError
347                                                     .getCode(),
348                                                 ErrorPacket.FORWARDCLOSECODE);
349       try {
350         ChannelUtils
351             .writeAbstractLocalPacket(localChannelReference, error, true);
352       } catch (final OpenR66ProtocolPacketException ignored) {
353         // nothing
354       }
355       localChannelReference.invalidateRequest(result);
356     }
357     localChannelReference.close();
358   }

367    * @throws OpenR66RunnerErrorException

374     return RetrieveRunner.writeWhenPossible(block, localChannelReference, null);

377   /**
378    * Utility method for send through mode
379    *
380    * @param data the data byte, if null it is the last block
381    *
382    * @return the DataBlock associated to the data
383    */
384   public DataBlock transformToDataBlock(byte[] data) {
385     final DataBlockk.html#DataBlock">DataBlock block = new DataBlock();
386     if (data == null) {
387       // last block
388       block.setEOF(true);
389     } else {
390       block.setBlock(Unpooled.wrappedBuffer(data));
391     }
392     return block;
