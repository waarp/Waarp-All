92    protected String applyTransferSubstitutions(final String line) {
93      final Object[] argFormat = COMPILE_BLANK.split(argTransfer);
94      if (argFormat != null && argFormat.length > 0) {
95        try {
96          return String.format(line, argFormat);
97        } catch (final Exception e) {
98          // ignored error since bad argument in static rule info
99          logger.error("Bad format in Rule: {" + line + "} " + e.getMessage());
100       }
101     }
102     return line;
103   }

105   /**
106    * Generates a Command line object from rule and transfer data
107    *
108    * @param line the command to process as a string
109    */
110   protected CommandLine buildCommandLine(final String line) {
111     if (line.contains(NOWAIT)) {
112       waitForValidation = false;
113     }
114     if (line.contains(LOCALEXEC)) {
115       useLocalExec = true;
116     }
117 
118     final String replacedLine =
119         COMPILE_REPLACE_ALL.matcher(line).replaceAll("\\${$1}");
120 
121     final CommandLine commandLine =
122         CommandLine.parse(replacedLine, getSubstitutionMap());
123 
124     final File exec = new File(commandLine.getExecutable());
125     if (exec.isAbsolute() && !exec.canExecute()) {
126       logger.error("Exec command is not executable: " + line);
127       final R66Result result =
128           new R66Result(session, false, ErrorCode.CommandNotFound,
129                         session.getRunner());
130       futureCompletion.setResult(result);
131       futureCompletion.cancel();
132       return null;
133     }
134 
135     return commandLine;
136   }

170   private void substituteErrorCode(final Map<String, Object> rv) {
171     if (session.getLocalChannelReference() == null) {
172       rv.put(COMPILE_HASH.matcher(ERRORMSG)
173                          .replaceAll(Matcher.quoteReplacement("")), "NoError");
174       rv.put(COMPILE_HASH.matcher(ERRORCODE)
175                          .replaceAll(Matcher.quoteReplacement("")), "-");
176       rv.put(COMPILE_HASH.matcher(ERRORSTRCODE)
177                          .replaceAll(Matcher.quoteReplacement("")),
178              ErrorCode.Unknown.name());
179     } else {
180       try {
181         rv.put(COMPILE_HASH.matcher(ERRORMSG)
182                            .replaceAll(Matcher.quoteReplacement("")),
183                session.getLocalChannelReference().getErrorMessage());
184       } catch (final NullPointerException e) {
185         rv.put(COMPILE_HASH.matcher(ERRORMSG)
186                            .replaceAll(Matcher.quoteReplacement("")),
187                "NoError");
188       }
189       try {
190         rv.put(COMPILE_HASH.matcher(ERRORCODE)
191                            .replaceAll(Matcher.quoteReplacement("")),
192                session.getLocalChannelReference().getCurrentCode().getCode());
193       } catch (final NullPointerException e) {
194         rv.put(COMPILE_HASH.matcher(ERRORCODE)
195                            .replaceAll(Matcher.quoteReplacement("")), "-");
196       }
197       try {
198         rv.put(COMPILE_HASH.matcher(ERRORSTRCODE)
199                            .replaceAll(Matcher.quoteReplacement("")),
200                session.getLocalChannelReference().getCurrentCode().name());
201       } catch (final NullPointerException e) {
202         rv.put(COMPILE_HASH.matcher(ERRORSTRCODE)
203                            .replaceAll(Matcher.quoteReplacement("")),
204                ErrorCode.Unknown.name());
205       }
206     }

324   private void substituteHost(final Map<String, Object> rv) {
325     if (session.getAuth() != null) {
326       rv.put(COMPILE_HASH.matcher(REMOTEHOST)
327                          .replaceAll(Matcher.quoteReplacement("")),
328              session.getAuth().getUser());
329       String localhost;
330       try {
331         localhost =
332             Configuration.configuration.getHostId(session.getAuth().isSsl());
333       } catch (final OpenR66ProtocolNoSslException e) {
334         // replace by standard name
335         localhost = Configuration.configuration.getHostId();
336       }
337       rv.put(COMPILE_HASH.matcher(LOCALHOST)
338                          .replaceAll(Matcher.quoteReplacement("")), localhost);
339     }
340     if (session.getRemoteAddress() != null) {
341       rv.put(COMPILE_HASH.matcher(REMOTEHOSTADDR)
342                          .replaceAll(Matcher.quoteReplacement("")),
343              session.getRemoteAddress().toString());
344       rv.put(COMPILE_HASH.matcher(LOCALHOSTADDR)
345                          .replaceAll(Matcher.quoteReplacement("")),
346              session.getLocalAddress().toString());
347     } else {
348       rv.put(COMPILE_HASH.matcher(REMOTEHOSTADDR)
349                          .replaceAll(Matcher.quoteReplacement("")), "unknown");
350       rv.put(COMPILE_HASH.matcher(LOCALHOSTADDR)
351                          .replaceAll(Matcher.quoteReplacement("")), "unknown");
352     }
353   }

355   private void substituteDate(final Map<String, Object> rv) {
356     DateFormat dateFormat = new SimpleDateFormat("yyyyMMdd");
357     final Date date = new Date();
358     rv.put(COMPILE_HASH.matcher(DATE).replaceAll(Matcher.quoteReplacement("")),
359            dateFormat.format(date));
360     dateFormat = new SimpleDateFormat("HHmmss");
361     rv.put(COMPILE_HASH.matcher(HOUR).replaceAll(Matcher.quoteReplacement("")),
362            dateFormat.format(date));
363   }

365   private DbTaskRunner substituteRunner(final Map<String, Object> rv) {
366     final DbTaskRunner runner = session.getRunner();
367     if (runner != null) {
368       rv.put(COMPILE_HASH.matcher(ORIGINALFULLPATH)
369                          .replaceAll(Matcher.quoteReplacement("")),
370              runner.getOriginalFilename());
371       rv.put(COMPILE_HASH.matcher(ORIGINALFILENAME)
372                          .replaceAll(Matcher.quoteReplacement("")),
373              R66File.getBasename(runner.getOriginalFilename()));
374       rv.put(
375           COMPILE_HASH.matcher(RULE).replaceAll(Matcher.quoteReplacement("")),
376           runner.getRuleId());
377     }
378     if (runner != null) {
379       rv.put(COMPILE_HASH.matcher(TRANSFERID)
380                          .replaceAll(Matcher.quoteReplacement("")),
381              runner.getSpecialId());
382       rv.put(COMPILE_HASH.matcher(REQUESTERHOST)
383                          .replaceAll(Matcher.quoteReplacement("")),
384              runner.getRequester());
385       rv.put(COMPILE_HASH.matcher(REQUESTEDHOST)
386                          .replaceAll(Matcher.quoteReplacement("")),
387              runner.getRequested());
388       rv.put(COMPILE_HASH.matcher(FULLTRANSFERID)
389                          .replaceAll(Matcher.quoteReplacement("")),
390              runner.getSpecialId() + "_" + runner.getRequester() + '_' +
391              runner.getRequested());
392       rv.put(COMPILE_HASH.matcher(RANKTRANSFER)
393                          .replaceAll(Matcher.quoteReplacement("")),
394              Integer.toString(runner.getRank()));
395     }
396     return runner;
397   }

399   private void substituteFile(final Map<String, Object> rv) {
400     File trueFile = null;
401     if (session.getFile() != null) {
402       trueFile = session.getFile().getTrueFile();
403     }
404     if (trueFile != null) {
405       rv.put(COMPILE_HASH.matcher(TRUEFULLPATH)
406                          .replaceAll(Matcher.quoteReplacement("")),
407              trueFile.getAbsolutePath());
408       rv.put(COMPILE_HASH.matcher(TRUEFILENAME)
409                          .replaceAll(Matcher.quoteReplacement("")),
410              R66Dir.getFinalUniqueFilename(session.getFile()));
411       rv.put(COMPILE_HASH.matcher(FILESIZE)
412                          .replaceAll(Matcher.quoteReplacement("")),
413              Long.toString(trueFile.length()));
414     } else {
415       rv.put(COMPILE_HASH.matcher(TRUEFULLPATH)
416                          .replaceAll(Matcher.quoteReplacement("")), "nofile");
417       rv.put(COMPILE_HASH.matcher(TRUEFILENAME)
418                          .replaceAll(Matcher.quoteReplacement("")), "nofile");
419       rv.put(COMPILE_HASH.matcher(FILESIZE)
420                          .replaceAll(Matcher.quoteReplacement("")), "0");
421     }
422   }

439     PrepareCommandExec(final String finalname, final boolean noOutput,

441       this.finalname = finalname;
442       this.noOutput = noOutput;
443       this.waitForValidation = waitForValidation;
444     }

469

471       return watchdog;

473 
474     public PrepareCommandExec invoke() {
475       commandLine = buildCommandLine(finalname);
476       if (commandLine == null) {
477         myResult = true;
478         return this;
479       }
480 
481       defaultExecutor = new DefaultExecutor();
482       if (noOutput) {
483         pumpStreamHandler = new PumpStreamHandler(null, null);
484       } else {
485         inputStream = new PipedInputStream();
486         outputStream = null;
487         try {
488           outputStream = new PipedOutputStream(inputStream);
489         } catch (final IOException e1) {
490           FileUtils.close(inputStream);
491           logger.error(
492               "Exception: " + e1.getMessage() + " Exec in error with " +
493               commandLine, e1);
494           futureCompletion.setFailure(e1);
495           myResult = true;
496           return this;
497         }
498         pumpStreamHandler = new PumpStreamHandler(outputStream, null);
499       }
500       defaultExecutor.setStreamHandler(pumpStreamHandler);
501       final int[] correctValues = { 0, 1 };
502       defaultExecutor.setExitValues(correctValues);
503       watchdog = null;
504       if (delay > 0 && waitForValidation) {
505         watchdog = new ExecuteWatchdog(delay);
506         defaultExecutor.setWatchdog(watchdog);
507       }
508       myResult = false;
509       return this;
510     }
511   }

518     private final DefaultExecutor defaultExecutor;

526     ExecuteCommand(final CommandLine commandLine,
527                    final DefaultExecutor defaultExecutor,
528                    final PipedInputStream inputStream,
529                    final PipedOutputStream outputStream,
530                    final PumpStreamHandler pumpStreamHandler,
531                    final Thread thread) {
532       this.commandLine = commandLine;
533       this.defaultExecutor = defaultExecutor;
534       this.inputStream = inputStream;
535       this.outputStream = outputStream;
536       this.pumpStreamHandler = pumpStreamHandler;
537       this.thread = thread;
538     }

540     boolean isError() {
541       return myResult;
542     }

544     public int getStatus() {
545       return status;
546     }
547 
548     public ExecuteCommand invoke() {
549       status = -1;
550       try {
551         status = defaultExecutor.execute(commandLine);//NOSONAR
552       } catch (final ExecuteException e) {
553         if (e.getExitValue() == -559038737) {
554           // Cannot run immediately so retry once
555           try {
556             Thread.sleep(Configuration.RETRYINMS);
557           } catch (final InterruptedException e1) {//NOSONAR
558             SysErrLogger.FAKE_LOGGER.ignoreLog(e1);
559           }
560           try {
561             status = defaultExecutor.execute(commandLine);//NOSONAR
562           } catch (final ExecuteException e1) {
563             closeAllForExecution(true);
564             finalizeFromError(thread, status, commandLine, e1);
565             myResult = true;
566             return this;
567           } catch (final IOException e1) {
568             closeAllForExecution(true);
569             logger.error(
570                 "IOException: " + e.getMessage() + " . Exec in error with " +
571                 commandLine);
572             futureCompletion.setFailure(e);
573             myResult = true;
574             return this;
575           }
576         } else {
577           closeAllForExecution(true);
578           finalizeFromError(thread, status, commandLine, e);
579           myResult = true;
580           return this;
581         }
582       } catch (final IOException e) {
583         closeAllForExecution(true);
584         logger.error(
585             "IOException: " + e.getMessage() + " . Exec in error with " +
586             commandLine);
587         futureCompletion.setFailure(e);
588         myResult = true;
589         return this;
590       }
591       closeAllForExecution(false);
592       if (thread != null) {
593         try {
594           if (delay > 0) {
595             thread.join(delay);
596           } else {
597             thread.join();
598           }
599         } catch (final InterruptedException e) {//NOSONAR
600           SysErrLogger.FAKE_LOGGER.ignoreLog(e);
601           Thread.currentThread().interrupt();
602         }
603       }
604       FileUtils.close(inputStream);
605       myResult = false;
606       return this;
607     }
608 
609     private void closeAllForExecution(final boolean interrupt) {
610       FileUtils.close(outputStream);
611       if (interrupt && thread != null) {
612         thread.interrupt();
613       }
614       FileUtils.close(inputStream);
615       try {
616         pumpStreamHandler.stop();
617       } catch (final IOException ignored) {
618         // nothing
619       }
620     }
