20  package org.waarp.openr66.context.filesystem;
21  
22  import org.waarp.common.command.exception.CommandAbstractException;
23  import org.waarp.common.command.exception.Reply502Exception;
24  import org.waarp.common.file.filesystembased.FilesystemBasedRestartImpl;
25  import org.waarp.openr66.context.R66Session;
26  
27  /**
28   * Restart implementation
29   */
30  public class R66Restart extends FilesystemBasedRestartImpl {
31  
32    /**
33     * @param session
34     */
35    public R66Restart(final R66Session session) {
36      super(session);
37    }
38  
39    @Override
40    public final boolean restartMarker(final String marker)
41        throws CommandAbstractException {
42      final long newposition;
43      try {
44        newposition = Long.parseLong(marker);
45      } catch (final NumberFormatException e) {
46        throw new Reply502Exception(
47            "Marker must be length in byte as a position");
48      }
49      return restartMarker(newposition);
50    }
51  
52    /**
53     * Same as restartMarker with String
54     *
55     * @param newposition
56     *
57     * @return True if OK
58     */
59    public final boolean restartMarker(final long newposition) {
60      position = newposition;
61      setSet(true);
62      return true;
63    }
64  }
