52    /**
53     * Gets the version of the current PostgreSQL Server
54     *
55     * @throws WaarpDatabaseNoConnectionException
56     */
57    private int getServerVersion(DbSession session)
58        throws WaarpDatabaseNoConnectionException {
59      int serverVersion = 0;
60      final DbRequestequest.html#DbRequest">DbRequest request = new DbRequest(session);
61      try {
62        request.select("SHOW server_version_num");
63        request.getNext();
64        serverVersion = request.getResultSet().getInt("server_version_num");
65      } catch (final WaarpDatabaseNoConnectionException e) {
66        SysErrLogger.FAKE_LOGGER.syserr(e);
67      } catch (final WaarpDatabaseSqlException e) {
68        SysErrLogger.FAKE_LOGGER.syserr(e);
69        // XXX FIX no return
70      } catch (final SQLException e) {
71        SysErrLogger.FAKE_LOGGER.syserr(e);
72      } finally {
73        request.close();
74      }
75      return serverVersion;
76    }

78    @Override
79    public void createTables(DbSession session)
80        throws WaarpDatabaseNoConnectionException {
81      // Create tables: configuration, hosts, rules, runner, cptrunner
82      final String createTableH2 = "CREATE TABLE ";
83      final String primaryKey = " PRIMARY KEY ";
84      final String notNull = " NOT NULL ";
85      DbRequest request = DbModelFactoryR66
86          .subCreateTableMariaDbMySQLPostgreSQL(dbTypeResolver, session,
87                                                createTableH2, primaryKey,
88                                                notNull);
89      if (request == null) {
90        return;
91      }
92      StringBuilder action;
93      // Index Runner
94      action = new StringBuilder(
95          "CREATE INDEX IDX_RUNNER ON " + DbTaskRunner.table + '(');
96      final DbTaskRunner.Columns[] icolumns = DbTaskRunner.indexes;
97      for (int i = 0; i < icolumns.length - 1; i++) {
98        action.append(icolumns[i].name()).append(", ");
99      }
100     action.append(icolumns[icolumns.length - 1].name()).append(')');
101     SysErrLogger.FAKE_LOGGER.sysout(action);
102     try {
103       request.query(action.toString());
104     } catch (final WaarpDatabaseNoConnectionException e) {
105       SysErrLogger.FAKE_LOGGER.syserr(e);
106       request = null;
107       return;
108     } catch (final WaarpDatabaseSqlException e) {
109       SysErrLogger.FAKE_LOGGER.syserr(e);
110       // XXX FIX no return
111     } finally {
112       if (request != null) {
113         request.close();
114       }
115     }
116 
117     // cptrunner
118     action = new StringBuilder(
119         "CREATE SEQUENCE " + DbTaskRunner.fieldseq + " MINVALUE " +
120         (ILLEGALVALUE + 1) + " START WITH " + (ILLEGALVALUE + 1));
121     SysErrLogger.FAKE_LOGGER.sysout(action);
122     try {
123       request.query(action.toString());
124     } catch (final WaarpDatabaseNoConnectionException e) {
125       SysErrLogger.FAKE_LOGGER.syserr(e);
126       return;
127     } catch (final WaarpDatabaseSqlException e) {
128       SysErrLogger.FAKE_LOGGER.syserr(e);
129       // XXX FIX no return
130     } finally {
131       request.close();
132     }
133 
134     DbHostConfiguration.updateVersionDb(Configuration.configuration.getHostId(),
135                                         R66Versions.V2_4_25.getVersion());
136   }

139   public boolean upgradeDb(DbSession session, String version)
140       throws WaarpDatabaseNoConnectionException {
141     if (PartnerConfiguration
142         .isVersion2GEQVersion1(version, R66Versions.V2_4_13.getVersion())) {
143       SysErrLogger.FAKE_LOGGER.sysout(
144           version + " to " + R66Versions.V2_4_13.getVersion() + "? " + true);
145 
146       String createTableH2 = "CREATE TABLE ";
147       final int serverVersion = getServerVersion(session);
148       if (serverVersion >= 90100) {
149         createTableH2 = "CREATE TABLE IF NOT EXISTS ";
150       }
151       final String primaryKey = " PRIMARY KEY ";
152       final String notNull = " NOT NULL ";
153 
154       // HostConfiguration
155       StringBuilder action =
156           new StringBuilder(createTableH2 + DbHostConfiguration.table + '(');
157       final DbHostConfiguration.Columns[] chcolumns =
158           DbHostConfiguration.Columns.values();
159       for (int i = 0; i < chcolumns.length - 1; i++) {
160         action.append(chcolumns[i].name())
161               .append(DBType.getType(DbHostConfiguration.dbTypes[i]))
162               .append(notNull).append(", ");
163       }
164       action.append(chcolumns[chcolumns.length - 1].name()).append(
165           DBType.getType(DbHostConfiguration.dbTypes[chcolumns.length - 1]))
166             .append(primaryKey).append(')');
167       SysErrLogger.FAKE_LOGGER.sysout(action);
168       final DbRequestequest.html#DbRequest">DbRequest request = new DbRequest(session);
169       try {
170         request.query(action.toString());
171       } catch (final WaarpDatabaseSqlException e) {
172         SysErrLogger.FAKE_LOGGER.syserr(e);
173         return false;
174       } finally {
175         request.close();
176       }
177     }
178     if (PartnerConfiguration
179         .isVersion2GEQVersion1(version, R66Versions.V2_4_17.getVersion())) {
180       SysErrLogger.FAKE_LOGGER.sysout(
181           version + " to " + R66Versions.V2_4_17.getVersion() + "? " + true);
182       final String command =
183           "DO $$ BEGIN " + "ALTER TABLE " + DbTaskRunner.table +
184           " ADD COLUMN " + DbTaskRunner.Columns.TRANSFERINFO.name() + ' ' +
185           DBType.getType(DbTaskRunner.dbTypes[DbTaskRunner.Columns.TRANSFERINFO
186               .ordinal()]) + " DEFAULT '{}' NOT NULL; " +
187           "EXCEPTION WHEN duplicate_column THEN END $$";
188       final DbRequestequest.html#DbRequest">DbRequest request = new DbRequest(session);
189       try {
190         request.query(command);
191       } catch (final WaarpDatabaseSqlException e) {
192         SysErrLogger.FAKE_LOGGER.syserr(e);
193         // return false
194       } finally {
195         request.close();
196       }
197     }
198     if (PartnerConfiguration
199         .isVersion2GEQVersion1(version, R66Versions.V2_4_23.getVersion())) {
200       SysErrLogger.FAKE_LOGGER.sysout(
201           version + " to " + R66Versions.V2_4_23.getVersion() + "? " + true);
202       String command =
203           "DO $$ BEGIN " + "ALTER TABLE " + DbHostAuth.table + " ADD COLUMN " +
204           DbHostAuth.Columns.ISACTIVE.name() + ' ' + DBType.getType(
205               DbHostAuth.dbTypes[DbHostAuth.Columns.ISACTIVE.ordinal()]) +
206           " DEFAULT " + true + " NOT NULL; " +
207           "EXCEPTION WHEN duplicate_column THEN END $$";
208       DbRequestequest.html#DbRequest">DbRequest request = new DbRequest(session);
209       try {
210         request.query(command);
211       } catch (final WaarpDatabaseSqlException e) {
212         SysErrLogger.FAKE_LOGGER.syserr(e);
213         // return false
214       } finally {
215         request.close();
216       }
217       command =
218           "DO $$ BEGIN " + "ALTER TABLE " + DbHostAuth.table + " ADD COLUMN " +
219           DbHostAuth.Columns.ISPROXIFIED.name() + ' ' + DBType.getType(
220               DbHostAuth.dbTypes[DbHostAuth.Columns.ISPROXIFIED.ordinal()]) +
221           " DEFAULT " + false + " NOT NULL; " +
222           "EXCEPTION WHEN duplicate_column THEN END $$";
223       request = new DbRequest(session);
224       try {
225         request.query(command);
226       } catch (final WaarpDatabaseSqlException e) {
227         SysErrLogger.FAKE_LOGGER.syserr(e);
228         // return false
229       } finally {
230         request.close();
231       }
232     }
233     if (PartnerConfiguration
234         .isVersion2GTVersion1(version, R66Versions.V2_4_25.getVersion())) {
235       SysErrLogger.FAKE_LOGGER.sysout(
236           version + " to " + R66Versions.V2_4_25.getVersion() + "? " + true);
237       final String command =
238           "ALTER TABLE " + DbTaskRunner.table + " ALTER COLUMN " +
239           DbTaskRunner.Columns.FILENAME.name() + " TYPE " + DBType.getType(
240               DbTaskRunner.dbTypes[DbTaskRunner.Columns.FILENAME.ordinal()]) +
241           ',' + " ALTER COLUMN " + DbTaskRunner.Columns.FILENAME.name() +
242           " SET NOT NULL, " + " ALTER COLUMN " +
243           DbTaskRunner.Columns.ORIGINALNAME.name() + " TYPE " + DBType.getType(
244               DbTaskRunner.dbTypes[DbTaskRunner.Columns.ORIGINALNAME
245                   .ordinal()]) + ',' + " ALTER COLUMN " +
246           DbTaskRunner.Columns.FILENAME.name() + " SET NOT NULL ";
247       final DbRequestequest.html#DbRequest">DbRequest request = new DbRequest(session);
248       try {
249         SysErrLogger.FAKE_LOGGER.sysout("Command: " + command);
250         request.query(command);
251       } catch (final WaarpDatabaseSqlException e) {
252         SysErrLogger.FAKE_LOGGER.syserr(e);
253         return false;
254       } finally {
255         request.close();
256       }
257     }
258     DbHostConfiguration.updateVersionDb(Configuration.configuration.getHostId(),
259                                         R66Versions.V2_4_25.getVersion());
260     return true;
261   }
