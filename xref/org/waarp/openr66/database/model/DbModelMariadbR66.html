20  package org.waarp.openr66.database.model;
21  
22  import org.waarp.common.database.DbSession;
23  import org.waarp.common.database.exception.WaarpDatabaseNoConnectionException;
24  import org.waarp.common.database.model.DbModelMariadb;
25  
26  /**
27   * MariaDB Database Model implementation
28   */
29  public class DbModelMariadbR66 extends DbModelMariadb {
30    /**
31     * Create the object and initialize if necessary the driver
32     *
33     * @param dbserver
34     * @param dbuser
35     * @param dbpasswd
36     *
37     * @throws WaarpDatabaseNoConnectionException
38     */
39    public DbModelMariadbR66(final String dbserver, final String dbuser,
40                             final String dbpasswd)
41        throws WaarpDatabaseNoConnectionException {
42      super(dbserver, dbuser, dbpasswd);
43    }
44  
45    @Override
46    public void createTables(final DbSession session)
47        throws WaarpDatabaseNoConnectionException {
48      DbModelFactoryR66.createTableMariaDbMySQL(dbTypeResolver, session);
49    }
50  
51    @Override
52    public boolean upgradeDb(final DbSession session, final String version)
53        throws WaarpDatabaseNoConnectionException {
54      return DbModelFactoryR66
55          .upgradeDbMariaDbMySQL(dbTypeResolver, session, version);
56    }
57  
58    @Override
59    public boolean needUpgradeDb(final DbSession session, final String version,
60                                 final boolean tryFix)
61        throws WaarpDatabaseNoConnectionException {
62      return DbModelFactoryR66
63          .needUpgradeDbAllDb(dbTypeResolver, session, version);
64    }
65  
66  }
