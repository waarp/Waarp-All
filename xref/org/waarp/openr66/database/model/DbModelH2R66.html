56    @Override
57    public void createTables(final DbSession session)
58        throws WaarpDatabaseNoConnectionException {
59      // Create tables: configuration, hosts, rules, runner, cptrunner
60      final String createTableH2 = "CREATE TABLE IF NOT EXISTS ";
61      final String primaryKey = " PRIMARY KEY ";
62      final String notNull = " NOT NULL ";
63      final DbRequest request = DbModelFactoryR66
64          .subCreateTableMariaDbMySQLPostgreSQL(dbTypeResolver, session,
65                                                createTableH2, primaryKey,
66                                                notNull);
67      if (request == null) {
68        return;
69      }
70      StringBuilder action;
71      // Index Runner
72      action = new StringBuilder(
73          "CREATE INDEX IF NOT EXISTS IDX_RUNNER ON " + DbTaskRunner.table + '(');
74      final DbTaskRunner.Columns[] icolumns = DbTaskRunner.indexes;
75      for (int i = 0; i < icolumns.length - 1; i++) {
76        action.append(icolumns[i].name()).append(", ");
77      }
78      action.append(icolumns[icolumns.length - 1].name()).append(')');
79      SysErrLogger.FAKE_LOGGER.sysout(action);
80      try {
81        request.query(action.toString());
82      } catch (final WaarpDatabaseNoConnectionException e) {
83        SysErrLogger.FAKE_LOGGER.syserr(e);
84        return;
85      } catch (final WaarpDatabaseSqlException e) {
86        SysErrLogger.FAKE_LOGGER.syserr(e);
87        // XXX FIX no return
88      } finally {
89        request.close();
90      }
91  
92      // cptrunner
93      action = new StringBuilder(
94          "CREATE SEQUENCE IF NOT EXISTS " + DbTaskRunner.fieldseq +
95          " START WITH " + (ILLEGALVALUE + 1) + " MINVALUE " +
96          (ILLEGALVALUE + 1));
97      SysErrLogger.FAKE_LOGGER.sysout(action);
98      try {
99        request.query(action.toString());
100     } catch (final WaarpDatabaseNoConnectionException e) {
101       SysErrLogger.FAKE_LOGGER.syserr(e);
102       return;
103     } catch (final WaarpDatabaseSqlException e) {
104       // version <= 1.2.173
105       action = new StringBuilder(
106           "CREATE SEQUENCE IF NOT EXISTS " + DbTaskRunner.fieldseq +
107           " START WITH " + (ILLEGALVALUE + 1));
108       SysErrLogger.FAKE_LOGGER.sysout(action);
109       try {
110         request.query(action.toString());
111       } catch (final WaarpDatabaseNoConnectionException e2) {
112         SysErrLogger.FAKE_LOGGER.syserr(e2);
113         return;
114       } catch (final WaarpDatabaseSqlException e2) {
115         SysErrLogger.FAKE_LOGGER.syserr(e2);
116         // XXX FIX no return
117       } finally {
118         request.close();
119       }
120       // XXX FIX no return
121     } finally {
122       request.close();
123     }
124 
125     DbHostConfiguration.updateVersionDb(Configuration.configuration.getHostId(),
126                                         R66Versions.V2_4_25.getVersion());
127   }
