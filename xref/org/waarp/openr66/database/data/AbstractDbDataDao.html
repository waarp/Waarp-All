48        WaarpLoggerFactory.getLogger(AbstractDbDataDao.class);

79    /**
80     * Test the existence of the current object
81     *
82     * @return True if the object exists
83     *
84     * @throws WaarpDatabaseException
85     */
86    @Override
87    public boolean exist() throws WaarpDatabaseException {
88      AbstractDAO<E> abstractDAO = null;
89      try {
90        abstractDAO = getDao(true);
91        return abstractDAO.exist(getPrimaryKey());
92      } catch (final DAOConnectionException e) {
93        throw new WaarpDatabaseNoConnectionException(e);
94      } finally {
95        DAOFactory.closeDAO(abstractDAO);
96      }
97    }

101    *
102    * @throws WaarpDatabaseException
103    */
104   @Override
105   public void select() throws WaarpDatabaseException {
106     AbstractDAO<E> abstractDAO = null;
107     try {
108       abstractDAO = getDao(true);
109       pojo = abstractDAO.select(getPrimaryKey());
110       isSaved = true;
111     } catch (final DAOConnectionException e) {
112       throw new WaarpDatabaseNoConnectionException(e);
113     } catch (final DAONoDataException e) {
114       throw new WaarpDatabaseNoDataException((e));
115     } finally {
116       DAOFactory.closeDAO(abstractDAO);
117     }
118   }

120   /**
121    * Insert object into table
122    *
123    * @throws WaarpDatabaseException
124    */
125   @Override
126   public void insert() throws WaarpDatabaseException {
127     if (isSaved) {
128       return;
129     }
130     checkValues();
131     AbstractDAO<E> abstractDAO = null;
132     try {
133       abstractDAO = getDao(false);
134       abstractDAO.insert(pojo);
135       isSaved = true;
136     } catch (final DAOConnectionException e) {
137       throw new WaarpDatabaseNoConnectionException(e);
138     } finally {
139       DAOFactory.closeDAO(abstractDAO);
140     }
141   }

206   /**
207    * Create the equivalent object in Json (no database access)
208    *
209    * @return The ObjectNode Json equivalent
210    */
211   @Override
212   public ObjectNode getJson() {
213     final ObjectNode node = JsonHandler.createObjectNode();
214     node.put(JSON_MODEL, getClass().getSimpleName());
215     final String json = JsonHandler.writeAsString(pojo);
216     final ObjectNode subnode = JsonHandler.getFromString(json);
217     if (subnode != null) {
218       for (final Iterator<Entry<String, JsonNode>> it = subnode.fields();
219            it.hasNext(); ) {
220         final Entry<String, JsonNode> entry = it.next();
221         node.set(entry.getKey(), entry.getValue());
222       }
223     }
224     return node;
225   }

227   protected abstract void setFromJson(String field, JsonNode value)
228       throws WaarpDatabaseSqlException;

230   /**
231    * Set the values from the Json node to the current object (no database
232    * access)
233    *
234    * @param node
235    * @param ignorePrimaryKey True will ignore primaryKey from Json
236    *
237    * @throws WaarpDatabaseSqlException
238    */
239   @Override
240   public void setFromJson(final ObjectNode node, final boolean ignorePrimaryKey)
241       throws WaarpDatabaseSqlException {
242     boolean foundPrimaryKey = false;
243     for (final Iterator<Entry<String, JsonNode>> it = node.fields();
244          it.hasNext(); ) {
245       final Entry<String, JsonNode> entry = it.next();
246       logger.debug("{} = {}", entry.getKey(), entry.getValue());
247       if ("UPDATEDINFO".equalsIgnoreCase(entry.getKey())) {
248         continue;
249       }
250       if (getPrimaryField().equalsIgnoreCase(entry.getKey())) {
251         if (ignorePrimaryKey) {
252           continue;
253         } else {
254           foundPrimaryKey = !entry.getValue().isNull() &&
255                             !entry.getValue().asText().isEmpty();
256         }
257       }
258       setFromJson(entry.getKey(), entry.getValue());
259       isSaved = false;
260     }
261     checkValues();
262     if (!ignorePrimaryKey && foundPrimaryKey) {
263       try {
264         insert();
265       } catch (final WaarpDatabaseException e) {
266         try {
267           update();
268         } catch (final WaarpDatabaseException ex) {
269           logger.error("Cannot save item: {}", ex.getMessage());
270         }
271       }
272     }
273   }
