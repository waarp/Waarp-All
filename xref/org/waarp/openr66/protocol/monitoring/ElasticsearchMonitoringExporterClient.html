21  package org.waarp.openr66.protocol.monitoring;
22  
23  import com.fasterxml.jackson.databind.node.ObjectNode;
24  import org.joda.time.DateTime;
25  
26  import java.io.Closeable;
27  
28  public interface ElasticsearchMonitoringExporterClient extends Closeable {
29  
30    /**
31     * Will use a "bulk" request to upsert data into ElasticSearch
32     *
33     * @param monitoredTransfers the Json objet to push as POST
34     * @param start the DateTime for the 'from' interval
35     * @param stop the DateTime for the 'to' interval
36     * @param serverId the serverId that is sending this monitoring information
37     *
38     * @return True if the Bulk upsert succeeded
39     */
40    boolean post(final ObjectNode monitoredTransfers, final DateTime start,
41                 final DateTime stop, final String serverId);
42  }
