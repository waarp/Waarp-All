103                  e1 != null? e1.getMessage() : "no exception",
104                  runner != null? runner.toShortString() : "no runner");
105     logger.debug("DEBUG Full stack", e1);
106     localChannelReference
107         .invalidateRequest(new R66Result(e1, session, true, code, null));
108 
109     if (packet.isToValidate()) {
110       // / answer with a wrong request since runner is not set on remote host
111       if (runner != null) {
112         if (runner.isSender()) {
113           // In case Wildcard was used
114           logger.debug("New FILENAME: {}", runner.getOriginalFilename());
115           packet.setFilename(runner.getOriginalFilename());
116           logger.debug("Rank set: " + runner.getRank());
117           packet.setRank(runner.getRank());
118         } else {
119           logger.debug("Rank set: " + runner.getRank());
120           packet.setRank(runner.getRank());
121         }
122       }
123       packet.validate();
124       packet.setCode(code.code);
125       session.newState(ERROR);
126       ChannelUtils
127           .writeAbstractLocalPacket(localChannelReference, packet, true);
128     } else {
129       session.newState(ERROR);
130       final ErrorPacketcalhandler/packet/ErrorPacket.html#ErrorPacket">ErrorPacket error = new ErrorPacket(
131           "TaskRunner initialisation in error: " +
132           (e1 != null? e1.getMessage() : "Unknown Error") + " for " + packet +
133           " since " + code.getMesg(), code.getCode(),
134           ErrorPacket.FORWARDCLOSECODE);
135       ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
136     }
137     session.setStatus(47);

141   /**
142    * Receive a request of Transfer
143    *
144    * @param packet
145    *
146    * @throws OpenR66ProtocolNoDataException
147    * @throws OpenR66ProtocolPacketException
148    * @throws OpenR66ProtocolBusinessException
149    * @throws OpenR66ProtocolSystemException
150    * @throws OpenR66RunnerErrorException
151    */
152   public void request(RequestPacket packet)
153       throws OpenR66ProtocolNoDataException, OpenR66ProtocolPacketException,
154              OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
155              OpenR66ProtocolBusinessException {
156     session.setStatus(99);
157     if (!session.isAuthenticated()) {
158       session.setStatus(48);
159       throw new OpenR66ProtocolNotAuthenticatedException(
160           Messages.getString("LocalServerHandler.3")); //$NON-NLS-1$
161     }
162     if (packet.isToValidate()) {
163       session.newState(REQUESTR);
164     }
165     // XXX validLimit only on requested side
166     if (checkRequest(packet)) {
167       return;
168     }
169     DbRule rule;
170     try {
171       rule = new DbRule(packet.getRulename());
172     } catch (final WaarpDatabaseException e) {
173       logger.info("Rule is unknown: " + packet.getRulename() + " {}",
174                   e.getMessage());
175       session.setStatus(49);
176       endInitRequestInError(ErrorCode.QueryRemotelyUnknown, null,
177                             new OpenR66ProtocolBusinessException(
178                                 Messages.getString("LocalServerHandler.9") +
179                                 //$NON-NLS-1$
180                                 packet.getRulename()), packet);
181       return;
182     }
183     packet = computeBlockSizeFromRequest(packet, rule);
184     DbTaskRunner runner;
185     // requested
186     final boolean isRetrieve = DbTaskRunner.getSenderByRequestPacket(packet);
187     if (packet.getSpecialId() != ILLEGALVALUE) {
188       runner = getPreviousDbTaskRunnerFromRequest(packet, rule, isRetrieve);
189       if (runner == null) {
190         return;
191       }
192     } else {
193       // Very new request
194       // should not be the case (the requester should always set the id)
195       logger.error("NO TransferID specified: SHOULD NOT BE THE CASE");
196       try {
197         runner = new DbTaskRunner(session, rule, isRetrieve, packet);
198       } catch (final WaarpDatabaseException e) {
199         session.setStatus(37);
200         endInitRequestInError(ErrorCode.QueryRemotelyUnknown, null,
201                               new OpenR66DatabaseGlobalException(e), packet);
202         return;
203       }
204       packet.setSpecialId(runner.getSpecialId());
205     }
206     logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(),
207                  runner);
208     // Check now if request is a valid one
209     if (packet.getCode() != ErrorCode.InitOk.code) {
210       createErrorFromRequestInitKo(packet, runner);
211       return;
212     }
213     // Receiver can specify a rank different from database
214     setRankAtStartupFromRequest(packet, runner);
215     runner.setBlocksize(packet.getBlocksize());
216     try {
217       runner.update();
218     } catch (WaarpDatabaseException ignored) {
219       // Ignore
220     }
221     logger.debug(
222         "Filesize: " + packet.getOriginalSize() + ':' + runner.isSender());
223     boolean shouldInformBack = false;
224     try {
225       session.setRunner(runner);
226       // Fix to ensure that recv request are not trying to access to not chroot files
227       if (Configuration.configuration.isChrootChecked() &&
228           packet.isToValidate() && runner.isSender()) {
229         session.startup(true);
230       } else {
231         session.startup(false);
232       }
233       if (runner.isSender() && !runner.isSendThrough()) {
234         if (packet.getOriginalSize() != runner.getOriginalSize()) {
235           packet.setOriginalSize(runner.getOriginalSize());
236           shouldInformBack = true;
237           logger.debug("Filesize2: " + packet.getOriginalSize() + ':' +
238                        runner.isSender());
239         }
240       }
241     } catch (final OpenR66RunnerErrorException e) {
242       try {
243         runner.saveStatus();
244       } catch (final OpenR66RunnerErrorException e1) {
245         logger.error("Cannot save Status: " + runner, e1);
246       }
247       if (runner.getErrorInfo() == ErrorCode.InitOk ||
248           runner.getErrorInfo() == ErrorCode.PreProcessingOk ||
249           runner.getErrorInfo() == ErrorCode.TransferOk) {
250         runner.setErrorExecutionStatus(ErrorCode.ExternalOp);
251       }
252       logger.error("PreTask in error {}", e.getMessage(), e);
253       errorToSend("PreTask in error: " + e.getMessage(), runner.getErrorInfo(),
254                   38);
255       return;
256     }
257     setFileSizeFromRequest(packet, runner, shouldInformBack);
258     session.setReady(true);
259     Configuration.configuration.getLocalTransaction()
260                                .setFromId(runner, localChannelReference);
261 
262     // Set read/write limit
263     final long remoteLimit = packet.getLimit();
264     long localLimit = localChannelReference.getChannelLimit(runner.isSender());
265     // Take the minimum speed
266     logger.trace("Received limit {}", packet.getLimit());
267     logger.trace("Local limit {}",
268                  localChannelReference.getChannelLimit(runner.isSender()));
269     if (localLimit <= 0) {
270       localLimit = remoteLimit;
271     } else if (remoteLimit > 0 && remoteLimit < localLimit) {
272       localLimit = remoteLimit;
273     }
274     localChannelReference.setChannelLimit(runner.isSender(), localLimit);
275     packet.setLimit(localLimit);
276 
277     // inform back
278     informBackFromRequest(packet, runner);
279     // if retrieve => START the retrieve operation except if in Send Through mode
280     sendDataFromRequest(runner);
281     session.setStatus(39);
282   }

284   private RequestPacketopenr66/protocol/localhandler/packet/RequestPacket.html#RequestPacket">RequestPacket computeBlockSizeFromRequest(RequestPacket packet,
285                                                     final DbRule rule)
286       throws OpenR66ProtocolNotAuthenticatedException {
287     int blocksize = packet.getBlocksize();
288     if (packet.isToValidate()) {
289       if (!rule.checkHostAllow(session.getAuth().getUser())) {
290         session.setStatus(30);
291         throw new OpenR66ProtocolNotAuthenticatedException(
292             Messages.getString("LocalServerHandler.10")); //$NON-NLS-1$
293       }
294       // Check if the blocksize is greater than local value
295       if (Configuration.configuration.getBlockSize() < blocksize) {
296         logger.warn("Blocksize is greater than allowed {} < {}",
297                     Configuration.configuration.getBlockSize(), blocksize);
298         blocksize = Configuration.configuration.getBlockSize();
299         final String sep = localChannelReference.getPartner().getSeperator();
300         packet = new RequestPacket(packet.getRulename(), packet.getMode(),
301                                    packet.getFilename(), blocksize,
302                                    packet.getRank(), packet.getSpecialId(),
303                                    packet.getFileInformation(),
304                                    packet.getOriginalSize(), sep);
305       }
306     }
307     if (!RequestPacket.isCompatibleMode(rule.getMode(), packet.getMode())) {
308       // not compatible Rule and mode in request
309       throw new OpenR66ProtocolNotAuthenticatedException(
310           Messages.getString("LocalServerHandler.12") + rule.getMode() + " vs "
311           //$NON-NLS-1$
312           + packet.getMode());
313     }
314     session.setBlockSize(blocksize);
315     return packet;
316   }

318   private boolean checkRequest(final RequestPacket packet)
319       throws OpenR66ProtocolPacketException {
320     if (packet.isToValidate()) {
321       if (Configuration.configuration.isShutdown()) {
322         logger.warn(Messages.getString("LocalServerHandler.7") //$NON-NLS-1$
323                     + packet.getRulename() + " from " + session.getAuth());
324         session.setStatus(100);
325         endInitRequestInError(ErrorCode.ServerOverloaded, null,
326                               new OpenR66ProtocolNotYetConnectionException(
327                                   "All new Request blocked"), packet);
328         session.setStatus(100);
329         return true;
330       }
331       if (Configuration.configuration.getConstraintLimitHandler()
332                                      .checkConstraints()) {
333         requestCheckConstraintsTrue(packet);
334         return true;
335       }
336     } else if (packet.getCode() == ErrorCode.ServerOverloaded.code) {
337       // XXX unvalid limit on requested host received
338       logger.info("TaskRunner initialisation in error: " +
339                   ErrorCode.ServerOverloaded.getMesg());
340       localChannelReference.invalidateRequest(
341           new R66Result(null, session, true, ErrorCode.ServerOverloaded, null));
342       session.setStatus(101);
343       ChannelCloseTimer.closeFutureTransaction(this);
344       return true;
345     }
346     return false;
347   }

349   private static void setRankAtStartupFromRequest(final RequestPacket packet,
350                                                   final DbTaskRunner runner) {
351     if (runner.isSender()) {
352       logger.debug("Rank was: " + runner.getRank() + " -> " + packet.getRank());
353       runner.setRankAtStartup(packet.getRank());
354     } else {
355       if (runner.getRank() > packet.getRank()) {
356         logger.debug(
357             "Recv Rank was: " + runner.getRank() + " -> " + packet.getRank());
358         // if receiver, change only if current rank is upper proposed rank
359         runner.setRankAtStartup(packet.getRank());
360       }
361       if (packet.getOriginalSize() > 0) {
362         runner.setOriginalSize(packet.getOriginalSize());
363       }
364     }
365   }

367   private void setFileSizeFromRequest(final RequestPacket packet,
368                                       final DbTaskRunner runner,
369                                       boolean shouldInformBack)
370       throws OpenR66ProtocolPacketException {
371     logger.debug(
372         "Filesize: " + packet.getOriginalSize() + ':' + runner.isSender());
373     if (!shouldInformBack) {
374       shouldInformBack =
375           !packet.getFileInformation().equals(runner.getFileInformation());
376     }
377     if (runner.isFileMoved() && runner.isSender() && runner.isInTransfer() &&
378         runner.getRank() == 0 && !packet.isToValidate()) {
379       // File was moved during PreTask and very beginning of the transfer
380       // and the remote host has already received the request packet
381       // => Informs the receiver of the new name
382       sendFilenameFilesizeChanging(packet, runner,
383                                    "Will send a modification of filename due to pretask: ",
384                                    "Change Filename by Pre action on sender");
385     } else if (!packet.getFilename().equals(runner.getOriginalFilename()) &&
386                runner.isSender() && runner.isInTransfer() &&
387                runner.getRank() == 0 && !packet.isToValidate()) {
388       // File was modify at the very beginning (using wildcards)
389       // and the remote host has already received the request packet
390       // => Informs the receiver of the new name
391       sendFilenameFilesizeChanging(packet, runner,
392                                    "Will send a modification of filename due to wildcard: ",
393                                    "Change Filename by Wildcard on sender");
394     } else if (runner.isSelfRequest() && runner.isSender() &&
395                runner.isInTransfer() && runner.getRank() == 0 &&
396                !packet.isToValidate()) {
397       // FIX SelfRequest
398       // File could be modified at the very beginning (using wildcards)
399       // and the remote host has already received the request packet
400       // => Informs the receiver of the new name
401       sendFilenameFilesizeChanging(packet, runner,
402                                    "Will send a modification of filename due to wildcard in SelfMode: ",
403                                    "Change Filename by Wildcard on sender in SelfMode");
404     } else if (shouldInformBack && !packet.isToValidate()) {
405       // Was only (shouldInformBack)
406       // File length is now known, so inform back
407       sendFilenameFilesizeChanging(packet, runner,
408                                    "Will send a modification of filesize or fileInfo: ",
409                                    "Change Filesize / FileInfo on sender");
410     }
411   }

413   private void informBackFromRequest(final RequestPacket packet,
414                                      final DbTaskRunner runner)
415       throws OpenR66ProtocolPacketException {
416     if (packet.isToValidate()) {
417       if (Configuration.configuration.getMonitoring() != null) {
418         Configuration.configuration.getMonitoring().lastInActiveTransfer =
419             System.currentTimeMillis();
420       }
421       if (runner.isSender()) {
422         // In case Wildcard was used
423         logger.debug("New FILENAME: {}", runner.getOriginalFilename());
424         packet.setFilename(runner.getOriginalFilename());
425         logger.debug("Rank set: " + runner.getRank());
426         packet.setRank(runner.getRank());
427       } else {
428         logger.debug("Rank set: " + runner.getRank());
429         packet.setRank(runner.getRank());
430       }
431       packet.validate();
432       session.newState(REQUESTD);
433       ChannelUtils
434           .writeAbstractLocalPacket(localChannelReference, packet, true);
435     } else {
436       session.newState(REQUESTD);
437       // requester => might be a client
438       // Save the runner into the session and validate the request so begin transfer
439       session.getLocalChannelReference().getFutureRequest().setRunner(runner);
440       localChannelReference.getFutureValidRequest().setSuccess();
441       if (Configuration.configuration.getMonitoring() != null) {
442         Configuration.configuration.getMonitoring().lastOutActiveTransfer =
443             System.currentTimeMillis();
444       }
445     }
446   }

448   private void sendDataFromRequest(final DbTaskRunner runner) {
449     if (runner.isSender()) {
450       if (runner.isSendThrough()) {
451         // it is legal to send data from now
452         logger.debug("Now ready to continue with send through");
453         localChannelReference.validateEndTransfer(
454             new R66Result(session, false, ErrorCode.PreProcessingOk, runner));
455       } else {
456         // Automatically send data now
457         logger.debug("Now ready to continue with runRetrieve");
458         NetworkTransaction.runRetrieve(session);
459       }
460     }

482   private DbTaskRunner getPreviousDbTaskRunnerFromRequest(
483       final RequestPacket packet, final DbRule rule, final boolean isRetrieve)
484       throws OpenR66ProtocolPacketException {
485     final DbTaskRunner runner;// Reload or create
486     final String requested = DbTaskRunner.getRequested(session, packet);
487     final String requester = DbTaskRunner.getRequester(session, packet);
488     logger.debug("DEBUG: " + packet.getSpecialId() + ':' + isRetrieve);
489     if (packet.isToValidate()) {
490       // Id could be a creation or a reload
491       // Try reload
492       runner =
493           reloadDbTaskRunner(packet, rule, isRetrieve, requested, requester);
494 
495       final LocalChannelReference lcr =
496           Configuration.configuration.getLocalTransaction().getFromRequest(
497               requested + ' ' + requester + ' ' + packet.getSpecialId());
498       if (checkRunnerConsistency(packet, runner, lcr)) {
499         return null;
500       }
501 
502       if (runner.isAllDone()) {
503         // truly an error since done
504         session.setStatus(31);
505         endInitRequestInError(ErrorCode.QueryAlreadyFinished, runner,
506                               new OpenR66ProtocolBusinessQueryAlreadyFinishedException(
507                                   Messages.getString("LocalServerHandler.13")
508                                   //$NON-NLS-1$
509                                   + packet.getSpecialId()), packet);
510         return null;
511       }
512       if (lcr != null) {
513         // truly an error since still running
514         session.setStatus(32);
515         endInitRequestInError(ErrorCode.QueryStillRunning, runner,
516                               new OpenR66ProtocolBusinessQueryStillRunningException(
517                                   Messages.getString("LocalServerHandler.14")
518                                   //$NON-NLS-1$
519                                   + packet.getSpecialId()), packet);
520         return null;
521       }
522       logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(),
523                    runner);
524       // ok to restart
525       try {
526         if (runner.restart(false)) {
527           runner.saveStatus();
528         }
529       } catch (final OpenR66RunnerErrorException ignored) {
530         // nothing
531       }
532       // Change the SpecialID! => could generate an error ?
533       logger.trace("Here was sepSpecialId without condition");
534       if (packet.getSpecialId() == ILLEGALVALUE) {
535         packet.setSpecialId(runner.getSpecialId());
536       }
537     } else {
538       // Id should be a reload
539       runner = reloadDbTaskRunnerFromId(packet, rule, isRetrieve, requested,
540                                         requester);
541       if (runner == null) {
542         return null;
543       }
544     }
545     return runner;
546   }

548   private boolean checkRunnerConsistency(final RequestPacket packet,
549                                          final DbTaskRunner runner,
550                                          final LocalChannelReference lcr)
551       throws OpenR66ProtocolPacketException {
552     // Check correctness of packet received vs current LCR
553     if (runner == null) {
554       logger.info("Id is unknown: {}}", packet.getSpecialId());
555       return true;
556     }
557     if (lcr != null && localChannelReference != null &&
558         !runner.isSelfRequest() &&
559         (!lcr.getLocalId().equals(localChannelReference.getLocalId()) ||
560          !lcr.getRemoteId().equals(localChannelReference.getRemoteId()))) {
561       logger.warn("LocalChannelReference differs: {}\n\t {}\n\tWill while " +
562                   "runner is AllDone: {}", localChannelReference, lcr,
563                   runner.isAllDone());
564       logger.info("Id is unknown: {}}", packet.getSpecialId());
565       if (runner.isAllDone()) {
566         try {
567           lcr.getServerHandler().tryFinalizeRequest(
568               new R66Result(lcr.getSession(), false, ErrorCode.Internal,
569                             runner));
570         } catch (OpenR66RunnerErrorException ignore) {
571           SysErrLogger.FAKE_LOGGER.ignoreLog(ignore);
572         } catch (OpenR66ProtocolSystemException ignore) {
573           SysErrLogger.FAKE_LOGGER.ignoreLog(ignore);
574         }
575         lcr.close();
576         if (localChannelReference.getClientRunner() != null &&
577             localChannelReference.getClientRunner().getTaskRunner() != null &&
578             localChannelReference.getClientRunner().getTaskRunner()
579                                  .isAllDone()) {
580           localChannelReference.close();
581         }
582         return true;
583       }
584       return false;
585     }
586     return false;
587   }

589   private DbTaskRunner reloadDbTaskRunnerFromId(final RequestPacket packet,
590                                                 final DbRule rule,
591                                                 final boolean isRetrieve,
592                                                 final String requested,
593                                                 final String requester)
594       throws OpenR66ProtocolPacketException {
595     DbTaskRunner runner;
596     try {
597       runner = new DbTaskRunner(session, rule, packet.getSpecialId(), requester,
598                                 requested);
599     } catch (final WaarpDatabaseException e) {
600       if (localChannelReference.getDbSession() == null) {
601         // Special case of no database client
602         try {
603           runner = new DbTaskRunner(session, rule, isRetrieve, packet);
604           logger.debug("Runner before any action: {} {}",
605                        runner.shallIgnoreSave(), runner);
606         } catch (final WaarpDatabaseException e1) {
607           session.setStatus(35);
