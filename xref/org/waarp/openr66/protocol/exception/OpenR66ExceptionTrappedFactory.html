38  /**
39   * Class that filter exceptions
40   */
41  public final class OpenR66ExceptionTrappedFactory {
42    /**
43     * Internal Logger
44     */
45    private static final WaarpLogger logger =
46        WaarpLoggerFactory.getLogger(OpenR66ExceptionTrappedFactory.class);
47  
48    private OpenR66ExceptionTrappedFactory() {
49    }
50  
51    /**
52     * @param channel
53     * @param throwable
54     *
55     * @return the OpenR66Exception corresponding to the ExceptionEvent, or null
56     *     if the exception should be
57     *     ignored
58     */
59    public static OpenR66Exception getExceptionFromTrappedException(
60        Channel channel, Throwable throwable) {
61      if (throwable instanceof ConnectException) {
62        final ConnectException e2 = (ConnectException) throwable;
63        logger.debug("Connection impossible since {} with Channel {}",
64                     e2.getMessage(), channel);
65        return new OpenR66ProtocolNoConnectionException("Connection impossible",
66                                                        e2);
67      } else if (throwable instanceof ChannelException) {
68        final ChannelException e2 = (ChannelException) throwable;
69        logger.info(
70            "Connection (example: timeout) impossible since {} with Channel {}",
71            e2.getMessage(), channel);
72        return new OpenR66ProtocolNetworkException(
73            "Connection (example: timeout) impossible", e2);
74      } else if (throwable instanceof CancelledKeyException) {
75        final CancelledKeyException e2 = (CancelledKeyException) throwable;
76        logger.error("Connection aborted since {}", e2.getMessage());
77        // Is it really what we should do ?
78        // Yes, No action
79        return null;
80      } else if (throwable instanceof ClosedChannelException) {
81        logger.debug("Connection closed before end");
82        return new OpenR66ProtocolBusinessNoWriteBackException(
83            "Connection closed before end", throwable);
84      } else if (throwable instanceof IllegalMonitorStateException) {
85        logger.debug("Try to release a lock incorrectly", throwable);
86        return new OpenR66ProtocolBusinessNoWriteBackException(
87            "Ignored exception", throwable);
88      } else if (throwable instanceof OpenR66ProtocolBusinessCancelException) {
89        final OpenR66ProtocolBusinessCancelException e2 =
90            (OpenR66ProtocolBusinessCancelException) throwable;
91        logger.debug("Request is canceled: {}", e2.getMessage());
92        return e2;
93      } else if (throwable instanceof OpenR66ProtocolBusinessStopException) {
94        final OpenR66ProtocolBusinessStopException e2 =
95            (OpenR66ProtocolBusinessStopException) throwable;
96        logger.debug("Request is stopped: {}", e2.getMessage());
97        return e2;
98      } else if (throwable instanceof OpenR66ProtocolBusinessQueryAlreadyFinishedException) {
99        final OpenR66ProtocolBusinessQueryAlreadyFinishedException e2 =
100           (OpenR66ProtocolBusinessQueryAlreadyFinishedException) throwable;
101       logger.debug("Request is already finished: {}", e2.getMessage());
102       return e2;
103     } else if (throwable instanceof OpenR66ProtocolBusinessQueryStillRunningException) {
104       final OpenR66ProtocolBusinessQueryStillRunningException e2 =
105           (OpenR66ProtocolBusinessQueryStillRunningException) throwable;
106       logger.debug("Request is still running: {}", e2.getMessage());
107       return e2;
108     } else if (throwable instanceof OpenR66ProtocolBusinessRemoteFileNotFoundException) {
109       final OpenR66ProtocolBusinessRemoteFileNotFoundException e2 =
110           (OpenR66ProtocolBusinessRemoteFileNotFoundException) throwable;
111       logger.debug("Remote server did not find file: {}", e2.getMessage());
112       return e2;
113     } else if (throwable instanceof OpenR66ProtocolBusinessNoWriteBackException) {
114       final OpenR66ProtocolBusinessNoWriteBackException e2 =
115           (OpenR66ProtocolBusinessNoWriteBackException) throwable;
116       logger.error("Command Error Reply: {}", e2.getMessage());
117       return e2;
118     } else if (throwable instanceof OpenR66ProtocolShutdownException) {
119       final OpenR66ProtocolShutdownException e2 =
120           (OpenR66ProtocolShutdownException) throwable;
121       logger.debug("Command Shutdown {}", e2.getMessage());
122       return e2;
123     } else if (throwable instanceof OpenR66Exception) {
124       final OpenR66Exception../../../org/waarp/openr66/protocol/exception/OpenR66Exception.html#OpenR66Exception">OpenR66Exception e2 = (OpenR66Exception) throwable;
125       logger.debug("Command Error Reply: {}", e2.getMessage());
126       return e2;
127     } else if (throwable instanceof BindException) {
128       final BindException e2 = (BindException) throwable;
129       logger.debug("Address already in use {}", e2.getMessage());
130       return new OpenR66ProtocolNetworkException("Address already in use", e2);
131     } else if (throwable instanceof NotYetConnectedException) {
132       final NotYetConnectedException e2 = (NotYetConnectedException) throwable;
133       logger.debug("Timeout occurs {}", e2.getMessage());
134       return new OpenR66ProtocolNetworkException("Timeout occurs", e2);
135     } else if (throwable instanceof NullPointerException) {
136       final NullPointerException e2 = (NullPointerException) throwable;
137       logger.error("Null pointer Exception", e2);
138       return new OpenR66ProtocolSystemException("Null Pointer Exception", e2);
139     } else if (throwable instanceof SSLException) {
140       final SSLException e2 = (SSLException) throwable;
141       logger.debug("Connection aborted since SSL Error {} with Channel {}",
142                    e2.getMessage(), channel);
143       return new OpenR66ProtocolBusinessNoWriteBackException(
144           "SSL Connection aborted", e2);
145     } else if (throwable instanceof IOException) {
146       final IOException e2 = (IOException) throwable;
147       logger
148           .debug("Connection aborted since {} with Channel {}", e2.getMessage(),
149                  channel);
150       if (channel.isActive()) {
151         return new OpenR66ProtocolSystemException(
152             "Connection aborted due to " + e2.getMessage(), e2);
153       } else {
154         return new OpenR66ProtocolBusinessNoWriteBackException(
155             "Connection aborted due to " + e2.getMessage(), e2);
156       }
157     } else if (throwable instanceof RejectedExecutionException) {
158       final RejectedExecutionException e2 =
159           (RejectedExecutionException) throwable;
160       logger
161           .debug("Connection aborted since {} with Channel {}", e2.getMessage(),
162                  channel);
163       if (channel.isActive()) {
164         return new OpenR66ProtocolSystemException("Execution aborted", e2);
165       } else {
166         return new OpenR66ProtocolBusinessNoWriteBackException(
167             "Execution aborted", e2);
168       }
169     } else if (throwable instanceof OutOfMemoryError) {
170       final OpenR66ProtocolShutdownException e2 =
171           new OpenR66ProtocolShutdownException("Restart since OOME raized",
172                                                throwable);
173       logger.debug("Force Shutdown and Restart : {}", e2.getMessage());
174       if (Configuration.configuration.getR66Mib() != null) {
175         Configuration.configuration.getR66Mib().notifyWarning(
176             "OOME so shutdown and restart", throwable.getMessage());
177       }
178       WaarpShutdownHook.setRestart(true);
179       return e2;
180     } else {
181       logger.error(
182           "Unexpected exception from Outband" + " Ref Channel: " + channel,
183           throwable);
184     }
185     if (Configuration.configuration.getR66Mib() != null) {
186       Configuration.configuration.getR66Mib()
187                                  .notifyWarning("Unexpected exception",
188                                                 throwable.getMessage());
189     }
190     return new OpenR66ProtocolSystemException(
191         "Unexpected exception: " + throwable.getMessage(), throwable);
192   }
