33  /**
34   * A list of all constants of the RESTv2 API.
35   * <p>
36   * This includes the URI of all the entry points, the name of HTTP headers
37   * specific to the API, the name of
38   * the JSON objects' fields, and a {@link DAOFactory} to create DAOs for access
39   * to the database.
40   */
41  public final class RestConstants {
42  
43    static {
44      try {
45        DAOFactory.initialize(ConnectionFactory.getInstance());
46      } catch (Throwable ignored) {//NOSONAR
47        SysErrLogger.FAKE_LOGGER //NOSONAR
48                                 .syserr("Error during static execution",//NOSONAR
49                                         ignored);//NOSONAR
50      }
51      DAO_FACTORY = DAOFactory.getInstance();
52    }
53  
54    /**
55     * Makes the default constructor of this utility class inaccessible.
56     */
57    private RestConstants() throws InstantiationException {
58      throw new InstantiationException(
59          getClass().getName() + " cannot be instantiated.");
60    }
