70    /**
71     * Converts the given {@link Business} object into an {@link ObjectNode}.
72     *
73     * @param hostConfig the HostConfig object to convert
74     *
75     * @return the converted ObjectNode
76     */
77    public static ObjectNode businessToNode(Business hostConfig) {
78      final ArrayNode business = getBusinessArray(hostConfig);
79      final ArrayNode roles = getRolesArray(hostConfig);
80      final ArrayNode aliasArray = getAliasArray(hostConfig);
81  
82      final ObjectNode node = new ObjectNode(JsonNodeFactory.instance);
83      node.putArray(BUSINESS).addAll(business);
84      node.putArray(ROLES).addAll(roles);
85      node.putArray(ALIASES).addAll(aliasArray);
86      node.put(OTHERS, hostConfig.getOthers());
87  
88      return node;
89    }

91    /**
92     * Converts the given {@link ObjectNode} into a {@link Business} object.
93     *
94     * @param object the ObjectNode to convert
95     *
96     * @return the corresponding HostConfig object
97     *
98     * @throws RestErrorException if the given ObjectNode does not
99     *     represent a Business object
100    */
101   public static Business nodeToNewBusiness(ObjectNode object) {
102     final Business emptyBusiness =
103         new Business(SERVER_NAME, "", "<roles></roles>", "<aliases></aliases>",
104                      "<root><version></version></root>");
105 
106     return nodeToUpdatedBusiness(object, emptyBusiness);
107   }

109   /**
110    * Returns the given {@link Business} object updated with the values defined
111    * in the {@link ObjectNode}
112    * parameter. All fields missing from the ObjectNode parameter will stay
113    * unchanged in the returned Business
114    * object.
115    *
116    * @param object the ObjectNode to convert.
117    * @param oldBusiness the Business object to update.
118    *
119    * @return the updated Business object
120    *
121    * @throws RestErrorException if the given ObjectNode does not
122    *     represent a Business object
123    */
124   public static Business nodeToUpdatedBusiness(ObjectNode object,
125                                                Business oldBusiness) {
126     final List<RestError> errors = new ArrayList<RestError>();
127 
128     final Iterator<Map.Entry<String, JsonNode>> fields = object.fields();
129     while (fields.hasNext()) {
130       final Map.Entry<String, JsonNode> field = fields.next();
131       final String name = field.getKey();
132       final JsonNode value = field.getValue();
133 
134       if (name.equalsIgnoreCase(BUSINESS)) {
135         if (value.isArray()) {
136           final Businesses businessList = new Businesses();
137           final Iterator<JsonNode> business = value.elements();
138           while (business.hasNext()) {
139             final JsonNode businessName = business.next();
140             if (businessName.isTextual()) {
141               businessList.business.add(businessName.asText());
142             } else {
143               errors
144                   .add(ILLEGAL_FIELD_VALUE(BUSINESS, businessName.toString()));
145             }
146           }
147           oldBusiness.setBusiness(XmlUtils.objectToXml(businessList));
148         } else {
149           errors.add(ILLEGAL_FIELD_VALUE(BUSINESS, value.toString()));
150         }
151       } else if (name.equalsIgnoreCase(ROLES)) {
152         if (value.isArray()) {
153           try {
154             final Roles roles = nodeToRoles((ArrayNode) value);
155             oldBusiness.setRoles(XmlUtils.objectToXml(roles));
156           } catch (final RestErrorException e) {
157             errors.addAll(e.errors);
158           }
159         } else {
160           errors.add(ILLEGAL_FIELD_VALUE(ROLES, value.toString()));
161         }
162       } else if (name.equalsIgnoreCase(ALIASES)) {
163         if (value.isArray()) {
164           try {
165             final Aliases aliases = nodeToAliasList((ArrayNode) value);
166             oldBusiness.setAliases(XmlUtils.objectToXml(aliases));
167           } catch (final RestErrorException e) {
168             errors.addAll(e.errors);
169           }
170         } else {
171           errors.add(ILLEGAL_FIELD_VALUE(ALIASES, value.toString()));
172         }
173       } else if (name.equalsIgnoreCase(OTHERS)) {
174         if (value.isTextual() &&
175             value.asText().matches("<root><version>.+</version></root>")) {
176           oldBusiness.setOthers(value.asText());
177 
178         } else {
179           errors.add(ILLEGAL_FIELD_VALUE(ALIASES, value.toString()));
180         }
181       } else {
182         errors.add(UNKNOWN_FIELD(name));
183       }
184     }
185 
186     if (errors.isEmpty()) {
187       return oldBusiness;
188     } else {
189       throw new RestErrorException(errors);
190     }
191   }

193   /**
194    * Returns the requested host's list of roles on the server.
195    *
196    * @param hostName the desired host's name
197    *
198    * @return the host's list of roles
199    *
200    * @throws InternalServerErrorException if an unexpected error
201    *     occurred
202    */
203   public static List<ROLE> getRoles(String hostName) {
204     ArrayNode array;
205     BusinessDAO businessDAO = null;
206     try {
207       businessDAO = DAO_FACTORY.getBusinessDAO();
208       final Business config = businessDAO.select(SERVER_NAME);
209       array = getRolesArray(config);
210     } catch (final DAOConnectionException e) {
211       throw new InternalServerErrorException(e);
212     } catch (final DAONoDataException e) {
213       throw new InternalServerErrorException(e);
214     } finally {
215       DAOFactory.closeDAO(businessDAO);
216     }
217 
218     final Roles roles = nodeToRoles(array);
219 
220     for (final RoleEntry role : roles.roles) {
221       if (role.hostName.equals(hostName)) {
222         return role.roleList;
223       }
224     }
225     return null;
226   }

230   /**
231    * Converts and returns the list of aliases of the given {@link Business}
232    * object into an {@link ArrayNode}.
233    *
234    * @param hostConfig the server's HostConfig object
235    *
236    * @return the server's list of known aliases
237    */
238   private static ArrayNode getAliasArray(Business hostConfig) {
239     final ArrayNode array = new ArrayNode(JsonNodeFactory.instance);
240     if (hostConfig.getAliases() != null) {
241       try {
242         final Aliases aliases =
243             XmlUtils.xmlToObject(hostConfig.getAliases(), Aliases.class);
244 
245         for (final AliasEntry alias : aliases.aliases) {
246           final ObjectNode node = new ObjectNode(JsonNodeFactory.instance);
247           final ArrayNode aliasIds = new ArrayNode(JsonNodeFactory.instance);
248 
249           for (final String aliasId : alias.aliasList) {
250             aliasIds.add(aliasId);
251           }
252 
253           node.put(HOST_NAME, alias.hostName);
254           node.putArray(ALIAS_LIST).addAll(aliasIds);
255 
256           array.add(node);
257         }
258       } catch (InternalServerErrorException ignore) { //NOSONAR
259         SysErrLogger.FAKE_LOGGER.syserr("No Alias definition");
260       }
261     }
262     return array;
263   }

265   /**
266    * Converts and returns the list of roles of the given {@link Business}
267    * object into an {@link ArrayNode}.
268    *
269    * @param hostConfig the server's HostConfig object
270    *
271    * @return the server's list of known aliases
272    */
273   private static ArrayNode getRolesArray(Business hostConfig) {
274     final ArrayNode array = new ArrayNode(JsonNodeFactory.instance);
275     if (hostConfig.getRoles() != null) {
276       try {
277         final Roles roles =
278             XmlUtils.xmlToObject(hostConfig.getRoles(), Roles.class);
279 
280         for (final RoleEntry role : roles.roles) {
281           final ObjectNode node = new ObjectNode(JsonNodeFactory.instance);
282           final ArrayNode roleTypes = new ArrayNode(JsonNodeFactory.instance);
283 
284           for (final ROLE roleType : role.roleList) {
285             roleTypes.add(roleType.name());
286           }
287 
288           node.put(HOST_NAME, role.hostName);
289           node.putArray(ROLE_LIST).addAll(roleTypes);
290 
291           array.add(node);
292         }
293       } catch (InternalServerErrorException ignore) {//NOSONAR
294         SysErrLogger.FAKE_LOGGER.syserr("No Roles definition");
295       }
296     }
297     return array;
298   }

300   /**
301    * Converts and returns the list of business partners of the given {@link
302    * Business} object into an
303    * {@link ArrayNode}.
304    *
305    * @param hostConfig the server's HostConfig object
306    *
307    * @return the server's list of known aliases
308    */
309   private static ArrayNode getBusinessArray(Business hostConfig) {
310     final ArrayNode array = new ArrayNode(JsonNodeFactory.instance);
311     if (hostConfig.getBusiness() != null) {
312       try {
313         final Businesses business =
314             XmlUtils.xmlToObject(hostConfig.getBusiness(), Businesses.class);
315 
316         for (final String businessId : business.business) {
317           array.add(businessId);
318         }
319       } catch (InternalServerErrorException ignore) {//NOSONAR
320         SysErrLogger.FAKE_LOGGER.syserr("No Business definition");
321       }
322     }
323     return array;
324   }

326   /**
327    * Converts the given {@link ArrayNode} into an {@link Aliases} object.
328    *
329    * @param array the JSON array of aliases
330    *
331    * @return the XML serializable list of aliases
332    *
333    * @throws RestErrorException if the given ArrayNode does not
334    *     represent a list of aliases
335    */
336   private static Aliases nodeToAliasList(ArrayNode array) {
337     final List<AliasEntry> aliases = new ArrayList<AliasEntry>();
338     final List<RestError> errors = new ArrayList<RestError>();
339 
340     final Iterator<JsonNode> elements = array.elements();
341     while (elements.hasNext()) {
342       final JsonNode element = elements.next();
343       final AliasEntry alias = new AliasEntry();
344 
345       if (!element.isObject()) {
346         errors.add(ILLEGAL_PARAMETER_VALUE(ALIASES, element.toString()));
347         continue;
348       }
349 
350       final Iterator<Map.Entry<String, JsonNode>> fields = element.fields();
351       while (fields.hasNext()) {
352         final Map.Entry<String, JsonNode> field = fields.next();
353         final String name = field.getKey();
354         final JsonNode value = field.getValue();
355 
356         if (name.equalsIgnoreCase(HOST_NAME)) {
357           if (value.isTextual()) {
358             alias.hostName = value.asText();
359           } else {
360             errors.add(ILLEGAL_PARAMETER_VALUE(HOST_NAME, value.toString()));
361           }
362         } else if (name.equalsIgnoreCase(ALIAS_LIST)) {
363           if (value.isArray()) {
364             final Iterator<JsonNode> aliasList = value.elements();
365             while (aliasList.hasNext()) {
366               final JsonNode aliasId = aliasList.next();
367               if (aliasId.isTextual()) {
368                 alias.aliasList.add(aliasId.asText());
369               } else {
370                 errors.add(
371                     ILLEGAL_PARAMETER_VALUE(ALIAS_LIST, aliasId.toString()));
372               }
373             }
374           } else {
375             errors.add(ILLEGAL_PARAMETER_VALUE(ALIAS_LIST, value.toString()));
376           }
377         } else {
378           errors.add(UNKNOWN_FIELD(name));
379         }
380       }
381 
382       if (alias.hostName.isEmpty()) {
383         errors.add(MISSING_FIELD(HOST_NAME));
384       }
385       if (alias.aliasList.isEmpty()) {
386         errors.add(MISSING_FIELD(ALIAS_LIST));
387       }
388       aliases.add(alias);
389     }
390 
391     if (errors.isEmpty()) {
392       return new Aliases(aliases);
393     } else {
394       throw new RestErrorException(errors);
395     }
396   }

398   /**
399    * Converts the given {@link ArrayNode} into an {@link Roles} object.
400    *
401    * @param array the JSON list of roles
402    *
403    * @return the XML serializable list of roles
404    *
405    * @throws RestErrorException if the given ArrayNode does not
406    *     represent a list of roles
407    */
408   private static Roles nodeToRoles(ArrayNode array) {
409     final List<RoleEntry> roles = new ArrayList<RoleEntry>();
410     final List<RestError> errors = new ArrayList<RestError>();
411 
412     final Iterator<JsonNode> elements = array.elements();
413     while (elements.hasNext()) {
414       final JsonNode element = elements.next();
415       final RoleEntry role = new RoleEntry();
416 
417       if (!element.isObject()) {
418         errors.add(ILLEGAL_PARAMETER_VALUE(ROLES, element.toString()));
419         continue;
420       }
421 
422       final Iterator<Map.Entry<String, JsonNode>> fields = element.fields();
423       while (fields.hasNext()) {
424         final Map.Entry<String, JsonNode> field = fields.next();
425         final String name = field.getKey();
426         final JsonNode value = field.getValue();
427 
428         if (name.equalsIgnoreCase(HOST_NAME)) {
429           if (value.isTextual()) {
430             role.hostName = value.asText();
431           } else {
432             errors.add(ILLEGAL_PARAMETER_VALUE(HOST_NAME, value.toString()));
433           }
434         } else if (name.equalsIgnoreCase(ROLE_LIST)) {
435           if (value.isArray()) {
436             final Iterator<JsonNode> roleTypes = value.elements();
437             while (roleTypes.hasNext()) {
438               final JsonNode roleType = roleTypes.next();
439               if (roleType.isTextual()) {
440                 try {
441                   role.roleList.add(ROLE.valueOf(roleType.asText()));
442                 } catch (final IllegalArgumentException e) {
443                   errors.add(
444                       ILLEGAL_PARAMETER_VALUE(ROLE_LIST, roleType.toString()));
445                 }
446               } else {
447                 errors.add(
448                     ILLEGAL_PARAMETER_VALUE(ROLE_LIST, roleType.toString()));
449               }
450             }
451           } else {
452             errors.add(ILLEGAL_PARAMETER_VALUE(ROLE_LIST, value.toString()));
453           }
454         } else {
455           errors.add(UNKNOWN_FIELD(name));
456         }
457       }
458 
459       if (role.hostName.isEmpty()) {
460         errors.add(MISSING_FIELD(HOST_NAME));
461       }
462       if (role.roleList.isEmpty()) {
463         errors.add(MISSING_FIELD(ROLE_LIST));
464       }
465       roles.add(role);
466     }
467 
468     if (errors.isEmpty()) {
469       return new Roles(roles);
470     } else {
471       throw new RestErrorException(errors);
472     }
473   }
