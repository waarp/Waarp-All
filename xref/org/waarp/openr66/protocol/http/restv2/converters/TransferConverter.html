71    /**
72     * All the possible ways to order a list of transfer objects.
73     */
74    public enum Order {
75      /**
76       * By transferId, in ascending order.
77       */
78      ascId(ID_FIELD, true),
79      /**
80       * By transferId, in descending order.
81       */
82      descId(ID_FIELD, false),
83      /**
84       * By fileName, in ascending order.
85       */
86      ascFile(ORIGINAL_NAME_FIELD, true),
87      /**
88       * By fileName, in descending order.
89       */
90      descFile(ORIGINAL_NAME_FIELD, false),
91      /**
92       * By starting date, in ascending order.
93       */
94      ascStart(TRANSFER_START_FIELD, true),
95      /**
96       * By starting date, in descending order.
97       */
98      descStart(TRANSFER_START_FIELD, false),
99      /**
100      * By end date, in ascending order.
101      */
102     ascStop(TRANSFER_STOP_FIELD, true),
103     /**
104      * By end date, in descending order.
105      */
106     descStop(TRANSFER_STOP_FIELD, false);
107 
108     /**
109      * The name of the database column used for sorting.
110      */
111     public final String column;
112     /**
113      * If the order is ascending or descending.
114      */
115     public final boolean ascend;
116 
117     Order(final String column, final boolean ascend) {
118       this.column = column;
119       this.ascend = ascend;
120     }
121   }

125   /**
126    * Returns an {@link ObjectNode} representing the {@link Transfer} object
127    * given as parameter.
128    *
129    * @param transfer the Transfer object to serialize
130    *
131    * @return the corresponding ObjectNode
132    */
133   public static ObjectNode transferToNode(final Transfer transfer) {
134     final ObjectNode node = new ObjectNode(JsonNodeFactory.instance);
135     node.put(TRANSFER_ID, transfer.getId());
136     node.put(GLOBAL_STEP, transfer.getGlobalStep().toString());
137     node.put(GLOBAL_LAST_STEP, transfer.getLastGlobalStep().toString());
138     node.put(STEP, transfer.getStep());
139     node.put(RANK, transfer.getRank());
140     node.put(UPDATED_INFO, transfer.getUpdatedInfo().toString());
141     node.put(STEP_STATUS, transfer.getStepStatus().toString());
142     node.put(ERROR_CODE, transfer.getInfoStatus().code);
143     node.put(ERROR_MESSAGE, transfer.getInfoStatus().getMesg());
144     node.put(ORIGINAL_FILENAME, transfer.getOriginalName());
145     node.put(FILENAME, transfer.getFilename());
146     node.put(RULE, transfer.getRule());
147     node.put(BLOCK_SIZE, transfer.getBlockSize());
148     node.put(FILE_INFO, transfer.getFileInfo());
149     node.put(TRANSFER_INFO, transfer.getTransferInfo());
150     node.put(START, new DateTime(transfer.getStart()).toString());
151     node.put(STOP, new DateTime(transfer.getStop()).toString());
152     node.put(REQUESTED, transfer.getRequested());
153     node.put(REQUESTER, transfer.getRequester());
154     node.put(RETRIEVE, transfer.getRetrieveMode());
155 
156     return node;
157   }

159   /**
160    * Initialize a {@link Transfer} object using the values of the given {@link
161    * ObjectNode}.
162    *
163    * @param object the ObjectNode to convert
164    *
165    * @return the new Transfer object
166    *
167    * @throws RestErrorException if the given ObjectNode does not
168    *     represent a Transfer object
169    * @throws InternalServerErrorException if an unexpected error
170    *     occurred
171    */
172   public static Transfer nodeToNewTransfer(final ObjectNode object) {
173     final Transfer defaultTransfer =
174         new Transfer(null, null, -1, false, null, null, ZERO_COPY_CHUNK_SIZE);
175     defaultTransfer.setRequester(serverName());
176     defaultTransfer.setOwnerRequest(serverName());
177     defaultTransfer.setBlockSize(Configuration.configuration.getBlockSize());
178     defaultTransfer.setTransferInfo("{}");
179     defaultTransfer.setStart(new Timestamp(DateTime.now().getMillis()));
180     final Transfer transfer = parseNode(object, defaultTransfer);
181 
182     ModeTrans mode;
183     RuleDAO ruleDAO = null;
184     try {
185       ruleDAO = DAO_FACTORY.getRuleDAO();
186       final Rule rule = ruleDAO.select(transfer.getRule());
187       mode = ModeTrans.fromCode(rule.getMode());
188     } catch (final DAOConnectionException e) {
189       throw new InternalServerErrorException(e);
190     } catch (final DAONoDataException e) {
191       throw new InternalServerErrorException(e);
192     } finally {
193       DAOFactory.closeDAO(ruleDAO);
194     }
195 
196     transfer.setRetrieveMode(
197         mode == ModeTrans.receive || mode == ModeTrans.receiveMD5);
198     transfer.setTransferMode(mode.code);
199     transfer.setStop(transfer.getStart());
200     transfer.setUpdatedInfo(UpdatedInfo.TOSUBMIT);
201 
202     return transfer;
203   }

207   /**
208    * Tells if the given rule exists in the database.
209    *
210    * @param rule the name of the rule
211    *
212    * @return {@code true} if the rule exists, {@code false} otherwise.
213    */
214   private static boolean ruleExists(final String rule) {
215     RuleDAO ruleDAO = null;
216     try {
217       ruleDAO = DAO_FACTORY.getRuleDAO();
218       return ruleDAO.exist(rule);
219     } catch (final DAOConnectionException e) {
220       throw new InternalServerErrorException(e);
221     } finally {
222       DAOFactory.closeDAO(ruleDAO);
223     }
224   }

245   /**
246    * Tells if the given host is allowed to use given rule.
247    *
248    * @param host the name of the host
249    * @param rule the name of the rule
250    *
251    * @return {@code true} if the host is allowed to use the rule, {@code
252    *     false} otherwise
253    */
254   private static boolean canUseRule(final String host, final String rule) {
255     RuleDAO ruleDAO = null;
256     try {
257       ruleDAO = DAO_FACTORY.getRuleDAO();
258       final List<String> hostIds = ruleDAO.select(rule).getHostids();
259       return !hostIds.isEmpty() && !hostIds.contains(host);
260     } catch (final DAOConnectionException e) {
261       throw new InternalServerErrorException(e);
262     } catch (final DAONoDataException e) {
263       throw new InternalServerErrorException(e);
264     } finally {
265       DAOFactory.closeDAO(ruleDAO);
266     }
267   }

269   /**
270    * Returns a list of {@link RestError} corresponding to all the fields
271    * required to initialize a transfer that
272    * are missing from the given {@link Transfer} object. If no fields are
273    * missing, an empty list is returned.
274    *
275    * @param transfer the Transfer object to check.
276    *
277    * @return the list of all missing fields
278    */
279   private static List<RestError> checkRequiredFields(final Transfer transfer) {
280     final List<RestError> errors = new ArrayList<RestError>();
281     if (transfer.getRule() == null || transfer.getRule().isEmpty()) {
282       errors.add(MISSING_FIELD(RULE));
283     }
284     if (transfer.getOriginalName() == null ||
285         transfer.getOriginalName().isEmpty()) {
286       errors.add(MISSING_FIELD(FILENAME));
287     }
288     if (transfer.getRequested() == null || transfer.getRequested().isEmpty()) {
289       errors.add(MISSING_FIELD(REQUESTED));
290     }
291 
292     return errors;
293   }

295   /**
296    * Fills the fields of the given {@link Transfer} object with the values
297    * extracted from the {@link ObjectNode}
298    * parameter, and returns the result.
299    *
300    * @param object the ObjectNode from which the values should be
301    *     extracted
302    * @param transfer the Transfer object whose fields will be filled
303    *
304    * @return the filled Transfer object
305    *
306    * @throws RestErrorException if the given ObjectNode does not
307    *     represent a Transfer object.
308    */
309   private static Transfer parseNode(final ObjectNode object,
310                                     final Transfer transfer) {
311     final List<RestError> errors = new ArrayList<RestError>();
312 
313     final Iterator<Map.Entry<String, JsonNode>> fields = object.fields();
314     while (fields.hasNext()) {
315       final Map.Entry<String, JsonNode> field = fields.next();
316       final String name = field.getKey();
317       final JsonNode value = field.getValue();
318 
319       if (name.equalsIgnoreCase(RULE)) {
320         if (value.isTextual()) {
321           if (ruleExists(value.asText())) {
322             transfer.setRule(value.asText());
323           } else {
324             errors.add(UNKNOWN_RULE(value.asText()));
325           }
326         } else {
327           errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
328         }
329       } else if (name.equalsIgnoreCase(FILENAME)) {
330         if (value.isTextual()) {
331           transfer.setOriginalName(value.asText());
332           transfer.setFilename(value.asText());
333         } else {
334           errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
335         }
336       } else if (name.equalsIgnoreCase(REQUESTED)) {
337         if (value.isTextual()) {
338           if (hostExists(value.asText())) {
339             transfer.setRequested(value.asText());
340             try {
341               transfer.setRequester(
342                   Configuration.configuration.getHostId(value.asText()));
343             } catch (final WaarpDatabaseException e) {
344               // Ignore !!
345             }
346           } else {
347             errors.add(UNKNOWN_HOST(value.asText()));
348           }
349         } else {
350           errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
351         }
352       } else if (name.equalsIgnoreCase(BLOCK_SIZE)) {
353         if (value.canConvertToInt() && value.asInt() > 0) {
354           transfer.setBlockSize(value.asInt());
355         } else {
356           errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
357         }
358       } else if (name.equalsIgnoreCase(FILE_INFO)) {
359         if (value.isTextual()) {
360           transfer.setFileInfo(value.asText());
361         } else {
362           errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
363         }
364       } else if (name.equalsIgnoreCase(TRANSFER_INFO)) {
365         if (value.isTextual()) {
366           transfer.setTransferInfo(value.asText());
367         } else {
368           errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
369         }
370       } else if (name.equalsIgnoreCase(START)) {
371         if (value.isTextual()) {
372           try {
373             final DateTime start = DateTime.parse(value.asText());
374             if (start.isBeforeNow()) {
375               errors.add(ILLEGAL_FIELD_VALUE(name, value.asText()));
376             } else {
377               transfer.setStart(new Timestamp(start.getMillis()));
378             }
379           } catch (final IllegalArgumentException e) {
380             errors.add(ILLEGAL_FIELD_VALUE(name, value.asText()));
381           }
382         } else {
383           errors.add(ILLEGAL_FIELD_VALUE(name, value.toString()));
384         }
385       }
386     }
387 
388     // check that both hosts are allowed to use the transfer rule
389     final String rule = transfer.getRule();
390     final String requested = transfer.getRequested();
391     final String requester = transfer.getRequester();
392     if (rule != null && !requested.isEmpty() && canUseRule(requested, rule)) {
393       errors.add(RULE_NOT_ALLOWED(requested, rule));
394     }
395     if (rule != null && !requester.isEmpty() && canUseRule(requester, rule)) {
396       errors.add(RULE_NOT_ALLOWED(requester, rule));
397     }
398 
399     errors.addAll(checkRequiredFields(transfer));
400 
401     if (errors.isEmpty()) {
402       return transfer;
403     } else {
404       throw new RestErrorException(errors);
405     }
406   }
