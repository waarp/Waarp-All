143   public NetworkTransaction() {
144     networkChannelGroup = new DefaultChannelGroup("NetworkChannels",
145                                                   Configuration.configuration
146                                                       .getHandlerGroup()
147                                                       .next());
148     final NetworkServerInitializer networkServerInitializer =
149         new NetworkServerInitializer(false);
150     clientBootstrap = new Bootstrap();
151     WaarpNettyUtil.setBootstrap(clientBootstrap, Configuration.configuration
152         .getNetworkWorkerGroup(), (int) Configuration.configuration
153         .getTimeoutCon());
154     clientBootstrap.handler(networkServerInitializer);
155     Configuration.configuration.setupLimitHandler();
156     clientSslBootstrap = new Bootstrap();
157     if (Configuration.configuration.isUseSSL() &&
158         Configuration.configuration.getHostSslId() != null) {
159       final NetworkSslServerInitializer networkSslServerInitializer =
160           new NetworkSslServerInitializer(true);
161       WaarpNettyUtil.setBootstrap(clientSslBootstrap,
162                                   Configuration.configuration
163                                       .getNetworkWorkerGroup(),
164                                   (int) Configuration.configuration
165                                       .getTimeoutCon());
166       clientSslBootstrap.handler(networkSslServerInitializer);
167     } else {
168       if (Configuration.configuration.isWarnOnStartup()) {
169         logger.warn("No SSL support configured");
170       } else {
171         logger.info("No SSL support configured");
172       }
173     }
174   }

301   private static WaarpLock getChannelLock(SocketAddress socketAddress) {
302     lockOfLock.lock();
303     try {
304       if (socketAddress == null) {
305         // should not
306         logger.info("SocketAddress empty here !");
307         return emptyLock;
308       }
309       WaarpLock socketLock = getLockNCR(socketAddress);
310       if (socketLock == null) {
311         socketLock = new WaarpLock(true);
312       }
313       // update TTL
314       addLockNCR(socketAddress, socketLock);
315       return socketLock;
316     } finally {
317       lockOfLock.unlock();
318     }
319   }
320 
321   private static void removeChannelLock() {
322     lockOfLock.lock();
323     try {
324       reentrantLockOnSocketAddressConcurrentHashMap.forceClearOldest();
325     } finally {
326       lockOfLock.unlock();
327     }
328   }

351   /**
352    * Create a connection to the specified socketAddress with multiple retries
353    *
354    * @param socketAddress
355    * @param isSSL
356    * @param futureRequest
357    *
358    * @return the LocalChannelReference
359    *
360    * @throws OpenR66ProtocolNotAuthenticatedException
361    */
362   public LocalChannelReference createConnectionWithRetryWithAuthenticationException(
363       SocketAddress socketAddress, boolean isSSL, R66Future futureRequest)
364       throws OpenR66ProtocolNotAuthenticatedException {
365     LocalChannelReference localChannelReference = null;
366     for (int i = 0; i < Configuration.RETRYNB; i++) {
367       if (WaarpShutdownHook.isShutdownStarting()) {
368         logger.error("Cannot connect : Local system in shutdown");
369         break;
370       }
371       try {
372         localChannelReference =
373             createConnection(socketAddress, isSSL, futureRequest);
374         break;
375       } catch (final OpenR66ProtocolRemoteShutdownException e) {
376         logger.error("Cannot connect : {}", e.getMessage());
377         logger.debug(e);
378         break;
379       } catch (final OpenR66ProtocolNoConnectionException e) {
380         logger.error("Cannot connect : {}", e.getMessage());
381         logger.debug(e);
382         break;
383       } catch (final OpenR66ProtocolNotAuthenticatedException e) {
384         logger.error("Cannot be authenticated : {}", e.getMessage());
385         logger.debug(e);
386         throw e;
387       } catch (final OpenR66ProtocolNetworkException e) {
388         // Can retry
389         logger.error("Cannot connect : {}. Will retry", e.getMessage());
390         logger.debug(e);
391         try {
392           Thread.sleep(Configuration.configuration.getDelayRetry());
393         } catch (final InterruptedException e1) {//NOSONAR
394           SysErrLogger.FAKE_LOGGER.ignoreLog(e1);
395         }
396       }
397     }
398     if (localChannelReference != null) {
399       logger.info("Connected");
400     }
401     return localChannelReference;
402   }

404   /**
405    * Create a connection to the specified socketAddress
406    *
407    * @param socketAddress
408    * @param isSSL
409    * @param futureRequest
410    *
411    * @return the LocalChannelReference
412    *
413    * @throws OpenR66ProtocolNetworkException
414    * @throws OpenR66ProtocolRemoteShutdownException
415    * @throws OpenR66ProtocolNoConnectionException
416    * @throws OpenR66ProtocolNotAuthenticatedException
417    */
418   private LocalChannelReference createConnection(SocketAddress socketAddress,
419                                                  boolean isSSL,
420                                                  R66Future futureRequest)
421       throws OpenR66ProtocolNetworkException,
422              OpenR66ProtocolRemoteShutdownException,
423              OpenR66ProtocolNoConnectionException,
424              OpenR66ProtocolNotAuthenticatedException {
425     NetworkChannelReference networkChannelReference = null;
426     LocalChannelReference localChannelReference;
427     boolean ok = false;
428     // check valid limit on server side only (could be the initiator but not a client)
429     final DbHostAuth auth =
430         isSSL? Configuration.configuration.getHostSslAuth() :
431             Configuration.configuration.getHostAuth();
432     if (!auth.isClient()) {
433       boolean valid = false;
434       for (int i = 0; i < Configuration.RETRYNB * 2; i++) {
435         if (Configuration.configuration.getConstraintLimitHandler()
436                                        .checkConstraintsSleep(i)) {
437           logger.debug("Constraints exceeded: " + i);
438         } else {
439           logger.debug("Constraints NOT exceeded");
440           valid = true;
441           break;
442         }
443       }
444       if (!valid) {
445         // Limit is locally exceeded
446         throw new OpenR66ProtocolNetworkException(
447             "Cannot connect to remote server due to local overload");
448       }
449     }
450     try {
451       // Ensure networkChannelReference and localChannelReference are linked
452       synchronized (emptyLock) {
453         networkChannelReference = createNewConnection(socketAddress, isSSL);
454         try {
455           localChannelReference =
456               Configuration.configuration.getLocalTransaction().createNewClient(
457                   networkChannelReference, ChannelUtils.NOCHANNEL,
458                   futureRequest, isSSL);
459         } catch (final OpenR66ProtocolSystemException e) {
460           throw new OpenR66ProtocolNetworkException(e);
461         } catch (final NullPointerException e) {
462           throw new OpenR66ProtocolNetworkException(e);
463         }
464       }
465       ok = true;
466     } finally {
467       if (!ok && networkChannelReference != null) {
468         checkClosingNetworkChannel(networkChannelReference, null);
469       }
470     }
471     if (localChannelReference.getFutureValidateStartup().isSuccess()) {
472       sendValidationConnection(localChannelReference);
473     } else {
474       final OpenR66ProtocolNetworkException exc =
475           new OpenR66ProtocolNetworkException("Startup is invalid");
476       logger.debug("Startup is Invalid", exc);
477       final R66Result finalValue =
478           new R66Result(exc, null, true, ErrorCode.ConnectionImpossible, null);
479       localChannelReference.invalidateRequest(finalValue);
480       localChannelReference.close();
481       throw exc;
482     }
483     return localChannelReference;
484   }

486   /**
487    * @param socketServerAddress
488    * @param isSSL
489    *
490    * @return the NetworkChannelReference
491    *
492    * @throws OpenR66ProtocolNetworkException
493    * @throws OpenR66ProtocolRemoteShutdownException
494    * @throws OpenR66ProtocolNoConnectionException
495    */
496   private NetworkChannelReference createNewConnection(
497       SocketAddress socketServerAddress, boolean isSSL)
498       throws OpenR66ProtocolNetworkException,
499              OpenR66ProtocolRemoteShutdownException,
500              OpenR66ProtocolNoConnectionException {
501     final WaarpLock socketLock = getChannelLock(socketServerAddress);
502     NetworkChannelReference networkChannelReference;
503     socketLock.lock();
504     try {
505       try {
506         networkChannelReference = getRemoteChannel(socketServerAddress);
507       } catch (final OpenR66ProtocolNoDataException e1) {
508         networkChannelReference = null;
509       }
510       if (networkChannelReference != null) {
511         networkChannelReference.use();
512         logger.info("Already Connected: {}", networkChannelReference);
513         return networkChannelReference;
514       }
515       logger.debug("NEW PHYSICAL CONNECTION REQUIRED");
516       ChannelFuture channelFuture = null;
517       for (int i = 0; i < Configuration.RETRYNB; i++) {
518         if (WaarpShutdownHook.isShutdownStarting()) {
519           throw new OpenR66ProtocolNoConnectionException(
520               "Local system in shutdown");
521         }
522         try {
523           if (isSSL) {
524             if (Configuration.configuration.getHostSslId() != null) {
525               channelFuture = clientSslBootstrap.connect(socketServerAddress);
526             } else {
527               throw new OpenR66ProtocolNoConnectionException("No SSL support");
528             }
529           } else {
530             channelFuture = clientBootstrap.connect(socketServerAddress);
531           }
532         } catch (final ChannelPipelineException e) {
533           throw new OpenR66ProtocolNoConnectionException(
534               "Cannot connect to remote server due to a channel exception");
535         }
536         WaarpNettyUtil.awaitOrInterrupted(channelFuture,
537                                           Configuration.configuration
538                                               .getTimeoutCon());
539         if (channelFuture.isSuccess()) {
540           final Channel channel = channelFuture.channel();
541           if (isSSL &&
542               !NetworkSslServerHandler.isSslConnectedChannel(channel)) {
543             logger.debug("KO CONNECT since SSL handshake is over");
544             channel.close();
545             throw new OpenR66ProtocolNoConnectionException(
546                 "Cannot finish connect to remote server");
547           }
548           if (networkChannelGroup != null) {
549             networkChannelGroup.add(channel);
550           }
551           networkChannelReference =
552               new NetworkChannelReference(channel, socketLock, isSSL);
553           addNCR(networkChannelReference);
554           logger.info("New Real Connection: {}", networkChannelReference);
555           return networkChannelReference;
556         } else {
557           try {
558             Thread.sleep(Configuration.RETRYINMS);
559           } catch (final InterruptedException e) {//NOSONAR
560             SysErrLogger.FAKE_LOGGER.ignoreLog(e);
561           }
562           if (!channelFuture.isDone()) {
563             throw new OpenR66ProtocolNoConnectionException(
564                 "Cannot connect to remote server since received an " +
565                 "interruption");
566           }
567           if (channelFuture.cause() instanceof ConnectException) {
568             logger.debug("KO CONNECT:" + channelFuture.cause().getMessage());
569             throw new OpenR66ProtocolNoConnectionException(
570                 channelFuture.cause().getMessage(), channelFuture.cause());
571           } else {
572             logger.debug("KO CONNECT but retry", channelFuture.cause());
573           }
574         }
575       }
576       if (channelFuture != null) {
577         throw new OpenR66ProtocolNetworkException(
578             "Cannot connect to remote server", channelFuture.cause());
579       } else {
580         throw new OpenR66ProtocolNetworkException(
581             "Cannot connect to remote server");
582       }
583     } finally {
584       socketLock.unlock();
585     }
586   }
