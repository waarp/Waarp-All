132   @Override
133   public void run() {
134     if (Configuration.configuration.isShutdown() || Thread.interrupted()) {
135       taskRunner.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
136       taskRunner.forceSaveStatus();
137       return;
138     }
139     try {
140       if (activeRunners != null) {
141         activeRunners.add(this);
142       }
143       // fix for SelfRequest
144       if (taskRunner.isSelfRequest()) {
145         taskRunner.setSenderByRequestToValidate(false);
146       }
147       R66Future transfer;
148       try {
149         transfer = runTransfer();
150       } catch (final OpenR66RunnerErrorException e) {
151         logger.error("Runner Error: {} {}", e.getMessage(),
152                      taskRunner.toShortString());
153         return;
154       } catch (final OpenR66ProtocolNoConnectionException e) {
155         logger.error("No connection Error {}", e.getMessage());
156         if (localChannelReference != null) {
157           localChannelReference
158               .setErrorMessage(ErrorCode.ConnectionImpossible.getMesg(),
159                                ErrorCode.ConnectionImpossible);
160         }
161         taskRunner.setErrorTask(localChannelReference);
162         try {
163           taskRunner.forceSaveStatus();
164           taskRunner.run();
165         } catch (final OpenR66RunnerErrorException e1) {
166           changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
167                             true);
168         }
169         return;
170       } catch (final OpenR66ProtocolPacketException e) {
171         logger.error("Protocol Error", e);
172         return;
173       } catch (final OpenR66ProtocolNotYetConnectionException e) {
174         logger.warn("No connection warning {}", e.getMessage());
175         return;
176       }
177       final R66Result result = transfer.getResult();
178       if (result != null) {
179         if (result.getCode() == ErrorCode.QueryAlreadyFinished) {
180           logger.warn(Messages.getString(TRANSFER_STATUS) +
181                       (transfer.isSuccess()?
182                           Messages.getString(REQUEST_INFORMATION_SUCCESS) :
183                           Messages.getString(REQUEST_INFORMATION_FAILURE)) +
184                       "     " + ErrorCode.QueryAlreadyFinished.getMesg() + ':' +
185                       (result != null? result.toString() : NO_RESULT));
186         } else {
187           if (transfer.isSuccess()) {
188             logger.info(Messages.getString(TRANSFER_STATUS) +
189                         Messages.getString(REQUEST_INFORMATION_SUCCESS) +
190                         "     " +
191                         (result != null? result.toString() : NO_RESULT));
192           } else {
193             logger.error(Messages.getString(TRANSFER_STATUS) +
194                          Messages.getString(REQUEST_INFORMATION_FAILURE) +
195                          "     " +
196                          (result != null? result.toString() : NO_RESULT));
197           }
198         }
199       } else {
200         if (transfer.isSuccess()) {
201           logger.warn(Messages.getString(TRANSFER_STATUS) +
202                       Messages.getString(REQUEST_INFORMATION_SUCCESS) +
203                       "     no result");
204         } else {
205           logger.error(Messages.getString(TRANSFER_STATUS) +
206                        Messages.getString(REQUEST_INFORMATION_FAILURE) +
207                        "     no result");
208         }
209       }
210     } finally {
211       if (activeRunners != null) {
212         activeRunners.remove(this);
213       }
214       setName("Finished_" + nameTask);
215     }
216   }

218   /**
219    * @param runner
220    * @param limit
221    *
222    * @return True if the task was run less than limit, else False
223    */
224   public boolean incrementTaskRunnerTry(DbTaskRunner runner, int limit) {
225     final String key = runner.getKey();
226     Integer tries = taskRunnerRetryHashMap.get(key);
227     logger.debug("try to find integer: " + tries);
228     if (tries == null) {
229       tries = Integer.valueOf(1);
230     } else {
231       tries += 1;
232     }
233     if (limit <= tries || !Thread.interrupted()) {
234       taskRunnerRetryHashMap.remove(key);
235       return false;
236     } else {
237       taskRunnerRetryHashMap.put(key, tries);
238       return true;
239     }
240   }

242   /**
243    * True transfer run (can be called directly to enable exception outside any
244    * executors)
245    *
246    * @return The R66Future of the transfer operation
247    *
248    * @throws OpenR66RunnerErrorException
249    * @throws OpenR66ProtocolNoConnectionException
250    * @throws OpenR66ProtocolPacketException
251    * @throws OpenR66ProtocolNotYetConnectionException
252    */
253   public R66Future runTransfer()
254       throws OpenR66RunnerErrorException, OpenR66ProtocolNoConnectionException,
255              OpenR66ProtocolPacketException,
256              OpenR66ProtocolNotYetConnectionException {
257     logger.debug("Start attempt Transfer");
258     localChannelReference = initRequest();
259     localChannelReference.getFutureValidRequest().awaitOrInterruptible(
260         Configuration.configuration.getTimeoutCon());
261     if (localChannelReference.getFutureValidRequest().isSuccess()) {
262       return finishTransfer(localChannelReference);
263     } else if (
264         localChannelReference.getFutureValidRequest().getResult() != null &&
265         localChannelReference.getFutureValidRequest().getResult().getCode() ==
266         ErrorCode.ServerOverloaded) {
267       return tryAgainTransferOnOverloaded(true, localChannelReference);
268     } else {
269       return finishTransfer(localChannelReference);
270     }
271   }

273   /**
274    * In case an overloaded signal is returned by the requested
275    *
276    * @param retry if True, it will retry in case of overloaded remote
277    *     server, else it just stops
278    * @param localChannelReference
279    *
280    * @return The R66Future of the transfer operation
281    *
282    * @throws OpenR66RunnerErrorException
283    * @throws OpenR66ProtocolNoConnectionException
284    * @throws OpenR66ProtocolPacketException
285    * @throws OpenR66ProtocolNotYetConnectionException
286    */
287   public R66Future tryAgainTransferOnOverloaded(boolean retry,
288                                                 LocalChannelReference localChannelReference)
289       throws OpenR66RunnerErrorException, OpenR66ProtocolNoConnectionException,
290              OpenR66ProtocolPacketException,
291              OpenR66ProtocolNotYetConnectionException {
292     if (this.localChannelReference == null) {
293       this.localChannelReference = localChannelReference;
294     }
295     final boolean incRetry =
296         incrementTaskRunnerTry(taskRunner, Configuration.RETRYNB);
297     logger.debug("tryAgainTransferOnOverloaded: " + retry + ':' + incRetry);
298     switch (taskRunner.getUpdatedInfo()) {
299       case DONE:
300       case INERROR:
301       case INTERRUPTED:
302         break;
303       default:
304         changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ServerOverloaded,
305                           true);
306     }
307     // redo if possible
308     if (retry && incRetry) {
309       try {
310         Thread.sleep(Configuration.configuration.getConstraintLimitHandler()
311                                                 .getSleepTime());
312       } catch (final InterruptedException e) {//NOSONAR
313         SysErrLogger.FAKE_LOGGER.ignoreLog(e);
314       }
315       return runTransfer();
316     } else {
317       if (localChannelReference == null) {
318         taskRunner.setLocalChannelReference(new LocalChannelReference());
319       }
320       taskRunner.getLocalChannelReference()
321                 .setErrorMessage(ErrorCode.ConnectionImpossible.getMesg(),
322                                  ErrorCode.ConnectionImpossible);
323       taskRunner.setErrorTask(localChannelReference);
324       taskRunner.run();
325       throw new OpenR66ProtocolNoConnectionException(
326           "End of retry on ServerOverloaded");
327     }
328   }

330   /**
331    * Finish the transfer (called at the end of runTransfer)
332    *
333    * @param localChannelReference
334    *
335    * @return The R66Future of the transfer operation
336    *
337    * @throws OpenR66ProtocolNotYetConnectionException
338    * @throws OpenR66ProtocolPacketException
339    * @throws OpenR66ProtocolNoConnectionException
340    * @throws OpenR66RunnerErrorException
341    */
342   public R66Future finishTransfer(LocalChannelReference localChannelReference)
343       throws OpenR66RunnerErrorException {
344     if (this.localChannelReference == null) {
345       this.localChannelReference = localChannelReference;
346     }
347     final R66Future transfer = localChannelReference.getFutureRequest();
348     transfer.awaitOrInterruptible();
349     taskRunnerRetryHashMap.remove(taskRunner.getKey());
350     logger.info("Request done with {}",
351                 transfer.isSuccess()? "success" : "error");
352     localChannelReference.close();
353     // now reload TaskRunner if it still exists (light client can forget it)
354     final boolean isSender = taskRunner.isSender();
355     if (transfer.isSuccess()) {
356       try {
357         taskRunner.select();
358       } catch (final WaarpDatabaseException e) {
359         logger.debug("Not a problem but cannot find at the end the task", e);
360         taskRunner.setFrom(transfer.getRunner());
361       }
362       taskRunner.setSender(isSender);
363       changeUpdatedInfo(UpdatedInfo.DONE, ErrorCode.CompleteOk, false);
364     } else {
365       try {
366         taskRunner.select();
367       } catch (final WaarpDatabaseException e) {
368         logger.debug("Not a problem but cannot find at the end the task");
369         taskRunner.setFrom(transfer.getRunner());
370       }
371       taskRunner.setSender(isSender);
372       // Case when we were interrupted
373       if (transfer.getResult() == null) {
374         switch (taskRunner.getUpdatedInfo()) {
375           case DONE:
376             final R66Result ok =
377                 new R66Result(null, true, ErrorCode.CompleteOk, taskRunner);
378             transfer.setResult(ok);
379             transfer.setSuccess();
380             changeUpdatedInfo(UpdatedInfo.DONE, ErrorCode.CompleteOk, false);
381             break;
382           case INERROR:
383           case INTERRUPTED:
384           default:
385             final R66Result error =
386                 new R66Result(null, true, ErrorCode.Internal, taskRunner);
387             transfer.setResult(error);
388             transfer.cancel();
389             changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.Internal, false);
390         }
391         return transfer;
392       }
393       if (transfer.getResult().getCode() == ErrorCode.QueryAlreadyFinished) {
394         // check if post task to execute
395         logger.warn("WARN QueryAlreadyFinished:     " + transfer + "     " +
396                     taskRunner.toShortString());
397         try {
398           TransferUtils
399               .finalizeTaskWithNoSession(taskRunner, localChannelReference);
400         } catch (final OpenR66RunnerErrorException e) {
401           taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
402           taskRunner.forceSaveStatus();
403         }
404       } else {
405         switch (taskRunner.getUpdatedInfo()) {
406           case DONE:
407           case INERROR:
408           case INTERRUPTED:
409           case TOSUBMIT:
410             break;
411           default:
412             changeUpdatedInfo(UpdatedInfo.INERROR,
413                               transfer.getResult().getCode(), false);
414         }
415       }
416     }
417     return transfer;
418   }

420   /**
421    * Initialize the request
422    *
423    * @return the localChannelReference holding the transfer request
424    *
425    * @throws OpenR66ProtocolNoConnectionException
426    * @throws OpenR66RunnerErrorException
427    * @throws OpenR66ProtocolPacketException
428    * @throws OpenR66ProtocolNotYetConnectionException
429    */
430   public LocalChannelReference initRequest()
431       throws OpenR66ProtocolNoConnectionException, OpenR66RunnerErrorException,
432              OpenR66ProtocolPacketException,
433              OpenR66ProtocolNotYetConnectionException {
434     changeUpdatedInfo(UpdatedInfo.RUNNING, ErrorCode.Running, true);
435     final long id = taskRunner.getSpecialId();
436     String tid;
437     if (id == DbConstantR66.ILLEGALVALUE) {
438       tid = "Runner_" + taskRunner.getRuleId() + '_' + taskRunner.getMode() +
439             "_NEWTRANSFER";
440     } else {
441       tid = "Runner_" + taskRunner.getRuleId() + '_' + taskRunner.getMode() +
442             '_' + id;
443     }
444     setName(tid);
445     logger.debug("Will run {}", taskRunner);
446     boolean restartPost = false;
447     if (taskRunner.getGloballaststep() == TASKSTEP.POSTTASK.ordinal()) {
448       // Send a validation to requested
449       if (!taskRunner.isSelfRequested()) {
450         // restart
451         restartPost = true;
452       }
453     }
454     if (taskRunner.isSelfRequested()) {
455       // Don't have to restart a task for itself (or should use requester)
456       logger.warn("Requested host cannot initiate itself the request");
457       changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.LoopSelfRequestedHost,
458                         true);
459       throw new OpenR66ProtocolNoConnectionException(
460           "Requested host cannot initiate itself the request");
461     }
462     DbHostAuth host;
463     try {
464       host = new DbHostAuth(taskRunner.getRequested());
465     } catch (final WaarpDatabaseException e) {
466       logger.error(
467           "Requested host cannot be found: " + taskRunner.getRequested());
468       changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.NotKnownHost, true);
469       throw new OpenR66ProtocolNoConnectionException(
470           "Requested host cannot be found " + taskRunner.getRequested());
471     }
472     if (host.isClient()) {
473       logger.warn("Cannot initiate a connection with a client: {}", host);
474       changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
475                         true);
476       throw new OpenR66ProtocolNoConnectionException(
477           "Cannot connect to client " + host);
478     }
479     SocketAddress socketAddress = host.getSocketAddress();
480     final boolean isSSL = host.isSsl();
481 
482     LocalChannelReference localChannelReference = networkTransaction
483         .createConnectionWithRetry(socketAddress, isSSL, futureRequest);
484     taskRunner.setLocalChannelReference(localChannelReference);
485     if (localChannelReference == null) {
486       // propose to redo
487       // See if reprogramming is ok (not too many tries)
488       String retry;
489       //FIXME Do not stop: true forced
490       if (incrementTaskRunnerTry(taskRunner, Configuration.RETRYNB) || true) {
491         logger.debug("Will retry since Cannot connect to {}", host);
492         retry = " but will retry";
493         // now wait
494         try {
495           Thread.sleep(Configuration.configuration.getDelayRetry());
496         } catch (final InterruptedException e) {//NOSONAR
497           SysErrLogger.FAKE_LOGGER.ignoreLog(e);
498           logger.debug(
499               "Will not retry since limit of connection attemtps is reached for {}",
500               host);
501           retry = " and retries limit is reached so stop here";
502           changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
503                             true);
504           taskRunner.setLocalChannelReference(new LocalChannelReference());
505           throw new OpenR66ProtocolNoConnectionException(
506               CANNOT_CONNECT_TO_SERVER + host + retry);
507         }
508         changeUpdatedInfo(UpdatedInfo.TOSUBMIT, ErrorCode.ConnectionImpossible,
509                           true);
510         throw new OpenR66ProtocolNotYetConnectionException(
511             CANNOT_CONNECT_TO_SERVER + host + retry);
512       } else {
513         logger.debug(
514             "Will not retry since limit of connection attemtps is reached for {}",
515             host);
516         retry = " and retries limit is reached so stop here";
517         changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
518                           true);
519         taskRunner.setLocalChannelReference(new LocalChannelReference());
520         // set this server as being in shutdown status
521         NetworkTransaction.proposeShutdownNetworkChannel(socketAddress, isSSL);
522         throw new OpenR66ProtocolNoConnectionException(
523             CANNOT_CONNECT_TO_SERVER + host + retry);
524       }
525     }
526     if (handler != null) {
527       localChannelReference.setRecvThroughHandler(handler);
528     }
529     localChannelReference.setSendThroughMode(isSendThroughMode);
530     if (restartPost) {
531       RequestPacket request = taskRunner.getRequest();
532       logger.debug("Will send request {} ", request);
533       localChannelReference.setClientRunner(this);
534       localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
535       try {
536         ChannelUtils
537             .writeAbstractLocalPacket(localChannelReference, request, true);
538       } catch (final OpenR66ProtocolPacketException e) {
539         // propose to redo
540         logger.warn("Cannot transfer request to " + host);
541         changeUpdatedInfo(UpdatedInfo.INTERRUPTED, ErrorCode.Internal, true);
542         localChannelReference.close();
543         throw e;
544       }
545       logger.debug("Wait for request to {}", host);
546       return localChannelReference;
547     }
548     // If Requester is NOT Sender, and if TransferTask then decrease now if
549     // possible the rank
550     if (!taskRunner.isSender() &&
551         taskRunner.getGloballaststep() == TASKSTEP.TRANSFERTASK.ordinal()) {
552       logger
553           .debug("Requester is not Sender so decrease if possible the rank {}",
554                  taskRunner);
555       taskRunner.restartRank();
556       taskRunner.forceSaveStatus();
557       logger.debug(
558           "Requester is not Sender so new rank is " + taskRunner.getRank() +
559           " {}", taskRunner);
560     }
561     RequestPacket request = taskRunner.getRequest();
562     request
563         .setLimit(localChannelReference.getChannelLimit(taskRunner.isSender()));
564     localChannelReference.setClientRunner(this);
565     logger.debug("Will send request {} {}", request, localChannelReference);
566     localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
567     try {
568       ChannelUtils
569           .writeAbstractLocalPacket(localChannelReference, request, true);
570     } catch (final OpenR66ProtocolPacketException e) {
571       // propose to redo
572       logger.warn("Cannot transfer request to " + host);
573       changeUpdatedInfo(UpdatedInfo.INTERRUPTED, ErrorCode.Internal, true);
574       localChannelReference.close();
575       throw e;
576     }
577     logger.debug("Wait for request to {} {} {}", host, localChannelReference,
578                  request);
579     return localChannelReference;
580   }

582   /**
583    * Change the UpdatedInfo of the current runner
584    *
585    * @param info
586    */
587   public void changeUpdatedInfo(AbstractDbData.UpdatedInfo info, ErrorCode code,
588                                 boolean force) {
589     taskRunner.changeUpdatedInfo(info);
590     taskRunner.setErrorExecutionStatus(code);
591     if (force) {
592       taskRunner.forceSaveStatus();
593     } else {
594       try {
595         taskRunner.saveStatus();
596       } catch (final OpenR66RunnerErrorException ignored) {
597         // nothing
598       }
599     }
600   }
