153   @Override
154   public void run() {
155     if (Configuration.configuration.isShutdown() || Thread.interrupted()) {
156       taskRunner.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
157       taskRunner.forceSaveStatus();
158       return;
159     }
160     boolean status = false;
161     try {
162       if (activeRunners != null) {
163         activeRunners.add(this);
164       }
165       // fix for SelfRequest
166       if (taskRunner.isSelfRequest()) {
167         taskRunner.setSenderByRequestToValidate(false);
168       }
169       // Try to check if file still exists in send not self not through mode
170       if (taskRunner.isSender() && !taskRunner.isSelfRequest() &&
171           !taskRunner.isSendThrough()) {
172         try {
173           final R66Sessionion.html#R66Session">R66Session session = new R66Session(false);
174           session.setReady(true);
175           final boolean ssl = Configuration.configuration.isUseSSL();
176           session.getAuth().specialNoSessionAuth(ssl,
177                                                  Configuration.configuration.getHostId(
178                                                      ssl));
179           final DbTaskRunner reloaded =
180               new DbTaskRunner(session, taskRunner.getRule(),
181                                taskRunner.getSpecialId(),
182                                taskRunner.getRequester(),
183                                taskRunner.getRequested());
184           reloaded.setSender(taskRunner.isSender());
185           session.setRunner(reloaded);
186           session.setBlockSize(reloaded.getBlocksize());
187           final File file = new File(reloaded.getFullFilePath());
188           if (!file.isFile()) {
189             logger.warn("File not found: {}", file.getAbsolutePath());
190             // File does no more exist => error
191             reloaded.changeUpdatedInfo(UpdatedInfo.INERROR);
192             reloaded.setErrorExecutionStatus(ErrorCode.FileNotFound);
193             logger.error("Runner Error: {} {}",
194                          ErrorCode.FileNotFound.getMesg(),
195                          taskRunner.toShortString());
196             reloaded.setErrorTask();
197             reloaded.update();
198             return;
199           }
200           status = true;
201         } catch (final CommandAbstractException e) {
202           if (Configuration.configuration.isShutdown()) {
203             // ignore since shutdown
204             logger.warn(e.getMessage());
205           } else {
206             // Wrong path? Ignore
207             logger.warn(e);
208           }
209         } catch (final OpenR66RunnerErrorException e) {
210           if (Configuration.configuration.isShutdown()) {
211             // ignore since shutdown
212             logger.warn(e.getMessage());
213           } else {
214             // Wrong run error? Ignore
215             logger.warn(e);
216           }
217         } catch (final WaarpDatabaseException e) {
218           if (Configuration.configuration.isShutdown()) {
219             // ignore since shutdown
220             logger.warn(e.getMessage());
221           } else {
222             // Wrong dbtask? Ignore
223             logger.warn(e);
224           }
225         } catch (final OpenR66ProtocolNoSslException e) {
226           if (Configuration.configuration.isShutdown()) {
227             // ignore since shutdown
228             logger.warn(e.getMessage());
229           } else {
230             // Wrong ssl? Ignore
231             logger.warn(e);
232           }
233         }
234       } else {
235         status = true;
236       }
237       if (Configuration.configuration.isShutdown() || Thread.interrupted() ||
238           !status) {
239         taskRunner.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
240         taskRunner.forceSaveStatus();
241         return;
242       }
243       final R66Future transfer;
244       try {
245         transfer = runTransfer();
246       } catch (final OpenR66RunnerErrorException e) {
247         logger.error("Runner Error: {} {}", e.getMessage(),
248                      taskRunner.toShortString());
249         return;
250       } catch (final OpenR66ProtocolNoConnectionException e) {
251         logger.error("No connection Error {}", e.getMessage());
252         if (localChannelReference != null) {
253           localChannelReference.setErrorMessage(
254               ErrorCode.ConnectionImpossible.getMesg(),
255               ErrorCode.ConnectionImpossible);
256         }
257         taskRunner.setErrorTask();
258         try {
259           taskRunner.forceSaveStatus();
260           taskRunner.run();
261         } catch (final OpenR66RunnerErrorException e1) {
262           changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
263                             true);
264         }
265         return;
266       } catch (final OpenR66ProtocolPacketException e) {
267         logger.error("Protocol Error", e);
268         return;
269       } catch (final OpenR66ProtocolNotYetConnectionException e) {
270         logger.warn("No connection warning {}", e.getMessage());
271         return;
272       }
273       final R66Result result = transfer.getResult();
274       if (result != null) {
275         if (result.getCode() == ErrorCode.QueryAlreadyFinished) {
276           logger.warn(Messages.getString(TRANSFER_STATUS) +
277                       (transfer.isSuccess()?
278                           Messages.getString(REQUEST_INFORMATION_SUCCESS) :
279                           Messages.getString(REQUEST_INFORMATION_FAILURE)) +
280                       "     " + ErrorCode.QueryAlreadyFinished.getMesg() + ':' +
281                       result.toString());
282         } else {
283           if (transfer.isSuccess()) {
284             logger.info("{}{}     {}", Messages.getString(TRANSFER_STATUS),
285                         Messages.getString(REQUEST_INFORMATION_SUCCESS),
286                         result);
287           } else {
288             logger.error(Messages.getString(TRANSFER_STATUS) +
289                          Messages.getString(REQUEST_INFORMATION_FAILURE) +
290                          "     " + result.toString());
291           }
292         }
293       } else {
294         if (transfer.isSuccess()) {
295           logger.warn(Messages.getString(TRANSFER_STATUS) +
296                       Messages.getString(REQUEST_INFORMATION_SUCCESS) +
297                       "     no result");
298         } else {
299           logger.error(Messages.getString(TRANSFER_STATUS) +
300                        Messages.getString(REQUEST_INFORMATION_FAILURE) +
301                        "     no result");
302         }
303       }
304     } finally {
305       if (activeRunners != null) {
306         activeRunners.remove(this);
307       }
308       setName("Finished_" + nameTask);
309     }
310   }

312   /**
313    * @param runner
314    * @param limit
315    *
316    * @return True if the task was run less than limit, else False
317    */
318   public final boolean incrementTaskRunnerTry(final DbTaskRunner runner,
319                                               final int limit) {
320     if (!isLimitRetryConnection()) {
321       return true;
322     }
323     final String key = runner.getKey();
324     Integer tries = taskRunnerRetryHashMap.get(key);
325     logger.debug("try to find integer: {}", tries);
326     if (tries == null) {
327       tries = 1;
328     } else {
329       tries += 1;
330     }
331     logger.debug("Check: {} vs {}: {}", tries, limit, limit <= tries);
332     if (limit <= tries || Thread.interrupted()) {
333       taskRunnerRetryHashMap.remove(key);
334       return false;
335     } else {
336       taskRunnerRetryHashMap.put(key, tries);
337       return true;
338     }
339   }

341   /**
342    * True transfer run (can be called directly to enable exception outside any
343    * executors)
344    *
345    * @return The R66Future of the transfer operation
346    *
347    * @throws OpenR66RunnerErrorException
348    * @throws OpenR66ProtocolNoConnectionException
349    * @throws OpenR66ProtocolPacketException
350    * @throws OpenR66ProtocolNotYetConnectionException
351    */
352   public final R66Future runTransfer()
353       throws OpenR66RunnerErrorException, OpenR66ProtocolNoConnectionException,
354              OpenR66ProtocolPacketException,
355              OpenR66ProtocolNotYetConnectionException {
356     logger.debug("Start attempt Transfer");
357     localChannelReference = initRequest();
358     localChannelReference.getFutureValidRequest().awaitOrInterruptible(
359         Configuration.configuration.getTimeoutCon());
360     if (localChannelReference.getFutureValidRequest().isSuccess()) {
361       return finishTransfer(localChannelReference);
362     } else if (
363         localChannelReference.getFutureValidRequest().getResult() != null &&
364         localChannelReference.getFutureValidRequest().getResult().getCode() ==
365         ErrorCode.ServerOverloaded) {
366       return tryAgainTransferOnOverloaded(true, localChannelReference);
367     } else {
368       return finishTransfer(localChannelReference);
369     }
370   }

372   /**
373    * In case an overloaded signal is returned by the requested
374    *
375    * @param retry if True, it will retry in case of overloaded remote
376    *     server, else it just stops
377    * @param localChannelReference
378    *
379    * @return The R66Future of the transfer operation
380    *
381    * @throws OpenR66RunnerErrorException
382    * @throws OpenR66ProtocolNoConnectionException
383    * @throws OpenR66ProtocolPacketException
384    * @throws OpenR66ProtocolNotYetConnectionException
385    */
386   public final R66Future tryAgainTransferOnOverloaded(final boolean retry,
387                                                       final LocalChannelReference localChannelReference)
388       throws OpenR66RunnerErrorException, OpenR66ProtocolNoConnectionException,
389              OpenR66ProtocolPacketException,
390              OpenR66ProtocolNotYetConnectionException {
391     if (this.localChannelReference == null) {
392       this.localChannelReference = localChannelReference;
393     }
394     final boolean incRetry =
395         incrementTaskRunnerTry(taskRunner, Configuration.RETRYNB);
396     logger.debug("tryAgainTransferOnOverloaded: {}:{}", retry, incRetry);
397     switch (taskRunner.getUpdatedInfo()) {
398       case DONE:
399       case INERROR:
400       case INTERRUPTED:
401         break;
402       default:
403         changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ServerOverloaded,
404                           true);
405     }
406     // redo if possible
407     if (retry && incRetry) {
408       try {
409         Thread.sleep(Configuration.configuration.getConstraintLimitHandler()
410                                                 .getSleepTime());
411       } catch (final InterruptedException e) {//NOSONAR
412         if (localChannelReference == null) {
413           taskRunner.setLocalChannelReference(new LocalChannelReference());
414         }
415         taskRunner.getLocalChannelReference()
416                   .setErrorMessage(ErrorCode.ConnectionImpossible.getMesg(),
417                                    ErrorCode.ConnectionImpossible);
418         taskRunner.setErrorTask();
419         taskRunner.run();
420         throw new OpenR66ProtocolNoConnectionException(
421             "End of retry on ServerOverloaded due to interruption");
422       }
423       return runTransfer();
424     } else {
425       if (localChannelReference == null) {
426         taskRunner.setLocalChannelReference(new LocalChannelReference());
427       }
428       taskRunner.getLocalChannelReference()
429                 .setErrorMessage(ErrorCode.ConnectionImpossible.getMesg(),
430                                  ErrorCode.ConnectionImpossible);
431       taskRunner.setErrorTask();
432       taskRunner.run();
433       throw new OpenR66ProtocolNoConnectionException(
434           "End of retry on ServerOverloaded");
435     }
436   }

438   /**
439    * Finish the transfer (called at the end of runTransfer)
440    *
441    * @param localChannelReference
442    *
443    * @return The R66Future of the transfer operation
444    */
445   public final R66Future finishTransfer(
446       final LocalChannelReference localChannelReference) {
447     if (this.localChannelReference == null) {
448       this.localChannelReference = localChannelReference;
449     }
450     final R66Future transfer = localChannelReference.getFutureRequest();
451     transfer.awaitOrInterruptible();
452     taskRunnerRetryHashMap.remove(taskRunner.getKey());
453     logger.info("Request done with {}",
454                 transfer.isSuccess()? "success" : "error");
455     localChannelReference.close();
456     // now reload TaskRunner if it still exists (light client can forget it)
457     final boolean isSender = taskRunner.isSender();
458     if (transfer.isSuccess()) {
459       try {
460         taskRunner.select();
461       } catch (final WaarpDatabaseException e) {
462         logger.debug("Not a problem but cannot find at the end the task", e);
463         taskRunner.setFrom(transfer.getRunner());
464       }
465       taskRunner.setSender(isSender);
466       changeUpdatedInfo(UpdatedInfo.DONE, ErrorCode.CompleteOk, false);
467     } else {
468       try {
469         taskRunner.select();
470       } catch (final WaarpDatabaseException e) {
471         logger.debug("Not a problem but cannot find at the end the task");
472         taskRunner.setFrom(transfer.getRunner());
473       }
474       taskRunner.setSender(isSender);
475       // Case when we were interrupted
476       if (transfer.getResult() == null) {
477         switch (taskRunner.getUpdatedInfo()) {
478           case DONE:
479             final R66Result ok =
480                 new R66Result(null, true, ErrorCode.CompleteOk, taskRunner);
481             transfer.setResult(ok);
482             transfer.setSuccess();
483             changeUpdatedInfo(UpdatedInfo.DONE, ErrorCode.CompleteOk, false);
484             break;
485           case INERROR:
486           case INTERRUPTED:
487           default:
488             final R66Result error =
489                 new R66Result(null, true, ErrorCode.Internal, taskRunner);
490             transfer.setResult(error);
491             transfer.cancel();
492             changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.Internal, false);
493         }
494         return transfer;
495       }
496       if (transfer.getResult().getCode() == ErrorCode.QueryAlreadyFinished) {
497         // check if post task to execute
498         logger.warn("WARN QueryAlreadyFinished:     " + transfer + "     " +
499                     taskRunner.toShortString());
500         try {
501           TransferUtils.finalizeTaskWithNoSession(taskRunner,
502                                                   localChannelReference);
503         } catch (final OpenR66RunnerErrorException e) {
504           taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
505           taskRunner.forceSaveStatus();
506         }
507       } else {
508         switch (taskRunner.getUpdatedInfo()) {
509           case DONE:
510           case INERROR:
511           case INTERRUPTED:
512           case TOSUBMIT:
513             break;
514           default:
515             changeUpdatedInfo(UpdatedInfo.INERROR,
516                               transfer.getResult().getCode(), false);
517         }
518       }
519     }
520     return transfer;
521   }

523   /**
524    * Initialize the request
525    *
526    * @return the localChannelReference holding the transfer request
527    *
528    * @throws OpenR66ProtocolNoConnectionException
529    * @throws OpenR66ProtocolPacketException
530    * @throws OpenR66ProtocolNotYetConnectionException
531    */
532   public final LocalChannelReference initRequest()
533       throws OpenR66ProtocolNoConnectionException,
534              OpenR66ProtocolPacketException,
535              OpenR66ProtocolNotYetConnectionException {
536     changeUpdatedInfo(UpdatedInfo.RUNNING, ErrorCode.Running, true);
537     final long id = taskRunner.getSpecialId();
538     final String tid;
539     if (id == DbConstantR66.ILLEGALVALUE) {
540       tid = "Runner_" + taskRunner.getRuleId() + '_' + taskRunner.getMode() +
541             "_NEWTRANSFER";
542     } else {
543       tid = "Runner_" + taskRunner.getRuleId() + '_' + taskRunner.getMode() +
544             '_' + id;
545     }
546     setName(tid);
547     logger.debug("Will run {}", taskRunner);
548     boolean restartPost = false;
549     if (taskRunner.getGloballaststep() == TASKSTEP.POSTTASK.ordinal()) {
550       // Send a validation to requested
551       if (!taskRunner.isRequestOnRequested()) {
552         // restart
553         restartPost = true;
554       }
555     }
556     if (taskRunner.isRequestOnRequested()) {
557       // Don't have to restart a task for itself (or should use requester)
558       logger.warn("Requested host cannot initiate itself the request");
559       changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.LoopSelfRequestedHost,
560                         true);
561       throw new OpenR66ProtocolNoConnectionException(
562           "Requested host cannot initiate itself the request");
563     }
564     final DbHostAuth host;
565     try {
566       host = new DbHostAuth(taskRunner.getRequested());
567     } catch (final WaarpDatabaseException e) {
568       logger.error(
569           "Requested host cannot be found: " + taskRunner.getRequested());
570       changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.NotKnownHost, true);
571       throw new OpenR66ProtocolNoConnectionException(
572           "Requested host cannot be found " + taskRunner.getRequested());
573     }
574     if (host.isClient()) {
575       logger.warn("Cannot initiate a connection with a client: {}", host);
576       changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
577                         true);
578       throw new OpenR66ProtocolNoConnectionException(
579           "Cannot connect to client " + host);
580     }
581     final SocketAddress socketAddress = host.getSocketAddress();
582     final boolean isSSL = host.isSsl();
583 
584     final LocalChannelReference localChannelReferenceTemp;
585     try {
586       localChannelReferenceTemp =
587           networkTransaction.createConnectionWithRetryWithAuthenticationException(
588               socketAddress, isSSL, futureRequest);
589     } catch (final OpenR66ProtocolNotAuthenticatedException e1) {
590       changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.BadAuthent, true);
591       taskRunner.setLocalChannelReference(new LocalChannelReference());
592       throw new OpenR66ProtocolNoConnectionException(
593           CANNOT_CONNECT_TO_SERVER + host +
594           " cannot be authenticated so stop retry here", e1);
595     }
596     taskRunner.setLocalChannelReference(localChannelReferenceTemp);
597     if (localChannelReferenceTemp == null) {
598       // propose to redo
599       String retry;
600       if (incrementTaskRunnerTry(taskRunner, Configuration.RETRYNB)) {
601 
602         logger.debug("Will retry since Cannot connect to {}", host);
603         retry = " but will retry";
604         // now wait
605         try {
606           Thread.sleep(Configuration.configuration.getDelayRetry());
607         } catch (final InterruptedException e) {//NOSONAR
608           SysErrLogger.FAKE_LOGGER.ignoreLog(e);
609           logger.info(
610               "Will not retry since an interruption occurs while connection to {}",
611               host);
612           retry = " and retries gets an interruption so stop here";
613           changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
614                             true);
615           taskRunner.setLocalChannelReference(new LocalChannelReference());
616           throw new OpenR66ProtocolNoConnectionException(
617               CANNOT_CONNECT_TO_SERVER + host + retry);
618         }
619         changeUpdatedInfo(UpdatedInfo.TOSUBMIT, ErrorCode.ConnectionImpossible,
620                           true);
621         throw new OpenR66ProtocolNotYetConnectionException(
622             CANNOT_CONNECT_TO_SERVER + host + retry);
623       } else {
624         logger.info(
625             "Will not retry since limit of connection attemtps is reached for {}",
626             host);
627         retry = " and retries reach step limit so stop here";
628         changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.ConnectionImpossible,
629                           true);
630         taskRunner.setLocalChannelReference(new LocalChannelReference());
631         throw new OpenR66ProtocolNoConnectionException(
632             CANNOT_CONNECT_TO_SERVER + host + retry);
633       }
634     }
635     if (handler != null) {
636       localChannelReferenceTemp.setRecvThroughHandler(handler);
637     }
638     localChannelReferenceTemp.setSendThroughMode(isSendThroughMode);
639     if (restartPost) {
640       final RequestPacket request = taskRunner.getRequest();
641       logger.debug("Will send request {} ", request);
642       localChannelReferenceTemp.setClientRunner(this);
643       localChannelReferenceTemp.sessionNewState(R66FiniteDualStates.REQUESTR);
644       try {
645         ChannelUtils.writeAbstractLocalPacket(localChannelReferenceTemp,
646                                               request, false);
647       } catch (final OpenR66ProtocolPacketException e) {
648         // propose to redo
649         logger.warn("Cannot transfer request to " + host);
650         changeUpdatedInfo(UpdatedInfo.INTERRUPTED, ErrorCode.Internal, true);
651         localChannelReferenceTemp.close();
652         throw e;
653       }
654       logger.debug("Wait for request to {}", host);
655       return localChannelReferenceTemp;
656     }
657     // If Requester is NOT Sender, and if TransferTask then decrease now if
658     // possible the rank
659     if (!taskRunner.isSender() &&
660         taskRunner.getGloballaststep() == TASKSTEP.TRANSFERTASK.ordinal()) {
661       logger.debug(
662           "Requester is not Sender so decrease if possible the rank {}",
663           taskRunner);
664       taskRunner.restartRank();
665       taskRunner.forceSaveStatus();
666       logger.info("Requester is not Sender so new rank is {} {}",
667                   taskRunner.getRank(), taskRunner);
668     }
669     final RequestPacket request = taskRunner.getRequest();
670     request.setLimit(
671         localChannelReferenceTemp.getChannelLimit(taskRunner.isSender()));
672     localChannelReferenceTemp.setClientRunner(this);
673     logger.debug("Will send request {} {}", request, localChannelReferenceTemp);
674     localChannelReferenceTemp.sessionNewState(R66FiniteDualStates.REQUESTR);
675     try {
676       ChannelUtils.writeAbstractLocalPacket(localChannelReferenceTemp, request,
677                                             false);
678     } catch (final OpenR66ProtocolPacketException e) {
679       // propose to redo
680       logger.warn("Cannot transfer request to " + host);
681       changeUpdatedInfo(UpdatedInfo.INTERRUPTED, ErrorCode.Internal, true);
682       localChannelReferenceTemp.close();
683       throw e;
684     }
685     logger.debug("Wait for request to {} {} {}", host,
686                  localChannelReferenceTemp, request);
687     return localChannelReferenceTemp;
688   }

694    */

696                                       final ErrorCode code,
697                                       final boolean force) {
698     taskRunner.changeUpdatedInfo(info);
699     taskRunner.setErrorExecutionStatus(code);
700     if (force) {

702     } else {

704         taskRunner.saveStatus();

706         // nothing
707       }
708     }
