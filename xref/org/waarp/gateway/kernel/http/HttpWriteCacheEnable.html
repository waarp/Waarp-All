58  /**
59   * Utility class to write external file with cache enable properties
60   */
61  public final class HttpWriteCacheEnable {
62    /**
63     * US locale - all HTTP dates are in english
64     */
65    public static final Locale LOCALE_US = Locale.US;
66  
67    /**
68     * GMT timezone - all HTTP dates are on GMT
69     */
70    public static final TimeZone GMT_ZONE = TimeZone.getTimeZone("GMT");
71  
72    /**
73     * format for RFC 1123 date string -- "Sun, 06 Nov 1994 08:49:37 GMT"
74     */
75    public static final String RFC1123_PATTERN = "EEE, dd MMM yyyyy HH:mm:ss z";
76  
77    private static final ArrayList<String> cache_control;
78  
79    private static final int MAX_AGE_SECOND = 604800;
80  
81    static {
82      cache_control = new ArrayList<String>(2);
83      cache_control.add(HttpHeaderValues.PUBLIC.toString());
84      cache_control.add(HttpHeaderValues.MAX_AGE + "=" + MAX_AGE_SECOND);// 1 week
85      cache_control.add(HttpHeaderValues.MUST_REVALIDATE.toString());
86    }
87  
88    /**
89     * set MIME TYPE if possible
90     */
91    public static final MimetypesFileTypeMap mimetypesFileTypeMap =
92        new MimetypesFileTypeMap();
93  
94    static {
95      mimetypesFileTypeMap.addMimeTypes("text/css css CSS");
96      mimetypesFileTypeMap.addMimeTypes("text/javascript js JS");
97      // Official but not supported mimetypesFileTypeMap.addMimeTypes("application/javascript js JS")
98      mimetypesFileTypeMap.addMimeTypes("application/json json JSON map MAP");
99      mimetypesFileTypeMap.addMimeTypes("text/plain txt text TXT");
100     mimetypesFileTypeMap.addMimeTypes("text/html htm html HTM HTML htmls htx");
101     mimetypesFileTypeMap.addMimeTypes("image/jpeg jpe jpeg jpg JPG");
102     mimetypesFileTypeMap.addMimeTypes("image/png png PNG");
103     mimetypesFileTypeMap.addMimeTypes("image/gif gif GIF");
104     mimetypesFileTypeMap.addMimeTypes("image/x-icon ico ICO");
105   }

110   /**
111    * Write a file, taking into account cache enabled and removing session
112    * cookie
113    *
114    * @param request
115    * @param ctx
116    * @param filename
117    * @param cookieNameToRemove
118    */
119   public static void writeFile(final HttpRequest request,
120                                final ChannelHandlerContext ctx,
121                                final String filename,
122                                final String cookieNameToRemove) {
123     // Convert the response content to a ByteBuf.
124     HttpResponse response;
125     final boolean keepAlive = HttpUtil.isKeepAlive(request);
126     final File file = new File(filename);
127     if (!file.isFile() || !file.canRead()) {
128       SysErrLogger.FAKE_LOGGER.syserr("Cannot read " + file.getAbsolutePath());
129       sendError(request, ctx, cookieNameToRemove, keepAlive);
130       return;
131     }
132     final DateFormat rfc1123Format =
133         new SimpleDateFormat(RFC1123_PATTERN, LOCALE_US);
134     rfc1123Format.setTimeZone(GMT_ZONE);
135     final Date lastModifDate = new Date(file.lastModified());
136     if (request.headers().contains(HttpHeaderNames.IF_MODIFIED_SINCE)) {
137       final String sdate =
138           request.headers().get(HttpHeaderNames.IF_MODIFIED_SINCE);
139       try {
140         final Date ifmodif = rfc1123Format.parse(sdate);
141         if (ifmodif.after(lastModifDate)) {
142           response = new DefaultHttpResponse(HTTP_1_1,
143                                              HttpResponseStatus.NOT_MODIFIED);
144           handleCookies(request, response, cookieNameToRemove);
145           ctx.writeAndFlush(response);
146           return;
147         }
148       } catch (final ParseException ignored) {
149         // nothing
150       }
151     }
152     final int fileLength = (int) file.length();
153     final ByteBuf byteBuf =
154         ByteBufAllocator.DEFAULT.buffer(fileLength, fileLength);
155     FileInputStream inputStream = null;
156     try {
157       inputStream = new FileInputStream(file);
158       byteBuf.writeBytes(inputStream, fileLength);
159     } catch (final FileNotFoundException e) {
160       byteBuf.release();
161       sendError(request, ctx, cookieNameToRemove, keepAlive);
162       return;
163     } catch (final IOException e) {
164       byteBuf.release();
165       sendError(request, ctx, cookieNameToRemove, keepAlive);
166       return;
167     }
168 
169     response =
170         new DefaultFullHttpResponse(HTTP_1_1, HttpResponseStatus.OK, byteBuf);
171     response.headers().set(HttpHeaderNames.CONTENT_LENGTH, fileLength);
172     setContentTypeHeader(response, file);
173     setDateAndCacheHeaders(response, rfc1123Format, lastModifDate);
174 
175     if (!keepAlive) {
176       response.headers()
177               .set(HttpHeaderNames.CONNECTION, HttpHeaderValues.CLOSE);
178     } else if (request.protocolVersion().equals(HTTP_1_0)) {
179       response.headers()
180               .set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
181     }
182 
183     // Write the initial line and the header.
184     final ChannelFuture sendFileFuture = ctx.writeAndFlush(response);
185 
186     // Decide whether to close the connection or not.
187     if (!keepAlive) {
188       // Close the connection when the whole content is written out.
189       sendFileFuture.addListener(ChannelFutureListener.CLOSE);
190     }
191   }

193   /**
194    * Sets the content type header for the HTTP Response
195    *
196    * @param response HTTP response
197    * @param file file to extract content type
198    */
199   private static void setContentTypeHeader(final HttpResponse response,
200                                            final File file) {
201     response.headers().set(HttpHeaderNames.CONTENT_TYPE,
202                            mimetypesFileTypeMap.getContentType(file.getPath()));
203   }
204 
205   /**
206    * Sets the Date and Cache headers for the HTTP Response
207    *
208    * @param response HTTP response
209    * @param rfc1123Format
210    * @param lastModifDate
211    */
212   private static void setDateAndCacheHeaders(final HttpResponse response,
213                                              final DateFormat rfc1123Format,
214                                              final Date lastModifDate) {
215     // Date header
216     final Calendar time = new GregorianCalendar();
217     response.headers()
218             .set(HttpHeaderNames.DATE, rfc1123Format.format(time.getTime()));
219 
220     // Add cache headers
221     time.add(Calendar.SECOND, MAX_AGE_SECOND);
222     response.headers()
223             .set(HttpHeaderNames.EXPIRES, rfc1123Format.format(time.getTime()));
224     response.headers().set(HttpHeaderNames.CACHE_CONTROL, cache_control);
225     response.headers().set(HttpHeaderNames.LAST_MODIFIED,
226                            rfc1123Format.format(lastModifDate));
227   }
228 
229   private static void sendError(final HttpRequest request,
230                                 final ChannelHandlerContext ctx,
231                                 final String cookieNameToRemove,
232                                 final boolean keepAlive) {
233     final HttpResponse response;
234     response =
235         new DefaultFullHttpResponse(HTTP_1_1, HttpResponseStatus.NOT_FOUND);
236     response.headers().add(HttpHeaderNames.CONTENT_LENGTH, 0);
237     if (!keepAlive) {
238       response.headers()
239               .set(HttpHeaderNames.CONNECTION, HttpHeaderValues.CLOSE);
240     } else if (request.protocolVersion().equals(HTTP_1_0)) {
241       response.headers()
242               .set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
243     }
244     handleCookies(request, response, cookieNameToRemove);
245     final ChannelFuture sendFileFuture = ctx.writeAndFlush(response);
246     // Decide whether to close the connection or not.
247     if (!keepAlive) {
248       // Close the connection when the whole content is written out.
249       sendFileFuture.addListener(ChannelFutureListener.CLOSE);
250     }
251   }

253   /**
254    * Remove the given named cookie
255    *
256    * @param request
257    * @param response
258    * @param cookieNameToRemove
259    */
260   public static void handleCookies(final HttpRequest request,
261                                    final HttpResponse response,
262                                    final String cookieNameToRemove) {
263     final String cookieString = request.headers().get(HttpHeaderNames.COOKIE);
264     if (cookieString != null) {
265       final Set<Cookie> cookies = ServerCookieDecoder.LAX.decode(cookieString);
266       if (!cookies.isEmpty()) {
267         // Reset the sessions if necessary.
268         // Remove all Session for images
269         for (final Cookie cookie : cookies) {
270           if (cookie.name().equalsIgnoreCase(cookieNameToRemove)) {
271             // nothing
272           } else {
273             response.headers().add(HttpHeaderNames.SET_COOKIE,
274                                    ServerCookieEncoder.LAX.encode(cookie));
275           }
276         }
277       }
278     }
