116   protected static AbstractHttpField loadHttpPage(ConfigHttpField fieldValue)
117       throws InvalidArgumentException {
118     return new DefaultHttpField(fieldValue.FIELDNAME, fieldValue.FIELDTYPE,
119                                 fieldValue.FIELDINFO, fieldValue.FIELDVALUE,
120                                 fieldValue.FIELDVISIBILITY,
121                                 fieldValue.FIELDMANDATORY,
122                                 fieldValue.FIELDCOOKIESET,
123                                 fieldValue.FIELDTOVALIDATE,
124                                 fieldValue.FIELDPOSITION, fieldValue.FIELDRANK);
125   }
126 
127   protected static HttpPage loadHttpConfiguration(ConfigHttpPage cpage)
128       throws InvalidArgumentException, ClassNotFoundException,
129              InstantiationException, IllegalAccessException {
130     List<ConfigHttpField> list = cpage.FIELD;
131     final LinkedHashMap<String, AbstractHttpField> linkedHashMap =
132         new LinkedHashMap<String, AbstractHttpField>(list.size());
133     // Now read the configuration
134     for (final ConfigHttpField fieldValue : list) {
135       final AbstractHttpField field = loadHttpPage(fieldValue);
136       linkedHashMap.put(field.getFieldname(), field);
137     }
138     list.clear();
139     list = null;
140     return new HttpPage(cpage.PAGENAME, cpage.FILEFORM, cpage.HEADER,
141                         cpage.FOOTER, cpage.BEGINFORM, cpage.ENDFORM,
142                         cpage.NEXTINFORM, cpage.URI, cpage.PAGEROLE,
143                         cpage.ERRORPAGE, cpage.CLASSNAME, linkedHashMap);
144   }

146   /**
147    * Initiate the configuration from the json file for Http server
148    *
149    * @param filename
150    *
151    * @return the List<HttpPage> if OK
152    *
153    * @throws InvalidArgumentException
154    * @throws ClassNotFoundException
155    * @throws IllegalAccessException
156    * @throws InstantiationException
157    */
158   public static HttpPageHandler setConfigurationHttpServerFromJson(
159       String filename) throws InvalidArgumentException, ClassNotFoundException,
160                               InstantiationException, IllegalAccessException {
161     final File file = new File(filename);
162     ConfigHttpPages cpages;
163     try {
164       cpages = JsonHandler.mapper.readValue(file, ConfigHttpPages.class);
165     } catch (final JsonParseException e) {
166       logger.error(UNABLE_TO_READ_THE_JSON_CONFIG_FILE + filename, e);
167       throw new InvalidArgumentException(UNABLE_TO_READ_JSON_FILE + filename,
168                                          e);
169     } catch (final JsonMappingException e) {
170       logger.error(UNABLE_TO_READ_THE_JSON_CONFIG_FILE + filename, e);
171       throw new InvalidArgumentException(UNABLE_TO_READ_JSON_FILE + filename,
172                                          e);
173     } catch (final IOException e) {
174       logger.error(UNABLE_TO_READ_THE_JSON_CONFIG_FILE + filename, e);
175       throw new InvalidArgumentException(UNABLE_TO_READ_JSON_FILE + filename,
176                                          e);
177     }
178     final HashMap<String, HttpPage> pages =
179         new HashMap<String, HttpPage>(cpages.PAGE.size());
180     // Now read the configuration
181     for (final ConfigHttpPage cpage : cpages.PAGE) {
182       final HttpPage page = loadHttpConfiguration(cpage);
183       pages.put(page.getUri(), page);
184     }
185     cpages.PAGE.clear();
186     return new HttpPageHandler(pages);
187   }

189   protected static void addToField(List<ConfigHttpField> fields,
190                                    AbstractHttpField field) {
191     final ConfigHttpField cfield = new ConfigHttpField();
192     cfield.FIELDNAME = field.getFieldname();
193     cfield.FIELDTYPE = field.getFieldtype();
194     cfield.FIELDINFO = field.getFieldinfo();
195     cfield.FIELDVALUE = field.fieldvalue;
196     cfield.FIELDVISIBILITY = field.isFieldvisibility();
197     cfield.FIELDMANDATORY = field.isFieldmandatory();
198     cfield.FIELDCOOKIESET = field.isFieldcookieset();
199     cfield.FIELDTOVALIDATE = field.isFieldtovalidate();
200     cfield.FIELDPOSITION = field.getFieldposition();
201     cfield.FIELDRANK = field.getFieldrank();
202     fields.add(cfield);
203   }

205   protected static void addToElement(List<ConfigHttpPage> pages,
206                                      HttpPage page) {
207     final ConfigHttpPage cpage = new ConfigHttpPage();
208     cpage.PAGENAME = page.getPagename();
209     cpage.FILEFORM = page.getFileform();
210     cpage.HEADER = page.getHeader();
211     cpage.FOOTER = page.getFooter();
212     cpage.BEGINFORM = page.getBeginform();
213     cpage.ENDFORM = page.getEndform();
214     cpage.NEXTINFORM = page.getNextinform();
215     cpage.URI = page.getUri();
216     cpage.PAGEROLE = page.getPagerole();
217     cpage.ERRORPAGE = page.getErrorpage();
218     cpage.CLASSNAME = page.getClassname();
219     cpage.FIELD = new ArrayList<ConfigHttpField>();
220     for (final AbstractHttpField field : page.getFields().values()) {
221       addToField(cpage.FIELD, field);
222     }
223     pages.add(cpage);
224   }

226   public static void exportConfiguration(HttpPageHandler httpPageHandler,
227                                          String filename)
228       throws HttpIncorrectRequestException {
229     final ConfigHttpPages cpages = new ConfigHttpPages();
230     cpages.PAGE = new ArrayList<ConfigHttpPage>();
231     for (final HttpPage page : httpPageHandler.getHashmap().values()) {
232       addToElement(cpages.PAGE, page);
233     }
234     final File file = new File(filename);
235     try {
236       JsonHandler.mapper.writerWithDefaultPrettyPrinter()
237                         .writeValue(file, cpages);
238     } catch (final JsonGenerationException e) {
239       throw new HttpIncorrectRequestException(CANNOT_WRITE_FILE + filename, e);
240     } catch (final JsonMappingException e) {
241       throw new HttpIncorrectRequestException(CANNOT_WRITE_FILE + filename, e);
242     } catch (final IOException e) {
243       throw new HttpIncorrectRequestException(CANNOT_WRITE_FILE + filename, e);
244     }
245   }
