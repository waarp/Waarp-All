66  public class HttpRestClientSimpleResponseHandler
67      extends SimpleChannelInboundHandler<HttpObject> {
68    /**
69     * Internal Logger
70     */
71    private static final WaarpLogger logger =
72        WaarpLoggerFactory.getLogger(HttpRestClientSimpleResponseHandler.class);
73  
74    public static final AttributeKey<RestFuture> RESTARGUMENT =
75        AttributeKey.valueOf("RestClient.Argument");
76  
77    private ByteBuf cumulativeBody;
78    protected JsonNode jsonObject;
79  
80    protected void actionFromResponse(Channel channel) {
81      final RestArgument/rest/RestArgument.html#RestArgument">RestArgument ra = new RestArgument((ObjectNode) jsonObject);
82      if (jsonObject == null) {
83        logger.warn("Recv: EMPTY");
84      } else {
85        logger.warn(ra.prettyPrint());
86      }
87      final RestFuture restFuture = channel.attr(RESTARGUMENT).get();
88      restFuture.setRestArgument(ra);
89      if (ra.getStatusCode() == HttpResponseStatus.OK.code()) {
90        restFuture.setSuccess();
91      } else {
92        logger.error("Error: " + ra.getStatusMessage());
93        restFuture.cancel();
94        if (channel.isActive()) {
95          WaarpSslUtility.closingSslChannel(channel);
96        }
97      }
98    }
99  
100   @Override
101   protected void channelRead0(ChannelHandlerContext ctx, HttpObject msg)
102       throws Exception {
103     if (msg instanceof HttpResponse) {
104       final HttpResponse response = (HttpResponse) msg;
105       final HttpResponseStatus status = response.status();
106       logger.debug(HttpHeaderNames.REFERER + ": " +
107                    response.headers().get(HttpHeaderNames.REFERER) +
108                    " STATUS: " + status);
109     }
110     if (msg instanceof HttpContent) {
111       final HttpContent chunk = (HttpContent) msg;
112       if (chunk instanceof LastHttpContent) {
113         final ByteBuf content = chunk.content();
114         if (content != null && content.isReadable()) {
115           if (cumulativeBody != null) {
116             cumulativeBody = Unpooled.wrappedBuffer(cumulativeBody, content);
117           } else {
118             cumulativeBody = content;
119           }
120         }
121         // get the Json equivalent of the Body
122         if (cumulativeBody == null) {
123           jsonObject = JsonHandler.createObjectNode();
124         } else {
125           try {
126             final String json = cumulativeBody.toString(WaarpStringUtils.UTF8);
127             jsonObject = JsonHandler.getFromString(json);
128           } catch (final Throwable e2) {
129             logger.warn("Error", e2);
130             throw new HttpIncorrectRequestException(e2);
131           }
132           cumulativeBody = null;
133         }
134         actionFromResponse(ctx.channel());
135       } else {
136         final ByteBuf content = chunk.content();
137         if (content != null && content.isReadable()) {
138           if (cumulativeBody != null) {
139             cumulativeBody = Unpooled.wrappedBuffer(cumulativeBody, content);
140           } else {
141             cumulativeBody = content;
142           }
143         }
144       }
145     }
146   }
147 
148   @Override
149   public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause)
150       throws Exception {
151     final RestFuture restFuture = ctx.channel().attr(RESTARGUMENT).get();
152     if (cause instanceof ClosedChannelException) {
153       restFuture.setFailure(cause);
154       logger.debug("Close before ending");
155       return;
156     } else if (cause instanceof ConnectException) {
157       restFuture.setFailure(cause);
158       if (ctx.channel().isActive()) {
159         logger.debug("Will close");
160         WaarpSslUtility.closingSslChannel(ctx.channel());
161       }
162       return;
163     }
164     restFuture.setFailure(cause);
165     logger.error("Error", cause);
166     WaarpSslUtility.closingSslChannel(ctx.channel());
167   }
