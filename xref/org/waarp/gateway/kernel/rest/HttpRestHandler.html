96    public enum METHOD {
97      /**
98       * REST: Standard GET item
99       * <p>
100      * The GET method means retrieve whatever information (in the form of
101      * an
102      * entity) is identified by the
103      * Request-URI. If the Request-URI refers to a data-producing process,
104      * it is
105      * the produced data which shall be
106      * returned as the entity in the response and not the source text of
107      * the
108      * process, unless that text happens to
109      * be the output of the process.
110      */
111     GET(HttpMethod.GET),
112     /**
113      * REST: Update existing item
114      * <p>
115      * The PUT method requests that the enclosed entity be stored under the
116      * supplied Request-URI.
117      */
118     PUT(HttpMethod.PUT),
119     /**
120      * REST: Create a new item
121      * <p>
122      * The POST method is used to request that the origin server accept the
123      * entity enclosed in the request as a
124      * new subordinate of the resource identified by the Request-URI in the
125      * Request-Line.
126      */
127     POST(HttpMethod.POST),
128     /**
129      * REST: Delete existing item
130      * <p>
131      * The DELETE method requests that the origin server delete the resource
132      * identified by the Request-URI.
133      */
134     DELETE(HttpMethod.DELETE),
135     /**
136      * REST: what options are supported for the URI
137      * <p>
138      * The OPTIONS method represents a request for information about the
139      * communication options available on the
140      * request/response chain identified by the Request-URI. This method
141      * allows
142      * the client to determine the
143      * options and/or requirements associated with a resource, or the
144      * capabilities of a server, without implying a
145      * resource action or initiating a resource retrieval.
146      */
147     OPTIONS(HttpMethod.OPTIONS),
148     /**
149      * REST: as GET but no BODY (existence ? metadata ?)
150      * <p>
151      * The HEAD method is identical to GET except that the server MUST NOT
152      * return a message-body in the response.
153      */
154     HEAD(HttpMethod.HEAD),
155     /**
156      * REST: should not be used, use POST instead
157      * <p>
158      * The PATCH method requests that a set of changes described in the
159      * request
160      * entity be applied to the resource
161      * identified by the Request-URI.
162      */
163     PATCH(HttpMethod.PATCH),
164     /**
165      * REST: unknown usage
166      * <p>
167      * The TRACE method is used to invoke a remote, application-layer
168      * loop-back
169      * of the request message.
170      */
171     TRACE(HttpMethod.TRACE),
172     /**
173      * REST: unknown
174      * <p>
175      * This specification reserves the method name CONNECT for use with a
176      * proxy
177      * that can dynamically switch to
178      * being a tunnel
179      */
180     CONNECT(HttpMethod.CONNECT);
181 
182     public final HttpMethod method;
183 
184     METHOD(HttpMethod method) {
185       this.method = method;
186     }
187   }

289     if (arguments != null) {
290       arguments.clean();
291       arguments = null;

293     if (response != null) {
294       response.clean();
295       response = null;
296     }

299       decoder = null;

347   /**
348    * Method to set Cookies in httpResponse from response ObjectNode
349    *
350    * @param httpResponse
351    */
352   protected void setCookies(FullHttpResponse httpResponse) {
353     if (response == null) {
354       return;
355     }
356     final ObjectNode cookieON = response.getCookieArgs();
357     if (!cookieON.isMissingNode()) {
358       final Iterator<Entry<String, JsonNode>> iter = cookieON.fields();
359       while (iter.hasNext()) {
360         final Entry<String, JsonNode> entry = iter.next();
361         httpResponse.headers().add(HttpHeaderNames.SET_COOKIE,
362                                    ServerCookieEncoder.LAX
363                                        .encode(entry.getKey(),
364                                                entry.getValue().asText()));
365       }
366     }
367   }

369   /**
370    * Could be overwritten if necessary
371    *
372    * @return RestMethodHandler associated with the current context
373    *
374    * @throws HttpIncorrectRequestException
375    * @throws HttpMethodNotAllowedRequestException
376    * @throws HttpForbiddenRequestException
377    */
378   protected RestMethodHandler getHandler()
379       throws HttpMethodNotAllowedRequestException,
380              HttpForbiddenRequestException {
381     final METHOD method = arguments.getMethod();
382     final String uri = arguments.getBaseUri();
383     boolean restFound = false;
384     RestMethodHandler handlerNew = restHashMap.get(uri);
385     if (handlerNew != null) {
386       handlerNew.checkHandlerSessionCorrectness(this, arguments, response);
387       if (handlerNew.isMethodIncluded(method)) {
388         restFound = true;
389       }
390     }
391     if (handlerNew == null && method == METHOD.OPTIONS) {
392       handlerNew = rootHandler;
393       // use Options default handler
394       restFound = true;
395     }
396     if (!restFound) {
397       throw new HttpMethodNotAllowedRequestException(
398           "No Method found for that URI: " + uri);
399     }
400     return handlerNew;
401   }

403   @Override
404   protected void channelRead0(ChannelHandlerContext ctx, HttpObject msg)
405       throws Exception {
406     logger.debug("Msg Received");
407     try {
408       if (msg instanceof HttpRequest) {
409         initialize();
410         request = (HttpRequest) msg;
411         arguments.setRequest(request);
412         final Iterator<Entry<CharSequence, CharSequence>> iterator =
413             request.headers().iteratorCharSequence();
414         arguments.setHeaderArgs(iterator);
415         arguments.setCookieArgs(request.headers().get(HttpHeaderNames.COOKIE));
416         logger.debug("DEBUG: {}", arguments);
417         checkConnection(ctx);
418         handler = getHandler();
419         if (arguments.getMethod() == METHOD.OPTIONS) {
420           response.setFromArgument(arguments);
421           handler.optionsCommand(this, arguments, response);
422           finalizeSend(ctx);
423           return;
424         }
425         if (request instanceof FullHttpRequest) {
426           if (handler.isBodyJsonDecoded()) {
427             final ByteBuf buffer = ((FullHttpRequest) request).content();
428             jsonObject = getBodyJsonArgs(buffer);
429           } else {
430             // decoder for 1 chunk
431             createDecoder();
432             // Not chunk version
433             readAllHttpData();
434           }
435           response.setFromArgument(arguments);
436           handler.endParsingRequest(this, arguments, response, jsonObject);
437           finalizeSend(ctx);
438           return;
439         }
440         // no body yet
441         if (!handler.isBodyJsonDecoded()) {
442           createDecoder();
443         }
444       } else {
445         // New chunk is received
446         if (handler != null) {
447           bodyChunk(ctx, (HttpContent) msg);
448         }
449       }
450     } catch (final HttpIncorrectRequestException e1) {
451       // real error => 400
452       if (handler != null) {
453         status =
454             handler.handleException(this, arguments, response, jsonObject, e1);
455       }
456       if (status == HttpResponseStatus.OK) {
457         status = HttpResponseStatus.BAD_REQUEST;
458       }
459       logger.warn("Error", e1);
460       if (response.getDetail().isEmpty()) {
461         response.setDetail(e1.getMessage());
462       }
463       if (handler != null) {
464         finalizeSend(ctx);
465       } else {
466         forceClosing(ctx);
467       }
468     } catch (final HttpMethodNotAllowedRequestException e1) {
469       if (handler != null) {
470         status =
471             handler.handleException(this, arguments, response, jsonObject, e1);
472       }
473       if (status == HttpResponseStatus.OK) {
474         status = HttpResponseStatus.METHOD_NOT_ALLOWED;
475       }
476       logger.warn("Error", e1);
477       if (response.getDetail().isEmpty()) {
478         response.setDetail(e1.getMessage());
479       }
480       if (handler != null) {
481         finalizeSend(ctx);
482       } else {
483         forceClosing(ctx);
484       }
485     } catch (final HttpForbiddenRequestException e1) {
486       if (handler != null) {
487         status =
488             handler.handleException(this, arguments, response, jsonObject, e1);
489       }
490       if (status == HttpResponseStatus.OK) {
491         status = HttpResponseStatus.FORBIDDEN;
492       }
493       logger.warn("Error", e1);
494       if (response.getDetail().isEmpty()) {
495         response.setDetail(e1.getMessage());
496       }
497       if (handler != null) {
498         finalizeSend(ctx);
499       } else {
500         forceClosing(ctx);
501       }
502     } catch (final HttpInvalidAuthenticationException e1) {
503       if (handler != null) {
504         status =
505             handler.handleException(this, arguments, response, jsonObject, e1);
506       }
507       if (status == HttpResponseStatus.OK) {
508         status = HttpResponseStatus.UNAUTHORIZED;
509       }
510       logger.warn("Error", e1);
511       if (response.getDetail().isEmpty()) {
512         response.setDetail(e1.getMessage());
513       }
514       if (handler != null) {
515         finalizeSend(ctx);
516       } else {
517         forceClosing(ctx);
518       }
519     } catch (final HttpNotFoundRequestException e1) {
520       if (handler != null) {
521         status =
522             handler.handleException(this, arguments, response, jsonObject, e1);
523       }
524       if (status == HttpResponseStatus.OK) {
525         status = HttpResponseStatus.NOT_FOUND;
526       }
527       logger.warn("Error", e1);
528       if (response.getDetail().isEmpty()) {
529         response.setDetail(e1.getMessage());
530       }
531       if (handler != null) {
532         finalizeSend(ctx);
533       } else {
534         forceClosing(ctx);
535       }
536     }
537   }

539   /**
540    * Create the decoder
541    *
542    * @throws HttpIncorrectRequestException
543    */
544   protected void createDecoder() throws HttpIncorrectRequestException {
545     final HttpMethod method = request.method();
546     if (!method.equals(HttpMethod.HEAD)) {
547       // in order decoder allows to parse
548       request.setMethod(HttpMethod.POST);
549     }
550     try {
551       decoder = new HttpPostRequestDecoder(factory, request);
552     } catch (final ErrorDataDecoderException e1) {
553       status = HttpResponseStatus.NOT_ACCEPTABLE;
554       throw new HttpIncorrectRequestException(e1);
555     } catch (final Exception e1) {
556       // GETDOWNLOAD Method: should not try to create a HttpPostRequestDecoder
557       // So OK but stop here
558       status = HttpResponseStatus.NOT_ACCEPTABLE;
559       throw new HttpIncorrectRequestException(e1);
560     }
561   }

563   /**
564    * Read all InterfaceHttpData from finished transfer
565    *
566    * @throws HttpIncorrectRequestException
567    */
568   protected void readAllHttpData() throws HttpIncorrectRequestException {
569     List<InterfaceHttpData> datas;
570     try {
571       datas = decoder.getBodyHttpDatas();
572     } catch (final NotEnoughDataDecoderException e1) {
573       // Should not be!
574       logger.warn("decoder issue", e1);
575       status = HttpResponseStatus.NOT_ACCEPTABLE;
576       throw new HttpIncorrectRequestException(e1);
577     }
578     for (final InterfaceHttpData data : datas) {
579       readHttpData(data);
580     }
581   }

583   /**
584    * Read one Data
585    *
586    * @param data
587    *
588    * @throws HttpIncorrectRequestException
589    */
590   protected void readHttpData(InterfaceHttpData data)
591       throws HttpIncorrectRequestException {
592     if (data.getHttpDataType() == HttpDataType.Attribute) {
593       final ObjectNode body = arguments.getBody();
594       try {
595         body.put(data.getName(), ((Attribute) data).getValue());
596       } catch (final IOException e) {
597         throw new HttpIncorrectRequestException("Bad reading", e);
598       }
599     } else if (data.getHttpDataType() == HttpDataType.FileUpload) {
600       final FileUpload fileUpload = (FileUpload) data;
601       if (fileUpload.isCompleted()) {
602         handler.getFileUpload(this, fileUpload, arguments, response);
603       } else {
604         logger.warn("File still pending but should not");
605         fileUpload.delete();
606         status = HttpResponseStatus.INTERNAL_SERVER_ERROR;
607         throw new HttpIncorrectRequestException(
608             "File still pending but should not");
609       }
610     } else {
611       logger.warn("Unknown element: " + data);
612     }
613   }

615   /**
616    * To allow quick answer even if in very bad shape
617    *
618    * @param ctx
619    */
620   protected void forceClosing(ChannelHandlerContext ctx) {
621     if (status == HttpResponseStatus.OK) {
622       status = HttpResponseStatus.INTERNAL_SERVER_ERROR;
623     }
624     if (ctx.channel().isActive()) {
625       setWillClose(true);
626       final String answer =
627           "<html><body>Error " + status.reasonPhrase() + "</body></html>";
628       final FullHttpResponse httpResponse = getResponse(
629           Unpooled.wrappedBuffer(answer.getBytes(WaarpStringUtils.UTF8)));
630       httpResponse.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/html");
631       httpResponse.headers().set(HttpHeaderNames.REFERER, request.uri());
632       final ChannelFuture future = ctx.writeAndFlush(httpResponse);
633       logger.debug("Will close");
634       future.addListener(WaarpSslUtility.SSLCLOSE);
635     }
636     clean();
637   }

639   /**
640    * @param content
641    *
642    * @return the Http Response according to the status and the content if not
643    *     null (setting the CONTENT_LENGTH)
644    */
645   public FullHttpResponse getResponse(ByteBuf content) {
646     // Decide whether to close the connection or not.
647     if (request == null) {
648       FullHttpResponse httpResponse;
649       if (content == null) {
650         httpResponse =
651             new DefaultFullHttpResponse(HttpVersion.HTTP_1_0, status);
652       } else {
653         httpResponse =
654             new DefaultFullHttpResponse(HttpVersion.HTTP_1_0, status, content);
655         httpResponse.headers().set(HttpHeaderNames.CONTENT_LENGTH,
656                                    content.array().length);
657       }
658       setCookies(httpResponse);
659       setWillClose(true);
660       return httpResponse;
661     }
662     boolean keepAlive = HttpUtil.isKeepAlive(request);
663     setWillClose(isWillClose() || status != HttpResponseStatus.OK ||
664                  HttpHeaderValues.CLOSE.contentEqualsIgnoreCase(
665                      request.headers().get(HttpHeaderNames.CONNECTION)) ||
666                  request.protocolVersion().equals(HttpVersion.HTTP_1_0) &&
667                  !keepAlive);
668     if (isWillClose()) {
669       keepAlive = false;
670     }
671     // Build the response object.
672     FullHttpResponse httpResponse;
673     if (content != null) {
674       httpResponse =
675           new DefaultFullHttpResponse(request.protocolVersion(), status,
676                                       content);
677       httpResponse.headers()
678                   .set(HttpHeaderNames.CONTENT_LENGTH, content.array().length);
679     } else {
680       httpResponse =
681           new DefaultFullHttpResponse(request.protocolVersion(), status);
682     }
683     if (keepAlive) {
684       httpResponse.headers()
685                   .set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
686     }
687     setCookies(httpResponse);
688     return httpResponse;
689   }

691   /**
692    * Method that get a chunk of data
693    *
694    * @param ctx
695    * @param chunk
696    *
697    * @throws HttpIncorrectRequestException
698    * @throws HttpInvalidAuthenticationException
699    * @throws HttpNotFoundRequestException
700    */
701   protected void bodyChunk(ChannelHandlerContext ctx, HttpContent chunk)
702       throws HttpIncorrectRequestException, HttpInvalidAuthenticationException,
703              HttpNotFoundRequestException {
704     // New chunk is received: only for Post!
705     if (handler.isBodyJsonDecoded()) {
706       final ByteBuf buffer = chunk.content();
707       if (cumulativeBody != null) {
708         if (buffer.isReadable()) {
709           cumulativeBody.writeBytes(buffer);
710         }
711       } else {
712         cumulativeBody = Unpooled.buffer().writeBytes(buffer);
713       }
714     } else {
715       try {
716         decoder.offer(chunk);
717       } catch (final ErrorDataDecoderException e1) {
718         status = HttpResponseStatus.NOT_ACCEPTABLE;
719         throw new HttpIncorrectRequestException(e1);
720       }
721       // example of reading chunk by chunk (minimize memory usage due to
722       // Factory)
723       readHttpDataChunkByChunk();
724     }
725     // example of reading only if at the end
726     if (chunk instanceof LastHttpContent) {
727       if (handler.isBodyJsonDecoded()) {
728         jsonObject = getBodyJsonArgs(cumulativeBody);
729         cumulativeBody.release();
730         cumulativeBody = null;
731       }
732       response.setFromArgument(arguments);
733       handler.endParsingRequest(this, arguments, response, jsonObject);
734       finalizeSend(ctx);
735     }
736   }

738   protected void finalizeSend(ChannelHandlerContext ctx) {
739     ChannelFuture future;
740     if (arguments.getMethod() == METHOD.OPTIONS) {
741       future = handler.sendOptionsResponse(this, ctx, response, status);
742     } else {
743       future = handler
744           .sendResponse(this, ctx, arguments, response, jsonObject, status);
745     }
746     if (future != null) {
747       future.addListener(WaarpSslUtility.SSLCLOSE);
748     }
749     clean();
750     logger.debug("Cleaned");
751   }

768   /**
769    * Read request by chunk and getting values from chunk to chunk
770    *
771    * @throws HttpIncorrectRequestException
772    */
773   protected void readHttpDataChunkByChunk()
774       throws HttpIncorrectRequestException {
775     try {
776       while (decoder.hasNext()) {
777         final InterfaceHttpData data = decoder.next();
778         if (data != null) {
779           // new value
780           readHttpData(data);
781         }
782       }
783     } catch (final EndOfDataDecoderException e1) {
784       // end
785     }
786   }

788   @Override
789   public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause)
790       throws Exception {
791     if (ctx.channel().isActive()) {
792       if (cause != null && cause.getMessage() != null) {
793         logger.warn("Exception {}", cause.getMessage(), cause);
794       } else {
795         logger.warn("Exception Received", cause);
796       }
797       if (cause instanceof ClosedChannelException ||
798           cause instanceof IOException) {
799         return;
800       }
801       if (handler != null) {
802         status = handler.handleException(this, arguments, response, jsonObject,
803                                          (Exception) cause);
804       }
805       if (status == HttpResponseStatus.OK) {
806         status = HttpResponseStatus.INTERNAL_SERVER_ERROR;
807       }
808       if (handler != null) {
809         finalizeSend(ctx);
810       } else {
811         forceClosing(ctx);
812       }
813     }
814   }
