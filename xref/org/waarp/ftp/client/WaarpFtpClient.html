41  /**
42   * FTP Client using Apache Commons net FTP client (not working using FTPS or
43   * FTPSE)

45  public class WaarpFtpClient {
46    private static final String LOGIN_IN_ERROR = "Login in error";
47  
48    /**
49     * Internal Logger
50     */
51    private static final WaarpLogger logger =
52        WaarpLoggerFactory.getLogger(WaarpFtpClient.class);

54    final String server;
55    int port = 21;
56    final String user;
57    final String pwd;
58    final String acct;
59    int timeout;
60    boolean isPassive;
61    final int ssl; // -1 native, 1 auth
62    protected final FTPClient ftpClient;
63    protected String result;

65    /**
66     * WARNING: SSL mode (FTPS and FTPSE) are not working due to a bug in Apache
67     * Commons-Net
68     *
69     * @param server
70     * @param port
71     * @param user
72     * @param pwd
73     * @param acct
74     * @param isPassive
75     * @param ssl
76     * @param timeout
77     */
78    public WaarpFtpClient(final String server, final int port, final String user,
79                          final String pwd, final String acct,
80                          final boolean isPassive, final int ssl,
81                          final int controlTimeout, final int timeout) {
82      this.server = server;
83      this.port = port;
84      this.user = user;
85      this.pwd = pwd;
86      this.acct = acct;
87      this.isPassive = isPassive;
88      this.ssl = ssl;
89      if (this.ssl != 0) {
90        // implicit or explicit
91        ftpClient = new FTPSClient(this.ssl == -1);
92        ((FTPSClient) ftpClient)
93            .setTrustManager(TrustManagerUtils.getAcceptAllTrustManager());
94      } else {
95        ftpClient = new FTPClient();
96      }
97      if (controlTimeout > 0) {
98        ftpClient.setControlKeepAliveTimeout(controlTimeout / 1000);
99        ftpClient.setControlKeepAliveReplyTimeout(controlTimeout);
100     }
101     if (timeout > 0) {
102       ftpClient.setDataTimeout(timeout);
103     }
104     ftpClient.addProtocolCommandListener(
105         new PrintCommandListener(new PrintWriter(System.out), true));
106   }

108   /**
109    * Try to connect to the server and goes with the authentication
110    *
111    * @return True if connected and authenticated, else False
112    */
113   public boolean connect() {
114     result = null;
115     boolean isActive = false;
116     try {
117       try {
118         ftpClient.connect(server, port);
119       } catch (final SocketException e) {
120         result = "Connection in error";
121         logger.error(result, e);
122         return false;
123       } catch (final IOException e) {
124         result = "Connection in error";
125         logger.error(result, e);
126         return false;
127       }
128       final int reply = ftpClient.getReplyCode();
129       if (!FTPReply.isPositiveCompletion(reply)) {
130         disconnect();
131         result = "Connection in error: " + reply;
132         logger.error(result);
133         return false;
134       }
135       try {
136         if (acct == null) {
137           // no account
138           if (!ftpClient.login(user, pwd)) {
139             logout();
140             result = LOGIN_IN_ERROR;
141             logger.error(result);
142             return false;
143           }
144         } else if (!ftpClient.login(user, pwd, acct)) {
145           logout();
146           result = LOGIN_IN_ERROR;
147           logger.error(result);
148           return false;
149         }
150       } catch (final IOException e) {
151         result = LOGIN_IN_ERROR;
152         logger.error(result, e);
153         return false;
154       }
155       try {
156         ftpClient.setFileType(FTP.BINARY_FILE_TYPE);
157       } catch (final IOException e1) {
158         result = "Set BINARY in error";
159         logger.error(result, e1);
160         return false;
161       }
162       changeMode(isPassive);
163       ftpClient.setUseEPSVwithIPv4(false);
164       if (ssl == 1) {
165         // now send request for PROT (AUTH already sent)
166         try {
167           ((FTPSClient) ftpClient).execPBSZ(0);
168           logger.debug("PBSZ 0");
169           ((FTPSClient) ftpClient).execPROT("P");
170           logger.debug("Info: {}",
171                        ((FTPSClient) ftpClient).getEnableSessionCreation());
172         } catch (final IOException e) {
173           logout();
174           result = "Explicit SSL in error";
175           logger.error(result, e);
176           return false;
177         }
178       }
179       isActive = true;
180       return true;
181     } finally {
182       if (!isActive && ftpClient.getDataConnectionMode() ==
183                        FTPClient.ACTIVE_LOCAL_DATA_CONNECTION_MODE) {
184         disconnect();
185       }
186     }
187   }

189   /**
190    * Logout from the Control connection
191    */
192   public void logout() {
193     try {
194       ftpClient.logout();
195     } catch (final IOException e) {
196       if (ftpClient.getDataConnectionMode() ==
197           FTPClient.ACTIVE_LOCAL_DATA_CONNECTION_MODE) {
198         try {
199           ftpClient.disconnect();
200         } catch (final IOException f) {
201           // do nothing
202         }
203       }
204     }
205   }

207   /**
208    * Disconnect the Ftp Client
209    */
210   public void disconnect() {
211     try {
212       ftpClient.disconnect();
213     } catch (final IOException e) {
214       logger.debug("Disconnection in error", e);
215     }
216   }

254   /**
255    * Change the FileType of Transfer (Binary true, ASCII false)
256    *
257    * @param binaryTransfer
258    *
259    * @return True if the change is OK
260    */
261   public boolean changeFileType(final boolean binaryTransfer) {
262     result = null;
263     try {
264       if (binaryTransfer) {
265         return ftpClient.setFileType(FTP.BINARY_FILE_TYPE);
266       } else {
267         return ftpClient.setFileType(FTP.ASCII_FILE_TYPE);
268       }
269     } catch (final IOException e) {
270       result = "FileType in error";
271       logger.warn(result, e);
272       return false;
273     }
274   }

276   /**
277    * Change to passive (true) or active (false) mode
278    *
279    * @param passive
280    */
281   public void changeMode(final boolean passive) {
282     isPassive = passive;
283     if (isPassive) {
284       ftpClient.enterLocalPassiveMode();
285     } else {
286       ftpClient.enterLocalActiveMode();
287     }
288   }

290   /**
291    * Ask to transfer a file
292    *
293    * @param local local filepath (limited to path if get, else full
294    *     path)
295    * @param remote filename
296    * @param getStoreOrAppend -1 = get, 1 = store, 2 = append
297    *
298    * @return True if the file is correctly transfered
299    */
300   public boolean transferFile(final String local, final String remote,
301                               final int getStoreOrAppend) {
302     result = null;
303     final boolean status;
304     FileOutputStream output = null;
305     FileInputStream fileInputStream = null;
306     try {
307       if (getStoreOrAppend > 0) {
308         fileInputStream = new FileInputStream(local);
309         if (getStoreOrAppend == 1) {
310           status = ftpClient.storeFile(remote, fileInputStream);
311         } else {
312           // append
313           status = ftpClient.appendFile(remote, fileInputStream);
314         }
315         if (!status) {
316           result = "Cannot finalize store like operation";
317           logger.error(result);
318           return false;
319         }
320       } else {
321         output = new FileOutputStream(new File(local, remote));
322         status = ftpClient.retrieveFile(remote, output);
323         if (!status) {
324           result = "Cannot finalize retrieve like operation";
325           logger.error(result);
326           return false;
327         }
328       }
329       return true;
330     } catch (final IOException e) {
331       result = "Cannot finalize operation";
332       logger.error(result, e);
333       return false;
334     } finally {
335       FileUtils.close(output);
336       FileUtils.close(fileInputStream);
337     }
338   }

340   /**
341    * @return the list of files as returned by the FTP command
342    */
343   public String[] listFiles() {
344     try {
345       final FTPFile[] list = ftpClient.listFiles();
346       final String[] results = new String[list.length];
347       int i = 0;
348       for (final FTPFile file : list) {
349         results[i] = file.toFormattedString();
350         i++;
351       }
352       return results;
353     } catch (final IOException e) {
354       result = "Cannot finalize transfer operation";
355       logger.error(result, e);
356       return null;
357     }
358   }

360   /**
361    * @param feature
362    *
363    * @return True if the feature is listed
364    */
365   public boolean featureEnabled(final String feature) {
366     try {
367       SysErrLogger.FAKE_LOGGER.sysout(ftpClient.features());
368       if (ftpClient.featureValue(feature) == null) {
369         final String resultNew = ftpClient.getReplyString();
370         return resultNew.contains(feature.toUpperCase());
371       }
372       return true;
373     } catch (final IOException e) {
374       result = "Cannot execute operation Feature";
375       logger.error(result, e);
376       return false;
377     }
378   }

380   /**
381    * @param params
382    *
383    * @return the string lines returned by the command params
384    */
385   public String[] executeCommand(final String params) {
386     result = null;
387     try {
388       SysErrLogger.FAKE_LOGGER.sysout(params);
389       final int pos = params.indexOf(' ');
390       String command = params;
391       String args = null;
392       if (pos > 0) {
393         command = params.substring(0, pos);
394         args = params.substring(pos + 1);
395       }
396       String[] results = ftpClient.doCommandAsStrings(command, args);
397       if (results == null) {
398         results = new String[1];
399         results[0] = ftpClient.getReplyString();
400       }
401       return results;
402     } catch (final IOException e) {
403       result = "Cannot execute operation Site";
404       logger.error(result, e);
405       return null;
406     }
407   }
