20  package org.waarp.ftp.core.command.rfc3659;
21  
22  import org.waarp.common.command.exception.CommandAbstractException;
23  import org.waarp.ftp.core.command.AbstractCommand;
24  
25  import java.util.List;
26  
27  /**
28   * MLSD command
29   */
30  public class MLSD extends AbstractCommand {
31    @Override
32    public final void exec() throws CommandAbstractException {
33      final String path;
34      if (!hasArg()) {
35        path = getSession().getDir().getPwd();
36      } else {
37        path = getArg();
38      }
39      final List<String> filesInfo = getSession().getDir().listFull(path, false);
40      getSession().openDataConnection();
41      getSession().getDataConn().getFtpTransferControl()
42                  .setNewFtpTransfer(getCode(), filesInfo, path);
43    }
44  }
