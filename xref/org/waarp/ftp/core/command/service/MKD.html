20  package org.waarp.ftp.core.command.service;
21  
22  import org.waarp.common.command.ReplyCode;
23  import org.waarp.common.command.exception.CommandAbstractException;
24  import org.waarp.common.command.exception.Reply501Exception;
25  import org.waarp.ftp.core.command.AbstractCommand;
26  
27  /**
28   * MKD command
29   */
30  public class MKD extends AbstractCommand {
31    @Override
32    public final void exec() throws CommandAbstractException {
33      // First Check if any argument
34      if (!hasArg()) {
35        throw new Reply501Exception("Need a path as argument");
36      }
37      final String path = getArg();
38      final String newpath = getSession().getDir().mkdir(path);
39      getSession().setReplyCode(ReplyCode.REPLY_257_PATHNAME_CREATED,
40                                '"' + newpath + "\" is created");
41    }
42  
43  }
