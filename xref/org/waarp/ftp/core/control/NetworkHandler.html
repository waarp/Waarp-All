121   /**
122    * Run firstly executeChannelClosed.
123    */
124   @Override
125   public void channelInactive(final ChannelHandlerContext ctx)
126       throws Exception {
127     if (session == null || session.getDataConn() == null ||
128         session.getDataConn().getFtpTransferControl() == null) {
129       super.channelInactive(ctx);
130       return;
131     }
132     // Wait for any command running before closing (bad client sometimes
133     // don't wait for answer)
134     int limit = 100;
135     while (session.getDataConn().getFtpTransferControl()
136                   .isFtpTransferExecuting()) {
137       Thread.sleep(WaarpNettyUtil.MINIMAL_DELAY_MS);
138       limit--;
139       if (limit <= 0) {
140         logger.warn("Waiting for transfer finished but 1s is not enough");
141         break; // wait at most 1s
142       }
143     }
144     businessHandler.executeChannelClosed();
145     // release file and other permanent objects
146     businessHandler.clear();
147     session.clear();
148     super.channelInactive(ctx);
149   }

151   /**
152    * Initialize the Handler.
153    */
154   @Override
155   public void channelActive(final ChannelHandlerContext ctx) throws Exception {
156     this.ctx = ctx;
157     final Channel channel = ctx.channel();
158     controlChannel = channel;
159     session.setControlConnected();
160     FtpChannelUtils.addCommandChannel(channel, session.getConfiguration());
161     if (isStillAlive(ctx)) {
162       // Make the first execution ready
163       final AbstractCommand command = new ConnectionCommand(getFtpSession());
164       session.setNextCommand(command);
165       // This command can change the next Command
166       businessHandler.executeChannelConnected(channel);
167       // Answer ready to continue from first command = Connection
168       messageRunAnswer(ctx);
169       getFtpSession().setReady(true);
170     }
171   }

173   /**
174    * If the service is going to shutdown, it sends back a 421 message to the
175    * connection
176    *
177    * @return True if the service is alive, else False if the system is going
178    *     down
179    */
180   private boolean isStillAlive(final ChannelHandlerContext ctx) {
181     if (session.getConfiguration().isShutdown()) {
182       session.setExitErrorCode("Service is going down: disconnect");
183       writeFinalAnswer(ctx);
184       return false;
185     }
186     return true;
187   }

189   /**
190    * Default exception task: close the current connection after calling
191    * exceptionLocalCaught and writing if
192    * possible the current replyCode.
193    */
194   @Override
195   public void exceptionCaught(final ChannelHandlerContext ctx,
196                               final Throwable cause) {
197     this.ctx = ctx;
198     final Channel channel = ctx.channel();
199     if (session == null) {
200       // should not be
201       logger.warn("NO SESSION", cause);
202       return;
203     }
204     if (cause instanceof ConnectException) {
205       final ConnectException e2 = (ConnectException) cause;
206       logger.warn("Connection impossible since {} with Channel {}",
207                   e2.getMessage(), channel);
208     } else if (cause instanceof ChannelException) {
209       final ChannelException e2 = (ChannelException) cause;
210       logger.warn(
211           "Connection (example: timeout) impossible since {} with Channel {}",
212           e2.getMessage(), channel);
213     } else if (cause instanceof ClosedChannelException) {
214       logger.debug("Connection closed before end");
215       session.setExitErrorCode(INTERNAL_ERROR_DISCONNECT);
216       if (channel.isActive()) {
217         writeFinalAnswer(ctx);
218       }
219       return;
220     } else if (cause instanceof CommandAbstractException) {
221       // FTP Exception: not close if not necessary
222       final CommandAbstractException/org/waarp/common/command/exception/CommandAbstractException.html#CommandAbstractException">CommandAbstractException e2 = (CommandAbstractException) cause;
223       logger.warn("Command Error Reply {}", e2.getMessage());
224       session.setReplyCode(e2);
225       businessHandler.afterRunCommandKo(e2);
226       if (channel.isActive()) {
227         writeFinalAnswer(ctx);
228       }
229       return;
230     } else if (cause instanceof NullPointerException) {
231       final NullPointerException e2 = (NullPointerException) cause;
232       logger.warn("Null pointer Exception: " + ctx.channel(), e2);
233       try {
234         session.setExitErrorCode(INTERNAL_ERROR_DISCONNECT);
235         if (businessHandler != null && session.getDataConn() != null) {
236           businessHandler.exceptionLocalCaught(cause);
237           if (channel.isActive()) {
238             writeFinalAnswer(ctx);
239           }
240         }
241       } catch (final NullPointerException ignored) {
242         // nothing
243       }
244       return;
245     } else if (cause instanceof IOException) {
246       final IOException e2 = (IOException) cause;
247       logger
248           .warn("Connection aborted since {} with Channel {}", e2.getMessage(),
249                 channel);
250       logger.warn(cause);
251     } else if (cause instanceof RejectedExecutionException) {
252       logger.debug("Rejected execution (shutdown) from {}", channel);
253       return;
254     } else {
255       logger.warn("Unexpected exception from Outband Ref Channel: " + channel +
256                   " Exception: " + cause.getMessage(), cause);
257     }
258     session.setExitErrorCode(INTERNAL_ERROR_DISCONNECT);
259     businessHandler.exceptionLocalCaught(cause);
260     if (channel.isActive()) {
261       writeFinalAnswer(ctx);
262     }
263   }

265   /**
266    * Simply call messageRun with the received message
267    */
268   @Override
269   public void channelRead0(final ChannelHandlerContext ctx, final String e) {
270     this.ctx = ctx;
271     if (isStillAlive(ctx)) {
272       // First wait for the initialization to be fully done
273       if (!session.isReady()) {
274         session.setReplyCode(
275             ReplyCode.REPLY_421_SERVICE_NOT_AVAILABLE_CLOSING_CONTROL_CONNECTION,
276             null);
277         businessHandler.afterRunCommandKo(
278             new Reply421Exception(session.getReplyCode().getMesg()));
279         writeIntermediateAnswer(ctx);
280         return;
281       }
282       AbstractCommand command = FtpCommandCode.getFromLine(getFtpSession(), e);
283       logger.debug("RECVMSG: {} CMD: {} {}" + command.getCode(), e,
284                    command.getCommand());
285       // First check if the command is an ABORT, QUIT or STAT
286       if (!FtpCommandCode.isSpecialCommand(command.getCode())) {
287         // Now check if a transfer is on its way: illegal to have at
288         // same time two commands (except ABORT). Wait is at most 100x
289         // RETRYINMS=1s
290         final FtpTransferControl control =
291             session.getDataConn().getFtpTransferControl();
292         final boolean notFinished = control.waitFtpTransferExecuting();
293         if (notFinished) {
294           session.setReplyCode(ReplyCode.REPLY_503_BAD_SEQUENCE_OF_COMMANDS,
295                                "Previous transfer command is not finished yet");
296           businessHandler.afterRunCommandKo(
297               new Reply503Exception(session.getReplyCode().getMesg()));
298           writeIntermediateAnswer(ctx);
299           return;
300         }
301       }
302       // Default message
303       session.setReplyCode(ReplyCode.REPLY_200_COMMAND_OKAY, null);
304       // Special check for SSL AUTH/PBSZ/PROT/USER/PASS/ACCT
305       if (FtpCommandCode.isSslOrAuthCommand(command.getCode())) {
306         session.setNextCommand(command);
307         messageRunAnswer(ctx);
308         return;
309       }
310       if (session.getCurrentCommand().isNextCommandValid(command)) {
311         logger.debug("Previous: {} Next: {}",
312                      session.getCurrentCommand().getCode(), command.getCode());
313         session.setNextCommand(command);
314         messageRunAnswer(ctx);
315       } else {
316         if (!session.getAuth().isIdentified()) {
317           session.setReplyCode(ReplyCode.REPLY_530_NOT_LOGGED_IN, null);
318           session.setNextCommand(new USER());
319           writeFinalAnswer(ctx);
320           return;
321         }
322         command = new IncorrectCommand();
323         command.setArgs(getFtpSession(), e, null,
324                         FtpCommandCode.IncorrectSequence);
325         session.setNextCommand(command);
326         messageRunAnswer(ctx);
327       }
328     }
329   }

331   /**
332    * Write the current answer and eventually close channel if necessary (421
333    * or
334    * 221)
335    *
336    * @return True if the channel is closed due to the code
337    */
338   private boolean writeFinalAnswer(final ChannelHandlerContext ctx) {
339     if (session.getReplyCode() ==
340         ReplyCode.REPLY_421_SERVICE_NOT_AVAILABLE_CLOSING_CONTROL_CONNECTION ||
341         session.getReplyCode() ==
342         ReplyCode.REPLY_221_CLOSING_CONTROL_CONNECTION) {
343       session.getDataConn().getFtpTransferControl().clear();
344       writeIntermediateAnswer(ctx).addListener(WaarpSslUtility.SSLCLOSE);
345       return true;
346     }
347     writeIntermediateAnswer(ctx);
348     session.setCurrentCommandFinished();
349     return false;
350   }

384   /**
385    * Execute one command and write the following answer
386    */
387   private void messageRunAnswer(final ChannelHandlerContext ctx) {
388     boolean error = false;
389     logger.debug("Code: {}", session.getCurrentCommand().getCode());
390     try {
391       businessHandler.beforeRunCommand();
392       final AbstractCommand command = session.getCurrentCommand();
393       logger.debug("Run {}", command.getCommand());
394       command.exec();
395       businessHandler.afterRunCommandOk();
396     } catch (final CommandAbstractException e) {
397       logger.debug("Command in error", e);
398       error = true;
399       session.setReplyCode(e);
400       businessHandler.afterRunCommandKo(e);
401     }
402     logger.debug("Code: {} [{}]", session.getCurrentCommand().getCode(),
403                  session.getReplyCode());
404     if (error) {
405       if (session.getCurrentCommand().getCode() !=
406           FtpCommandCode.INTERNALSHUTDOWN) {
407         writeFinalAnswer(ctx);
408       }
409       // In error so Check that Data is closed
410       if (session.getDataConn().isActive()) {
411         logger.debug("Closing DataChannel while command is in error");
412         try {
413           session.getDataConn().getCurrentDataChannel().close();
414         } catch (final FtpNoConnectionException e) {
415           // ignore
416         }
417       }
418       return;
419     }
420     if (session.getCurrentCommand().getCode() == FtpCommandCode.AUTH ||
421         session.getCurrentCommand().getCode() == FtpCommandCode.CCC) {
422       controlChannel.config().setAutoRead(false);
423       final ChannelFuture future = writeIntermediateAnswer(ctx);
424       session.setCurrentCommandFinished();
425       if (session.getCurrentCommand().getCode() == FtpCommandCode.AUTH) {
426         logger.debug("SSL to be added to pipeline");
427         ChannelHandler sslHandler = ctx.pipeline().first();
428         if (sslHandler instanceof SslHandler) {
429           logger.debug("Already got a SslHandler");
430         } else {
431           logger.debug("Add Explicitely SSL support to Command");
432           // add the SSL support
433           sslHandler = FtpsInitializer.waarpSslContextFactory
434               .initInitializer(true, FtpsInitializer.waarpSslContextFactory
435                   .needClientAuthentication());
436           session.prepareSsl();
437           WaarpSslUtility.addSslHandler(future, ctx.pipeline(), sslHandler,
438                                         new GenericFutureListener<Future<? super Channel>>() {
439                                           @Override
440                                           public void operationComplete(
441                                               final Future<? super Channel> future) {
442                                             if (!future.isSuccess()) {
443                                               final String error2 =
444                                                   future.cause() != null?
445                                                       future.cause()
446                                                             .getMessage() :
447                                                       "During Handshake";
448                                               logger.error(
449                                                   "Cannot finalize Ssl Command channel {}",
450                                                   error2);
451                                               callForSnmp(
452                                                   "SSL Connection Error",
453                                                   error2);
454                                               session.setSsl(false);
455                                               ctx.close();
456                                             } else {
457                                               logger.debug(
458                                                   "End of initialization of SSL and command channel: {}",
459                                                   ctx.channel());
460                                               session.setSsl(true);
461                                             }
462                                           }
463                                         });
464         }
465       } else if (session.getCurrentCommand().getCode() == FtpCommandCode.CCC) {
466         logger.debug("SSL to be removed from pipeline");
467         // remove the SSL support
468         session.prepareSsl();
469         WaarpSslUtility.removingSslHandler(future, controlChannel, false);
470       }
471     } else if (session.getCurrentCommand().getCode() !=
472                FtpCommandCode.INTERNALSHUTDOWN) {
473       writeFinalAnswer(ctx);
474     }
475   }
