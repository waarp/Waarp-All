133            .append(FtpCommandCode.XCUP.name()).append('\n')
134            .append(FtpCommandCode.XCWD.name()).append('\n')
135            .append(FtpCommandCode.XMKD.name()).append('\n')
136            .append(FtpCommandCode.XPWD.name()).append('\n')
137            .append(FtpCommandCode.XRMD.name()).append('\n')
138            .append(FtpCommandCode.PASV.name()).append('\n')
139            .append(FtpCommandCode.ALLO.name()).append('\n')

148            .append(' ').append(FILENAME).append('\n')
149            .append(FtpCommandCode.SITE.name()).append(' ')

151            // .append(" \"filename\"")

190   /**
191    * @param args
192    *
193    * @return the string to return to the client for the FEAT command for the
194    *     MLSx argument
195    */
196   protected String getMLSxOptsMessage(final String[] args) {
197     String[] properties = PROPERTIES_0_LENGTH;
198     if (args.length >= 2) {
199       properties = args[1].split(";");
200     }
201 
202     final FilesystemBasedOptsMLSxImpl optsMLSx =
203         (FilesystemBasedOptsMLSxImpl) getFtpSession().getDir().getOptsMLSx();
204     optsMLSx.setOptsModify((byte) 0);
205     optsMLSx.setOptsPerm((byte) 0);
206     optsMLSx.setOptsSize((byte) 0);
207     optsMLSx.setOptsType((byte) 0);
208     for (final String propertie : properties) {
209       if ("modify".equalsIgnoreCase(propertie)) {
210         optsMLSx.setOptsModify((byte) 1);
211       } else if ("perm".equalsIgnoreCase(propertie)) {
212         optsMLSx.setOptsPerm((byte) 1);
213       } else if ("size".equalsIgnoreCase(propertie)) {
214         optsMLSx.setOptsSize((byte) 1);
215       } else if ("type".equalsIgnoreCase(propertie)) {
216         optsMLSx.setOptsType((byte) 1);
217       }
218     }
219     return args[0] + ' ' + FtpCommandCode.OPTS.name() + optsMLSx.getFeat();
220   }

263   /**
264    * This method is called for every received message before the execution of
265    * the command. If an exception is
266    * raised, the reply is immediate and no action taken.
267    *
268    * @throws CommandAbstractException
269    */
270   public abstract void beforeRunCommand() throws CommandAbstractException;
271 
272   /**
273    * This method is called for every received message after the execution of
274    * the
275    * command but before the final
276    * reply to the client. If an exception is raised, the reply is immediate.
277    * This is the last call before
278    * finishing the command.
279    *
280    * @throws CommandAbstractException
281    */
282   public abstract void afterRunCommandOk() throws CommandAbstractException;
283 
284   /**
285    * Run when a FTP exception is catch (the channel is not necessary closed
286    * after). This must set a correct
287    * answer and a correct code of reply. If the code of reply is 421, then the
288    * channel will be closed after this
289    * call. This is the last call before finishing the command.
290    *
291    * @param e
292    */
293   public abstract void afterRunCommandKo(CommandAbstractException e);
294 
295   /**
296    * Run when a transfer is finished (eventually in error) but before
297    * answering.
298    * Note that this is called only
299    * for a Transfer Request (or LIST) but called before afterRunCommandXX is
300    * called (Ok or Ko).
301    *
302    * @param transfer
303    *
304    * @throws CommandAbstractException
305    */
306   public abstract void afterTransferDoneBeforeAnswer(FtpTransfer transfer)
307       throws CommandAbstractException;
