88  class MD5State {
89    /**
90     * 128-bit state
91     */
92    final int[] state;
93  
94    /**
95     * 64-bit character count
96     */
97    long count;
98  
99    /**
100    * 64-byte buffer (512 bits) for storing to-be-hashed characters
101    */
102   final byte[] buffer;
103 
104   /**
105    * Constructor
106    */
107   MD5State() {
108     buffer = new byte[64];
109     count = 0;
110     state = new int[4];
111 
112     state[0] = 0x67452301;
113     state[1] = 0xefcdab89;
114     state[2] = 0x98badcfe;
115     state[3] = 0x10325476;
116 
117   }
118 
119   /**
120    * Create this State as a copy of another state
121    *
122    * @param from
123    */
124   MD5State" href="../../../../org/waarp/common/digest/MD5State.html#MD5State">MD5State(MD5State from) {
125     this();
126     System.arraycopy(from.buffer, 0, buffer, 0, buffer.length);
127     System.arraycopy(from.state, 0, state, 0, state.length);
128 
129     count = from.count;
130   }
131 }
