309   /**
310    * CR LF<br>
311    * A User Telnet MUST be able to send any of the forms: CR LF, CR NUL, and
312    * LF.
313    * A User Telnet on an ASCII host
314    * SHOULD have a user-controllable mode to send either CR LF or CR NUL when
315    * the user presses the "end-of-line"
316    * key, and CR LF SHOULD be the default.
317    */
318   public static final String CRLF = "\r\n";

320   /**
321    * Construct a Reply code from specific message
322    *
323    * @param code
324    * @param mesg
325    */
326   ReplyCode(int code, String mesg) {
327     this.code = code;
328     this.mesg = getFinalMsg(code, mesg);
329   }
330 
331   /**
332    * Return the final message formatted as needed from the code and the
333    * message
334    *
335    * @param code
336    * @param msg
337    *
338    * @return the final formatted message
339    */
340   public static String getFinalMsg(int code, String msg) {
341     final StringBuilder builder = new StringBuilder();
342     builder.append(code);
343     if (msg.indexOf('\n') == -1) {
344       builder.append(' ').append(msg).append(CRLF);
345     } else {
346       final String[] lines = msg.split("\n");
347       // first line
348       builder.append('-').append(lines[0]).append(CRLF);
349       // next lines
350       for (int i = 1; i < lines.length - 1; i++) {
351         final int firstBlank = lines[i].indexOf(' ');
352         if (firstBlank > 0) {
353           final String firstParam = lines[i].substring(0, firstBlank);
354           boolean isInt = false;
355           try {
356             Integer.parseInt(firstParam);
357             isInt = true;
358           } catch (final NumberFormatException e) {
359             // not a number
360           }
361           if (isInt) {
362             builder.append("  ");
363           }
364         }
365         builder.append(lines[i]).append(CRLF);
366       }
367       // last line
368       builder.append(code).append(' ').append(lines[lines.length - 1])
369              .append(CRLF);
370     }
371     return builder.toString();
372   }

374   /**
375    * Construct a Reply Code from its name in Enum structure
376    *
377    * @param code
378    */
379   ReplyCode(int code) {
380     this.code = code;
381     mesg = name().substring(6).replace('_', ' ') + CRLF;
382   }
