41  public final class JvmProcessId {
42    /**
43     * Definition for Machine Id replacing MAC address
44     */
45    private static final Pattern MACHINE_ID_PATTERN =
46        Pattern.compile("^(?:[0-9a-fA-F][:-]?){6,8}$");
47    private static final int MACHINE_ID_LEN = 6;
48    /**
49     * So MAX value on 3 bytes (64 system use 2^22 id)
50     */
51    private static final int MAX_PID = 4194304;
52    /**
53     * 2 bytes value maximum
54     */
55    static final int JVMPID;
56    private static final Object[] EMPTY_OBJECTS = new Object[0];
57    private static final Class<?>[] EMPTY_CLASSES = new Class<?>[0];
58    /**
59     * Try to get Mac Address but could be also changed dynamically
60     */
61    static byte[] mac;
62    static int macInt;
63  
64    static {
65      JVMPID = jvmProcessId();
66      mac = macAddress();
67      macInt = macAddressAsInt();
68    }

70    private JvmProcessId() {
71    }

74    /**
75     * @return the JVM Process ID
76     */
77    public static int jvmProcessId() {
78      // Note: may fail in some JVM implementations
79      // something like '<pid>@<hostname>', at least in SUN / Oracle JVMs
80      try {
81        final ClassLoader loader = getSystemClassLoader();
82        String value;
83        value =
84            jvmProcessIdManagementFactory(loader, EMPTY_OBJECTS, EMPTY_CLASSES);
85        final int atIndex = value.indexOf('@');
86        if (atIndex >= 0) {
87          value = value.substring(0, atIndex);
88        }
89        int processId = -1;
90        processId = parseProcessId(processId, value);
91        if (processId < 0 || processId > MAX_PID) {
92          processId = StringUtils.RANDOM.nextInt(MAX_PID + 1);
93        }
94        return processId;
95      } catch (final Throwable e) {//NOSONAR
96        SysErrLogger.FAKE_LOGGER.syserr(e);
97        return StringUtils.RANDOM.nextInt(MAX_PID + 1);
98      }
99    }

101   /**
102    * @return the mac address if possible, else random values
103    */
104   public static byte[] macAddress() {
105     try {
106       byte[] machineId = null;
107       final String customMachineId =
108           SystemPropertyUtil.get("org.waarp.machineId");
109       if (customMachineId != null &&
110           MACHINE_ID_PATTERN.matcher(customMachineId).matches()) {
111         machineId = parseMachineId(customMachineId);
112       }
113 
114       if (machineId == null) {
115         machineId = defaultMachineId();
116       }
117       return machineId;
118     } catch (final Throwable e) {//NOSONAR
119       return StringUtils.getRandom(MACHINE_ID_LEN);
120     }
121   }

123   /**
124    * @return MAC address as int (truncated to 4 bytes instead of 6)
125    */
126   public static int macAddressAsInt() {
127     return (mac[3] & 0xFF) << 24 | (mac[2] & 0xFF) << 16 |
128            (mac[1] & 0xFF) << 8 | mac[0] & 0xFF;
129   }

131   /**
132    * Up to the 6 first bytes will be used. If Null or less than 6 bytes, extra
133    * bytes will be randomly generated.
134    *
135    * @param mac the MAC address in byte format (up to the 6 first
136    *     bytes will
137    *     be used)
138    */
139   public static synchronized void setMac(final byte[] mac) {
140     if (mac == null) {
141       JvmProcessId.mac = StringUtils.getRandom(MACHINE_ID_LEN);
142     } else {
143       JvmProcessId.mac = Arrays.copyOf(mac, MACHINE_ID_LEN);
144       for (int i = mac.length; i < MACHINE_ID_LEN; i++) {
145         JvmProcessId.mac[i] = (byte) StringUtils.RANDOM.nextInt(256);
146       }
147     }
148     macInt = macAddressAsInt();
149   }

151   /**
152    * @return positive - current is better, 0 - cannot tell from MAC addr,
153    *     negative - candidate is better.
154    */
155   private static int compareAddresses(final byte[] current,
156                                       final byte[] candidate) {
157     if (candidate == null) {
158       return 1;
159     }
160     // Must be EUI-48 or longer.
161     if (candidate.length < 6) {
162       return 1;
163     }
164     // Must not be filled with only 0 and 1.
165     boolean onlyZeroAndOne = true;
166     for (final byte b : candidate) {
167       if (b != 0 && b != 1) {
168         onlyZeroAndOne = false;
169         break;
170       }
171     }
172     if (onlyZeroAndOne) {
173       return 1;
174     }
175     // Must not be a multicast address
176     if ((candidate[0] & 1) != 0) {
177       return 1;
178     }
179     // Prefer globally unique address.
180     if ((current[0] & 2) == 0) {
181       if ((candidate[0] & 2) == 0) {
182         // Both current and candidate are globally unique addresses.
183         return 0;
184       } else {
185         // Only current is globally unique.
186         return 1;
187       }
188     } else {
189       if ((candidate[0] & 2) == 0) {
190         // Only candidate is globally unique.
191         return -1;
192       } else {
193         // Both current and candidate are non-unique.
194         return 0;
195       }
196     }
197   }

199   /**
200    * @return positive - current is better, 0 - cannot tell, negative -
201    *     candidate
202    *     is better
203    */
204   private static int compareAddresses(final InetAddress current,
205                                       final InetAddress candidate) {
206     return scoreAddress(current) - scoreAddress(candidate);
207   }
208 
209   private static int scoreAddress(final InetAddress addr) {
210     if (addr.isAnyLocalAddress()) {
211       return 0;
212     }
213     if (addr.isMulticastAddress()) {
214       return 1;
215     }
216     if (addr.isLinkLocalAddress()) {
217       return 2;
218     }
219     if (addr.isSiteLocalAddress()) {
220       return 3;
221     }
222 
223     return 4;
224   }

226   // pulled from http://stackoverflow.com/questions/35842/how-can-a-java-program-get-its-own-process-id
227   private static ClassLoader getSystemClassLoader() {
228     if (System.getSecurityManager() == null) {
229       return ClassLoader.getSystemClassLoader();
230     } else {
231       return AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {
232         @Override
233         public ClassLoader run() {
234           return ClassLoader.getSystemClassLoader();
235         }
236       });
237     }
238   }

240   /**
241    * @param oldProcessId
242    * @param customProcessId
243    *
244    * @return the processId
245    */
246   private static int parseProcessId(final int oldProcessId,
247                                     final String customProcessId) {
248     int processId = oldProcessId;
249     try {
250       processId = Integer.parseInt(customProcessId);
251     } catch (final NumberFormatException e) {
252       // Malformed input.
253     }
254     if (processId < 0 || processId > MAX_PID) {
255       processId = -1;
256     }
257     return processId;
258   }

260   /**
261    * @param loader
262    * @param emptyObjects
263    * @param emptyClasses
264    *
265    * @return the processId as String
266    */
267   private static String jvmProcessIdManagementFactory(final ClassLoader loader,
268                                                       final Object[] emptyObjects,
269                                                       final Class<?>[] emptyClasses) {
270     String value;
271     try {
272       // Invoke
273       // java.lang.management.ManagementFactory.getRuntimeMXBean().getName()
274       final Class<?> mgmtFactoryType =
275           Class.forName("java.lang.management.ManagementFactory", true, loader);
276       final Class<?> runtimeMxBeanType =
277           Class.forName("java.lang.management.RuntimeMXBean", true, loader);
278 
279       final Method getRuntimeMXBean =
280           mgmtFactoryType.getMethod("getRuntimeMXBean", emptyClasses);
281       final Object bean = getRuntimeMXBean.invoke(null, emptyObjects);
282       final Method getName =
283           runtimeMxBeanType.getDeclaredMethod("getName", emptyClasses);
284       value = (String) getName.invoke(bean, emptyObjects);
285     } catch (final Exception e) {
286       SysErrLogger.FAKE_LOGGER.syserr(
287           "Unable to get PID, try another way: " + e.getMessage());
288 
289       try {
290         // Invoke android.os.Process.myPid()
291         final Class<?> processType =
292             Class.forName("android.os.Process", true, loader);
293         final Method myPid = processType.getMethod("myPid", emptyClasses);
294         value = myPid.invoke(null, emptyObjects).toString();
295       } catch (final Exception e2) {
296         SysErrLogger.FAKE_LOGGER.syserr(
297             "Unable to get PID: " + e2.getMessage());
298         value = "";
299       }
300     }
301     return value;
302   }

304   private static byte[] parseMachineId(String value) {
305     // Strip separators.
306     value = value.replaceAll("[:-]", "");
307 
308     final byte[] machineId = new byte[MACHINE_ID_LEN];
309     for (int i = 0; i < value.length() && i < MACHINE_ID_LEN; i += 2) {
310       machineId[i] = (byte) Integer.parseInt(value.substring(i, i + 2), 16);
311     }
312 
313     return machineId;
314   }

316   private static byte[] defaultMachineId() {
317     // Find the best MAC address available.
318     final byte[] notFound = { -1 };
319     byte[] bestMacAddr = notFound;
320     InetAddress bestInetAddr;
321     try {
322       bestInetAddr = InetAddress.getByAddress(new byte[] { 127, 0, 0, 1 });
323     } catch (final UnknownHostException e) {
324       // Never happens.
325       throw new IllegalArgumentException(e);
326     }
327 
328     // Retrieve the list of available network interfaces.
329     final Map<NetworkInterface, InetAddress> ifaces =
330         new LinkedHashMap<NetworkInterface, InetAddress>();
331     try {
332       for (final Enumeration<NetworkInterface> i =
333            NetworkInterface.getNetworkInterfaces(); i.hasMoreElements(); ) {
334         final NetworkInterface iface = i.nextElement();
335         // Use the interface with proper INET addresses only.
336         final Enumeration<InetAddress> addrs = iface.getInetAddresses();
337         if (addrs.hasMoreElements()) {
338           final InetAddress a = addrs.nextElement();
339           if (!a.isLoopbackAddress()) {
340             ifaces.put(iface, a);
341           }
342         }
343       }
344     } catch (final SocketException ignored) {
345       // nothing
346     }
347 
348     for (final Entry<NetworkInterface, InetAddress> entry : ifaces.entrySet()) {
349       final NetworkInterface iface = entry.getKey();
350       final InetAddress inetAddr = entry.getValue();
351       if (iface.isVirtual()) {
352         continue;
353       }
354 
355       final byte[] macAddr;
356       try {
357         macAddr = iface.getHardwareAddress();
358       } catch (final SocketException e) {
359         continue;
360       }
361 
362       boolean replace = false;
363       int res = compareAddresses(bestMacAddr, macAddr);
364       if (res < 0) {
365         // Found a better MAC address.
366         replace = true;
367       } else if (res == 0) {
368         // Two MAC addresses are of pretty much same quality.
369         res = compareAddresses(bestInetAddr, inetAddr);
370         if (res < 0) {
371           // Found a MAC address with better INET address.
372           replace = true;
373         } else if (res == 0) {
374           // Cannot tell the difference. Choose the longer one.
375           if (bestMacAddr.length < macAddr.length) {
376             replace = true;
377           }
378         }
379       }
380 
381       if (replace) {
382         bestMacAddr = macAddr;
383         bestInetAddr = inetAddr;
384       }
385     }
386 
387     if (bestMacAddr == notFound) {
388       bestMacAddr = StringUtils.getRandom(MACHINE_ID_LEN);
389     }
390     return bestMacAddr;
391   }
