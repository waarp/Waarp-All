82    /**
83     * Initialize empty KeyStore. No TrustStore is internally created.
84     *
85     * @param keyStorePasswd
86     * @param keyPassword
87     *
88     * @throws CryptoException
89     */
90    public WaarpSecureKeyStore(final String keyStorePasswd,
91                               final String keyPassword) throws CryptoException {
92      this.keyStorePasswd = keyStorePasswd;
93      this.keyPassword = keyPassword;
94      try {
95        keyStore = KeyStore.getInstance("JKS");
96      } catch (final KeyStoreException e) {
97        logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
98        throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
99      }
100     try {
101       // Empty keyStore created so null for the InputStream
102       keyStore.load(null, getKeyStorePassword());
103     } catch (final NoSuchAlgorithmException e) {
104       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
105       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
106     } catch (final CertificateException e) {
107       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
108       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
109     } catch (final FileNotFoundException e) {
110       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
111       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
112     } catch (final IOException e) {
113       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
114       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
115     }
116     initKeyManagerFactory();
117   }

134   /**
135    * Initialize the SecureKeyStore and TrustStore from files
136    *
137    * @param keyStoreFilename
138    * @param keyStorePasswd
139    * @param keyPassword
140    * @param trustStoreFilename if Null, no TrustKeyStore will be
141    *     created
142    * @param trustStorePasswd
143    * @param needClientAuthent True if the TrustStore is also used for
144    *     Client
145    *     Authentication
146    *
147    * @throws CryptoException
148    */
149   public WaarpSecureKeyStore(final String keyStoreFilename,
150                              final String keyStorePasswd,
151                              final String keyPassword,
152                              final String trustStoreFilename,
153                              final String trustStorePasswd,
154                              final boolean needClientAuthent)
155       throws CryptoException {
156     // Create the KeyStore
157     initKeyStore(keyStoreFilename, keyStorePasswd, keyPassword);
158     // Now create the TrustKeyStore
159     if (trustStoreFilename != null) {
160       initTrustStore(trustStoreFilename, trustStorePasswd, needClientAuthent);
161     } else {
162       initEmptyTrustStore();
163     }
164   }

166   /**
167    * Initialize the SecureKeyStore with no TrustStore from file
168    *
169    * @param keystoreFilename
170    * @param keystorePasswd
171    * @param keyPasswordNew
172    *
173    * @throws CryptoException
174    */
175   public void initKeyStore(final String keystoreFilename,
176                            final String keystorePasswd,
177                            final String keyPasswordNew) throws CryptoException {
178     keyStoreFilename = keystoreFilename;
179     keyStorePasswd = keystorePasswd;
180     keyPassword = keyPasswordNew;
181     // First keyStore itself
182     try {
183       keyStore = KeyStore.getInstance("JKS");
184     } catch (final KeyStoreException e) {
185       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
186       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
187     }
188     FileInputStream inputStream = null;
189     try {
190       inputStream = new FileInputStream(keyStoreFilename);
191       keyStore.load(inputStream, getKeyStorePassword());
192     } catch (final NoSuchAlgorithmException e) {
193       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
194       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
195     } catch (final CertificateException e) {
196       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
197       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
198     } catch (final FileNotFoundException e) {
199       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
200       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
201     } catch (final IOException e) {
202       logger.error(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
203       throw new CryptoException(CANNOT_CREATE_KEY_STORE_INSTANCE, e);
204     } finally {
205       FileUtils.close(inputStream);
206     }
207     initKeyManagerFactory();
208   }

210   /**
211    * Init KeyManagerFactory
212    *
213    * @throws CryptoException
214    */
215   void initKeyManagerFactory() throws CryptoException {
216     try {
217       keyManagerFactory = KeyManagerFactory
218           .getInstance(KeyManagerFactory.getDefaultAlgorithm());
219     } catch (final NoSuchAlgorithmException e) {
220       logger.error(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
221       throw new CryptoException(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
222     }
223     try {
224       keyManagerFactory.init(keyStore, getCertificatePassword());
225     } catch (final UnrecoverableKeyException e) {
226       logger.error(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
227       throw new CryptoException(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
228     } catch (final KeyStoreException e) {
229       logger.error(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
230       throw new CryptoException(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
231     } catch (final NoSuchAlgorithmException e) {
232       logger.error(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
233       throw new CryptoException(CANNOT_CREATE_KEY_MANAGER_FACTORY_INSTANCE, e);
234     }
235   }

244   public boolean deleteKeyFromKeyStore(final String alias) {

297         logger.error(CANNOT_SAVE_TO_FILE_KEY_STORE_INSTANCE, e);

309   /**
310    * Initialize the TrustStore from a filename and its password
311    *
312    * @param truststoreFilename
313    * @param truststorePasswd
314    * @param needClientAuthent True if the TrustStore is also to
315    *     authenticate
316    *     clients
317    *
318    * @throws CryptoException
319    */
320   public void initTrustStore(final String truststoreFilename,
321                              final String truststorePasswd,
322                              final boolean needClientAuthent)
323       throws CryptoException {
324     trustStorePasswd = truststorePasswd;
325     try {
326       keyTrustStore = KeyStore.getInstance("JKS");
327     } catch (final KeyStoreException e) {
328       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e);
329       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
330                                 e);
331     }
332     FileInputStream inputStream = null;
333     try {
334       inputStream = new FileInputStream(truststoreFilename);
335       keyTrustStore.load(inputStream, getKeyTrustStorePassword());
336     } catch (final NoSuchAlgorithmException e) {
337       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e);
338       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
339                                 e);
340     } catch (final CertificateException e) {
341       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e);
342       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
343                                 e);
344     } catch (final FileNotFoundException e) {
345       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e);
346       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
347                                 e);
348     } catch (final IOException e) {
349       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e);
350       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
351                                 e);
352     } finally {
353       FileUtils.close(inputStream);
354     }
355     final TrustManagerFactory trustManagerFactory;
356     try {
357       trustManagerFactory = TrustManagerFactory
358           .getInstance(KeyManagerFactory.getDefaultAlgorithm());
359     } catch (final NoSuchAlgorithmException e1) {
360       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e1);
361       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
362                                 e1);
363     }
364     try {
365       trustManagerFactory.init(keyTrustStore);
366     } catch (final KeyStoreException e1) {
367       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e1);
368       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
369                                 e1);
370     }
371     try {
372       secureTrustManagerFactory =
373           new WaarpSecureTrustManagerFactory(trustManagerFactory,
374                                              needClientAuthent);
375     } catch (final CryptoException e) {
376       logger.error(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE, e);
377       throw new CryptoException(CANNOT_CREATE_TRUST_MANAGER_FACTORY_INSTANCE,
378                                 e);
379     }
380   }

382   /**
383    * Initialize an empty TrustStore
384    *
385    * @return True if correctly initialized empty
386    */
387   public boolean initEmptyTrustStore() {
388     trustStorePasswd = "secret";//NOSONAR
389     try {
390       keyTrustStore = KeyStore.getInstance("JKS");
391     } catch (final KeyStoreException e) {
392       logger.error(CANNOT_CREATE_KEY_TRUST_STORE_INSTANCE, e);
393       return false;
394     }
395     try {
396       // Empty keyTrustStore created so null for the InputStream
397       keyTrustStore.load(null, getKeyTrustStorePassword());
398     } catch (final NoSuchAlgorithmException e) {
399       logger.error(CANNOT_CREATE_KEY_TRUST_STORE_INSTANCE, e);
400       return false;
401     } catch (final CertificateException e) {
402       logger.error(CANNOT_CREATE_KEY_TRUST_STORE_INSTANCE, e);
403       return false;
404     } catch (final FileNotFoundException e) {
405       logger.error(CANNOT_CREATE_KEY_TRUST_STORE_INSTANCE, e);
406       return false;
407     } catch (final IOException e) {
408       logger.error(CANNOT_CREATE_KEY_TRUST_STORE_INSTANCE, e);
409       return false;
410     }
411     secureTrustManagerFactory = new WaarpSecureTrustManagerFactory();
412     return true;
413   }

486   /**
487    * Load a certificate from a filename
488    *
489    * @param filename
490    *
491    * @return the X509 Certificate from filename
492    *
493    * @throws CertificateException
494    * @throws FileNotFoundException
495    */
496   public static Certificate loadX509Certificate(final String filename)
497       throws CertificateException, FileNotFoundException {
498     final CertificateFactory cf = CertificateFactory.getInstance("X.509");
499     final FileInputStream in = new FileInputStream(filename);
500     try {
501       return cf.generateCertificate(in);
502     } finally {
503       FileUtils.close(in);
504     }
505   }
