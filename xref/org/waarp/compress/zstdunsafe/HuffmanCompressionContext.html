34  package org.waarp.compress.zstdunsafe;
35  
36  class HuffmanCompressionContext {
37    private final HuffmanTableWriterWorkspace tableWriterWorkspace =
38        new HuffmanTableWriterWorkspace();
39    private final HuffmanCompressionTableWorkspace compressionTableWorkspace =
40        new HuffmanCompressionTableWorkspace();
41  
42    private HuffmanCompressionTable previousTable =
43        new HuffmanCompressionTable(Huffman.MAX_SYMBOL_COUNT);
44    private HuffmanCompressionTable temporaryTable =
45        new HuffmanCompressionTable(Huffman.MAX_SYMBOL_COUNT);
46  
47    private HuffmanCompressionTable previousCandidate = previousTable;
48    private HuffmanCompressionTable temporaryCandidate = temporaryTable;
49  
50    public HuffmanCompressionTable getPreviousTable() {
51      return previousTable;
52    }
53  
54    public HuffmanCompressionTable borrowTemporaryTable() {
55      previousCandidate = temporaryTable;
56      temporaryCandidate = previousTable;
57  
58      return temporaryTable;
59    }
60  
61    public void discardTemporaryTable() {
62      previousCandidate = previousTable;
63      temporaryCandidate = temporaryTable;
64    }
65  
66    public void saveChanges() {
67      temporaryTable = temporaryCandidate;
68      previousTable = previousCandidate;
69    }
70  
71    public HuffmanCompressionTableWorkspace getCompressionTableWorkspace() {
72      return compressionTableWorkspace;
73    }
74  
75    public HuffmanTableWriterWorkspace getTableWriterWorkspace() {
76      return tableWriterWorkspace;
77    }
78  }
