49  public class NetworkTransaction {
50    /**
51     * Internal Logger
52     */
53    private static final WaarpLogger logger =
54        WaarpLoggerFactory.getLogger(NetworkTransaction.class);
55  
56    private final Bootstrap clientBootstrap;
57    private final Bootstrap clientSslBootstrap;
58    private final ChannelGroup networkChannelGroup;
59  
60    public NetworkTransaction() {
61      networkChannelGroup = new DefaultChannelGroup("NetworkChannels",
62                                                    configuration
63                                                        .getSubTaskGroup()
64                                                        .next());
65      final NetworkServerInitializerProxy networkServerInitializerProxy =
66          new NetworkServerInitializerProxy(false);
67      clientBootstrap = new Bootstrap();
68      WaarpNettyUtil
69          .setBootstrap(clientBootstrap, configuration.getNetworkWorkerGroup(),
70                        (int) configuration.getTimeoutCon(),
71                        configuration.getBlockSize() + 64, false);
72      clientBootstrap.handler(networkServerInitializerProxy);
73      clientSslBootstrap = new Bootstrap();
74      if (configuration.isUseSSL() && configuration.getHostSslId() != null) {
75        final NetworkSslServerInitializerProxy networkSslServerInitializerProxy =
76            new NetworkSslServerInitializerProxy(true);
77        WaarpNettyUtil.setBootstrap(clientSslBootstrap,
78                                    configuration.getNetworkWorkerGroup(),
79                                    (int) configuration.getTimeoutCon(),
80                                    configuration.getBlockSize() + 64, false);
81        clientSslBootstrap.handler(networkSslServerInitializerProxy);
82      } else {
83        if (configuration.isWarnOnStartup()) {
84          logger.warn("No SSL support configured");
85        } else {
86          logger.info("No SSL support configured");
87        }
88      }
89    }

91    /**
92     * Create a connection to the specified socketAddress with multiple retries
93     *
94     * @param socketAddress
95     * @param isSSL
96     *
97     * @return the Channel
98     */
99    public Channel createConnectionWithRetry(final SocketAddress socketAddress,
100                                            final boolean isSSL) {
101     Channel channel = null;
102     OpenR66Exception lastException = null;
103     for (int i = 0; i < RETRYNB; i++) {
104       try {
105         channel = createConnection(socketAddress, isSSL);
106         break;
107       } catch (final OpenR66ProtocolNoConnectionException e1) {
108         lastException = e1;
109         channel = null;
110         break;
111       } catch (final OpenR66ProtocolNetworkException e1) {
112         // Can retry
113         lastException = e1;
114         channel = null;
115         try {
116           Thread.sleep(WAITFORNETOP * 5);
117         } catch (final InterruptedException e) {//NOSONAR
118           SysErrLogger.FAKE_LOGGER.ignoreLog(e);
119           break;
120         }
121       }
122     }
123     if (channel == null) {
124       if (lastException != null) {
125         logger.info("Cannot connect : {}", lastException.getMessage());
126       } else {
127         logger.info("Cannot connect!");
128       }
129     } else if (lastException != null) {
130       logger.debug("Connection retried since {}", lastException.getMessage());
131     }
132     return channel;
133   }

135   /**
136    * Create a connection to the specified socketAddress
137    *
138    * @param socketAddress
139    * @param isSSL
140    *
141    * @return the channel
142    *
143    * @throws OpenR66ProtocolNetworkException
144    * @throws OpenR66ProtocolNoConnectionException
145    */
146   private Channel createConnection(final SocketAddress socketAddress,
147                                    final boolean isSSL)
148       throws OpenR66ProtocolNetworkException,
149              OpenR66ProtocolNoConnectionException {
150     Channel channel = null;
151     boolean ok = false;
152     // check valid limit on server side only (could be the initiator but not a client)
153     boolean valid = false;
154     for (int i = 0; i < RETRYNB * 2; i++) {
155       if (configuration.getConstraintLimitHandler().checkConstraintsSleep(i)) {
156         logger.info("Constraints exceeded: {}", i);
157       } else {
158         logger.debug("Constraints NOT exceeded");
159         valid = true;
160         break;
161       }
162     }
163     if (!valid) {
164       // Limit is locally exceeded
165       logger.debug("Overloaded local system");
166       throw new OpenR66ProtocolNetworkException(
167           "Cannot connect to remote server due to local overload");
168     }
169     try {
170       channel = createNewConnection(socketAddress, isSSL);
171       ok = true;
172     } finally {
173       if (!ok) {
174         if (channel != null) {
175           if (channel.isOpen()) {
176             WaarpSslUtility.closingSslChannel(channel);
177           }
178           channel = null;
179         }
180       }
181     }
182     return channel;
183   }

185   /**
186    * @param socketServerAddress
187    * @param isSSL
188    *
189    * @return the channel
190    *
191    * @throws OpenR66ProtocolNetworkException
192    * @throws OpenR66ProtocolNoConnectionException
193    */
194   private Channel createNewConnection(final SocketAddress socketServerAddress,
195                                       final boolean isSSL)
196       throws OpenR66ProtocolNetworkException,
197 
198              OpenR66ProtocolNoConnectionException {
199     ChannelFuture channelFuture = null;
200     for (int i = 0; i < RETRYNB; i++) {
201       try {
202         if (isSSL) {
203           if (configuration.getHostSslId() != null) {
204             channelFuture = clientSslBootstrap.connect(socketServerAddress);
205           } else {
206             throw new OpenR66ProtocolNoConnectionException("No SSL support");
207           }
208         } else {
209           channelFuture = clientBootstrap.connect(socketServerAddress);
210         }
211       } catch (final ChannelPipelineException e) {
212         throw new OpenR66ProtocolNoConnectionException(
213             "Cannot connect to remote server due to a channel exception");
214       }
215       try {
216         channelFuture.await(configuration.getTimeoutCon() / 3);
217       } catch (final InterruptedException e1) {//NOSONAR
218         SysErrLogger.FAKE_LOGGER.ignoreLog(e1);
219       }
220       if (channelFuture.isSuccess()) {
221         final Channel channel = channelFuture.channel();
222         if (isSSL && !NetworkSslServerHandler.isSslConnectedChannel(channel)) {
223           logger.info("KO CONNECT since SSL handshake is over");
224           channel.close();
225           throw new OpenR66ProtocolNoConnectionException(
226               "Cannot finish connect to remote server");
227         }
228         networkChannelGroup.add(channel);
229         return channel;
230       } else {
231         try {
232           Thread.sleep(WAITFORNETOP * 2);
233         } catch (final InterruptedException e) {//NOSONAR
234           SysErrLogger.FAKE_LOGGER.ignoreLog(e);
235         }
236         if (!channelFuture.isDone()) {
237           throw new OpenR66ProtocolNoConnectionException(
238               "Cannot connect to remote server due to interruption");
239         }
240         if (channelFuture.cause() instanceof ConnectException) {
241           logger.debug("KO CONNECT: {}", channelFuture.cause().getMessage());
242           throw new OpenR66ProtocolNoConnectionException(
243               "Cannot connect to remote server", channelFuture.cause());
244         } else {
245           logger.debug("KO CONNECT but retry", channelFuture.cause());
246         }
247       }
248     }
249     final Throwable cause = channelFuture == null? null : channelFuture.cause();
250     throw new OpenR66ProtocolNetworkException("Cannot connect to remote server",
251                                               cause);
252   }

254   /**
255    * Close all Network Ttransaction
256    */
257   public void closeAll() {
258     logger.debug("close All Network Channels");
259     try {
260       Thread.sleep(RETRYINMS * 2);
261     } catch (final InterruptedException e) {//NOSONAR
262       SysErrLogger.FAKE_LOGGER.ignoreLog(e);
263     }
264     if (!configuration.isServer()) {
265       WaarpShutdownHook.shutdownHook.launchFinalExit();
266     }
267     for (final Channel channel : networkChannelGroup) {
268       WaarpSslUtility.closingSslChannel(channel);
269     }
270     WaarpNettyUtil.awaitOrInterrupted(networkChannelGroup.close());
271     try {
272       Thread.sleep(WAITFORNETOP);
273     } catch (final InterruptedException e) {//NOSONAR
274       SysErrLogger.FAKE_LOGGER.ignoreLog(e);
275     }
276     configuration.clientStop();
277     logger.debug("Last action before exit");
278     ChannelUtils.stopLogger();
279   }

281   /**

283    */
284   public int getNumberClients() {
285     return networkChannelGroup.size();
