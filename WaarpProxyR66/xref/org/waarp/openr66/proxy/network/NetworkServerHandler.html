55  /**
56   * Network Server Handler (Requester side)
57   */
58  public class NetworkServerHandler
59      extends SimpleChannelInboundHandler<NetworkPacket> {
60    /**
61     * Internal Logger
62     */
63    private static final WaarpLogger logger =
64        WaarpLoggerFactory.getLogger(NetworkServerHandler.class);
65  
66    /**
67     * The underlying Network Channel
68     */
69    private volatile Channel networkChannel;
70    /**
71     * The underlying Proxified associated Channel
72     */
73    private volatile Channel proxyChannel;
74    /**
75     * The associated bridge
76     */
77    private volatile ProxyBridge bridge;
78    /**
79     * Does this Handler is for SSL
80     */
81    protected volatile boolean isSSL;
82    /**
83     * Is this Handler a server side
84     */
85    protected final boolean isServer;
86    /**
87     * To handle the keep alive
88     */
89    private final AtomicInteger keepAlivedSent = new AtomicInteger();
90    /**
91     * Future to wait for Client to be setup
92     */
93    protected volatile R66Future clientFuture;
94  
95    /**
96     * @param isServer
97     */
98    public NetworkServerHandler(final boolean isServer) {
99      this.isServer = isServer;
100     if (!this.isServer) {
101       clientFuture = new R66Future(true);
102     }
103   }
104 
105   public void setBridge(final ProxyBridge bridge) {
106     this.bridge = bridge;
107     if (this.bridge != null) {
108       proxyChannel = bridge.getSource().getNetworkChannel();
109     }
110     clientFuture.setSuccess();
111     logger.info("setBridge: {} {}", isServer, (bridge != null?
112         bridge.getProxyEntry() + " proxyChannelId: " + proxyChannel.id() :
113         "nobridge"));
114   }

117    * @return the networkChannel
118    */
119   public Channel getNetworkChannel() {
120     return networkChannel;
121   }

123   public void close() {
124     WaarpSslUtility.closingSslChannel(networkChannel);
125   }

127   @Override
128   public void channelInactive(final ChannelHandlerContext ctx) {
129     if (proxyChannel != null) {
130       WaarpSslUtility.closingSslChannel(proxyChannel);
131     }

134   @Override
135   public void channelActive(final ChannelHandlerContext ctx) throws Exception {
136     try {
137       networkChannel = ctx.channel();
138       /*
139        * The associated Local Address
140        */
141       final SocketAddress localAddress = networkChannel.localAddress();
142       if (isServer) {
143         final ProxyEntry entry = ProxyEntry.get(localAddress.toString());
144         if (entry == null) {
145           // error
146           // XXX FIXME need to send error !
147           WaarpSslUtility.closingSslChannel(networkChannel);
148           logger.error("No proxy configuration found for: " + localAddress);
149           return;
150         }
151         bridge = new ProxyBridge(entry, this);
152         bridge.initializeProxy();
153         if (!bridge.waitForRemoteConnection()) {
154           logger.error("No connection for proxy: " + localAddress);
155           WaarpSslUtility.closingSslChannel(networkChannel);
156           return;
157         }
158         proxyChannel = bridge.getProxified().networkChannel;
159         logger.warn("Connected: " + isServer + ' ' + (bridge != null?
160             bridge.getProxyEntry() + " proxyChannelId: " + proxyChannel.id() :
161             "nobridge"));
162       } else {
163         clientFuture.awaitOrInterruptible(configuration.getTimeoutCon());
164         if (bridge == null || !clientFuture.isDone()) {
165           logger.error("No connection for proxy: " + localAddress);
166           WaarpSslUtility.closingSslChannel(networkChannel);
167           return;
168         }
169         bridge.remoteConnected();
170       }
171       logger.debug("Network Channel Connected: {} ", ctx.channel().id());
172     } finally {
173       ctx.read();
174     }
175   }

177   @Override
178   public void userEventTriggered(final ChannelHandlerContext ctx,
179                                  final Object evt) throws Exception {
180     if (configuration.isShutdown()) {
181       return;
182     }
183     if (evt instanceof IdleStateEvent) {
184       if (keepAlivedSent.get() > 0) {
185         if (keepAlivedSent.get() < 5) {
186           // ignore this time
187           keepAlivedSent.getAndIncrement();
188           return;
189         }
190         logger.error("Not getting KAlive: closing channel");
191         if (configuration.getR66Mib() != null) {
192           configuration.getR66Mib().notifyWarning("KeepAlive get no answer",
193                                                   "Closing network connection");
194         }
195         ChannelCloseTimer.closeFutureChannel(ctx.channel());
196       } else {
197         keepAlivedSent.set(1);
198         final KeepAlivePacket keepAlivePacket = new KeepAlivePacket();
199         final NetworkPacket response =
200             new NetworkPacket(ChannelUtils.NOCHANNEL, ChannelUtils.NOCHANNEL,
201                               keepAlivePacket, null);
202         logger.info("Write KAlive");
203         ctx.channel().writeAndFlush(response);
204       }
205     }
206   }
207 
208   public void resetKeepAlive() {
209     keepAlivedSent.set(0);
210   }
211 
212   @Override
213   public void channelRead0(final ChannelHandlerContext ctx,
214                            final NetworkPacket msg) {
215     try {
216       if (msg.getCode() == LocalPacketFactory.NOOPPACKET) {
217         resetKeepAlive();
218         // Will forward
219       } else if (msg.getCode() == LocalPacketFactory.CONNECTERRORPACKET) {
220         logger.debug("NetworkRecv: {}", msg);
221         // Special code to STOP here
222         if (msg.getLocalId() == ChannelUtils.NOCHANNEL) {
223           // No way to know what is wrong: close all connections with
224           // remote host
225           logger.error(
226               "Will close NETWORK channel, Cannot continue connection with remote Host: " +
227               msg + " : " + ctx.channel().remoteAddress());
228           WaarpSslUtility.closingSslChannel(ctx.channel());
229           msg.clear();
230           return;
231         }
232       } else if (msg.getCode() == LocalPacketFactory.KEEPALIVEPACKET) {
233         resetKeepAlive();
234         try {
235           final KeepAlivePacket keepAlivePacket =
236               (KeepAlivePacket) LocalPacketCodec
237                   .decodeNetworkPacket(msg.getBuffer());
238           if (keepAlivePacket.isToValidate()) {
239             keepAlivePacket.validate();
240             final NetworkPacket response =
241                 new NetworkPacket(ChannelUtils.NOCHANNEL,
242                                   ChannelUtils.NOCHANNEL, keepAlivePacket,
243                                   null);
244             logger.info("Answer KAlive");
245             ctx.channel().writeAndFlush(response);
246           } else {
247             logger.info("Get KAlive");
248           }
249         } catch (final OpenR66ProtocolPacketException ignored) {
250           // nothing
251         }
252         msg.clear();
253         return;
254       }
255       // forward message
256       resetKeepAlive();
257       if (proxyChannel != null) {
258         proxyChannel.writeAndFlush(msg);
259       } else {
260         msg.clear();
261       }
262     } finally {
263       ctx.read();
264     }
265   }

267   @Override
268   public void exceptionCaught(final ChannelHandlerContext ctx,
269                               final Throwable cause) {
270     final Channel channel = ctx.channel();
271     logger.debug("Network Channel Exception: {}", channel.id(), cause);
272     if (cause instanceof ReadTimeoutException) {
273       final ReadTimeoutException exception = (ReadTimeoutException) cause;
274       // No read for too long
275       logger.error("ReadTimeout so Will close NETWORK channel {}",
276                    exception.getClass().getName() + " : " +
277                    exception.getMessage());
278       ChannelCloseTimer.closeFutureChannel(channel);
279       return;
280     }
281     if (cause instanceof BindException) {
282       // received when not yet connected
283       logger.debug("BindException");
284       ChannelCloseTimer.closeFutureChannel(channel);
285       return;
286     }
287     final OpenR66Exception exception = OpenR66ExceptionTrappedFactory
288         .getExceptionFromTrappedException(channel, cause);
289     if (exception != null) {
290       if (exception instanceof OpenR66ProtocolBusinessNoWriteBackException) {
291         logger.debug("Will close NETWORK channel");
292         ChannelCloseTimer.closeFutureChannel(channel);
293         return;
294       } else if (exception instanceof OpenR66ProtocolNoConnectionException) {
295         logger.info("Connection impossible with NETWORK channel {}",
296                     exception.getMessage());
297         channel.close();
298         return;
299       } else {
300         logger.info("Network Channel Exception: {} {}", channel.id(),
301                     exception.getMessage());
302       }
303       final ConnectionErrorPacket errorPacket =
304           new ConnectionErrorPacket(exception.getMessage(), null);
305       writeError(channel, ChannelUtils.NOCHANNEL, ChannelUtils.NOCHANNEL,
306                  errorPacket);
307       if (proxyChannel != null) {
308         writeError(proxyChannel, ChannelUtils.NOCHANNEL, ChannelUtils.NOCHANNEL,
309                    errorPacket);
310       }
311       logger.debug("Will close NETWORK channel: {}", exception.getMessage());
312       ChannelCloseTimer.closeFutureChannel(channel);
313     } else {
314       // Nothing to do
315     }
316   }

318   /**
319    * Write error back to remote client
320    *
321    * @param channel
322    * @param remoteId
323    * @param localId
324    * @param error
325    */
326   void writeError(final Channel channel, final Integer remoteId,
327                   final Integer localId, final AbstractLocalPacket error) {
328     try {
329       if (channel.isActive()) {
330         NetworkPacket networkPacket = null;
331         try {
332           networkPacket = new NetworkPacket(localId, remoteId, error, null);
333         } catch (final OpenR66ProtocolPacketException ignored) {
334           // nothing
335         }
336         if (networkPacket != null) {
337           final NetworkPacket finalNP = networkPacket;
338           final ChannelFuture future = channel.writeAndFlush(networkPacket);
339           future.await(ConfigurationProxyR66.WAITFORNETOP);
340           future.addListener(new ChannelFutureListener() {
341             @Override
342             public void operationComplete(final ChannelFuture future) {
343               finalNP.clear();
344             }
345           });
346         }
347       }
348     } catch (final InterruptedException e) {//NOSONAR
349       SysErrLogger.FAKE_LOGGER.ignoreLog(e);
350     }
351   }
