62    private enum REQUEST {
63      Logon("Logon.html"), index("index.html"), error("error.html"),
64      System("System.html");

66      private final String header;

68      /**
69       * Constructor for a unique file
70       *
71       * @param uniquefile
72       */
73      REQUEST(final String uniquefile) {
74        header = uniquefile;

88    private String readFileHeaderInternal(final String filename) {
89      final String value;
90      try {
91        value = WaarpStringUtils.readFileException(filename);
92      } catch (final FileTransferException e) {
93        logger.error("Error while trying to read: " + filename, e);
94        return "";
95      }
96      final StringBuilder builder = new StringBuilder(value);
97      WaarpStringUtils.replace(builder, REPLACEMENT.XXXLOCALXXX.toString(),
98                               configuration.getLocalTransaction()
99                                            .getNumberLocalChannel() + " " +
100                              Thread.activeCount());
101     WaarpStringUtils.replace(builder, REPLACEMENT.XXXNETWORKXXX.toString(),
102                              Integer.toString(
103                                  configuration.getLocalTransaction()
104                                               .getNumberLocalChannel()));
105     WaarpStringUtils.replace(builder, REPLACEMENT.XXXHOSTIDXXX.toString(),
106                              configuration.getHostId());
107     if (authentHttp.isAuthenticated()) {
108       WaarpStringUtils.replace(builder, REPLACEMENT.XXXADMINXXX.toString(),
109                                Messages.getString(
110                                    "HttpSslHandler.1")); //$NON-NLS-1$
111     } else {
112       WaarpStringUtils.replace(builder, REPLACEMENT.XXXADMINXXX.toString(),
113                                Messages.getString(
114                                    "HttpSslHandler.0")); //$NON-NLS-1$
115     }
116     final TrafficCounter trafficCounter =
117         configuration.getGlobalTrafficShapingHandler().trafficCounter();
118     WaarpStringUtils.replace(builder, REPLACEMENT.XXXBANDWIDTHXXX.toString(),
119                              Messages.getString("HttpSslHandler.IN") +
120                              (trafficCounter.lastReadThroughput() >> 20) +
121                              //$NON-NLS-1$
122                              Messages.getString("HttpSslHandler.MOPS") +
123                              //$NON-NLS-1$
124                              Messages.getString("HttpSslHandler.OUT") +
125                              //$NON-NLS-1$
126                              (trafficCounter.lastWriteThroughput() >> 20) +
127                              Messages.getString(
128                                  "HttpSslHandler.MOPS")); //$NON-NLS-1$
129     WaarpStringUtils.replace(builder, REPLACEMENT.XXXLANGXXX.toString(), lang);
130     return builder.toString();
131   }

133   private String indexProxy() {
134     final String index = REQUEST.index.readFileUnique(this);
135     final StringBuilder builder = new StringBuilder(index);
136     WaarpStringUtils.replaceAll(builder, REPLACEMENT.XXXHOSTIDXXX.toString(),
137                                 configuration.getHostId());

148   private void replaceStringSystem(final StringBuilder builder) {
149     WaarpStringUtils
150         .replace(builder, REPLACEMENT.XXXXSESSIONLIMITWXXX.toString(),
151                  Long.toString(configuration.getServerChannelWriteLimit()));
152     WaarpStringUtils
153         .replace(builder, REPLACEMENT.XXXXSESSIONLIMITRXXX.toString(),
154                  Long.toString(configuration.getServerChannelReadLimit()));
155     WaarpStringUtils.replace(builder, REPLACEMENT.XXXXDELAYCOMMDXXX.toString(),
156                              Long.toString(configuration.getDelayCommander()));
157     WaarpStringUtils.replace(builder, REPLACEMENT.XXXXDELAYRETRYXXX.toString(),
158                              Long.toString(configuration.getDelayRetry()));
159     WaarpStringUtils
160         .replace(builder, REPLACEMENT.XXXXCHANNELLIMITWXXX.toString(),
161                  Long.toString(configuration.getServerGlobalWriteLimit()));
162     WaarpStringUtils
163         .replace(builder, REPLACEMENT.XXXXCHANNELLIMITRXXX.toString(),
164                  Long.toString(configuration.getServerGlobalReadLimit()));
165     WaarpStringUtils.replace(builder, "XXXBLOCKXXX",
166                              configuration.isShutdown()? CHECKED2 : "");
167     switch (WaarpLoggerFactory.getLogLevel()) {
168       case DEBUG:
169         WaarpStringUtils.replace(builder, XXXLEVEL1XXX2, CHECKED2);
170         WaarpStringUtils.replace(builder, XXXLEVEL2XXX2, "");
171         WaarpStringUtils.replace(builder, XXXLEVEL3XXX2, "");
172         WaarpStringUtils.replace(builder, XXXLEVEL4XXX2, "");
173         break;
174       case INFO:
175         WaarpStringUtils.replace(builder, XXXLEVEL1XXX2, "");
176         WaarpStringUtils.replace(builder, XXXLEVEL2XXX2, CHECKED2);
177         WaarpStringUtils.replace(builder, XXXLEVEL3XXX2, "");
178         WaarpStringUtils.replace(builder, XXXLEVEL4XXX2, "");
179         break;
180       case WARN:
181         WaarpStringUtils.replace(builder, XXXLEVEL1XXX2, "");
182         WaarpStringUtils.replace(builder, XXXLEVEL2XXX2, "");
183         WaarpStringUtils.replace(builder, XXXLEVEL3XXX2, CHECKED2);
184         WaarpStringUtils.replace(builder, XXXLEVEL4XXX2, "");
185         break;
186       case ERROR:
187         WaarpStringUtils.replace(builder, XXXLEVEL1XXX2, "");
188         WaarpStringUtils.replace(builder, XXXLEVEL2XXX2, "");
189         WaarpStringUtils.replace(builder, XXXLEVEL3XXX2, "");
190         WaarpStringUtils.replace(builder, XXXLEVEL4XXX2, CHECKED2);
191         break;
192       default:
193         WaarpStringUtils.replace(builder, XXXLEVEL1XXX2, "");
194         WaarpStringUtils.replace(builder, XXXLEVEL2XXX2, "");
195         WaarpStringUtils.replace(builder, XXXLEVEL3XXX2, "");
196         WaarpStringUtils.replace(builder, XXXLEVEL4XXX2, "");
197         break;
198 
199     }

202   private String System() {
203     getParamsProxy();
204     if (params == null) {
205       final String system = REQUEST.System.readFileUnique(this);
206       final StringBuilder builder = new StringBuilder(system);
207       replaceStringSystem(builder);
208       langHandle(builder);
209       return builder.toString();
210     }
211     String extraInformation = null;
212     if (params.containsKey("ACTION")) {
213       final List<String> action = params.get("ACTION");
214       for (final String act : action) {
215         if ("Language".equalsIgnoreCase(act)) {
216           lang = getTrimValue("change");
217           final String sys = getTrimValue("changesys");
218           Messages.init(new Locale(sys));
219           extraInformation =
220               Messages.getString("HttpSslHandler.LangIs") + "Web: " + lang +
221               " OpenR66: " //$NON-NLS-1$
222               + Messages.getSlocale();
223         } else if ("Level".equalsIgnoreCase(act)) {
224           final String loglevel = getTrimValue("loglevel");
225           WaarpLogLevel level = WaarpLogLevel.WARN;
226           if ("debug".equalsIgnoreCase(loglevel)) {
227             level = WaarpLogLevel.DEBUG;
228           } else if ("info".equalsIgnoreCase(loglevel)) {
229             level = WaarpLogLevel.INFO;
230           } else if ("warn".equalsIgnoreCase(loglevel)) {
231             level = WaarpLogLevel.WARN;
232           } else if ("error".equalsIgnoreCase(loglevel)) {
233             level = WaarpLogLevel.ERROR;
234           }
235           WaarpLoggerFactory.setLogLevel(level);
236           extraInformation = Messages.getString("HttpSslHandler.LangIs") +
237                              level.name(); //$NON-NLS-1$
238         } else if ("Disconnect".equalsIgnoreCase(act)) {
239           String logon = logon();
240           logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(),
241                                    Messages
242                                        .getString("HttpSslHandler.DisActive"));
243           newSession = true;
244           clearSession();
245           forceClose = true;
246           return logon;
247         } else if ("Shutdown".equalsIgnoreCase(act)) {
248           final String error;
249           if (configuration.getShutdownConfiguration().serviceFuture != null) {
250             error =
251                 error(Messages.getString("HttpSslHandler.38")); //$NON-NLS-1$
252           } else {
253             error =
254                 error(Messages.getString("HttpSslHandler.37")); //$NON-NLS-1$
255           }
256           WaarpShutdownHook.setRestart(false);
257           newSession = true;
258           clearSession();
259           forceClose = true;
260           shutdown = true;
261           return error;
262         } else if ("Restart".equalsIgnoreCase(act)) {
263           String error;
264           if (configuration.getShutdownConfiguration().serviceFuture != null) {
265             error =
266                 error(Messages.getString("HttpSslHandler.38")); //$NON-NLS-1$
267           } else {
268             error = error(Messages.getString("HttpSslHandler.39")
269                           //$NON-NLS-1$
270                           + configuration.getTimeoutCon() * 2 / 1000 + Messages
271                               .getString("HttpSslHandler.40")); //$NON-NLS-1$
272           }
273           error = error.replace("XXXRELOADHTTPXXX",
274                                 "HTTP-EQUIV=\"refresh\" CONTENT=\"" +
275                                 configuration.getTimeoutCon() * 2 / 1000 + '"');
276           WaarpShutdownHook.setRestart(true);
277           newSession = true;
278           clearSession();
279           forceClose = true;
280           shutdown = true;
281           return error;
282         } else if ("Validate".equalsIgnoreCase(act)) {
283           final String bsessionr = getTrimValue("BSESSR");
284           long lsessionr = configuration.getServerChannelReadLimit();
285           long lglobalr;
286           long lsessionw;
287           long lglobalw;
288           try {
289             if (bsessionr != null) {
290               lsessionr = (Long.parseLong(bsessionr) / 10) * 10;
291             }
292             final String bglobalr = getTrimValue("BGLOBR");
293             lglobalr = configuration.getServerGlobalReadLimit();
294             if (bglobalr != null) {
295               lglobalr = (Long.parseLong(bglobalr) / 10) * 10;
296             }
297             final String bsessionw = getTrimValue("BSESSW");
298             lsessionw = configuration.getServerChannelWriteLimit();
299             if (bsessionw != null) {
300               lsessionw = (Long.parseLong(bsessionw) / 10) * 10;
301             }
302             final String bglobalw = getTrimValue("BGLOBW");
303             lglobalw = configuration.getServerGlobalWriteLimit();
304             if (bglobalw != null) {
305               lglobalw = (Long.parseLong(bglobalw) / 10) * 10;
306             }
307             configuration
308                 .changeNetworkLimit(lglobalw, lglobalr, lsessionw, lsessionr,
309                                     configuration.getDelayLimit());
310             final String dcomm = getTrimValue("DCOM");
311             if (dcomm != null) {
312               configuration.setDelayCommander(Long.parseLong(dcomm));
313               if (configuration.getDelayCommander() <= 100) {
314                 configuration.setDelayCommander(100);
315               }
316               configuration.reloadCommanderDelay();
317             }
318             final String dret = getTrimValue("DRET");
319             if (dret != null) {
320               configuration.setDelayRetry(Long.parseLong(dret));
321               if (configuration.getDelayRetry() <= 1000) {
322                 configuration.setDelayRetry(1000);
323               }
324             }
325             extraInformation =
326                 Messages.getString("HttpSslHandler.41"); //$NON-NLS-1$
327           } catch (final NumberFormatException e) {
328             extraInformation =
329                 Messages.getString("HttpSslHandler.42"); //$NON-NLS-1$
330           }
331         }
332       }
333     }
334     final String system = REQUEST.System.readFileUnique(this);
335     final StringBuilder builder = new StringBuilder(system);
336     replaceStringSystem(builder);
337     langHandle(builder);
338     if (extraInformation != null) {
339       builder.append(extraInformation);
340     }
341     return builder.toString();
342   }

344   private void getParamsProxy() {
345     if (request.method() == HttpMethod.GET) {
346       params = null;
347     } else if (request.method() == HttpMethod.POST) {
348       final ByteBuf content = request.content();
349       if (content.isReadable()) {
350         final String param = content.toString(WaarpStringUtils.UTF8);
351         final QueryStringDecoder queryStringDecoder2 =
352             new QueryStringDecoder("/?" + param);
353         params = queryStringDecoder2.parameters();
354       } else {
355         params = null;
356       }
357     }
358   }

364       admin = null;

372   private void checkAuthentProxy(final ChannelHandlerContext ctx) {
373     newSession = true;
374     if (request.method() == HttpMethod.GET) {
375       String logon = logon();
376       logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(), "");
377       responseContent.append(logon);
378       clearSession();
379       writeResponse(ctx);
380       return;
381     } else if (request.method() == HttpMethod.POST) {
382       getParamsProxy();
383       if (params == null) {
384         String logon = logon();
385         logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(),
386                                  Messages
387                                      .getString("HttpSslHandler.EmptyLogin"));
388         responseContent.append(logon);
389         clearSession();
390         writeResponse(ctx);
391         return;
392       }
393     }
394     boolean getMenu = false;
395     if (params != null && params.containsKey("Logon")) {
396       String name = null;
397       String password = null;
398       List<String> values;
399       if (!params.isEmpty()) {
400         // get values
401         if (params.containsKey("name")) {
402           values = params.get("name");
403           if (values != null) {
404             name = values.get(0);
405             if (name == null || name.isEmpty()) {
406               getMenu = true;
407             }
408           }
409         } else {
410           getMenu = true;
411         }
412         // search the nb param
413         if (!getMenu && params.containsKey("passwd")) {
414           values = params.get("passwd");
415           if (values != null) {
416             password = values.get(0);
417             getMenu = password == null || password.isEmpty();
418           } else {
419             getMenu = true;
420           }
421         } else {
422           getMenu = true;
423         }
424       } else {
425         getMenu = true;
426       }
427       if (!getMenu && name != null) {
428         if (logger.isDebugEnabled()) {
429           logger.debug("Name={} vs {} Passwd vs ", name,
430                        name.equals(configuration.getAdminName()), Arrays
431                            .equals(password.getBytes(WaarpStringUtils.UTF8),
432                                    configuration.getServerAdminKey()));
433         }
434         if (name.equals(configuration.getAdminName()) && Arrays
435             .equals(password.getBytes(WaarpStringUtils.UTF8),
436                     configuration.getServerAdminKey())) {
437           authentHttp.getAuth()
438                      .specialNoSessionAuth(true, configuration.getHostId());
439           authentHttp.setStatus(70);
440         } else {
441           getMenu = true;
442         }
443         if (!authentHttp.isAuthenticated()) {
444           authentHttp.setStatus(71);
445           logger.info("Still not authenticated: {}", authentHttp);
446           getMenu = true;
447         }
448       }
449     } else {
450       getMenu = true;
451     }
452     if (getMenu) {
453       String logon = logon();
454       logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(),
455                                Messages.getString("HttpSslHandler.BadLogin"));
456       responseContent.append(logon);
457       clearSession();
458     } else {
459       final String index = indexProxy();
460       responseContent.append(index);
461       clearSession();
462       admin = new DefaultCookie(R66SESSION + configuration.getHostId(),
463                                 configuration.getHostId() +
464                                 Long.toHexString(RANDOM.nextLong()));
465       sessions.put(admin.value(), authentHttp);
466       authentHttp.setStatus(72);
467       logger.debug("CreateSession: {}:{}", uriRequest, admin);
468     }
469     writeResponse(ctx);
470   }

472   @Override
473   protected void channelRead0(final ChannelHandlerContext ctx,
474                               final FullHttpRequest msg) {
475     final FullHttpRequest request = this.request = msg;
476     final QueryStringDecoder queryStringDecoder =
477         new QueryStringDecoder(request.uri());
478     uriRequest = queryStringDecoder.path();
479     logger.debug("Msg: {}", uriRequest);
480     if (uriRequest.contains("gre/") || uriRequest.contains("img/") ||
481         uriRequest.contains("res/") || uriRequest.contains("favicon.ico")) {
482       HttpWriteCacheEnable
483           .writeFile(request, ctx, configuration.getHttpBasePath() + uriRequest,
484                      R66SESSION + configuration.getHostId());
485       ctx.flush();
486       return;
487     }
488     checkSession(ctx.channel());
489     try {
490       if (!authentHttp.isAuthenticated()) {
491         logger.debug("Not Authent: {}:{}", uriRequest, authentHttp);
492         checkAuthentProxy(ctx);
493         return;
494       }
495       String find = uriRequest;
496       if (uriRequest.charAt(0) == '/') {
497         find = uriRequest.substring(1);
498       }
499       REQUEST req = REQUEST.index;
500       if (find.length() != 0) {
501         find = find.substring(0, find.indexOf('.'));
502         try {
503           req = REQUEST.valueOf(find);
504         } catch (final IllegalArgumentException e1) {
505           req = REQUEST.index;
506           logger.info("NotFound: {}:{}", find, uriRequest);
507         }
508       }
509       if (req == REQUEST.System) {
510         responseContent.append(System());
511       } else {
512         responseContent.append(indexProxy());
513       }
514       writeResponse(ctx);
515     } finally {
516       closeConnection();
517     }
518   }
