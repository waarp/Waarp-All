57    @Test
58    public void test98_Https() throws InterruptedException {
59      //Thread.sleep(1000000);
60      try {
61        // Test name: ProxyAdmin
62        // Step # | name | target | value | comment
63        // 1 | open | / |  |
64        driver.get("https://127.0.0.1:11187/");
65        // 7 | click | name=name |  |
66        driver.findElement(By.name("name")).click();
67        // 8 | type | name=name | monadmin |
68        driver.findElement(By.name("name")).sendKeys("monadmin");
69        // 9 | type | name=passwd | pwdhttp |
70        driver.findElement(By.name("passwd")).sendKeys("pwdhttp");
71        // 10 | sendKeys | name=passwd | ${KEY_ENTER} |
72        driver.findElement(By.name("passwd")).sendKeys(Keys.ENTER);
73        // 11 | click | linkText=SYSTEM |  |
74        driver.get("https://127.0.0.1:11187/System.html");
75        // 12 | click | linkText=START |  |
76        driver.get("https://127.0.0.1:11187/index.html");
77      } catch (NoSuchElementException e) {
78        e.printStackTrace();
79        fail(e.getMessage());
80      } catch (StaleElementReferenceException e) {
81        e.printStackTrace();
82        fail(e.getMessage());
83      } finally {
84        driver.get("https://127.0.0.1:11187/Logout.html");
85      }
86    }
