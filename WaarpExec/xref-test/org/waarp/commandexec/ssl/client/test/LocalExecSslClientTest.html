67   * LocalExecSsl client.
68   * <p>
69   * This class is an example of client.
70   * <p>
71   * No client authentication On a bi-core Centrino2 vPro: 5/s in 50 sequential,
72   * 29/s in 10 threads with 50
73   * sequential<br>
74   * With client authentication On a bi-core Centrino2 vPro: 3/s in 50 sequential,
75   * 27/s in 10 threads with 50
76   * sequential<br>
77   * No client authentication On a quad-core i7: 20/s in 50 sequential, 178/s in
78   * 10 threads with 50
79   * sequential<br>
80   * With client authentication On a quad-core i7: 17/s in 50 sequential, 176/s in
81   * 10 threads with 50
82   * sequential<br>
83   */

117   @Test
118   public void testSslClient() throws Exception {
119     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
120         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
121     ResourceLeakDetector.setLevel(Level.PARANOID);
122     WaarpSystemUtil.setJunit(true);
123     InetAddress addr;
124     final byte[] loop = { 127, 0, 0, 1 };
125     try {
126       addr = InetAddress.getByAddress(loop);
127     } catch (final UnknownHostException e) {
128       return;
129     }
130     address = new InetSocketAddress(addr, port);
131     // Configure the client.
132     bootstrap = new Bootstrap();
133     WaarpNettyUtil.setBootstrap(bootstrap, workerGroup, 1000);
134     // Configure the pipeline factory.
135     // First create the SSL part
136     // Load the KeyStore (No certificates)
137     final ClassLoader classLoader =
138         LocalExecSslClientTest.class.getClassLoader();
139     final String keyStoreFilename = "certs/testsslnocert.jks";
140     final URL url = classLoader.getResource(keyStoreFilename);
141     assertNotNull(url);
142     final File file = new File(url.getFile());
143     assertTrue("File Should exists", file.exists());
144     final String keyStorePasswd = "testsslnocert";
145     final String keyPassword = "testalias";
146     final WaarpSecureKeyStore waarpSecureKeyStore =
147         new WaarpSecureKeyStore(file.getAbsolutePath(), keyStorePasswd,
148                                 keyPassword);
149     final WaarpSecureKeyStore waarpSecureKeyStoreClient =
150         new WaarpSecureKeyStore(file.getAbsolutePath(), keyStorePasswd,
151                                 keyPassword);
152     // Include certificates
153     final String trustStoreFilename = "certs/testcert.jks";
154     final File file2 =
155         new File(classLoader.getResource(trustStoreFilename).getFile());
156     assertTrue("File2 Should exists", file2.exists());
157     final String trustStorePasswd = "testcert";
158     waarpSecureKeyStore.initTrustStore(file2.getAbsolutePath(),
159                                        trustStorePasswd, true);
160 
161     // configure the server
162     final ServerBootstrap bootstrapServer = new ServerBootstrap();
163     WaarpNettyUtil.setServerBootstrap(bootstrapServer, workerGroup, workerGroup,
164                                       1000);
165 
166     // Configure the pipeline factory.
167     final WaarpSslContextFactory waarpSslContextFactoryServer =
168         new WaarpSslContextFactory(waarpSecureKeyStore, true);
169     final LocalExecSslServerInitializer localExecServerInitializer =
170         new LocalExecSslServerInitializer(waarpSslContextFactoryServer,
171                                           LocalExecDefaultResult.MAXWAITPROCESS,
172                                           executor);
173     bootstrapServer.childHandler(localExecServerInitializer);
174 
175     // Bind and start to accept incoming connections only on local address.
176     final ChannelFuture future =
177         bootstrapServer.bind(new InetSocketAddress(addr, port));
178 
179     // Finalize client configuration
180     waarpSecureKeyStoreClient.initTrustStore(file2.getAbsolutePath(),
181                                              trustStorePasswd, false);
182     final WaarpSslContextFactory waarpSslContextFactoryClient =
183         new WaarpSslContextFactory(waarpSecureKeyStoreClient);
184 
185     localExecClientInitializer =
186         new LocalExecSslClientInitializer(waarpSslContextFactoryClient);
187     bootstrap.handler(localExecClientInitializer);
188 
189     // Wait for the server
190     future.sync();
191 
192     try {
193       // Parse options.
194       final LocalExecSslClientTest client = new LocalExecSslClientTest();
195       // run once
196       long first = System.currentTimeMillis();
197       if (client.connect()) {
198         client.runOnce();
199         client.disconnect();
200       }
201       long second = System.currentTimeMillis();
202       // print time for one exec
203       SysErrLogger.FAKE_LOGGER.sysout(
204           "1=Total time in ms: " + (second - first) + " or " +
205           1 * 1000 / (second - first) + " exec/s");
206       SysErrLogger.FAKE_LOGGER.sysout("Result: " + ok + ':' + ko);
207       ok = 0;
208       ko = 0;
209 
210       // Now run multiple within one thread
211       first = System.currentTimeMillis();
212       for (int i = 0; i < nit; i++) {
213         if (client.connect()) {
214           client.runOnce();
215           client.disconnect();
216         }
217       }
218       second = System.currentTimeMillis();
219       // print time for one exec
220       System.err.println(
221           nit + "=Total time in ms: " + (second - first) + " or " +
222           nit * 1000 / (second - first) + " exec/s");
223       System.err.println("Result: " + ok + ':' + ko);
224       ok = 0;
225       ko = 0;
226 
227       // Now run multiple within multiple threads
228       // Create multiple threads
229       final ExecutorService executorService = Executors.newFixedThreadPool(nth);
230       first = System.currentTimeMillis();
231       // Starts all thread with a default number of execution
232       for (int i = 0; i < nth; i++) {
233         executorService.submit(new LocalExecSslClientTest());
234       }
235       Thread.sleep(100);
236       executorService.shutdown();
237       while (!executorService.awaitTermination(200, TimeUnit.MILLISECONDS)) {
238         Thread.sleep(50);
239       }
240       second = System.currentTimeMillis();
241 
242       // print time for one exec
243       System.err.println(
244           nit * nth + "=Total time in ms: " + (second - first) + " or " +
245           nit * nth * 1000 / (second - first) + " exec/s");
246       System.err.println("Result: " + ok + ':' + ko);
247       ok = 0;
248       ko = 0;
249 
250       // run once
251       first = System.currentTimeMillis();
252       if (client.connect()) {
253         client.runFinal();
254         client.disconnect();
255       }
256       second = System.currentTimeMillis();
257       // print time for one exec
258       System.err.println("1=Total time in ms: " + (second - first) + " or " +
259                          1 * 1000 / (second - first) + " exec/s");
260       System.err.println("Result: " + ok + ':' + ko);
261       assertEquals(0, ko);
262       ok = 0;
263       ko = 0;
264     } finally {
265       future.channel().close();
266       // Shut down all thread pools to exit.
267       localExecClientInitializer.releaseResources();
268       localExecServerInitializer.releaseResources();
269       // Shut down all thread pools to exit.
270       workerGroup.shutdownGracefully();
271       executor.shutdownGracefully();
272       localExecClientInitializer.releaseResources();
273       LocalExecServerHandler.junitSetNotShutdown();
274     }
275   }

277   /**
278    * Connect to the Server
279    */
280   private boolean connect() {
281     // Start the connection attempt.
282     final ChannelFuture future = bootstrap.connect(address);
283 
284     // Wait until the connection attempt succeeds or fails.
285     channel = WaarpSslUtility.waitforChannelReady(future);
286     if (channel == null) {
287       System.err.println("Client Not Connected");
288       if (future.cause() != null) {
289         future.cause().printStackTrace();
290       }
291       fail("Cannot connect");
292       return false;
293     }
294     return true;
295   }
296 
297   /**
298    * Run method both for not threaded execution and threaded execution
299    */
300   private void runOnce() {
301     // Initialize the command context
302     final LocalExecSslClientHandler clientHandler =
303         (LocalExecSslClientHandler) channel.pipeline().last();
304     // Command to execute
305     final String line = command + ' ' + atomicInteger.incrementAndGet();
306     clientHandler.initExecClient(0, line);
307     // Wait for the end of the exec command
308     final LocalExecResult localExecResult = clientHandler.waitFor(10000);
309     final int status = localExecResult.getStatus();
310     if (status < 0) {
311       System.err.println(
312           "Status: " + status + "\nResult: " + localExecResult.getResult());
313       ko++;
314     } else {
315       ok++;
316       result = localExecResult;
317     }
318   }
319 
320   /**
321    * Disconnect from the server
322    */
323   private void disconnect() {
324     WaarpSslUtility.closingSslChannel(channel);
325     WaarpSslUtility.waitForClosingSslChannel(channel, 10000);
326   }
327 
328   /**
329    * Run method for closing Server
330    */
331   private void runFinal() {
332     // Initialize the command context
333     final LocalExecSslClientHandler clientHandler =
334         (LocalExecSslClientHandler) channel.pipeline().last();
335     // Command to execute
336     clientHandler.initExecClient(-1000, "stop");
337     // Wait for the end of the exec command
338     final LocalExecResult localExecResult = clientHandler.waitFor(10000);
339     final int status = localExecResult.getStatus();
340     if (status < 0) {
341       System.err.println(
342           "Status: " + status + "\nResult: " + localExecResult.getResult());
343       ok++;
344     } else {
345       ok++;
346       result = localExecResult;
347     }
348   }
349 
350   /**
351    * Run method for thread
352    */
353   @Override
354   public void run() {
355     if (connect()) {
356       for (int i = 0; i < nit; i++) {
357         runOnce();
358       }
359       disconnect();
360     }
