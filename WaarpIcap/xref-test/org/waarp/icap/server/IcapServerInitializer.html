21  package org.waarp.icap.server;
22  
23  import io.netty.buffer.ByteBuf;
24  import io.netty.buffer.Unpooled;
25  import io.netty.channel.Channel;
26  import io.netty.channel.ChannelInitializer;
27  import io.netty.channel.ChannelPipeline;
28  import io.netty.channel.group.ChannelGroup;
29  import io.netty.channel.group.DefaultChannelGroup;
30  import io.netty.channel.socket.SocketChannel;
31  import io.netty.handler.codec.DelimiterBasedFrameDecoder;
32  import io.netty.handler.codec.string.StringDecoder;
33  import io.netty.handler.codec.string.StringEncoder;
34  import io.netty.util.concurrent.EventExecutorGroup;
35  import org.waarp.common.utility.WaarpStringUtils;
36  
37  import static org.waarp.common.file.FileUtils.*;
38  
39  public class IcapServerInitializer extends ChannelInitializer<SocketChannel> {
40    private static final ByteBuf RNRN = Unpooled.wrappedBuffer(new byte[] {
41        '\r', '\n', '\r', '\n'
42    });
43  
44    private long delay;
45    protected final EventExecutorGroup eventExecutorGroup;
46    private final ChannelGroup channelGroup;
47  
48    /**
49     * Constructor with a specific default delay
50     *
51     * @param newdelay
52     * @param eventExecutorGroup
53     */
54    public IcapServerInitializer(long newdelay,
55                                 EventExecutorGroup eventExecutorGroup) {
56      delay = newdelay;
57      this.eventExecutorGroup = eventExecutorGroup;
58      channelGroup =
59          new DefaultChannelGroup("IcapServer", eventExecutorGroup.next());
60    }
61  
62    @Override
63    public void initChannel(SocketChannel ch) throws Exception {
64      // Create a default pipeline implementation.
65      final ChannelPipeline pipeline = ch.pipeline();
66  
67      // Add the text line codec combination first,
68      pipeline.addLast("framer",
69                       new DelimiterBasedFrameDecoder(ZERO_COPY_CHUNK_SIZE, false,
70                                                      RNRN));
71      pipeline.addLast(eventExecutorGroup, "decoder",
72                       new StringDecoder(WaarpStringUtils.UTF8));
73      pipeline.addLast(eventExecutorGroup, "encoder",
74                       new StringEncoder(WaarpStringUtils.UTF8));
75  
76      // and then business logic.
77      // Could change it with a new fixed delay if necessary at construction
78      pipeline.addLast(eventExecutorGroup, "icap",
79                       new IcapServerHandler(this, delay));
80    }
81  
82    /**
83     * Add a channel to the channel Group
84     *
85     * @param channel
86     */
87    public void addChannel(Channel channel) {
88      channelGroup.add(channel);
89    }
90  
91    /**
92     * Release internal resources
93     */
94    public void releaseResources() {
95      channelGroup.close();
96      eventExecutorGroup.shutdownGracefully();
97    }
98  }
