145   @Test
146   public void testConnect() throws Exception {
147     IcapClient icapClient = new IcapClient("127.0.0.1", 1234, "avscan");
148     assertNull(icapClient.getSubStringFromKeyIcap200());
149     assertNull(icapClient.getKeyIcap200());
150     assertNull(icapClient.getSubstringHttpStatus200());
151     assertNull(icapClient.getKeyIcap204());
152     assertNull(icapClient.getSubStringFromKeyIcap204());
153     assertEquals(0, icapClient.getPreviewSize());
154     assertEquals(Integer.MAX_VALUE, icapClient.getMaxSize());
155     assertEquals(8192, icapClient.getSendLength());
156     assertEquals(64 * 1024, icapClient.getReceiveLength());
157     icapClient.close();
158     try {
159       icapClient.connect();
160       fail("Should failed");
161     } catch (IcapException e) {
162       assertTrue(e.getError() == IcapError.ICAP_CANT_CONNECT);
163     }
164     icapClient.close();
165     logger.warn("Disconnection while not connected with no impact");
166     IcapServer.main(new String[] { "127.0.0.1", "9999" });
167     logger.warn("Server started");
168     Thread.sleep(100);
169     icapClient = new IcapClient("127.0.0.1", 9999, "avscan", 2048);
170     try {
171       icapClient.connect();
172       logger.warn("Connection OK");
173       assertTrue(icapClient.getPreviewSize() == 2048);
174       assertNull(icapClient.getFinalResult());
175       icapClient.close();
176       logger.warn("Disconnection OK");
177       icapClient.connect();
178       logger.warn("ReConnection OK");
179       icapClient.connect();
180       logger.warn("ReConnection without disconnection OK");
181       icapClient.close();
182       logger.warn("Disconnection OK");
183     } catch (IcapException e) {
184       logger.error(e);
185       fail("Should failed");
186     }
187     icapClient = new IcapClient("127.0.0.1", 9999, "avscan");
188     icapClient.close();
189     logger.warn("Disconnection while not connected with no impact");
190     try {
191       icapClient.connect();
192       logger.warn("Connection OK");
193       assertTrue(icapClient.getPreviewSize() == 4096);
194       assertTrue(icapClient.getFinalResult().containsKey("Preview"));
195       icapClient.close();
196       assertNotNull(icapClient.getFinalResult());
197       logger.warn("Disconnection OK");
198       icapClient.close();
199       logger.warn("Disconnection repeated with no impact");
200       icapClient.connect().connect().close();
201       logger.warn("ReConnection without disconnection OK");
202     } catch (IcapException e) {
203       logger.error(e);
204       fail("Should failed");
205     }
206     logger.warn("No preview");
207     IcapServerHandler.setIsPreviewOkTest(false);
208     IcapServerHandler.setFinalStatus(200);
209     try {
210       icapClient.connect();
211       fail("Should failed");
212     } catch (IcapException e) {
213       assertTrue(e.getError() == IcapError.ICAP_SERVER_MISSING_INFO);
214     }
215     IcapServerHandler.setFinalStatus(500);
216     logger.warn("OPTIONS error");
217     try {
218       icapClient.connect();
219       fail("Should failed");
220     } catch (IcapException e) {
221       assertTrue(e.getError() == IcapError.ICAP_SERVER_MISSING_INFO);
222     }
223   }

225   private void createFile(File file, int len) throws IOException {
226     file.delete();
227     byte[] from = new byte[len];
228     Arrays.fill(from, (byte) 'A');
229     FileOutputStream outputStream = new FileOutputStream(file);
230     outputStream.write(from);
231     outputStream.flush();
232     outputStream.close();
233   }
