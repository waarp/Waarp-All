56                                      "* -clear to specify uncrypted password shown as clear text";

72    /**
73     * @param args
74     *
75     * @throws Exception
76     */
77    public static void main(String[] args) throws Exception {
78      if (!loadOptions(args)) {
79        // Bad options
80        if (DetectionUtils.isJunit()) {
81          return;
82        }
83        System.exit(2);//NOSONAR
84      }
85      final WaarpPasswordd">WaarpPassword waarpPassword = new WaarpPassword();
86      if (po == null && pi == null) {
87        // stop
88        SysErrLogger.FAKE_LOGGER.syserr("Key written");
89        if (DetectionUtils.isJunit()) {
90          return;
91        }
92        System.exit(0);//NOSONAR
93      }
94      if (waarpPassword.clearPassword == null ||
95          waarpPassword.clearPassword.length() == 0) {
96        SysErrLogger.FAKE_LOGGER.syserr("Password to crypt:");
97        final String newp = waarpPassword.readString();
98        if (newp == null || newp.length() == 0) {
99          SysErrLogger.FAKE_LOGGER.syserr("No password as input");
100         if (DetectionUtils.isJunit()) {
101           return;
102         }
103         System.exit(4);//NOSONAR
104       }
105       waarpPassword.setClearPassword(newp);
106       if (po != null) {
107         waarpPassword.setPasswordFile(new File(po));
108         waarpPassword.savePasswordFile();
109       }
110       if (clearPasswordView) {
111         SysErrLogger.FAKE_LOGGER
112             .syserr("ClearPwd: " + waarpPassword.getClearPassword());
113         SysErrLogger.FAKE_LOGGER
114             .syserr("CryptedPwd: " + waarpPassword.getCryptedPassword());
115       }
116     }
117   }

119   public static boolean loadOptions(String[] args) {
120     desModel = true;
121     clearPasswordView = false;
122     ki = null;
123     ko = null;
124     pi = null;
125     po = null;
126     pwd = null;
127     cpwd = null;
128 
129     int i;
130     if (args.length == 0) {
131       SysErrLogger.FAKE_LOGGER.syserr(HELPOPTIONS);
132       return false;
133     }
134     if (!SystemPropertyUtil.isFileEncodingCorrect()) {
135       SysErrLogger.FAKE_LOGGER.syserr(
136           "Issue while trying to set UTF-8 as default file encoding: use -Dfile.encoding=UTF-8 as java command argument\n" +
137           "Currently file.encoding is: " +
138           SystemPropertyUtil.get(SystemPropertyUtil.FILE_ENCODING));
139       return false;
140     }
141     for (i = 0; i < args.length; i++) {
142       if ("-ki".equalsIgnoreCase(args[i])) {
143         i++;
144         if (i < args.length) {
145           ki = args[i];
146         } else {
147           SysErrLogger.FAKE_LOGGER.syserr("-ki needs a file as argument");
148           return false;
149         }
150       } else if ("-ko".equalsIgnoreCase(args[i])) {
151         i++;
152         if (i < args.length) {
153           ko = args[i];
154         } else {
155           SysErrLogger.FAKE_LOGGER.syserr("-ko needs a file as argument");
156           return false;
157         }
158       } else if ("-pi".equalsIgnoreCase(args[i])) {
159         i++;
160         if (i < args.length) {
161           pi = args[i];
162         } else {
163           SysErrLogger.FAKE_LOGGER.syserr("-pi needs a file as argument");
164           return false;
165         }
166       } else if ("-po".equalsIgnoreCase(args[i])) {
167         i++;
168         if (i < args.length) {
169           po = args[i];
170         } else {
171           SysErrLogger.FAKE_LOGGER.syserr("-po needs a file as argument");
172           return false;
173         }
174       } else if ("-des".equalsIgnoreCase(args[i])) {
175         desModel = true;
176       } else if ("-blf".equalsIgnoreCase(args[i])) {
177         desModel = false;
178       } else if ("-pwd".equalsIgnoreCase(args[i])) {
179         i++;
180         if (i < args.length) {
181           pwd = args[i];
182         } else {
183           SysErrLogger.FAKE_LOGGER.syserr("-pwd needs a password as argument");
184           return false;
185         }
186       } else if ("-cpwd".equalsIgnoreCase(args[i])) {
187         i++;
188         if (i < args.length) {
189           cpwd = args[i];
190         } else {
191           SysErrLogger.FAKE_LOGGER
192               .syserr("-cpwd needs a crypted password as argument");
193           return false;
194         }
195       } else if ("-clear".equalsIgnoreCase(args[i])) {
196         clearPasswordView = true;
197       } else {
198         SysErrLogger.FAKE_LOGGER.syserr("Unknown option: " + args[i]);
199         return false;
200       }
201     }
202     if (ki == null && ko == null) {
203       SysErrLogger.FAKE_LOGGER
204           .syserr("You must specify one of ki or ko options");
205       return false;
206     }
207     if (ki == null) {
208       ki = ko;
209     }
210     if (ki == null && (po != null || pi != null)) {
211       SysErrLogger.FAKE_LOGGER.syserr(
212           "If pi or po options are set, ki or ko options must be set also!\n");
213       return false;
214     }
215     if (pi == null && po == null && (pwd != null || cpwd != null)) {
216       SysErrLogger.FAKE_LOGGER.syserr(
217           "Cannot create a password if no password GGP file is specified with pi or po options");
218       return false;
219     }
220     return true;
221   }

223   public WaarpPassword() throws Exception {
224     if (desModel) {
225       currentKey = new Des();
226     } else {
227       currentKey = new Blowfish();
228     }
229     if (ko != null) {
230       createNewKey();
231       saveKey(new File(ko));
232     }
233     if (ki != null) {
234       loadKey(new File(ki));
235     }
236     if (pi != null) {
237       setPasswordFile(new File(pi));
238       loadPasswordFile();
239     }
240     if (pwd != null) {
241       setClearPassword(pwd);
242     }
243     if (cpwd != null) {
244       setCryptedPassword(cpwd);
245     }
246     if (po != null) {
247       setPasswordFile(new File(po));
248       savePasswordFile();
249     }
250     if (clearPassword != null) {
251       if (clearPasswordView) {
252         SysErrLogger.FAKE_LOGGER.syserr("ClearPwd: " + getClearPassword());
253       }
254       SysErrLogger.FAKE_LOGGER.syserr("CryptedPwd: " + getCryptedPassword());
255     }
256   }

258   private String readString() {
259     String read = "";
260     final InputStreamReader input =
261         new InputStreamReader(System.in, WaarpStringUtils.UTF8);
262     final BufferedReader reader = new BufferedReader(input);
263     try {
264       read = reader.readLine();
265     } catch (final IOException e) {
266       SysErrLogger.FAKE_LOGGER.syserr(e);
267     }
268     return read;
269   }

271   /**
272    * Create a new Key but do not save it on file
273    *
274    * @throws Exception
275    */
276   public void createNewKey() throws Exception {
277     try {
278       currentKey.generateKey();
279     } catch (final Exception e) {
280       throw new CryptoException("Create New Key in error", e);
281     }
282     if (clearPassword != null) {
283       setClearPassword(clearPassword);
284     }
285   }

287   /**
288    * @param file source file
289    *
290    * @throws CryptoException
291    */
292   public void loadKey(File file) throws CryptoException {
293     keyFile = file;
294     try {
295       currentKey.setSecretKey(file);
296     } catch (final IOException e) {
297       throw new CryptoException("Load Key in error", e);
298     }
299   }
300 
301   /**
302    * @param file destination file, if null previously set file is used
303    *
304    * @throws CryptoException
305    */
306   public void saveKey(File file) throws CryptoException {
307     if (file != null) {
308       keyFile = file;
309     }
310     try {
311       currentKey.saveSecretKey(keyFile);
312     } catch (final IOException e) {
313       throw new CryptoException("Save Key in error", e);
314     }
315   }

359   /**
360    * Save the Crypted Paswword to the File
361    *
362    * @throws IOException
363    */
364   public void savePasswordFile() throws IOException {
365     final FileOutputStream outputStream = new FileOutputStream(passwordFile);
366     try {
367       outputStream.write(cryptedPassword.getBytes(WaarpStringUtils.UTF8));
368     } finally {
369       FileUtils.close(outputStream);
370     }
371   }

373   /**
374    * Load the crypted password from the file
375    *
376    * @throws Exception
377    */
378   public void loadPasswordFile() throws Exception {
379     if (passwordFile.canRead()) {
380       final int len = (int) passwordFile.length();
381       final byte[] key = new byte[len];
382       FileInputStream inputStream;
383       inputStream = new FileInputStream(passwordFile);
384       DataInputStream dis = null;
385       try {
386         dis = new DataInputStream(inputStream);
387         dis.readFully(key);
388       } finally {
389         if (dis != null) {
390           FileUtils.close(dis);
391         } else {
392           FileUtils.close(inputStream);
393         }
394       }
395       setCryptedPassword(new String(key, WaarpStringUtils.UTF8));
396     } else {
397       throw new CryptoException("Cannot read crypto file");
398     }
399   }
