175   private void decode(final byte[] buffer, final int shift, final int[] out) {
176     // unrolled loop (original loop shown above)
177     out[0] = buffer[shift] & 0xff | (buffer[shift + 1] & 0xff) << 8 |
178              (buffer[shift + 2] & 0xff) << 16 | buffer[shift + 3] << 24;
179     out[1] = buffer[shift + 4] & 0xff | (buffer[shift + 5] & 0xff) << 8 |
180              (buffer[shift + 6] & 0xff) << 16 | buffer[shift + 7] << 24;
181     out[2] = buffer[shift + 8] & 0xff | (buffer[shift + 9] & 0xff) << 8 |
182              (buffer[shift + 10] & 0xff) << 16 | buffer[shift + 11] << 24;
183     out[3] = buffer[shift + 12] & 0xff | (buffer[shift + 13] & 0xff) << 8 |
184              (buffer[shift + 14] & 0xff) << 16 | buffer[shift + 15] << 24;
185     out[4] = buffer[shift + 16] & 0xff | (buffer[shift + 17] & 0xff) << 8 |
186              (buffer[shift + 18] & 0xff) << 16 | buffer[shift + 19] << 24;
187     out[5] = buffer[shift + 20] & 0xff | (buffer[shift + 21] & 0xff) << 8 |
188              (buffer[shift + 22] & 0xff) << 16 | buffer[shift + 23] << 24;
189     out[6] = buffer[shift + 24] & 0xff | (buffer[shift + 25] & 0xff) << 8 |
190              (buffer[shift + 26] & 0xff) << 16 | buffer[shift + 27] << 24;
191     out[7] = buffer[shift + 28] & 0xff | (buffer[shift + 29] & 0xff) << 8 |
192              (buffer[shift + 30] & 0xff) << 16 | buffer[shift + 31] << 24;
193     out[8] = buffer[shift + 32] & 0xff | (buffer[shift + 33] & 0xff) << 8 |
194              (buffer[shift + 34] & 0xff) << 16 | buffer[shift + 35] << 24;
195     out[9] = buffer[shift + 36] & 0xff | (buffer[shift + 37] & 0xff) << 8 |
196              (buffer[shift + 38] & 0xff) << 16 | buffer[shift + 39] << 24;
197     out[10] = buffer[shift + 40] & 0xff | (buffer[shift + 41] & 0xff) << 8 |
198               (buffer[shift + 42] & 0xff) << 16 | buffer[shift + 43] << 24;
199     out[11] = buffer[shift + 44] & 0xff | (buffer[shift + 45] & 0xff) << 8 |
200               (buffer[shift + 46] & 0xff) << 16 | buffer[shift + 47] << 24;
201     out[12] = buffer[shift + 48] & 0xff | (buffer[shift + 49] & 0xff) << 8 |
202               (buffer[shift + 50] & 0xff) << 16 | buffer[shift + 51] << 24;
203     out[13] = buffer[shift + 52] & 0xff | (buffer[shift + 53] & 0xff) << 8 |
204               (buffer[shift + 54] & 0xff) << 16 | buffer[shift + 55] << 24;
205     out[14] = buffer[shift + 56] & 0xff | (buffer[shift + 57] & 0xff) << 8 |
206               (buffer[shift + 58] & 0xff) << 16 | buffer[shift + 59] << 24;
207     out[15] = buffer[shift + 60] & 0xff | (buffer[shift + 61] & 0xff) << 8 |
208               (buffer[shift + 62] & 0xff) << 16 | buffer[shift + 63] << 24;
209   }

211   private void transform(final MD5State stat, final byte[] buffer,
212                          final int shift, final int[] decodeBuf) {
213     int a = stat.state[0];
214     int b = stat.state[1];
215     int c = stat.state[2];
216     int d = stat.state[3];
217 
218     decode(buffer, shift, decodeBuf);
219 
220     /* Round 1 */
221     a += (b & c | ~b & d) + decodeBuf[0] + 0xd76aa478; /* 1 */
222     a = (a << 7 | a >>> 25) + b;
223     d += (a & b | ~a & c) + decodeBuf[1] + 0xe8c7b756; /* 2 */
224     d = (d << 12 | d >>> 20) + a;
225     c += (d & a | ~d & b) + decodeBuf[2] + 0x242070db; /* 3 */
226     c = (c << 17 | c >>> 15) + d;
227     b += (c & d | ~c & a) + decodeBuf[3] + 0xc1bdceee; /* 4 */
228     b = (b << 22 | b >>> 10) + c;
229 
230     a += (b & c | ~b & d) + decodeBuf[4] + 0xf57c0faf; /* 5 */
231     a = (a << 7 | a >>> 25) + b;
232     d += (a & b | ~a & c) + decodeBuf[5] + 0x4787c62a; /* 6 */
233     d = (d << 12 | d >>> 20) + a;
234     c += (d & a | ~d & b) + decodeBuf[6] + 0xa8304613; /* 7 */
235     c = (c << 17 | c >>> 15) + d;
236     b += (c & d | ~c & a) + decodeBuf[7] + 0xfd469501; /* 8 */
237     b = (b << 22 | b >>> 10) + c;
238 
239     a += (b & c | ~b & d) + decodeBuf[8] + 0x698098d8; /* 9 */
240     a = (a << 7 | a >>> 25) + b;
241     d += (a & b | ~a & c) + decodeBuf[9] + 0x8b44f7af; /* 10 */
242     d = (d << 12 | d >>> 20) + a;
243     c += (d & a | ~d & b) + decodeBuf[10] + 0xffff5bb1; /* 11 */
244     c = (c << 17 | c >>> 15) + d;
245     b += (c & d | ~c & a) + decodeBuf[11] + 0x895cd7be; /* 12 */
246     b = (b << 22 | b >>> 10) + c;
247 
248     a += (b & c | ~b & d) + decodeBuf[12] + 0x6b901122; /* 13 */
249     a = (a << 7 | a >>> 25) + b;
250     d += (a & b | ~a & c) + decodeBuf[13] + 0xfd987193; /* 14 */
251     d = (d << 12 | d >>> 20) + a;
252     c += (d & a | ~d & b) + decodeBuf[14] + 0xa679438e; /* 15 */
253     c = (c << 17 | c >>> 15) + d;
254     b += (c & d | ~c & a) + decodeBuf[15] + 0x49b40821; /* 16 */
255     b = (b << 22 | b >>> 10) + c;
256 
257     /* Round 2 */
258     a += (b & d | c & ~d) + decodeBuf[1] + 0xf61e2562; /* 17 */
259     a = (a << 5 | a >>> 27) + b;
260     d += (a & c | b & ~c) + decodeBuf[6] + 0xc040b340; /* 18 */
261     d = (d << 9 | d >>> 23) + a;
262     c += (d & b | a & ~b) + decodeBuf[11] + 0x265e5a51; /* 19 */
263     c = (c << 14 | c >>> 18) + d;
264     b += (c & a | d & ~a) + decodeBuf[0] + 0xe9b6c7aa; /* 20 */
265     b = (b << 20 | b >>> 12) + c;
266 
267     a += (b & d | c & ~d) + decodeBuf[5] + 0xd62f105d; /* 21 */
268     a = (a << 5 | a >>> 27) + b;
269     d += (a & c | b & ~c) + decodeBuf[10] + 0x02441453; /* 22 */
270     d = (d << 9 | d >>> 23) + a;
271     c += (d & b | a & ~b) + decodeBuf[15] + 0xd8a1e681; /* 23 */
272     c = (c << 14 | c >>> 18) + d;
273     b += (c & a | d & ~a) + decodeBuf[4] + 0xe7d3fbc8; /* 24 */
274     b = (b << 20 | b >>> 12) + c;
275 
276     a += (b & d | c & ~d) + decodeBuf[9] + 0x21e1cde6; /* 25 */
277     a = (a << 5 | a >>> 27) + b;
278     d += (a & c | b & ~c) + decodeBuf[14] + 0xc33707d6; /* 26 */
279     d = (d << 9 | d >>> 23) + a;
280     c += (d & b | a & ~b) + decodeBuf[3] + 0xf4d50d87; /* 27 */
281     c = (c << 14 | c >>> 18) + d;
282     b += (c & a | d & ~a) + decodeBuf[8] + 0x455a14ed; /* 28 */
283     b = (b << 20 | b >>> 12) + c;
284 
285     a += (b & d | c & ~d) + decodeBuf[13] + 0xa9e3e905; /* 29 */
286     a = (a << 5 | a >>> 27) + b;
287     d += (a & c | b & ~c) + decodeBuf[2] + 0xfcefa3f8; /* 30 */
288     d = (d << 9 | d >>> 23) + a;
289     c += (d & b | a & ~b) + decodeBuf[7] + 0x676f02d9; /* 31 */
290     c = (c << 14 | c >>> 18) + d;
291     b += (c & a | d & ~a) + decodeBuf[12] + 0x8d2a4c8a; /* 32 */
292     b = (b << 20 | b >>> 12) + c;
293 
294     /* Round 3 */
295     a += (b ^ c ^ d) + decodeBuf[5] + 0xfffa3942; /* 33 */
296     a = (a << 4 | a >>> 28) + b;
297     d += (a ^ b ^ c) + decodeBuf[8] + 0x8771f681; /* 34 */
298     d = (d << 11 | d >>> 21) + a;
299     c += (d ^ a ^ b) + decodeBuf[11] + 0x6d9d6122; /* 35 */
300     c = (c << 16 | c >>> 16) + d;
301     b += (c ^ d ^ a) + decodeBuf[14] + 0xfde5380c; /* 36 */
302     b = (b << 23 | b >>> 9) + c;
303 
304     a += (b ^ c ^ d) + decodeBuf[1] + 0xa4beea44; /* 37 */
305     a = (a << 4 | a >>> 28) + b;
306     d += (a ^ b ^ c) + decodeBuf[4] + 0x4bdecfa9; /* 38 */
307     d = (d << 11 | d >>> 21) + a;
308     c += (d ^ a ^ b) + decodeBuf[7] + 0xf6bb4b60; /* 39 */
309     c = (c << 16 | c >>> 16) + d;
310     b += (c ^ d ^ a) + decodeBuf[10] + 0xbebfbc70; /* 40 */
311     b = (b << 23 | b >>> 9) + c;
312 
313     a += (b ^ c ^ d) + decodeBuf[13] + 0x289b7ec6; /* 41 */
314     a = (a << 4 | a >>> 28) + b;
315     d += (a ^ b ^ c) + decodeBuf[0] + 0xeaa127fa; /* 42 */
316     d = (d << 11 | d >>> 21) + a;
317     c += (d ^ a ^ b) + decodeBuf[3] + 0xd4ef3085; /* 43 */
318     c = (c << 16 | c >>> 16) + d;
319     b += (c ^ d ^ a) + decodeBuf[6] + 0x04881d05; /* 44 */
320     b = (b << 23 | b >>> 9) + c;
321 
322     a += (b ^ c ^ d) + decodeBuf[9] + 0xd9d4d039; /* 33 */
323     a = (a << 4 | a >>> 28) + b;
324     d += (a ^ b ^ c) + decodeBuf[12] + 0xe6db99e5; /* 34 */
325     d = (d << 11 | d >>> 21) + a;
326     c += (d ^ a ^ b) + decodeBuf[15] + 0x1fa27cf8; /* 35 */
327     c = (c << 16 | c >>> 16) + d;
328     b += (c ^ d ^ a) + decodeBuf[2] + 0xc4ac5665; /* 36 */
329     b = (b << 23 | b >>> 9) + c;
330 
331     /* Round 4 */
332     a += (c ^ (b | ~d)) + decodeBuf[0] + 0xf4292244; /* 49 */
333     a = (a << 6 | a >>> 26) + b;
334     d += (b ^ (a | ~c)) + decodeBuf[7] + 0x432aff97; /* 50 */
335     d = (d << 10 | d >>> 22) + a;
336     c += (a ^ (d | ~b)) + decodeBuf[14] + 0xab9423a7; /* 51 */
337     c = (c << 15 | c >>> 17) + d;
338     b += (d ^ (c | ~a)) + decodeBuf[5] + 0xfc93a039; /* 52 */
339     b = (b << 21 | b >>> 11) + c;
340 
341     a += (c ^ (b | ~d)) + decodeBuf[12] + 0x655b59c3; /* 53 */
342     a = (a << 6 | a >>> 26) + b;
343     d += (b ^ (a | ~c)) + decodeBuf[3] + 0x8f0ccc92; /* 54 */
344     d = (d << 10 | d >>> 22) + a;
345     c += (a ^ (d | ~b)) + decodeBuf[10] + 0xffeff47d; /* 55 */
346     c = (c << 15 | c >>> 17) + d;
347     b += (d ^ (c | ~a)) + decodeBuf[1] + 0x85845dd1; /* 56 */
348     b = (b << 21 | b >>> 11) + c;
349 
350     a += (c ^ (b | ~d)) + decodeBuf[8] + 0x6fa87e4f; /* 57 */
351     a = (a << 6 | a >>> 26) + b;
352     d += (b ^ (a | ~c)) + decodeBuf[15] + 0xfe2ce6e0; /* 58 */
353     d = (d << 10 | d >>> 22) + a;
354     c += (a ^ (d | ~b)) + decodeBuf[6] + 0xa3014314; /* 59 */
355     c = (c << 15 | c >>> 17) + d;
356     b += (d ^ (c | ~a)) + decodeBuf[13] + 0x4e0811a1; /* 60 */
357     b = (b << 21 | b >>> 11) + c;
358 
359     a += (c ^ (b | ~d)) + decodeBuf[4] + 0xf7537e82; /* 61 */
360     a = (a << 6 | a >>> 26) + b;
361     d += (b ^ (a | ~c)) + decodeBuf[11] + 0xbd3af235; /* 62 */
362     d = (d << 10 | d >>> 22) + a;
363     c += (a ^ (d | ~b)) + decodeBuf[2] + 0x2ad7d2bb; /* 63 */
364     c = (c << 15 | c >>> 17) + d;
365     b += (d ^ (c | ~a)) + decodeBuf[9] + 0xeb86d391; /* 64 */
366     b = (b << 21 | b >>> 11) + c;
367 
368     stat.state[0] += a;
369     stat.state[1] += b;
370     stat.state[2] += c;
371     stat.state[3] += d;
372   }

374   /**
375    * Updates hash with the bytebuffer given (using at maximum length bytes
376    * from
377    * that buffer)
378    *
379    * @param stat Which state is updated
380    * @param buffer Array of bytes to be hashed
381    * @param offset Offset to buffer array
382    * @param length Use at maximum `length' bytes (absolute maximum is
383    *     buffer.length)
384    */
385   void Update(final MD5State stat, final byte[] buffer, final int offset,
386               final int length) {
387     int index;
388     int partlen;
389     int i;
390     final int start;
391     finals = null;
392     int newlength = length;
393     /* Length can be told to be shorter, but not inter */
394     if (newlength + offset > buffer.length) {
395       newlength = buffer.length - offset;
396     }
397 
398     /* compute number of bytes mod 64 */
399 
400     index = (int) (stat.count & 0x3f);
401     stat.count += newlength;
402 
403     partlen = 64 - index;
404 
405     if (newlength >= partlen) {
406       // update state (using only Java) to reflect input
407 
408       final int[] decodeBuf = new int[16];
409       if (partlen == 64) {
410         partlen = 0;
411       } else {
412         for (i = 0; i < partlen; i++) {
413           stat.buffer[i + index] = buffer[i + offset];
414         }
415         transform(stat, stat.buffer, 0, decodeBuf);
416       }
417       for (i = partlen; i + 63 < newlength; i += 64) {
418         transform(stat, buffer, i + offset, decodeBuf);
419       }
420       index = 0;
421     } else {
422       i = 0;
423     }
424 
425     /* buffer remaining input */
426     if (i < newlength) {
427       start = i;
428       for (; i < newlength; i++) {
429         stat.buffer[index + i - start] = buffer[i + offset];
430       }
431     }
432   }

480   /**
481    * Update buffer with given string. Note that because the version of the
482    * s.getBytes() method without
483    * parameters is used to convert the string to a byte array, the results of
484    * this method may be different on
485    * different platforms. The s.getBytes() method converts the string into a
486    * byte array using the current
487    * platform's default character set and may therefore have different results
488    * on platforms with different
489    * default character sets. If a version that works consistently across
490    * platforms with different default
491    * character sets is desired, use the overloaded version of the Update()
492    * method which takes a string and a
493    * character encoding.
494    *
495    * @param s String to be update to hash (is used as s.getBytes())
496    */
497   void Update(final String s) {
498     final byte[] chars = s.getBytes(FilesystemBasedDigest.UTF8);
499     Update(chars, chars.length);
500   }

502   /**
503    * Update buffer with given string using the given encoding. If the given
504    * encoding is null, the encoding
505    * "UTF8" is used.
506    *
507    * @param s String to be update to hash (is used as
508    *     s.getBytes(charset_name))
509    * @param charsetName The character set to use to convert s to a
510    *     byte
511    *     array, or null if the "ISO8859_1"
512    *     character set is desired.
513    *
514    * @throws UnsupportedEncodingException If the named charset
515    *     is not supported.
516    */
517   void Update(final String s, final String charsetName)
518       throws UnsupportedEncodingException {
519     Charset newcharset = FilesystemBasedDigest.UTF8;
520     if (charsetName != null) {

540   /**
541    * Updates hash with given {@link ByteBuf} (from Netty)
542    *
543    * @param buffer ByteBuf to use for updating the hash and this
544    *     buffer will
545    *     not be changed
546    */
547   void Update(final ByteBuf buffer) {
548     final byte[] bytes;
549     int start = 0;
550     final int len = buffer.readableBytes();
551     if (buffer.hasArray()) {
552       start = buffer.arrayOffset();
553       bytes = buffer.array();
554     } else {
555       if (reusableBytes == null || reusableBytes.length != len) {
556         reusableBytes = new byte[len];
557       }
558       bytes = reusableBytes;
559       buffer.getBytes(start, bytes);
560     }
561     Update(state, bytes, start, len);
562   }

564   private byte[] encode(final int[] input, final int len) {
565     int i;
566     int j;
567     final byte[] out;
568 
569     out = new byte[len];
570 
571     for (i = j = 0; j < len; i++, j += 4) {
572       out[j] = (byte) (input[i] & 0xff);
573       out[j + 1] = (byte) (input[i] >>> 8 & 0xff);
574       out[j + 2] = (byte) (input[i] >>> 16 & 0xff);
575       out[j + 3] = (byte) (input[i] >>> 24 & 0xff);
576     }
577 
578     return out;
579   }

581   /**
582    * Returns array of bytes (16 bytes) representing hash as of the current
583    * state
584    * of this object. Note: getting a
585    * hash does not invalidate the hash object, it only creates a copy of the
586    * real state which is finalized.
587    *
588    * @return Array of 16 bytes, the hash of all updated bytes
589    */
590   synchronized byte[] Final() {
591     final byte[] bits;
592     final int index;
593     final int padlen;
594     final MD5State fin;
595 
596     if (finals == null) {
597       fin = new MD5State(state);
598 
599       final int[] countInts =
600           { (int) (fin.count << 3), (int) (fin.count >> 29) };
601       bits = encode(countInts, 8);
602 
603       index = (int) (fin.count & 0x3f);
604       padlen = index < 56? 56 - index : 120 - index;
605 
606       Update(fin, padding, 0, padlen);
607       Update(fin, bits, 0, 8);
608 
609       /* Update() sets finals to null */
610       finals = fin;
611     }
612 
613     return encode(finals.state, 16);
614   }

623    * number.
624    *
625    * @param hash Array of bytes to convert to hex-string
626    *
627    * @return Generated hex string
628    */
629   static String asHex(final byte[] hash) {
630     final char[] buf = new char[hash.length * 2];
631     for (int i = 0, x = 0; i < hash.length; i++) {
632       buf[x++] = HEX_CHARS[hash[i] >>> 4 & 0xf];
633       buf[x++] = HEX_CHARS[hash[i] & 0xf];
634     }
635     return new String(buf);

638   /**
639    * Turns String into array of bytes representing each couple of unsigned hex
640    * number as one byte.
641    *
642    * @param buf hex string
643    *
644    * @return Array of bytes converted from hex-string
645    */
646   static byte[] asByte(final String buf) {
647     final byte[] from = buf.getBytes(FilesystemBasedDigest.UTF8);
648     final byte[] hash = new byte[from.length / 2];
649     for (int i = 0, x = 0; i < hash.length; i++) {
650       byte code1 = from[x++];
651       byte code2 = from[x++];
652       if (code1 >= HEX_CHARS[10]) {
653         code1 -= HEX_CHARS[10] - 10;
654       } else {
655         code1 -= HEX_CHARS[0];
656       }
657       if (code2 >= HEX_CHARS[10]) {
658         code2 -= HEX_CHARS[10] - 10;
659       } else {
660         code2 -= HEX_CHARS[0];
661       }
662       hash[i] = (byte) ((code1 << 4) + (code2 & 0xFF));
663     }
664     return hash;
665   }
