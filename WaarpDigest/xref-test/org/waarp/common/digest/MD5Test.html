21  package org.waarp.common.digest;
22  
23  import io.netty.buffer.ByteBuf;
24  import io.netty.buffer.Unpooled;
25  import org.junit.Test;
26  
27  import java.io.UnsupportedEncodingException;
28  
29  import static org.junit.Assert.*;
30  
31  public class MD5Test {
32  
33    @Test
34    public void updateTest() throws UnsupportedEncodingException {
35      final String totest = "MonTest";
36      final MD5 md5 = new MD5(totest);
37      assertArrayEquals(md5.Final(), MD5.asByte(md5.asHex()));
38      final MD5 md52 = new MD5();
39      md52.Update(totest);
40      assertTrue(MD5.hashesEqual(md5.Final(), md52.Final()));
41      final MD5 md53 = new MD5();
42      final byte[] btotest = totest.getBytes();
43      md53.Update(btotest, btotest.length);
44      assertTrue(MD5.hashesEqual(md5.Final(), md53.Final()));
45      final MD5 md54 = new MD5();
46      md54.Update(btotest);
47      assertTrue(MD5.hashesEqual(md5.Final(), md54.Final()));
48      final MD5 md55 = new MD5();
49      md55.Update(totest, "UTF-8");
50      assertTrue(MD5.hashesEqual(md5.Final(), md55.Final()));
51      final MD5 md56 = new MD5();
52      for (final byte element : btotest) {
53        md56.Update(element);
54      }
55      assertTrue(MD5.hashesEqual(md5.Final(), md56.Final()));
56      final MD5 md57 = new MD5();
57      for (final byte element : btotest) {
58        md57.Update((int) element);
59      }
60      assertTrue(MD5.hashesEqual(md5.Final(), md57.Final()));
61      final ByteBuf buf = Unpooled.wrappedBuffer(btotest);
62      final MD5 md58 = new MD5();
63      md58.Update(buf);
64      assertTrue(MD5.hashesEqual(md5.Final(), md58.Final()));
65    }
66  
67    @Test
68    public void passwordTest() {
69      final String passwd = "totest";
70      final String pwd = MD5.passwdCrypt(passwd);
71      assertTrue(MD5.equalPasswd(passwd, pwd));
72      final byte[] bpwd = MD5.passwdCrypt(passwd.getBytes());
73      assertTrue(MD5.equalPasswd(passwd.getBytes(), bpwd));
74      assertTrue(MD5.hashesEqual(MD5.asByte(pwd), bpwd));
75    }
