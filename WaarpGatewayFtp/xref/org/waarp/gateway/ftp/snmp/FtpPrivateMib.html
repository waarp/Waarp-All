188     rootOIDWaarpGlobal = new OID(srootOIDWaarp + ".2");

198   /**
199    * Unregister and Register again the SNMPv2MIB with System adapted to this
200    * Mib
201    *
202    * @throws DuplicateRegistrationException
203    */
204   protected void agentRegisterSystem() throws DuplicateRegistrationException {
205     // Since BaseAgent registers some mibs by default we need to unregister
206     // one before we register our own sysDescr. Normally you would
207     // override that method and register the mibs that you need
208 
209     agent.unregisterManagedObject(agent.getSnmpv2MIB());
210 
211     // Register a system description, use one from you product environment
212     // to test with
213     snmpv2 = new SNMPv2MIB(new OctetString(SnmpName), ggObjectId,
214                            new Integer32(SnmpService));
215     snmpv2.setContact(new OctetString(SnmpDefaultAuthor));
216     snmpv2.setLocation(new OctetString(SnmpDefaultLocalization));
217     snmpv2.setName(new OctetString(SnmpVersion));
218     snmpv2.registerMOs(agent.getServer(), null);
219     if (logger.isDebugEnabled()) {
220       logger.debug(
221           "SNMPV2: " + snmpv2.getContact() + ':' + snmpv2.getDescr() + ':' +
222           snmpv2.getLocation() + ':' + snmpv2.getName() + ':' +
223           snmpv2.getObjectID() + ':' + snmpv2.getServices() + ':' +
224           snmpv2.getUpTime());
225     }
226     // Save UpTime reference since used everywhere
227     upTime = snmpv2.getSysUpTime();
228   }

230   /**
231    * Register this MIB
232    *
233    * @throws DuplicateRegistrationException
234    */
235   protected void agentRegisterWaarpMib() throws DuplicateRegistrationException {
236     logger.debug("registerGGMib");
237     // register Static info
238     rowInfo = new WaarpMORow(this, rootOIDWaarpInfo, WaarpDefinition,
239                              MibLevel.staticInfo.ordinal());
240     rowInfo.setValue(WaarpDefinitionIndex.applName.ordinal(),
241                      "Waarp " + "GatewayFtp");
242     rowInfo.setValue(WaarpDefinitionIndex.applServerName.ordinal(),
243                      FileBasedConfiguration.fileBasedConfiguration.getHostId());
244     rowInfo.setValue(WaarpDefinitionIndex.applVersion.ordinal(),
245                      Version.fullIdentifier());
246     rowInfo.setValue(WaarpDefinitionIndex.applDescription.ordinal(),
247                      "Waarp Gateway Ftp: FTP Gateway");
248     rowInfo.setValue(WaarpDefinitionIndex.applURL.ordinal(),
249                      "http://waarp.github.com/Waarp");
250     rowInfo.setValue(WaarpDefinitionIndex.applApplicationProtocol.ordinal(),
251                      applicationProtocol);
252 
253     rowInfo.registerMOs(agent.getServer(), null);
254     // register General info
255     rowGlobal = new WaarpMORow(this, rootOIDWaarpGlobal, WaarpGlobalValues,
256                                MibLevel.globalInfo.ordinal());
257     WaarpMOScalar memoryScalar =
258         rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryTotal.ordinal()];
259     memoryScalar.setValue(new MemoryGauge32(MemoryType.TotalMemory));
260     memoryScalar =
261         rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryFree.ordinal()];
262     memoryScalar.setValue(new MemoryGauge32(MemoryType.FreeMemory));
263     memoryScalar =
264         rowGlobal.getRow()[WaarpGlobalValuesIndex.memoryUsed.ordinal()];
265     memoryScalar.setValue(new MemoryGauge32(MemoryType.UsedMemory));
266     rowGlobal.registerMOs(agent.getServer(), null);
267     // setup UpTime to SysUpTime and change status
268     scalarUptime =
269         rowGlobal.getRow()[WaarpGlobalValuesIndex.applUptime.ordinal()];
270     scalarUptime.setValue(new WaarpUptime(upTime));
271     changeStatus(OperStatus.restarting);
272     changeStatus(OperStatus.up);
273     // register Detailed info
274     rowDetailed =
275         new WaarpMORow(this, rootOIDWaarpDetailed, WaarpDetailedValues,
276                        MibLevel.detailedInfo.ordinal());
277     rowDetailed.registerMOs(agent.getServer(), null);
278     // register Error info
279     rowError = new WaarpMORow(this, rootOIDWaarpError, WaarpErrorValues,
280                               MibLevel.errorInfo.ordinal());
281     rowError.registerMOs(agent.getServer(), null);
282   }

332   /**
333    * Change the status and the LastChange Timeticks
334    *
335    * @param status
336    */
337   public void changeStatus(OperStatus status) {
338     final WaarpMOScalar statusScalar =
339         rowGlobal.getRow()[WaarpGlobalValuesIndex.applOperStatus.ordinal()];
340     final Integer32 var = (Integer32) statusScalar.getValue();
341     if (var.getValue() != status.status) {
342       var.setValue(status.status);
343       final WaarpMOScalar lastTimeScalar =
344           rowGlobal.getRow()[WaarpGlobalValuesIndex.applLastChange.ordinal()];
345       final TimeTicks time = (TimeTicks) lastTimeScalar.getValue();
346       time.setValue(upTime.get().getValue());
347     }
348   }

402   /**
403    * Send a notification (trap or inform) for Warning/Error event on a single
404    * Transfer Task
405    *
406    * @param message
407    * @param runner
408    */
409   public void notifyInfoTask(String message, DbTransferLog runner) {
410     if (!TrapLevel.All.isLevelValid(agent.getTrapLevel())) {
411       return;
412     }
413     if (logger.isDebugEnabled()) {
414       logger.debug(
415           "Notify: " + NotificationElements.InfoTask + ':' + message + ':' +
416           runner);
417     }
418     long delay =
419         (runner.getStart().getTime() - agent.getUptimeSystemTime()) / 10;
420     if (delay < 0) {
421       delay = 0;
422     }
423     agent.getNotificationOriginator().notify(new OctetString("public"),
424                                              NotificationElements.InfoTask
425                                                  .getOID(rootOIDWaarpNotif),
426                                              new VariableBinding[] {
427                                                  new VariableBinding(
428                                                      NotificationElements.InfoTask
429                                                          .getOID(
430                                                              rootOIDWaarpNotif,
431                                                              1),
432                                                      new OctetString(
433                                                          NotificationElements.InfoTask
434                                                              .name())),
435                                                  new VariableBinding(
436                                                      NotificationElements.InfoTask
437                                                          .getOID(
438                                                              rootOIDWaarpNotif,
439                                                              1),
440                                                      new OctetString(message)),
441                                                  // Start of Task
442                                                  new VariableBinding(
443                                                      NotificationElements.InfoTask
444                                                          .getOID(
445                                                              rootOIDWaarpNotif,
446                                                              NotificationTasks.filenameInfo
447                                                                  .getOID()),
448                                                      new OctetString(
449                                                          runner.getFilename())),
450                                                  new VariableBinding(
451                                                      NotificationElements.InfoTask
452                                                          .getOID(
453                                                              rootOIDWaarpNotif,
454                                                              NotificationTasks.modeTransInfo
455                                                                  .getOID()),
456                                                      new OctetString(
457                                                          runner.getMode())),
458                                                  new VariableBinding(
459                                                      NotificationElements.InfoTask
460                                                          .getOID(
461                                                              rootOIDWaarpNotif,
462                                                              NotificationTasks.startTransInfo
463                                                                  .getOID()),
464                                                      new TimeTicks(delay)),
465                                                  new VariableBinding(
466                                                      NotificationElements.InfoTask
467                                                          .getOID(
468                                                              rootOIDWaarpNotif,
469                                                              NotificationTasks.infoStatusInfo
470                                                                  .getOID()),
471                                                      new OctetString(
472                                                          runner.getErrorInfo()
473                                                                .getMesg())),
474                                                  new VariableBinding(
475                                                      NotificationElements.InfoTask
476                                                          .getOID(
477                                                              rootOIDWaarpNotif,
478                                                              NotificationTasks.userIdInfo
479                                                                  .getOID()),
480                                                      new OctetString(
481                                                          runner.getUser())),
482                                                  new VariableBinding(
483                                                      NotificationElements.InfoTask
484                                                          .getOID(
485                                                              rootOIDWaarpNotif,
486                                                              NotificationTasks.accountId
487                                                                  .getOID()),
488                                                      new OctetString(
489                                                          runner.getAccount())),
490                                                  new VariableBinding(
491                                                      NotificationElements.InfoTask
492                                                          .getOID(
493                                                              rootOIDWaarpNotif,
494                                                              NotificationTasks.specialIdInfo
495                                                                  .getOID()),
496                                                      new OctetString(String
497                                                                          .valueOf(
498                                                                              runner
499                                                                                  .getSpecialId()))),
500                                                  // End of Task
501                                                  new VariableBinding(
502                                                      SnmpConstants.sysDescr,
503                                                      snmpv2.getDescr()),
504                                                  new VariableBinding(
505                                                      SnmpConstants.sysObjectID,
506                                                      snmpv2.getObjectID()),
507                                                  new VariableBinding(
508                                                      SnmpConstants.sysContact,
509                                                      snmpv2.getContact()),
510                                                  new VariableBinding(
511                                                      SnmpConstants.sysName,
512                                                      snmpv2.getName()),
513                                                  new VariableBinding(
514                                                      SnmpConstants.sysLocation,
515                                                      snmpv2.getLocation())
516                                              });
517   }

519   /**
520    * Trap/Notification
521    *
522    * @param element
523    * @param message
524    * @param message2
525    */
526   private void notify(NotificationElements element, String message,
527                       String message2) {
528     if (logger.isDebugEnabled()) {
529       logger.debug("Notify: " + element + ':' + message + ':' + message2);
530     }
531     agent.getNotificationOriginator()
532          .notify(new OctetString("public"), element.getOID(rootOIDWaarpNotif),
533                  new VariableBinding[] {
534                      new VariableBinding(element.getOID(rootOIDWaarpNotif, 1),
535                                          new OctetString(element.name())),
536                      new VariableBinding(element.getOID(rootOIDWaarpNotif, 1),
537                                          new OctetString(message)),
538                      new VariableBinding(element.getOID(rootOIDWaarpNotif, 1),
539                                          new OctetString(message2)),
540                      new VariableBinding(SnmpConstants.sysDescr,
541                                          snmpv2.getDescr()),
542                      new VariableBinding(SnmpConstants.sysObjectID,
543                                          snmpv2.getObjectID()),
544                      new VariableBinding(SnmpConstants.sysContact,
545                                          snmpv2.getContact()),
546                      new VariableBinding(SnmpConstants.sysName,
547                                          snmpv2.getName()),
548                      new VariableBinding(SnmpConstants.sysLocation,
549                                          snmpv2.getLocation())
550                  });
551   }

580       final int[] ids = { oid[0], rank };
581       return new OID(oidBase.getValue(), ids);
