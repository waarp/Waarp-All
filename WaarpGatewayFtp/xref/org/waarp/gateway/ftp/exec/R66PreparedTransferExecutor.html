46  /**
47   * R66PreparedTransferExecutor class. If the command starts with "REFUSED", the
48   * command will be refused for
49   * execution. If "REFUSED" is set, the command "RETR" or "STOR" like operations
50   * will be stopped at starting of
51   * command.
52   * <p>
53   * <p>
54   * <p>
55   * Format is like r66send command in any order except "-info" which should be
56   * the last item:<br>
57   * "-to Host -file FILE -rule RULE [-md5] [-nolog] [-start yyyyMMddHHmmss or
58   * -delay (delay or +delay)] [-info
59   * INFO]" <br>
60   * <br>
61   * INFO is the only one field that can contains blank character.<br>
62   * <br>
63   * The following replacement are done dynamically before the command is
64   * executed:<br>
65   * - #BASEPATH# is replaced by the full path for the root of FTP Directory<br>
66   * - #FILE# is replaced by the current file path relative to FTP Directory (so
67   * #BASEPATH##FILE# is the full
68   * path of the file)<br>
69   * - #USER# is replaced by the username<br>
70   * - #ACCOUNT# is replaced by the account<br>
71   * - #COMMAND# is replaced by the command issued for the file<br>
72   * - #SPECIALID# is replaced by the FTP id of the transfer (whatever in or
73   * out)<br>
74   * - #UUID# is replaced by a special UUID globally unique for the transfer, in
75   * general to be placed in -info
76   * part (for instance ##UUID## giving #uuid#)<br>
77   * <br>
78   * So for instance "-to Host -file #BASEPATH##FILE# -rule RULE [-md5] [-nolog]
79   * [-delay +delay] [-info ##UUID##
80   * #USER# #ACCOUNT# #COMMAND# INFO]" <br>
81   * will be a standard use of this function.
82   */
83  public class R66PreparedTransferExecutor extends AbstractExecutor {
84    private static final String CANNOT_GET_NEW_TASK = "Cannot get new task\n    ";

112   /**
113    * @param command
114    * @param delay
115    * @param futureCompletion
116    */
117   public R66PreparedTransferExecutor(String command, long delay,
118                                      WaarpFuture futureCompletion) {
119     final String[] args = BLANK.split(command);
120     for (int i = 0; i < args.length; i++) {
121       if ("-to".equalsIgnoreCase(args[i])) {
122         i++;
123         remoteHost = args[i];
124         if (Configuration.configuration.getAliases().containsKey(remoteHost)) {
125           remoteHost = Configuration.configuration.getAliases().get(remoteHost);
126         }
127       } else if ("-file".equalsIgnoreCase(args[i])) {
128         i++;
129         filename = args[i];
130       } else if ("-rule".equalsIgnoreCase(args[i])) {
131         i++;
132         rulename = args[i];
133       } else if ("-info".equalsIgnoreCase(args[i])) {
134         i++;
135         fileinfo = args[i];
136         i++;
137         while (i < args.length) {
138           fileinfo += ' ' + args[i];
139           i++;
140         }
141       } else if ("-md5".equalsIgnoreCase(args[i])) {
142         isMD5 = true;
143       } else if ("-block".equalsIgnoreCase(args[i])) {
144         i++;
145         blocksize = Integer.parseInt(args[i]);
146         if (blocksize < 100) {
147           logger.warn("Block size is too small: " + blocksize);
148           blocksize = Configuration.configuration.getBlockSize();
149         }
150       } else if ("-nolog".equalsIgnoreCase(args[i])) {
151         nolog = true;
152         i++;
153       } else if ("-start".equalsIgnoreCase(args[i])) {
154         i++;
155         final SimpleDateFormat dateFormat =
156             new SimpleDateFormat("yyyyMMddHHmmss");
157         Date date;
158         try {
159           date = dateFormat.parse(args[i]);
160           timestart = new Timestamp(date.getTime());
161         } catch (final ParseException ignored) {
162           // nothing
163         }
164       } else if ("-delay".equalsIgnoreCase(args[i])) {
165         i++;
166         if (args[i].charAt(0) == '+') {
167           timestart = new Timestamp(System.currentTimeMillis() +
168                                     Long.parseLong(args[i].substring(1)));
169         } else {
170           timestart = new Timestamp(Long.parseLong(args[i]));
171         }
172       }
173     }
174     if (fileinfo == null) {
175       fileinfo = "noinfo";
176     }
177     future = futureCompletion;
178   }

180   /**

184     this.dbsession = dbsession;

187   @Override
188   public void run() throws CommandAbstractException {
189     final String message =
190         "R66Prepared with -to " + remoteHost + " -rule " + rulename +
191         " -file " + filename + " -nolog: " + nolog + " -isMD5: " + isMD5 +
192         " -info " + fileinfo;
193     if (remoteHost == null || rulename == null || filename == null) {
194       logger.error(
195           "Mandatory argument is missing: -to " + remoteHost + " -rule " +
196           rulename + " -file " + filename);
197       throw new Reply421Exception(
198           "Mandatory argument is missing\n    " + message);
199     }
200     logger.debug(message);
201     DbRule rule;
202     try {
203       rule = new DbRule(rulename);
204     } catch (final WaarpDatabaseException e) {
205       logger.error("Cannot get Rule: " + rulename + " since {}\n    " + message,
206                    e.getMessage());
207       throw new Reply421Exception(
208           "Cannot get Rule: " + rulename + "\n    " + message);
209     }
210     int mode = rule.getMode();
211     if (isMD5) {
212       mode = RequestPacket.getModeMD5(mode);
213     }
214     final String sep = PartnerConfiguration.getSeparator(remoteHost);
215     long originalSize = -1;
216     if (RequestPacket.isSendMode(mode) && !RequestPacket.isThroughMode(mode)) {
217       final File file = new File(filename);
218       if (file.canRead()) {
219         originalSize = file.length();
220       }
221     }
222     final RequestPacket request =
223         new RequestPacket(rulename, mode, filename, blocksize, 0, ILLEGALVALUE,
224                           fileinfo, originalSize, sep);
225     // Not isRecv since it is the requester, so send => isRetrieve is true
226     final boolean isRetrieve = !RequestPacket.isRecvMode(request.getMode());
227     logger.debug("Will prepare: {}", request);
228     DbTaskRunner taskRunner;
229     try {
230       taskRunner =
231           new DbTaskRunner(rule, isRetrieve, request, remoteHost, timestart);
232     } catch (final WaarpDatabaseException e) {
233       logger.error("Cannot get new task since {}\n    " + message,
234                    e.getMessage());
235       throw new Reply421Exception(CANNOT_GET_NEW_TASK + message);
236     }
237     taskRunner.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
238     if (!taskRunner.forceSaveStatus()) {
239       try {
240         if (!taskRunner.specialSubmit()) {
241           logger.error("Cannot prepare task: " + message);
242           throw new Reply421Exception(CANNOT_GET_NEW_TASK + message);
243         }
244       } catch (final WaarpDatabaseException e) {
245         logger.error("Cannot prepare task since {}\n    " + message,
246                      e.getMessage());
247         throw new Reply421Exception(CANNOT_GET_NEW_TASK + message);
248       }
249     }
250     logger.debug("R66PreparedTransfer prepared: {}", request);
251     future.setSuccess();
252   }
