package org.waarp.snmp.utils;

import org.snmp4j.agent.mo.snmp.SysUpTime;
import org.snmp4j.smi.TimeTicks;

/**
 * Specific Value for TimeTicks for Uptime
 */
public class WaarpUptime extends TimeTicks {
  /**
   *
   */
  private static final long serialVersionUID = -6737574100905398690L;
  protected transient final SysUpTime type;

  protected final void setInternalValue() {
    setValue(type.get().getValue());
  }

  protected final void setInternalValue(final long value) {
    setValue(type.get().getValue());
  }

  /**
   * @param type SysUpTime from SNMPV2
   */
  public WaarpUptime(final SysUpTime type) {
    this.type = type;
    setInternalValue();
  }

  @Override
  public final Object clone() {
    setInternalValue();
    return super.clone();
  }

  @Override
  public final long getValue() {
    setInternalValue();
    return super.getValue();
  }

}
