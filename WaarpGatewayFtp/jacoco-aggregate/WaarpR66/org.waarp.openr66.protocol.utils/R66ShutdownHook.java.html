/**
 *
 */
public class R66ShutdownHook extends WaarpShutdownHook {

  /**
   * @param configuration
   */
  public R66ShutdownHook(ShutdownConfiguration configuration) {
    super(configuration);
  }

  @Override
  protected void exitService() {
    ChannelUtils.exit();
  }

}
