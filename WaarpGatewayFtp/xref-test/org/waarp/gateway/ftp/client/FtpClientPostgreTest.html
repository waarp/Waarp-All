120   public void testFtp4J(String server, int port, String username, String passwd,
121                         String account, int isSSL, String localFilename,
122                         int type, int delay, boolean shutdown, int numberThread,
123                         int numberIteration) {
124     // initiate Directories
125     final Ftp4JClientTransactionTest client =
126         new Ftp4JClientTransactionTest(server, port, username, passwd, account,
127                                        isSSL);
128 
129     client.setReportActiveExternalIPAddress("127.0.0.1");
130     client.setActiveDataTransferPortRange(57000, 63000);
131     logger.warn("First connexion");
132     if (!client.connect()) {
133       logger.error("Can't connect");
134       FtpClientTest.numberKO.incrementAndGet();
135       assertEquals("No KO", 0, FtpClientTest.numberKO.get());
136       return;
137     }
138     try {
139       logger.warn("Create Dirs");
140       for (int i = 0; i < numberThread; i++) {
141         client.makeDir("T" + i);
142       }
143       try {
144         Thread.sleep(10);
145       } catch (InterruptedException e) {
146         // Ignore
147       }
148       logger.warn("Feature commands");
149       System.err.println("SITE: " + client.featureEnabled("SITE"));
150       System.err.println("SITE CRC: " + client.featureEnabled("SITE XCRC"));
151       System.err.println("CRC: " + client.featureEnabled("XCRC"));
152       System.err.println("MD5: " + client.featureEnabled("XMD5"));
153       System.err.println("SHA1: " + client.featureEnabled("XSHA1"));
154       System.err.println("DIGEST: " + client.featureEnabled("XDIGEST"));
155     } finally {
156       logger.warn("Logout");
157       client.logout();
158     }
159     final ExecutorService executorService = Executors.newCachedThreadPool();
160     logger.warn("Will start {} Threads", numberThread);
161     final long date1 = System.currentTimeMillis();
162     for (int i = 0; i < numberThread; i++) {
163       executorService.execute(
164           new FtpClientThread("T" + i, server, port, username, passwd, account,
165                               localFilename, numberIteration, type, delay,
166                               isSSL));
167       Thread.yield();
168     }
169     try {
170       Thread.sleep(100);
171     } catch (final InterruptedException e1) {
172       e1.printStackTrace();
173     }
174     executorService.shutdown();
175     long date2 = 0;
176     try {
177       if (!executorService.awaitTermination(120000, TimeUnit.MILLISECONDS)) {
178         executorService.shutdownNow();
179         if (!executorService.awaitTermination(60, TimeUnit.SECONDS)) {
180           System.err.println("Really not shutdown normally");
181         }
182       }
183     } catch (final InterruptedException e) {
184       e.printStackTrace();
185       executorService.shutdownNow();
186     }
187     date2 = System.currentTimeMillis();
188 
189     logger.warn(
190         localFilename + ' ' + numberThread + ' ' + numberIteration + ' ' +
191         type + " Real: " + (date2 - date1) + " OK: " +
192         FtpClientTest.numberOK.get() + " KO: " + FtpClientTest.numberKO.get() +
193         " Trf/s: " + FtpClientTest.numberOK.get() * 1000 / (date2 - date1));
194     assertEquals("No KO", 0, FtpClientTest.numberKO.get());
195   }

197   @BeforeClass
198   public static void startServer() throws Exception {
199     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
200         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
201     ResourceLeakDetector.setLevel(
202         SystemPropertyUtil.get(IT_LONG_TEST, false)? Level.SIMPLE :
203             Level.PARANOID);
204     WaarpSystemUtil.setJunit(true);
205     // R66 Home
206     File home = new File("/tmp/FTP");
207     home.mkdirs();
208     FileUtils.forceDeleteRecursiveDir(home);
209     final File localFilename = new File("/tmp/ftpfile.bin");
210     FileTestUtils.createTestFile(localFilename, 100);
211     final ClassLoader classLoader = FtpClientPostgreTest.class.getClassLoader();
212     // Adapt config file to Database
213     File file =
214         new File(classLoader.getResource("Gg-FTP-postgre.xml").getFile());
215     if (!file.exists()) {
216       SysErrLogger.FAKE_LOGGER.syserr(
217           "Cannot find in  " + file.getAbsolutePath());
218       fail("Cannot find " + file.getAbsolutePath());
219     }
220     String content = WaarpStringUtils.readFile(file.getAbsolutePath());
221     SysErrLogger.FAKE_LOGGER.sysout(scenario.getJDC().getJdbcUrl());
222     String driver = scenario.getJDC().getDriverClassName();
223     String target = "notfound";
224     String jdbcUrl = scenario.getJDC().getJdbcUrl();
225     if (driver.equalsIgnoreCase("org.mariadb.jdbc.Driver")) {
226       target = "mariadb";
227     } else if (driver.equalsIgnoreCase("org.h2.Driver")) {
228       target = "h2";
229     } else if (driver.equalsIgnoreCase("oracle.jdbc.OracleDriver")) {
230       target = "oracle";
231       jdbcUrl = "jdbc:oracle:thin:@//localhost:1521/test";
232       SysErrLogger.FAKE_LOGGER.syserr(
233           jdbcUrl + " while should be something like " + jdbcUrl);
234       throw new UnsupportedOperationException(
235           "Unsupported Test for Oracle since wrong JDBC driver");
236     } else if (driver.equalsIgnoreCase("org.postgresql.Driver")) {
237       target = "postgresql";
238     } else if (DbModelMysql.MYSQL_DRIVER_JRE6.equalsIgnoreCase(driver) ||
239                DbModelMysql.MYSQL_DRIVER_JRE8.equalsIgnoreCase(driver)) {
240       target = "mysql";
241     } else {
242       SysErrLogger.FAKE_LOGGER.syserr("Cannot find driver for " + driver);
243     }
244     content = content.replace("XXXJDBCXXX", jdbcUrl);
245     content = content.replace("XXXDRIVERXXX", target);
246     SysErrLogger.FAKE_LOGGER.sysout(scenario.getJDC().getDriverClassName());
247     SysErrLogger.FAKE_LOGGER.sysout(target);
248     File fileTo = new File("/tmp/gg-ftp-db.xml");
249     fileTo.getParentFile().mkdirs();
250     FileWriter writer = null;
251     try {
252       writer = new FileWriter(fileTo);
253       writer.write(content);
254       writer.flush();
255     } catch (IOException e) {
256       e.printStackTrace();
257     } finally {
258       if (writer != null) {
259         FileUtils.close(writer);
260       }
261     }
262     if (file.exists()) {
263       driverType = DriverType.FIREFOX;
264       initiateWebDriver(file.getParentFile());
265     }
266     final File tmp = new File("/tmp");
267     final File[] files = tmp.listFiles(new FileFilter() {
268       @Override
269       public boolean accept(File file) {
270         return file.getName().startsWith("ggftp");
271       }
272     });
273     for (final File fileDb : files) {
274       fileDb.delete();
275     }
276     final FileBasedConfiguration configuration =
277         new FileBasedConfiguration(ExecGatewayFtpServer.class,
278                                    ExecBusinessHandler.class,
279                                    FileSystemBasedDataBusinessHandler.class,
280                                    new FilesystemBasedFileParameterImpl());
281     try {
282       if (!configuration.setConfigurationServerFromXml(
283           fileTo.getAbsolutePath())) {
284         System.err.println("Bad main configuration");
285         Assert.fail("Bad main configuration");
286       }
287       // Init database
288       try {
289         ServerInitDatabase.initdb();
290       } catch (final WaarpDatabaseNoConnectionException e) {
291         logger.error("Cannot connect to database");
292         return;
293       }
294       System.out.println("End creation");
295     } finally {
296       if (DbConstantFtp.gatewayAdmin != null) {
297         DbConstantFtp.gatewayAdmin.close();
298       }
299     }
300     logger.warn("Will start server");
301     key = configuration.getCryptoKey().decryptHexInString("c5f4876737cf351a");
302 
303     ExecGatewayFtpServer.main(new String[] { fileTo.getAbsolutePath() });
304     try {
305       Thread.sleep(500);
306     } catch (InterruptedException e) {
307       // Wait for server started
308     }
309     System.gc();
310     try {
311       Thread.sleep(500);
312     } catch (InterruptedException e) {
313       // Wait for server started
314     }
315   }

317   @AfterClass
318   public static void stopServer() throws InterruptedException {
319     logger.warn("Will shutdown from client");
320     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
321         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
322     try {
323       Thread.sleep(200);
324     } catch (final InterruptedException ignored) {
325     }
326     finalizeDriver();
327     try {
328       final Ftp4JClientTransactionTest client =
329           new Ftp4JClientTransactionTest("127.0.0.1", 2021, "fredo", key, "a",
330                                          0);
331       if (!client.connect()) {
332         logger.warn("Cant connect");
333       } else {
334         try {
335           final String[] results =
336               client.executeSiteCommand("internalshutdown pwdhttp");
337           System.err.print("SHUTDOWN: ");
338           for (final String string : results) {
339             System.err.println(string);
340           }
341         } finally {
342           client.disconnect();
343         }
344       }
345     } finally {
346       logger.warn("Will stop server");
347       WaarpSystemUtil.stopLogger(true);
348       FileBasedConfiguration.fileBasedConfiguration.setShutdown(true);
349       FileBasedConfiguration.fileBasedConfiguration.releaseResources();
350       try {
351         Thread.sleep(1000);
352       } catch (final InterruptedException ignored) {
353       }
354       File home = new File("/tmp/FTP");
355       FileUtils.forceDeleteRecursiveDir(home);
356     }
357   }
