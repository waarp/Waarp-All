90    public void testFtp4J(String server, int port, String username, String passwd,
91                          String account, int isSSL, String localFilename,
92                          int type, int delay, boolean shutdown, int numberThread,
93                          int numberIteration) {
94      // initiate Directories
95      final Ftp4JClientTransactionTest client =
96          new Ftp4JClientTransactionTest(server, port, username, passwd, account,
97                                         isSSL);
98  
99      client.setReportActiveExternalIPAddress("127.0.0.1");
100     client.setActiveDataTransferPortRange(57000, 63000);
101     logger.warn("First connexion");
102     if (!client.connect()) {
103       logger.error("Can't connect");
104       FtpClientTest.numberKO.incrementAndGet();
105       assertEquals("No KO", 0, numberKO.get());
106       return;
107     }
108     try {
109       logger.warn("Create Dirs");
110       for (int i = 0; i < numberThread; i++) {
111         client.makeDir("T" + i);
112       }
113       try {
114         Thread.sleep(10);
115       } catch (InterruptedException e) {
116         // Ignore
117       }
118       logger.warn("Feature commands");
119       System.err.println("SITE: " + client.featureEnabled("SITE"));
120       System.err.println("SITE CRC: " + client.featureEnabled("SITE XCRC"));
121       System.err.println("CRC: " + client.featureEnabled("XCRC"));
122       System.err.println("MD5: " + client.featureEnabled("XMD5"));
123       System.err.println("SHA1: " + client.featureEnabled("XSHA1"));
124       System.err.println("DIGEST: " + client.featureEnabled("XDIGEST"));
125     } finally {
126       logger.warn("Logout");
127       client.logout();
128     }
129     final ExecutorService executorService = Executors.newCachedThreadPool();
130     logger.warn("Will start {} Threads", numberThread);
131     final long date1 = System.currentTimeMillis();
132     for (int i = 0; i < numberThread; i++) {
133       executorService.execute(
134           new FtpClientThread("T" + i, server, port, username, passwd, account,
135                               localFilename, numberIteration, type, delay,
136                               isSSL));
137       Thread.yield();
138     }
139     try {
140       Thread.sleep(100);
141     } catch (final InterruptedException e1) {
142       e1.printStackTrace();
143     }
144     executorService.shutdown();
145     long date2 = 0;
146     try {
147       if (!executorService.awaitTermination(120000, TimeUnit.MILLISECONDS)) {
148         executorService.shutdownNow();
149         if (!executorService.awaitTermination(120, TimeUnit.SECONDS)) {
150           System.err.println("Really not shutdown normally");
151         }
152       }
153     } catch (final InterruptedException e) {
154       e.printStackTrace();
155       executorService.shutdownNow();
156     }
157     date2 = System.currentTimeMillis();
158 
159     logger.warn(
160         localFilename + ' ' + numberThread + ' ' + numberIteration + ' ' +
161         type + " Real: " + (date2 - date1) + " OK: " + numberOK.get() +
162         " KO: " + numberKO.get() + " Trf/s: " +
163         numberOK.get() * 1000 / (date2 - date1));
164     assertEquals("No KO", 0, numberKO.get());
165   }

167   @BeforeClass
168   public static void startServer() throws Exception {
169     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
170         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
171     ResourceLeakDetector.setLevel(
172         SystemPropertyUtil.get(IT_LONG_TEST, false)? Level.SIMPLE :
173             Level.PARANOID);
174     WaarpSystemUtil.setJunit(true);
175     // R66 Home
176     File home = new File("/tmp/FTP");
177     home.mkdirs();
178     FileUtils.forceDeleteRecursiveDir(home);
179     final File localFilename = new File("/tmp/ftpfile.bin");
180     FileTestUtils.createTestFile(localFilename, 100);
181     final ClassLoader classLoader = FtpClientTest.class.getClassLoader();
182     File file = new File(classLoader.getResource("Gg-FTP.xml").getFile());
183     if (file.exists()) {
184       driverType = DriverType.FIREFOX;
185       initiateWebDriver(file.getParentFile());
186     }
187     final File tmp = new File("/tmp");
188     final File[] files = tmp.listFiles(new FileFilter() {
189       @Override
190       public boolean accept(File file) {
191         return file.getName().startsWith("ggftp");
192       }
193     });
194     for (final File fileDb : files) {
195       fileDb.delete();
196     }
197     final FileBasedConfiguration configuration =
198         new FileBasedConfiguration(ExecGatewayFtpServer.class,
199                                    ExecBusinessHandler.class,
200                                    FileSystemBasedDataBusinessHandler.class,
201                                    new FilesystemBasedFileParameterImpl());
202     try {
203       if (!configuration.setConfigurationServerFromXml(
204           file.getAbsolutePath())) {
205         System.err.println("Bad main configuration");
206         Assert.fail("Bad main configuration");
207       }
208       // Init database
209       try {
210         ServerInitDatabase.initdb();
211       } catch (final WaarpDatabaseNoConnectionException e) {
212         logger.error("Cannot connect to database");
213         return;
214       }
215       System.out.println("End creation");
216     } finally {
217       if (DbConstantFtp.gatewayAdmin != null) {
218         DbConstantFtp.gatewayAdmin.close();
219       }
220     }
221     logger.warn("Will start server");
222     key = configuration.getCryptoKey().decryptHexInString("c5f4876737cf351a");
223 
224     ExecGatewayFtpServer.main(new String[] { file.getAbsolutePath() });
225     try {
226       Thread.sleep(500);
227     } catch (InterruptedException e) {
228       // Wait for server started
229     }
230     System.gc();
231     try {
232       Thread.sleep(500);
233     } catch (InterruptedException e) {
234       // Wait for server started
235     }
236   }

238   @AfterClass
239   public static void stopServer() throws InterruptedException {
240     logger.warn("Will shutdown from client");
241     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
242         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
243     try {
244       Thread.sleep(200);
245     } catch (final InterruptedException ignored) {
246     }
247     finalizeDriver();
248     try {
249       final Ftp4JClientTransactionTest client =
250           new Ftp4JClientTransactionTest("127.0.0.1", 2021, "fredo", key, "a",
251                                          0);
252       if (!client.connect()) {
253         logger.warn("Cant connect");
254       } else {
255         try {
256           final String[] results =
257               client.executeSiteCommand("internalshutdown pwdhttp");
258           System.err.print("SHUTDOWN: ");
259           for (final String string : results) {
260             System.err.println(string);
261           }
262         } finally {
263           client.disconnect();
264         }
265       }
266     } finally {
267       logger.warn("Will stop server");
268       WaarpSystemUtil.stopLogger(true);
269       FileBasedConfiguration.fileBasedConfiguration.setShutdown(true);
270       FileBasedConfiguration.fileBasedConfiguration.releaseResources();
271       try {
272         Thread.sleep(1000);
273       } catch (final InterruptedException ignored) {
274       }
275       File home = new File("/tmp/FTP");
276       FileUtils.forceDeleteRecursiveDir(home);
277     }
278   }
