106   public void testFtp4J(String server, int port, String username, String passwd,
107                         String account, int isSSL, String localFilename,
108                         int type, int delay, boolean shutdown, int numberThread,
109                         int numberIteration) {
110     // initiate Directories
111     final Ftp4JClientTransactionTest client =
112         new Ftp4JClientTransactionTest(server, port, username, passwd, account,
113                                        isSSL);
114 
115     client.setReportActiveExternalIPAddress("127.0.0.1");
116     client.setActiveDataTransferPortRange(57000, 63000);
117     logger.warn("First connexion");
118     if (!client.connect()) {
119       logger.error("Can't connect");
120       FtpClientTest.numberKO.incrementAndGet();
121       assertEquals("No KO", 0, numberKO.get());
122       return;
123     }
124     try {
125       logger.warn("Create Dirs");
126       for (int i = 0; i < numberThread; i++) {
127         client.makeDir("T" + i);
128       }
129       try {
130         Thread.sleep(10);
131       } catch (InterruptedException e) {
132         // Ignore
133       }
134       logger.warn("Feature commands");
135       System.err.println("SITE: " + client.featureEnabled("SITE"));
136       System.err.println("SITE CRC: " + client.featureEnabled("SITE XCRC"));
137       System.err.println("CRC: " + client.featureEnabled("XCRC"));
138       System.err.println("MD5: " + client.featureEnabled("XMD5"));
139       System.err.println("SHA1: " + client.featureEnabled("XSHA1"));
140       System.err.println("DIGEST: " + client.featureEnabled("XDIGEST"));
141     } finally {
142       logger.warn("Logout");
143       client.logout();
144     }
145     final ExecutorService executorService = Executors.newCachedThreadPool();
146     logger.warn("Will start {} Threads", numberThread);
147     final long date1 = System.currentTimeMillis();
148     for (int i = 0; i < numberThread; i++) {
149       executorService.execute(
150           new FtpClientThread("T" + i, server, port, username, passwd, account,
151                               localFilename, numberIteration, type, delay,
152                               isSSL));
153       Thread.yield();
154     }
155     try {
156       Thread.sleep(100);
157     } catch (final InterruptedException e1) {
158       e1.printStackTrace();
159     }
160     executorService.shutdown();
161     long date2 = 0;
162     try {
163       if (!executorService.awaitTermination(120000, TimeUnit.MILLISECONDS)) {
164         executorService.shutdownNow();
165         if (!executorService.awaitTermination(120, TimeUnit.SECONDS)) {
166           System.err.println("Really not shutdown normally");
167         }
168       }
169     } catch (final InterruptedException e) {
170       e.printStackTrace();
171       executorService.shutdownNow();
172     }
173     date2 = System.currentTimeMillis();
174 
175     logger.warn(
176         localFilename + ' ' + numberThread + ' ' + numberIteration + ' ' +
177         type + " Real: " + (date2 - date1) + " OK: " + numberOK.get() +
178         " KO: " + numberKO.get() + " Trf/s: " +
179         numberOK.get() * 1000 / (date2 - date1));
180     assertEquals("No KO", 0, numberKO.get());
181   }

252   @AfterClass
253   public static void stopServer() throws InterruptedException {
254     logger.warn("Will shutdown from client");
255     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
256         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
257     try {
258       Thread.sleep(200);
259     } catch (final InterruptedException ignored) {
260     }
261     finalizeDriver();
262     try {
263       final Ftp4JClientTransactionTest client =
264           new Ftp4JClientTransactionTest("127.0.0.1", 2021, "fredo", key, "a",
265                                          0);
266       if (!client.connect()) {
267         logger.warn("Cant connect");
268       } else {
269         try {
270           final String[] results =
271               client.executeSiteCommand("internalshutdown pwdhttp");
272           System.err.print("SHUTDOWN: ");
273           for (final String string : results) {
274             System.err.println(string);
275           }
276         } finally {
277           client.disconnect();
278         }
279       }
280     } finally {
281       logger.warn("Will stop server");
282       WaarpSystemUtil.stopLogger(true);
283       FileBasedConfiguration.fileBasedConfiguration.setShutdown(true);
284       FileBasedConfiguration.fileBasedConfiguration.releaseResources();
285       try {
286         Thread.sleep(1000);
287       } catch (final InterruptedException ignored) {
288       }
289       File home = new File("/tmp/FTP");
290       FileUtils.forceDeleteRecursiveDir(home);
291     }
292   }

322   public static void initiateWebDriver(File file) {
323     File libdir = file.getParentFile().getParentFile().getParentFile();
324     // test-classes -> target -> WaarpR66 -> lib -> geckodriver (linux x64)
325     File libPhantomJS = new File("/tmp/phantomjs-2.1.1");
326     if (!libPhantomJS.canRead()) {
327       File libPhantomJSZip = new File(libdir, "lib/phantomjs-2.1.1.bz2");
328       if (libPhantomJSZip.canRead()) {
329         FileUtils.uncompressedBz2File(libPhantomJSZip, libPhantomJS);
330         libPhantomJS.setExecutable(true);
331       }
332     }
333     assertTrue(libPhantomJS.exists());
334     System.setProperty("phantomjs.binary.path", libPhantomJS.getAbsolutePath());
335     try {
336       driver = initializeDriver();
337     } catch (InterruptedException e) {
338       e.printStackTrace();
339       fail(e.getMessage());
340     }
341   }

343   public static void reloadDriver() throws InterruptedException {
344     if (driver != null) {
345       finalizeDriver();
346     }
347     driver = initializeDriver();
348   }

350   public static WebDriver initializeDriver() throws InterruptedException {
351     WebDriver driver;
352     switch (driverType) {
353       case PHANTOMJS:
354       default:
355         driver = createPhantomJSDriver();
356     }
357     driver.manage().timeouts().implicitlyWait(5, TimeUnit.SECONDS);
358     //driver.manage().timeouts().pageLoadTimeout(10, TimeUnit.SECONDS);
359     Thread.sleep(10);
360     return driver;
361   }

363   private static WebDriver createPhantomJSDriver() {
364     DesiredCapabilities desiredCapabilities =
365         new DesiredCapabilities("phantomjs", "", Platform.ANY);
366     desiredCapabilities.setCapability(
367         PhantomJSDriverService.PHANTOMJS_EXECUTABLE_PATH_PROPERTY,
368         System.getProperty("phantomjs.binary.path"));
369     desiredCapabilities.setCapability(CapabilityType.ELEMENT_SCROLL_BEHAVIOR,
370                                       true);
371     desiredCapabilities.setCapability(CapabilityType.TAKES_SCREENSHOT, false);
372     desiredCapabilities.setCapability(
373         CapabilityType.ENABLE_PROFILING_CAPABILITY, false);
374     LoggingPreferences logPrefs = new LoggingPreferences();
375     logPrefs.enable(LogType.BROWSER, java.util.logging.Level.OFF);
376     logPrefs.enable(LogType.CLIENT, java.util.logging.Level.OFF);
377     logPrefs.enable(LogType.DRIVER, java.util.logging.Level.OFF);
378     logPrefs.enable(LogType.PERFORMANCE, java.util.logging.Level.OFF);
379     logPrefs.enable(LogType.PROFILER, java.util.logging.Level.OFF);
380     logPrefs.enable(LogType.SERVER, java.util.logging.Level.OFF);
381     desiredCapabilities.setCapability(CapabilityType.LOGGING_PREFS, logPrefs);
382     desiredCapabilities.setCapability(CapabilityType.HAS_NATIVE_EVENTS, true);
383     desiredCapabilities.setCapability(
384         PhantomJSDriverService.PHANTOMJS_GHOSTDRIVER_CLI_ARGS,
385         "--webdriver-loglevel=NONE");
386 
387     desiredCapabilities.setJavascriptEnabled(true);
388 
389     ArrayList<String> cliArgs = new ArrayList<String>();
390     cliArgs.add("--web-security=true");
391     cliArgs.add("--ignore-ssl-errors=true");
392     cliArgs.add("--webdriver-loglevel=NONE");
393     desiredCapabilities.setCapability(PhantomJSDriverService.PHANTOMJS_CLI_ARGS,
394                                       cliArgs);
395 
396     PhantomJSDriver phantomJSDriver = new PhantomJSDriver(desiredCapabilities);
397     phantomJSDriver.setLogLevel(java.util.logging.Level.OFF);
398     return phantomJSDriver;
399   }
