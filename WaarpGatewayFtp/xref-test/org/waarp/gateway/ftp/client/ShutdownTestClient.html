19    public static void main(String[] args) throws Exception {
20      if (args.length == 0) {
21        System.err.println("No argument: configuration file missing");
22        System.exit(3);
23      }
24      File file = new File(args[0]);
25      final FileBasedConfiguration configuration =
26          new FileBasedConfiguration(ExecGatewayFtpServer.class,
27                                     ExecBusinessHandler.class,
28                                     FileSystemBasedDataBusinessHandler.class,
29                                     new FilesystemBasedFileParameterImpl());
30      try {
31        if (!configuration.setConfigurationServerFromXml(
32            file.getAbsolutePath())) {
33          System.err.println("Bad main configuration");
34          System.exit(1);
35        }
36      } finally {
37        if (DbConstantFtp.gatewayAdmin != null) {
38          DbConstantFtp.gatewayAdmin.close();
39        }
40      }
41      System.err.println("Will start server");
42      String key =
43          configuration.getCryptoKey().decryptHexInString("c5f4876737cf351a");
44      final Ftp4JClientTransactionTest client =
45          new Ftp4JClientTransactionTest("127.0.0.1", 2021, "fredo", key, "a", 0);
46      if (!client.connect()) {
47        System.err.println("Cant connect");
48        System.exit(2);
49        return;
50      }
51      try {
52        final String[] results =
53            client.executeSiteCommand("internalshutdown pwdhttp");
54        System.err.print("SHUTDOWN: ");
55        for (final String string : results) {
56          System.err.println(string);
57        }
58      } finally {
59        client.disconnect();
60      }
61    }
