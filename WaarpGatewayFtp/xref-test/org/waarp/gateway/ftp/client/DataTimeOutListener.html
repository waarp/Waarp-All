20  package org.waarp.gateway.ftp.client;
21  
22  import it.sauronsoftware.ftp4j.FTPClient;
23  import it.sauronsoftware.ftp4j.FTPDataTransferListener;
24  import it.sauronsoftware.ftp4j.FTPIllegalReplyException;
25  import org.waarp.common.logging.WaarpLogger;
26  import org.waarp.common.logging.WaarpLoggerFactory;
27  
28  import java.io.IOException;
29  import java.util.Timer;
30  import java.util.TimerTask;
31  
32  /**
33   * Timeout listener for Data connection for FTP4J model
34   */
35  public class DataTimeOutListener implements FTPDataTransferListener {
36    /**
37     * Internal Logger
38     */
39    private static final WaarpLogger logger = WaarpLoggerFactory
40        .getLogger(org.waarp.ftp.client.DataTimeOutListener.class);
41  
42    private final FTPClient client;
43    private final Timer timer;
44    private final String command;
45    private final String file;
46    private long timeout = 10000;
47    private long last = System.currentTimeMillis();
48    private boolean finished;
49  
50    public DataTimeOutListener(FTPClient client, long timeout, String command,
51                               String file) {
52      this.client = client;
53      timer = new Timer(true);
54      this.timeout = timeout;
55      this.command = command;
56      this.file = file;
57    }
58  
59    private void renewTask() {
60      final TimerTask task = new TimerTask() {
61        @Override
62        public void run() {
63          if (finished) {
64            return;
65          }
66          final long now = System.currentTimeMillis();
67          if (now - last - timeout > 0) {
68            try {
69              logger
70                  .warn("Timeout during file transfer: " + command + ' ' + file);
71              client.abortCurrentDataTransfer(true);
72            } catch (final IOException ignored) {
73            } catch (final FTPIllegalReplyException ignored) {
74            }
75          } else {
76            renewTask();
77          }
78        }
79      };
80      timer.schedule(task, timeout);
81    }
82  
83    @Override
84    public void started() {
85      renewTask();
86      last = System.currentTimeMillis();
87    }
88  
89    @Override
90    public void transferred(int length) {
91      last = System.currentTimeMillis();
92    }
93  
94    @Override
95    public void completed() {
96      finished = true;
97      last = System.currentTimeMillis();
98      timer.cancel();
99    }
100 
101   @Override
102   public void aborted() {
103     finished = true;
104     last = System.currentTimeMillis();
105     timer.cancel();
106   }
107 
108   @Override
109   public void failed() {
110     finished = true;
111     last = System.currentTimeMillis();
112     timer.cancel();
113   }
114 
115 }
