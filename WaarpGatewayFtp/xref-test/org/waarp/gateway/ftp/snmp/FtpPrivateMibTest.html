90    @BeforeClass
91    public static void startServer() throws Exception {
92      WaarpLoggerFactory.setDefaultFactoryIfNotSame(
93          new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
94      ResourceLeakDetector.setLevel(Level.PARANOID);
95      WaarpSystemUtil.setJunit(true);
96      // R66 Home
97      File home = new File("/tmp/FTP");
98      home.mkdirs();
99      FileUtils.forceDeleteRecursiveDir(home);
100     final ClassLoader classLoader = FtpClientTest.class.getClassLoader();
101     File file = new File(classLoader.getResource("Gg-FTP.xml").getFile());
102     final File tmp = new File("/tmp");
103     final File[] files = tmp.listFiles(new FileFilter() {
104       @Override
105       public boolean accept(File file) {
106         return file.getName().startsWith("ggftp");
107       }
108     });
109     for (final File fileDb : files) {
110       fileDb.delete();
111     }
112     final FileBasedConfiguration configuration =
113         new FileBasedConfiguration(ExecGatewayFtpServer.class,
114                                    ExecBusinessHandler.class,
115                                    FileSystemBasedDataBusinessHandler.class,
116                                    new FilesystemBasedFileParameterImpl());
117     try {
118       if (!configuration.setConfigurationServerFromXml(
119           file.getAbsolutePath())) {
120         logger.error("Bad main configuration");
121         fail("Bad main configuration");
122       }
123       // Init database
124       try {
125         ServerInitDatabase.initdb();
126       } catch (final WaarpDatabaseNoConnectionException e) {
127         SysErrLogger.FAKE_LOGGER.syserr("Cannot connect to database");
128         return;
129       }
130       System.out.println("End creation");
131     } finally {
132       if (DbConstantFtp.gatewayAdmin != null) {
133         DbConstantFtp.gatewayAdmin.close();
134       }
135     }
136     logger.warn("Will start server");
137     key = configuration.getCryptoKey().decryptHexInString("c5f4876737cf351a");
138 
139     ExecGatewayFtpServer.main(new String[] { file.getAbsolutePath() });
140     try {
141       Thread.sleep(1000);
142     } catch (InterruptedException e) {
143       // Wait for server started
144     }
145   }

147   @AfterClass
148   public static void stopServer() throws InterruptedException {
149     logger.warn("Will shutdown from client");
150     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
151         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
152     try {
153       Thread.sleep(200);
154     } catch (final InterruptedException ignored) {
155     }
156     try {
157       final Ftp4JClientTransactionTest client =
158           new Ftp4JClientTransactionTest("127.0.0.1", 2021, "fredo", key, "a",
159                                          0);
160       if (!client.connect()) {
161         logger.warn("Cant connect");
162       } else {
163         try {
164           final String[] results =
165               client.executeSiteCommand("internalshutdown pwdhttp");
166           System.err.print("SHUTDOWN: ");
167           for (final String string : results) {
168             System.err.println(string);
169           }
170         } finally {
171           client.disconnect();
172         }
173       }
174     } finally {
175       logger.warn("Will stop server");
176       WaarpSystemUtil.stopLogger(true);
177       FileBasedConfiguration.fileBasedConfiguration.setShutdown(true);
178       FileBasedConfiguration.fileBasedConfiguration.releaseResources();
179       try {
180         Thread.sleep(1000);
181       } catch (final InterruptedException ignored) {
182       }
183       File home = new File("/tmp/FTP");
184       FileUtils.forceDeleteRecursiveDir(home);
185     }
186   }

188   @Test
189   public void allTests() throws Exception {
190     WaarpLoggerFactory.setDefaultFactoryIfNotSame(
191         new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
192     ResourceLeakDetector.setLevel(Level.PARANOID);
193     final ClassLoader classLoader = FtpPrivateMibTest.class.getClassLoader();
194     final File file =
195         new File(classLoader.getResource("snmpconfig.xml").getFile());
196     if (!file.exists()) {
197       System.err.println("Need SnmpConfig file");
198       return;
199     }
200     setUp(file.getAbsolutePath());
201     System.out.println("Test SysDescr");
202     verifySysDescr();
203     System.out.println("Test Table");
204     verifyTableContents();
205     Thread.sleep(1000);
206     sendNotification();
207     Thread.sleep(3000);
208     final long uptime = agent.getUptime();
209     final long systemUptime = agent.getUptimeSystemTime();
210     assertTrue(uptime < systemUptime);
211     addUser();
212     System.out.println("Stopping");
213     tearDown();
214   }

216   public static void setUp(String file) throws Exception {
217     // Setup the client to use our newly started agent
218     client = new WaarpSimpleSnmpClient("udp:127.0.0.1/2001", 1162);
219     // Create a monitor
220     DbConstant.admin = new DbAdmin();
221     FileBasedConfiguration.fileBasedConfiguration.setMonitoring(
222         new FtpMonitoring(null));
223     // Create a Mib
224     test = new FtpPrivateMib(
225         FileBasedConfiguration.fileBasedConfiguration.getServerPort());
226     // Set the default VariableFactory
227     WaarpMOFactory.setFactory(new FtpVariableFactory());
228     // Create the agent associated with the monitor and Mib
229     agent = new WaarpSnmpAgent(new File(file),
230                                FileBasedConfiguration.fileBasedConfiguration.getMonitoring(),
231                                test);
232     FileBasedConfiguration.fileBasedConfiguration.setAgentSnmp(agent);
233     agent.start();
234   }

324   private static boolean assertEquals(String a, String b) {
325     if (!a.equals(b)) {
326       System.err.println("Not Equals! " + a + " and " + b);
327       fail("Should be equals");
328       return false;
329     }
330     System.out.println("Equal: " + a);
331     return true;
332   }
