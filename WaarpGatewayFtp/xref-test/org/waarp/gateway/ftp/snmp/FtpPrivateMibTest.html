84    @BeforeClass
85    public static void startServer() throws Exception {
86      WaarpLoggerFactory
87          .setDefaultFactory(new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
88      ResourceLeakDetector.setLevel(Level.PARANOID);
89      DetectionUtils.setJunit(true);
90      // R66 Home
91      File home = new File("/tmp/FTP");
92      home.mkdirs();
93      FileUtils.forceDeleteRecursiveDir(home);
94      final ClassLoader classLoader = FtpClientTest.class.getClassLoader();
95      File file = new File(classLoader.getResource("Gg-FTP.xml").getFile());
96      final File tmp = new File("/tmp");
97      final File[] files = tmp.listFiles(new FileFilter() {
98        @Override
99        public boolean accept(File file) {
100         return file.getName().startsWith("ggftp");
101       }
102     });
103     for (final File fileDb : files) {
104       fileDb.delete();
105     }
106     final FileBasedConfiguration configuration =
107         new FileBasedConfiguration(ExecGatewayFtpServer.class,
108                                    ExecBusinessHandler.class,
109                                    FileSystemBasedDataBusinessHandler.class,
110                                    new FilesystemBasedFileParameterImpl());
111     try {
112       if (!configuration
113           .setConfigurationServerFromXml(file.getAbsolutePath())) {
114         logger.error("Bad main configuration");
115         fail("Bad main configuration");
116       }
117       // Init database
118       try {
119         ServerInitDatabase.initdb();
120       } catch (final WaarpDatabaseNoConnectionException e) {
121         SysErrLogger.FAKE_LOGGER.syserr("Cannot connect to database");
122         return;
123       }
124       System.out.println("End creation");
125     } finally {
126       if (DbConstantFtp.gatewayAdmin != null) {
127         DbConstantFtp.gatewayAdmin.close();
128       }
129     }
130     logger.warn("Will start server");
131     key = configuration.getCryptoKey().decryptHexInString("c5f4876737cf351a");
132 
133     ExecGatewayFtpServer.main(new String[] { file.getAbsolutePath() });
134   }

136   @AfterClass
137   public static void stopServer() throws InterruptedException {
138     logger.warn("Will shutdown from client");
139     try {
140       Thread.sleep(500);
141     } catch (final InterruptedException ignored) {//NOSONAR
142     }
143     try {
144       final Ftp4JClientTransactionTest client =
145           new Ftp4JClientTransactionTest("127.0.0.1", 2021, "fredo", key, "a",
146                                          0);
147       if (!client.connect()) {
148         logger.warn("Cant connect");
149         return;
150       }
151       try {
152         final String[] results =
153             client.executeSiteCommand("internalshutdown pwdhttp");
154         System.err.print("SHUTDOWN: ");
155         for (final String string : results) {
156           System.err.println(string);
157         }
158       } finally {
159         client.disconnect();
160       }
161     } finally {
162       logger.warn("Will stop server");
163       FileBasedConfiguration.fileBasedConfiguration.setShutdown(true);
164       FileBasedConfiguration.fileBasedConfiguration.releaseResources();
165       try {
166         Thread.sleep(1000);
167       } catch (final InterruptedException ignored) {//NOSONAR
168       }
169       File home = new File("/tmp/FTP");
170       FileUtils.forceDeleteRecursiveDir(home);
171     }
172   }

174   @Test
175   public void allTests() throws Exception {
176     WaarpLoggerFactory
177         .setDefaultFactory(new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
178     final ClassLoader classLoader = FtpPrivateMibTest.class.getClassLoader();
179     final File file =
180         new File(classLoader.getResource("snmpconfig.xml").getFile());
181     if (!file.exists()) {
182       System.err.println("Need SnmpConfig file");
183       return;
184     }
185     setUp(file.getAbsolutePath());
186     System.out.println("Test SysDescr");
187     verifySysDescr();
188     System.out.println("Test Table");
189     verifyTableContents();
190     Thread.sleep(1000);
191     sendNotification();
192     Thread.sleep(3000);
193     final long uptime = agent.getUptime();
194     final long systemUptime = agent.getUptimeSystemTime();
195     assertTrue(uptime < systemUptime);
196     addUser();
197     System.out.println("Stopping");
198     tearDown();
199   }

201   public static void setUp(String file) throws Exception {
202     // Setup the client to use our newly started agent
203     client = new WaarpSimpleSnmpClient("udp:127.0.0.1/2001", 1162);
204     // Create a monitor
205     DbConstant.admin = new DbAdmin();
206     FileBasedConfiguration.fileBasedConfiguration
207         .setMonitoring(new FtpMonitoring(null));
208     // Create a Mib
209     test = new FtpPrivateMib(
210         FileBasedConfiguration.fileBasedConfiguration.getServerPort());
211     // Set the default VariableFactory
212     WaarpMOFactory.setFactory(new FtpVariableFactory());
213     // Create the agent associated with the monitor and Mib
214     agent = new WaarpSnmpAgent(new File(file),
215                                FileBasedConfiguration.fileBasedConfiguration
216                                    .getMonitoring(), test);
217     FileBasedConfiguration.fileBasedConfiguration.setAgentSnmp(agent);
218     agent.start();
219   }

309   private static boolean assertEquals(String a, String b) {
310     if (!a.equals(b)) {
311       System.err.println("Not Equals! " + a + " and " + b);
312       fail("Should be equals");
313       return false;
314     }
315     System.out.println("Equal: " + a);
316     return true;
317   }
