91    protected static boolean getParams(String[] args) {
92      ParametersChecker.checkParameter(ParametersChecker.DEFAULT_ERROR, args);
93      if (args.length < 1) {
94        logger
95            .error(Messages.getString("Configuration.NeedConfig")); //$NON-NLS-1$
96        final JFileChooser chooser = new JFileChooser();
97        final int returnvval = chooser.showOpenDialog(null);
98        if (returnvval == JFileChooser.APPROVE_OPTION) {
99          final File file = chooser.getSelectedFile();
100         args = new String[] { file.getAbsolutePath() };
101       } else {
102         JOptionPane.showMessageDialog(null, Messages
103             .getString("Configuration.NeedConfig")); //$NON-NLS-1$
104         return false;
105       }
106     }
107     if (!FileBasedConfiguration
108         .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
109       logger
110           .error(Messages.getString("Configuration.NeedConfig")); //$NON-NLS-1$
111       return false;
112     }
113     Configuration.configuration.pipelineInit();
114     getEnvironnement().networkTransaction = new NetworkTransaction();
115     return true;
116   }

218     btnCheckPartners = new JButton("Check Partners");
219     btnCheckPartners.addActionListener(new ActionListener() {
220       @Override
221       public void actionPerformed(ActionEvent arg0) {
222         // Check all Known Partners
223         try {
224           TestPacket packet;
225           String myhost;
226           try {
227             myhost = getEnvironnement().hostId == null?
228                 InetAddress.getLocalHost().getHostName() :
229                 getEnvironnement().hostId;
230           } catch (final UnknownHostException e) {
231             myhost = Messages.getString("AdminGui.NameUnknown");
232           }
233           packet =
234               new TestPacket("MSG", "Administrator checking from " + myhost,
235                              100);
236           packet.retain();
237           StringBuilder result =
238               new StringBuilder(Messages.getString("AdminGui.CheckedHosts"));
239           for (final DbHostAuth host : DbHostAuth.getAllHosts()) {
240             final R66Future future = new R66Future(true);
241             final Message mesg =
242                 new Message(getEnvironnement().networkTransaction, future, host,
243                             packet);
244             mesg.run();
245             packet.retain();
246             future.awaitOrInterruptible();
247             if (future.isSuccess()) {
248               result.append("OK: ").append(host).append('\n');
249             } else {
250               result.append("KO: ").append(host).append('\n');
251             }
252           }
253           packet.clear();
254 
255           JOptionPane.showMessageDialog(null, result.toString());
256         } catch (final WaarpDatabaseNoConnectionException e) {
257           SysErrLogger.FAKE_LOGGER.syserr(e);
258         }
259       }
260     });

399   /**
400    * Quit application
401    */
402   private void quit() {
403     final List<AdminXample> list = new ArrayList<AdminXample>();
404     for (final AdminXample xample : xamples) {
405       if (xample.isStillLaunched()) {
406         list.add(xample);
407       }
408     }
409     for (final AdminXample xample : list) {
410       xample.exit();
411     }
412     final List<AdminUiPassword> list2 = new ArrayList<AdminUiPassword>();
413     list2.addAll(passwords);
414     for (final AdminUiPassword pwd : list2) {
415       pwd.exit(null);
416     }
417     if (R66ClientGui.window != null) {
418       R66ClientGui.window.frmRClientGui.dispose();
419       R66ClientGui.window = null;
420     }
421     frmWaarpRCentral.dispose();
422     System.exit(0);//NOSONAR
423   }
