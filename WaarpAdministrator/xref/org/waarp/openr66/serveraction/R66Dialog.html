20  package org.waarp.openr66.serveraction;
21  
22  import javax.swing.JButton;
23  import javax.swing.JDialog;
24  import javax.swing.JEditorPane;
25  import javax.swing.JPanel;
26  import javax.swing.JScrollPane;
27  import javax.swing.WindowConstants;
28  import javax.swing.border.EmptyBorder;
29  import java.awt.BorderLayout;
30  import java.awt.FlowLayout;
31  import java.awt.GridLayout;
32  import java.awt.event.ActionEvent;
33  import java.awt.event.ActionListener;
34  
35  /**
36   *
37   */
38  public class R66Dialog extends JDialog {
39  
40    /**
41     *
42     */
43    private static final long serialVersionUID = -6105635300084413738L;
44    private final JPanel contentPanel = new JPanel();
45    private JEditorPane textPaneDialog;
46  
47    /**
48     * Create the dialog.
49     */
50    public R66Dialog() {
51      setTitle("R66 Gui Dialog");
52      setBounds(100, 100, 426, 298);
53      getContentPane().setLayout(new BorderLayout());
54      contentPanel.setBorder(new EmptyBorder(5, 5, 5, 5));
55      getContentPane().add(contentPanel, BorderLayout.CENTER);
56      contentPanel.setLayout(new GridLayout(1, 0, 0, 0));
57      {
58        final JScrollPane scrollPane = new JScrollPane();
59        contentPanel.add(scrollPane);
60        {
61          setTextPaneDialog(new JEditorPane());
62          scrollPane.setViewportView(getTextPaneDialog());
63          getTextPaneDialog().setEditable(false);
64          getTextPaneDialog().setContentType("text/html");
65        }
66      }
67      {
68        final JPanel buttonPane = new JPanel();
69        buttonPane.setLayout(new FlowLayout(FlowLayout.CENTER));
70        getContentPane().add(buttonPane, BorderLayout.SOUTH);
71        {
72          final JButton okButton = new JButton("OK");
73          okButton.addActionListener(new ActionListener() {
74            @Override
75            public void actionPerformed(final ActionEvent e) {
76              AdminR66OperationsGui.getWindow().enableAllButtons();
77              dispose();
78            }
79          });
80          okButton.setActionCommand("OK");
81          buttonPane.add(okButton);
82          getRootPane().setDefaultButton(okButton);
83        }
84      }
85      setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
86    }
87  
88    /**
89     * @return the textPaneDialog
90     */
91    public JEditorPane getTextPaneDialog() {
92      return textPaneDialog;
93    }
94  
95    /**
96     * @param textPaneDialog the textPaneDialog to set
97     */
98    public void setTextPaneDialog(final JEditorPane textPaneDialog) {
99      this.textPaneDialog = textPaneDialog;
100   }
101 
102 }
