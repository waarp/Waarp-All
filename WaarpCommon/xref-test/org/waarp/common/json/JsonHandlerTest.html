42    /**
43     * Test method for {@link JsonHandler#createObjectNode()}.
44     */
45    @Test
46    public void testCreateObjectNode() {
47      final ObjectNode node = JsonHandler.createObjectNode();
48      JsonHandler.setValue(node, "boolean", true);
49      JsonHandler.setValue(node, "bytes", "bytes".getBytes());
50      JsonHandler.setValue(node, "double", 2.0);
51      JsonHandler.setValue(node, "int", 3);
52      JsonHandler.setValue(node, "long", 5L);
53      JsonHandler.setValue(node, "string", "string");
54      assertTrue(JsonHandler.exist(node, "boolean"));
55      assertTrue(JsonHandler.exist(node, "bytes"));
56      assertTrue(JsonHandler.exist(node, "double"));
57      assertTrue(JsonHandler.exist(node, "int"));
58      assertTrue(JsonHandler.exist(node, "long"));
59      assertTrue(JsonHandler.exist(node, "string"));
60      assertEquals(true, JsonHandler.getValue(node, "boolean", false));
61      assertArrayEquals("bytes".getBytes(),
62                        JsonHandler.getValue(node, "bytes", " ".getBytes()));
63      assertEquals(2.0, JsonHandler.getValue(node, "double", 1.0), 0.0);
64      assertEquals(3, (int) JsonHandler.getValue(node, "int", 2));
65      assertEquals(5L, (long) JsonHandler.getValue(node, "long", 0L));
66      assertEquals("string", JsonHandler.getValue(node, "string", "no"));
67      final String result = JsonHandler.writeAsString(node);
68      final ObjectNode node2 = JsonHandler.getFromString(result);
69      assertEquals(result, JsonHandler.writeAsString(node2));
70      final Map<String, Object> map = JsonHandler.getMapFromString(result);
71      assertEquals(true, map.get("boolean"));
72      assertEquals(2.0, (Double) map.get("double"), 0.0);
73      assertEquals(3, (int) (Integer) map.get("int"));
74      assertEquals(5L, (int) (Integer) map.get("long"));
75      assertEquals("string", map.get("string"));
76    }

81    @Test
82    public void testCreateArrayNode() {
83      final ArrayNode node = JsonHandler.createArrayNode();
84      node.add(true);
85      node.add("bytes".getBytes());
86      node.add(2.0);
87      node.add(3);
88      node.add(5L);
89      node.add("string");
90      assertEquals(6, node.size());
91      final String result = JsonHandler.writeAsString(node);
92      ArrayNode node2;
93      try {
94        node2 = (ArrayNode) JsonHandler.mapper.readTree(result);
95      } catch (final JsonProcessingException e) {
96        fail(e.getMessage());
97        return;
98      } catch (final Exception e) {
99        fail(e.getMessage());
100       return;
101     }
102     assertEquals(result, JsonHandler.writeAsString(node2));
103   }

132   @Test
133   public void testMapRendering() {
134     final String maps = "{\"foo\": \"bar\", \"foo2\": true}";
135     final String maps2 = "{\"foo\":\"bar\",\"foo2\":true}";
136     String unescaped = JsonHandler.unEscape(maps);
137     assertEquals(maps, unescaped);
138     Map<String, Object> map = JsonHandler.getMapFromString(maps);
139     assertEquals(true, map.get("foo2"));
140     assertEquals("bar", map.get("foo"));
141     String escaped = JsonHandler.writeAsStringEscaped(map);
142     assertEquals("{\"foo\":\"bar\",\"foo2\":true}", escaped);
143     String unescaped2 = JsonHandler.unEscape(escaped);
144     assertEquals(maps2, unescaped2);
145 
146     final String maps3 = "{\\\\\"foo\": \"bar\", \"foo2\": true}";
147     unescaped = JsonHandler.unEscape(maps3);
148     assertEquals(maps, unescaped);
149     map = JsonHandler.getMapFromString(maps);
150     assertEquals(true, map.get("foo2"));
151     assertEquals("bar", map.get("foo"));
152     escaped = JsonHandler.writeAsStringEscaped(map);
153     assertEquals("{\"foo\":\"bar\",\"foo2\":true}", escaped);
154     unescaped2 = JsonHandler.unEscape(escaped);
155     assertEquals(maps2, unescaped2);
156   }
