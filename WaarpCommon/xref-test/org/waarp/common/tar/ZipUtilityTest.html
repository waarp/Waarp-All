21  package org.waarp.common.tar;
22  
23  import org.junit.Test;
24  import org.waarp.common.file.FileUtils;
25  
26  import java.io.File;
27  import java.io.IOException;
28  import java.util.ArrayList;
29  import java.util.Collections;
30  import java.util.List;
31  
32  import static org.junit.Assert.*;
33  
34  public class ZipUtilityTest {
35  
36    @Test
37    public void createZipFromDirectory() throws IOException {
38      final File dir = new File("/tmp/tests/dir");
39      FileUtils.createDir(dir);
40      final File test = new File(dir, "test");
41      test.createNewFile();
42      final File test2 = new File(dir, "test2");
43      test2.createNewFile();
44      final File zip = new File(dir.getParentFile(), "testzip.zip");
45      assertTrue(ZipUtility.createZipFromDirectory(dir.getAbsolutePath(),
46                                                   zip.getAbsolutePath(), false));
47      FileUtils.forceDeleteRecursiveDir(dir);
48  
49      final List<String> list = ZipUtility.unZip(zip, dir);
50      assertEquals(2, list.size());
51      final File[] arrayFile = FileUtils.getFiles(dir);
52      assertEquals(2, arrayFile.length);
53      int cpt = 0;
54      for (final File file : arrayFile) {
55        for (final String name : list) {
56          if (name.equals(file.getName())) {
57            cpt++;
58            break;
59          }
60        }
61      }
62      assertEquals(2, cpt);
63      FileUtils.forceDeleteRecursiveDir(dir);
64      FileUtils.deleteDir(dir);
65      FileUtils.delete(zip);
66    }
67  
68    @Test
69    public void createZipFromFiles() throws IOException {
70      final File dir = new File("/tmp/tests/dir");
71      FileUtils.createDir(dir);
72      final File test = new File(dir, "test");
73      test.createNewFile();
74      final File test2 = new File(dir, "test2");
75      test2.createNewFile();
76      File[] arrayFile = dir.listFiles();
77      final List<File> listFile = new ArrayList<File>();
78      Collections.addAll(listFile, arrayFile);
79      final File zip = new File(dir.getParentFile(), "testzip.zip");
80      assertTrue(ZipUtility.createZipFromFiles(listFile, zip.getAbsolutePath()));
81      FileUtils.forceDeleteRecursiveDir(dir);
82  
83      List<String> list = ZipUtility.unZip(zip, dir);
84      assertEquals(2, list.size());
85      arrayFile = FileUtils.getFiles(dir);
86      assertEquals(2, arrayFile.length);
87      int cpt = 0;
88      for (final File file : arrayFile) {
89        for (final String name : list) {
90          if (name.equals(file.getName())) {
91            cpt++;
92            break;
93          }
94        }
95      }
96      assertEquals(2, cpt);
97      FileUtils.delete(zip);
98      assertTrue(ZipUtility.createZipFromFiles(arrayFile, zip.getAbsolutePath()));
99      FileUtils.forceDeleteRecursiveDir(dir);
100 
101     list = ZipUtility.unZip(zip, dir);
102     assertEquals(2, list.size());
103     final File[] arrayFiles2 = arrayFile = FileUtils.getFiles(dir);
104     assertEquals(2, arrayFiles2.length);
105     cpt = 0;
106     for (final File file : arrayFile) {
107       for (final File file2 : arrayFiles2) {
108         if (file2.getName().equals(file.getName())) {
109           cpt++;
110           break;
111         }
112       }
113     }
114     assertEquals(2, cpt);
115 
116     FileUtils.forceDeleteRecursiveDir(dir);
117     FileUtils.deleteDir(dir);
118     FileUtils.delete(zip);
119   }
