37  public class ZipUtilityTest {
38    @Rule(order = Integer.MIN_VALUE)
39    public TestWatcher watchman = new TestWatcherJunit4();

41  
42    @Test
43    public void createZipFromDirectory() throws IOException {
44      final File dir = new File("/tmp/tests/dir");
45      FileUtils.createDir(dir);
46      final File test = new File(dir, "test");
47      test.createNewFile();
48      final File test2 = new File(dir, "test2");
49      test2.createNewFile();
50      final File zip = new File(dir.getParentFile(), "testzip.zip");
51      assertTrue(ZipUtility.createZipFromDirectory(dir.getAbsolutePath(),
52                                                   zip.getAbsolutePath(), false));
53      FileUtils.forceDeleteRecursiveDir(dir);
54  
55      final List<String> list = ZipUtility.unZip(zip, dir);
56      assertEquals(2, list.size());
57      final File[] arrayFile = FileUtils.getFiles(dir);
58      assertEquals(2, arrayFile.length);
59      int cpt = 0;
60      for (final File file : arrayFile) {
61        for (final String name : list) {
62          if (name.equals(file.getName())) {
63            cpt++;
64            break;
65          }
66        }
67      }
68      assertEquals(2, cpt);
69      FileUtils.forceDeleteRecursiveDir(dir);
70      FileUtils.deleteDir(dir);
71      FileUtils.delete(zip);
72    }
73  
74    @Test
75    public void createZipFromFiles() throws IOException {
76      final File dir = new File("/tmp/tests/dir");
77      FileUtils.createDir(dir);
78      final File test = new File(dir, "test");
79      test.createNewFile();
80      final File test2 = new File(dir, "test2");
81      test2.createNewFile();
82      File[] arrayFile = dir.listFiles();
83      final List<File> listFile = new ArrayList<File>();
84      Collections.addAll(listFile, arrayFile);
85      final File zip = new File(dir.getParentFile(), "testzip.zip");
86      assertTrue(ZipUtility.createZipFromFiles(listFile, zip.getAbsolutePath()));
87      FileUtils.forceDeleteRecursiveDir(dir);
88  
89      List<String> list = ZipUtility.unZip(zip, dir);
90      assertEquals(2, list.size());
91      arrayFile = FileUtils.getFiles(dir);
92      assertEquals(2, arrayFile.length);
93      int cpt = 0;
94      for (final File file : arrayFile) {
95        for (final String name : list) {
96          if (name.equals(file.getName())) {
97            cpt++;
98            break;
99          }
100       }
101     }
102     assertEquals(2, cpt);
103     FileUtils.delete(zip);
104     assertTrue(ZipUtility.createZipFromFiles(arrayFile, zip.getAbsolutePath()));
105     FileUtils.forceDeleteRecursiveDir(dir);
106 
107     list = ZipUtility.unZip(zip, dir);
108     assertEquals(2, list.size());
109     final File[] arrayFiles2 = arrayFile = FileUtils.getFiles(dir);
110     assertEquals(2, arrayFiles2.length);
111     cpt = 0;
112     for (final File file : arrayFile) {
113       for (final File file2 : arrayFiles2) {
114         if (file2.getName().equals(file.getName())) {
115           cpt++;
116           break;
117         }
118       }
119     }
120     assertEquals(2, cpt);
121 
122     FileUtils.forceDeleteRecursiveDir(dir);
123     FileUtils.deleteDir(dir);
124     FileUtils.delete(zip);
125   }
