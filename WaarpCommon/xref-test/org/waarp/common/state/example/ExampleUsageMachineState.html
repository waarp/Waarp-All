20  package org.waarp.common.state.example;
21  
22  import org.waarp.common.exception.IllegalFiniteStateException;
23  import org.waarp.common.logging.WaarpLoggerFactory;
24  import org.waarp.common.logging.WaarpSlf4JLoggerFactory;
25  import org.waarp.common.state.MachineState;
26  import org.waarp.common.state.example.ExampleEnumState.ExampleTransition;
27  
28  import java.util.EnumSet;
29  import java.util.concurrent.ConcurrentHashMap;
30  
31  /**
32   * Example of usage for MachineState based on ExampleEnumState enum class
33   */
34  public class ExampleUsageMachineState {
35    /**
36     * An example of usage.
37     *
38     * @param args
39     */
40    @SuppressWarnings({ "unchecked", "rawtypes" })
41    public static void main(String[] args) {
42      WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
43  
44      // Example
45  
46      // First create a HashMap and fill it directly
47      final ConcurrentHashMap<ExampleEnumState, EnumSet<ExampleEnumState>>
48          stateMap =
49          new ConcurrentHashMap<ExampleEnumState, EnumSet<ExampleEnumState>>();
50      stateMap.put(ExampleTransition.tRUNNING.elt.getState(),
51                   ExampleTransition.tRUNNING.elt.getSet());
52      // Second create the MachineState with the right Map
53      final MachineState<ExampleEnumState> machineState1 =
54          new MachineState(ExampleEnumState.PAUSED, stateMap);
55      // Third, if not already done, fill the Map with the transitions
56      for (final ExampleTransition trans : ExampleTransition.values()) {
57        machineState1.addNewAssociation(trans.elt);
58      }
59  
60      // Or First create the MachineSate but empty
61      final MachineState<ExampleEnumState> machineState2 =
62          new MachineState(ExampleEnumState.PAUSED);
63      // Second fill the associations with transitions since none exist yet
64      for (final ExampleTransition trans : ExampleTransition.values()) {
65        machineState2.addNewAssociation(trans.elt);
66      }
67  
68      System.out.println("Machine1 states...");
69      changeState(machineState1, ExampleEnumState.CONFIGURING);
70      changeState(machineState1, ExampleEnumState.PAUSED);
71      changeState(machineState1, ExampleEnumState.RUNNING);
72      changeState(machineState1, ExampleEnumState.ENDED);
73      changeState(machineState1, ExampleEnumState.PAUSED);
74      changeState(machineState1, ExampleEnumState.RESET);
75      changeState(machineState1, ExampleEnumState.PAUSED);
76  
77      System.out.println("Machine2 states...");
78      changeState(machineState2, ExampleEnumState.CONFIGURING);
79      changeState(machineState2, ExampleEnumState.PAUSED);
80      changeState(machineState2, ExampleEnumState.RUNNING);
81      changeState(machineState2, ExampleEnumState.ENDED);
82      changeState(machineState2, ExampleEnumState.PAUSED);
83      changeState(machineState2, ExampleEnumState.RESET);
84      changeState(machineState2, ExampleEnumState.PAUSED);
85    }
86  
87    private static void changeState(MachineState<ExampleEnumState> mach,
88                                    ExampleEnumState desired) {
89      try {
90        printState(mach);
91        mach.setCurrent(desired);
92        printState(mach);
93      } catch (final IllegalFiniteStateException e) {
94        printWrongState(mach, desired);
95      }
96    }
97  
98    private static void printState(MachineState<ExampleEnumState> mach) {
99      System.out.println("State is " + mach.getCurrent());
100   }
101 
102   private static void printWrongState(MachineState<ExampleEnumState> mach,
103                                       ExampleEnumState desired) {
104     System.out.println(
105         "Cannot go from State " + mach.getCurrent() + " to State " + desired);
106   }
107 
108 }
