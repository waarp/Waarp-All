20  package org.waarp.common.state.example;
21  
22  import org.waarp.common.exception.IllegalFiniteStateException;
23  import org.waarp.common.logging.WaarpLoggerFactory;
24  import org.waarp.common.logging.WaarpSlf4JLoggerFactory;
25  import org.waarp.common.state.MachineState;
26  import org.waarp.common.state.example.ExampleEnumState.ExampleTransition;
27  
28  import java.util.EnumSet;
29  import java.util.concurrent.ConcurrentHashMap;
30  
31  /**
32   * Example of usage for MachineState based on ExampleEnumState enum class
33   */
34  public class ExampleUsageMachineState {
35    /**
36     * An example of usage.
37     *
38     * @param args
39     */
40    @SuppressWarnings({ "unchecked", "rawtypes" })
41    public static void main(String[] args) {
42      WaarpLoggerFactory.setDefaultFactoryIfNotSame(
43          new WaarpSlf4JLoggerFactory(null));
44  
45      // Example
46  
47      // First create a HashMap and fill it directly
48      final ConcurrentHashMap<ExampleEnumState, EnumSet<ExampleEnumState>>
49          stateMap =
50          new ConcurrentHashMap<ExampleEnumState, EnumSet<ExampleEnumState>>();
51      stateMap.put(ExampleTransition.tRUNNING.elt.getState(),
52                   ExampleTransition.tRUNNING.elt.getSet());
53      // Second create the MachineState with the right Map
54      final MachineState<ExampleEnumState> machineState1 =
55          new MachineState(ExampleEnumState.PAUSED, stateMap);
56      // Third, if not already done, fill the Map with the transitions
57      for (final ExampleTransition trans : ExampleTransition.values()) {
58        machineState1.addNewAssociation(trans.elt);
59      }
60  
61      // Or First create the MachineSate but empty
62      final MachineState<ExampleEnumState> machineState2 =
63          new MachineState(ExampleEnumState.PAUSED);
64      // Second fill the associations with transitions since none exist yet
65      for (final ExampleTransition trans : ExampleTransition.values()) {
66        machineState2.addNewAssociation(trans.elt);
67      }
68  
69      System.out.println("Machine1 states...");
70      changeState(machineState1, ExampleEnumState.CONFIGURING);
71      changeState(machineState1, ExampleEnumState.PAUSED);
72      changeState(machineState1, ExampleEnumState.RUNNING);
73      changeState(machineState1, ExampleEnumState.ENDED);
74      changeState(machineState1, ExampleEnumState.PAUSED);
75      changeState(machineState1, ExampleEnumState.RESET);
76      changeState(machineState1, ExampleEnumState.PAUSED);
77  
78      System.out.println("Machine2 states...");
79      changeState(machineState2, ExampleEnumState.CONFIGURING);
80      changeState(machineState2, ExampleEnumState.PAUSED);
81      changeState(machineState2, ExampleEnumState.RUNNING);
82      changeState(machineState2, ExampleEnumState.ENDED);
83      changeState(machineState2, ExampleEnumState.PAUSED);
84      changeState(machineState2, ExampleEnumState.RESET);
85      changeState(machineState2, ExampleEnumState.PAUSED);
86    }
87  
88    private static void changeState(MachineState<ExampleEnumState> mach,
89                                    ExampleEnumState desired) {
90      try {
91        printState(mach);
92        mach.setCurrent(desired);
93        printState(mach);
94      } catch (final IllegalFiniteStateException e) {
95        printWrongState(mach, desired);
96      }
97    }
98  
99    private static void printState(MachineState<ExampleEnumState> mach) {
100     System.out.println("State is " + mach.getCurrent());
101   }
102 
103   private static void printWrongState(MachineState<ExampleEnumState> mach,
104                                       ExampleEnumState desired) {
105     System.out.println(
106         "Cannot go from State " + mach.getCurrent() + " to State " + desired);
107   }
108 
109 }
