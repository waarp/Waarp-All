37    public TestWatcher watchman = new TestWatcherJunit4();

39    @Test
40    @SuppressWarnings({ "unchecked", "rawtypes" })
41    public void testMachineStateEnumStateConcurrentHashMapOfEnumStateEnumSetOfQ() {
42      // First create a HashMap and fill it directly
43      final ConcurrentHashMap<ExampleEnumState, EnumSet<ExampleEnumState>>
44          stateMap =
45          new ConcurrentHashMap<ExampleEnumState, EnumSet<ExampleEnumState>>();
46      stateMap.put(ExampleTransition.tRUNNING.elt.getState(),
47                   ExampleTransition.tRUNNING.elt.getSet());
48      // Second create the MachineState with the right Map
49      final MachineState<ExampleEnumState> machineState1 =
50          new MachineState(ExampleEnumState.PAUSED, stateMap);
51      // Third, if not already done, fill the Map with the transitions
52      for (final ExampleTransition trans : ExampleTransition.values()) {
53        machineState1.addNewAssociation(trans.elt);
54      }
55      System.out.println("Machine1 states...");
56      assertTrue(changeState(machineState1, ExampleEnumState.CONFIGURING));
57      assertTrue(changeState(machineState1, ExampleEnumState.PAUSED));
58      assertTrue(changeState(machineState1, ExampleEnumState.RUNNING));
59      assertTrue(changeState(machineState1, ExampleEnumState.ENDED));
60      assertFalse(changeState(machineState1, ExampleEnumState.PAUSED));
61      assertTrue(changeState(machineState1, ExampleEnumState.RESET));
62      assertTrue(changeState(machineState1, ExampleEnumState.PAUSED));
63    }
64  
65    private static boolean changeState(MachineState<ExampleEnumState> mach,
66                                       ExampleEnumState desired) {
67      try {
68        printState(mach);
69        mach.setCurrent(desired);
70        printState(mach);
71        return true;
72      } catch (final IllegalFiniteStateException e) {
73        printWrongState(mach, desired);
74        return false;
75      }
76    }
77  
78    private static final void printState(MachineState<ExampleEnumState> mach) {
79      System.out.println("State is " + mach.getCurrent());
80    }
81  
82    private static final void printWrongState(MachineState<ExampleEnumState> mach,
83                                              ExampleEnumState desired) {
84      System.out.println(
85          "Cannot go from State " + mach.getCurrent() + " to State " + desired);
86    }
87  
88    @Test
89    @SuppressWarnings({ "unchecked", "rawtypes" })
90    public void testMachineStateEnumState() {
91      // Or First create the MachineSate but empty
92      final MachineState<ExampleEnumState> machineState2 =
93          new MachineState(ExampleEnumState.PAUSED);
94      // Second fill the associations with transitions since none exist yet
95      for (final ExampleTransition trans : ExampleTransition.values()) {
96        machineState2.addNewAssociation(trans.elt);
97      }
98      System.out.println("Machine2 states...");
99      assertTrue(changeState(machineState2, ExampleEnumState.CONFIGURING));
100     assertTrue(changeState(machineState2, ExampleEnumState.PAUSED));
101     assertTrue(changeState(machineState2, ExampleEnumState.RUNNING));
102     assertTrue(changeState(machineState2, ExampleEnumState.ENDED));
103     assertFalse(changeState(machineState2, ExampleEnumState.PAUSED));
104     assertTrue(changeState(machineState2, ExampleEnumState.RESET));
105     assertTrue(changeState(machineState2, ExampleEnumState.PAUSED));
106   }

108   public enum ExampleEnumState {
109     RUNNING, PAUSED, CONFIGURING, RESET, ENDED;
110 
111     public enum ExampleTransition {
112       tRUNNING(RUNNING, EnumSet.of(PAUSED, ENDED)),
113       tPAUSED(PAUSED, EnumSet.of(RUNNING, RESET, CONFIGURING)),
114       tENDED(ENDED, EnumSet.of(RESET)),
115       tCONFIGURING(CONFIGURING, EnumSet.of(PAUSED)),
116       tRESET(RESET, EnumSet.of(PAUSED, RESET));
117 
118       public Transition<ExampleEnumState> elt;
119 
120       ExampleTransition(ExampleEnumState state, EnumSet<ExampleEnumState> set) {
121         elt = new Transition<ExampleEnumState>(state, set);
122       }
123     }
124
