360   public void testSetRoleDefault() {
361     for (final ROLE roletoset : ROLE.values()) {
362       final RoleDefault role = new RoleDefault();
363       checkRole(role, ROLE.NOACCESS);
364       role.addRole(roletoset);
365       checkRole(role, roletoset);
366       final RoleDefault role2 = new RoleDefault();
367       checkRole(role2, ROLE.NOACCESS);
368       role2.setRoleDefault(role);
369       checkRole(role2, roletoset);
370       role.clear();
371       checkRole(role, ROLE.NOACCESS);
372       checkRole(role2, roletoset);
373     }
