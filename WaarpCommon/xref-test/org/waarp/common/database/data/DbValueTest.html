43    @Rule(order = Integer.MIN_VALUE)
44    public TestWatcher watchman = new TestWatcherJunit4();

46    @Test
47    public void testDbValue() throws WaarpDatabaseSqlException, IOException {
48      DbValue valueS = new DbValue("string");
49      DbValue valueVl = new DbValue("string", true);
50      DbValue valueI = new DbValue(0);
51      DbValue valueB = new DbValue((byte) 0);
52      DbValue valueT = new DbValue(true);
53      DbValue valueSh = new DbValue((short) 0);
54      DbValue valueL = new DbValue(0L);
55      DbValue valueF = new DbValue((float) 0.1);
56      DbValue valueD = new DbValue(0.1D);
57      DbValue valueA = new DbValue("string".getBytes(WaarpStringUtils.UTF8));
58      DbValue valueDa = new DbValue(new Date(0));
59      DbValue valueTm = new DbValue(new Timestamp(0));
60      DbValue valueDaS = new DbValue(new java.sql.Date(0));
61      DbValue valueClob = new DbValue(new StringReader("clob"), null);
62      DbValue valueBlob =
63          new DbValue(new ByteArrayInputStream("blob".getBytes()), null);
64  
65      DbValue valueS2 = new DbValue("string", "S2");
66      DbValue valueVl2 = new DbValue("string", "Vl2", true);
67      DbValue valueI2 = new DbValue(0, "I2");
68      DbValue valueB2 = new DbValue((byte) 0, "B2");
69      DbValue valueT2 = new DbValue(true, "T2");
70      DbValue valueSh2 = new DbValue((short) 0, "Sh2");
71      DbValue valueL2 = new DbValue(0L, "L2");
72      DbValue valueF2 = new DbValue((float) 0.1, "F2");
73      DbValue valueD2 = new DbValue(0.1D, "F2");
74      DbValue valueA2 =
75          new DbValue("string".getBytes(WaarpStringUtils.UTF8), "A2");
76      DbValue valueDa2 = new DbValue(new Date(0), "Da2");
77      DbValue valueTm2 = new DbValue(new Timestamp(0), "Ta2");
78      DbValue valueDaS2 = new DbValue(new java.sql.Date(0), "DaS2");
79      DbValue valueClob2 = new DbValue(new StringReader("clob"), "clob2");
80      DbValue valueBlob2 =
81          new DbValue(new ByteArrayInputStream("blob".getBytes()), "blob2");
82  
83      assertEquals(valueS.getType(), valueS2.getType());
84      assertEquals(valueVl.getType(), valueVl2.getType());
85      assertEquals(valueI.getType(), valueI2.getType());
86      assertEquals(valueB.getType(), valueB2.getType());
87      assertEquals(valueT.getType(), valueT2.getType());
88      assertEquals(valueSh.getType(), valueSh2.getType());
89      assertEquals(valueL.getType(), valueL2.getType());
90      assertEquals(valueF.getType(), valueF2.getType());
91      assertEquals(valueD.getType(), valueD2.getType());
92      assertEquals(valueA.getType(), valueA2.getType());
93      assertEquals(valueDa.getType(), valueDa2.getType());
94      assertEquals(valueTm.getType(), valueTm2.getType());
95      assertEquals(valueDaS.getType(), valueDaS2.getType());
96      assertEquals(valueClob.getType(), valueClob2.getType());
97      assertEquals(valueBlob.getType(), valueBlob2.getType());
98  
99      assertEquals(valueS.getValue(), valueS2.getValue());
100     assertEquals(valueVl.getValue(), valueVl2.getValue());
101     assertEquals(valueI.getValue(), valueI2.getValue());
102     assertEquals(valueB.getValue(), valueB2.getValue());
103     assertEquals(valueT.getValue(), valueT2.getValue());
104     assertEquals(valueSh.getValue(), valueSh2.getValue());
105     assertEquals(valueL.getValue(), valueL2.getValue());
106     assertEquals(valueF.getValue(), valueF2.getValue());
107     assertEquals(valueD.getValue(), valueD2.getValue());
108     assertArrayEquals((byte[]) valueA.getValue(), (byte[]) valueA2.getValue());
109     assertEquals(valueDa.getValue(), valueDa2.getValue());
110     assertEquals(valueTm.getValue(), valueTm2.getValue());
111     assertEquals(valueDaS.getValue(), valueDaS2.getValue());
112 
113     assertEquals(valueS.getValueAsString(), valueS2.getValueAsString());
114     assertEquals(valueVl.getValueAsString(), valueVl2.getValueAsString());
115     assertEquals(valueI.getValueAsString(), valueI2.getValueAsString());
116     assertEquals(valueB.getValueAsString(), valueB2.getValueAsString());
117     assertEquals(valueT.getValueAsString(), valueT2.getValueAsString());
118     assertEquals(valueSh.getValueAsString(), valueSh2.getValueAsString());
119     assertEquals(valueL.getValueAsString(), valueL2.getValueAsString());
120     assertEquals(valueF.getValueAsString(), valueF2.getValueAsString());
121     assertEquals(valueD.getValueAsString(), valueD2.getValueAsString());
122     assertEquals(valueA.getValueAsString(), valueA2.getValueAsString());
123     assertEquals(valueDa.getValueAsString(), valueDa2.getValueAsString());
124     assertEquals(valueTm.getValueAsString(), valueTm2.getValueAsString());
125     assertEquals(valueDaS.getValueAsString(), valueDaS2.getValueAsString());
126     assertEquals(valueClob.getValueAsString(), valueClob2.getValueAsString());
127     assertEquals(valueBlob.getValueAsString(), valueBlob2.getValueAsString());
128 
129 
130     valueS.setValue(valueS2.getValue());
131     valueS.setValue((String) valueS2.getValue());
132     valueVl.setValue((String) valueVl2.getValue());
133     valueI.setValue((Integer) valueI2.getValue());
134     valueB.setValue((Byte) valueB2.getValue());
135     valueT.setValue((Boolean) valueT2.getValue());
136     valueSh.setValue((Short) valueSh2.getValue());
137     valueL.setValue((Long) valueL2.getValue());
138     valueF.setValue((Float) valueF2.getValue());
139     valueD.setValue((Double) valueD2.getValue());
140     valueA.setValue((byte[]) valueA2.getValue());
141     valueDa.setValue((Date) valueDa2.getValue());
142     valueTm.setValue((Timestamp) valueTm2.getValue());
143     valueDaS.setValue((java.sql.Date) valueDaS2.getValue());
144     valueClob.setValue((Reader) valueClob2.getValue());
145     valueBlob.setValue((InputStream) valueBlob2.getValue());
146 
147     assertEquals(valueS.getValue(), valueS2.getValue());
148     assertEquals(valueVl.getValue(), valueVl2.getValue());
149     assertEquals(valueI.getValue(), valueI2.getValue());
150     assertEquals(valueB.getValue(), valueB2.getValue());
151     assertEquals(valueT.getValue(), valueT2.getValue());
152     assertEquals(valueSh.getValue(), valueSh2.getValue());
153     assertEquals(valueL.getValue(), valueL2.getValue());
154     assertEquals(valueF.getValue(), valueF2.getValue());
155     assertEquals(valueD.getValue(), valueD2.getValue());
156     assertEquals(valueA.getValue(), valueA2.getValue());
157     assertEquals(valueDa.getValue(), valueDa2.getValue());
158     assertEquals(valueTm.getValue(), valueTm2.getValue());
159     assertEquals(valueDaS.getValue(), valueDaS2.getValue());
160     assertEquals(valueClob.getValue(), valueClob2.getValue());
161     assertEquals(valueBlob.getValue(), valueBlob2.getValue());
162 
163     valueS.setValueFromString(valueS2.getValueAsString());
164     valueVl.setValueFromString(valueVl2.getValueAsString());
165     valueI.setValueFromString(valueI2.getValueAsString());
166     valueB.setValueFromString(valueB2.getValueAsString());
167     valueT.setValueFromString(valueT2.getValueAsString());
168     valueSh.setValueFromString(valueSh2.getValueAsString());
169     valueL.setValueFromString(valueL2.getValueAsString());
170     valueF.setValueFromString(valueF2.getValueAsString());
171     valueD.setValueFromString(valueD2.getValueAsString());
172     valueA.setValueFromString(valueA2.getValueAsString());
173     valueDa.setValueFromString(valueDa2.getValueAsString());
174     valueTm.setValueFromString(valueTm2.getValueAsString());
175     valueDaS.setValueFromString(valueDaS2.getValueAsString());
176     File file = new File("/tmp/clob.txt");
177     FileOutputStream outputStream = new FileOutputStream(file);
178     outputStream.write("content".getBytes(WaarpStringUtils.UTF8));
179     outputStream.flush();
180     outputStream.close();
181     valueClob.setValueFromString(file.getAbsolutePath());
182     valueBlob.setValueFromString(file.getAbsolutePath());
183 
184     assertEquals(valueS.getValue(), valueS2.getValue());
185     assertEquals(valueVl.getValue(), valueVl2.getValue());
186     assertEquals(valueI.getValue(), valueI2.getValue());
187     assertEquals(valueB.getValue(), valueB2.getValue());
188     assertEquals(valueT.getValue(), valueT2.getValue());
189     assertEquals(valueSh.getValue(), valueSh2.getValue());
190     assertEquals(valueL.getValue(), valueL2.getValue());
191     assertEquals(valueF.getValue(), valueF2.getValue());
192     assertEquals(valueD.getValue(), valueD2.getValue());
193     assertArrayEquals((byte[]) valueA.getValue(), (byte[]) valueA2.getValue());
194     // Zone issue
195     //assertEquals(valueDa.getValue(), valueDa2.getValue());
196     //assertEquals(valueTm.getValue(), valueTm2.getValue());
197     //assertEquals(valueDaS.getValue(), valueDaS2.getValue());
