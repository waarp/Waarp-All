41  public class DbValueTest {
42    @Rule(order = Integer.MIN_VALUE)
43    public TestWatcher watchman = new TestWatcherJunit4();
44  
45    @Test
46    public void testDbValue() throws WaarpDatabaseSqlException, IOException {
47      DbValue valueS = new DbValue("string");
48      DbValue valueVl = new DbValue("string", true);
49      DbValue valueI = new DbValue(0);
50      DbValue valueB = new DbValue((byte) 0);
51      DbValue valueT = new DbValue(true);
52      DbValue valueSh = new DbValue((short) 0);
53      DbValue valueL = new DbValue(0L);
54      DbValue valueF = new DbValue((float) 0.1);
55      DbValue valueD = new DbValue(0.1D);
56      DbValue valueA = new DbValue("string".getBytes());
57      DbValue valueDa = new DbValue(new Date(0));
58      DbValue valueTm = new DbValue(new Timestamp(0));
59      DbValue valueDaS = new DbValue(new java.sql.Date(0));
60      DbValue valueClob = new DbValue(new StringReader("clob"), null);
61      DbValue valueBlob =
62          new DbValue(new ByteArrayInputStream("blob".getBytes()), null);
63  
64      DbValue valueS2 = new DbValue("string", "S2");
65      DbValue valueVl2 = new DbValue("string", "Vl2", true);
66      DbValue valueI2 = new DbValue(0, "I2");
67      DbValue valueB2 = new DbValue((byte) 0, "B2");
68      DbValue valueT2 = new DbValue(true, "T2");
69      DbValue valueSh2 = new DbValue((short) 0, "Sh2");
70      DbValue valueL2 = new DbValue(0L, "L2");
71      DbValue valueF2 = new DbValue((float) 0.1, "F2");
72      DbValue valueD2 = new DbValue(0.1D, "F2");
73      DbValue valueA2 = new DbValue("string".getBytes(), "A2");
74      DbValue valueDa2 = new DbValue(new Date(0), "Da2");
75      DbValue valueTm2 = new DbValue(new Timestamp(0), "Ta2");
76      DbValue valueDaS2 = new DbValue(new java.sql.Date(0), "DaS2");
77      DbValue valueClob2 = new DbValue(new StringReader("clob"), "clob2");
78      DbValue valueBlob2 =
79          new DbValue(new ByteArrayInputStream("blob".getBytes()), "blob2");
80  
81      assertEquals(valueS.getType(), valueS2.getType());
82      assertEquals(valueVl.getType(), valueVl2.getType());
83      assertEquals(valueI.getType(), valueI2.getType());
84      assertEquals(valueB.getType(), valueB2.getType());
85      assertEquals(valueT.getType(), valueT2.getType());
86      assertEquals(valueSh.getType(), valueSh2.getType());
87      assertEquals(valueL.getType(), valueL2.getType());
88      assertEquals(valueF.getType(), valueF2.getType());
89      assertEquals(valueD.getType(), valueD2.getType());
90      assertEquals(valueA.getType(), valueA2.getType());
91      assertEquals(valueDa.getType(), valueDa2.getType());
92      assertEquals(valueTm.getType(), valueTm2.getType());
93      assertEquals(valueDaS.getType(), valueDaS2.getType());
94      assertEquals(valueClob.getType(), valueClob2.getType());
95      assertEquals(valueBlob.getType(), valueBlob2.getType());
96  
97      assertEquals(valueS.getValue(), valueS2.getValue());
98      assertEquals(valueVl.getValue(), valueVl2.getValue());
99      assertEquals(valueI.getValue(), valueI2.getValue());
100     assertEquals(valueB.getValue(), valueB2.getValue());
101     assertEquals(valueT.getValue(), valueT2.getValue());
102     assertEquals(valueSh.getValue(), valueSh2.getValue());
103     assertEquals(valueL.getValue(), valueL2.getValue());
104     assertEquals(valueF.getValue(), valueF2.getValue());
105     assertEquals(valueD.getValue(), valueD2.getValue());
106     assertArrayEquals((byte[]) valueA.getValue(), (byte[]) valueA2.getValue());
107     assertEquals(valueDa.getValue(), valueDa2.getValue());
108     assertEquals(valueTm.getValue(), valueTm2.getValue());
109     assertEquals(valueDaS.getValue(), valueDaS2.getValue());
110 
111     assertEquals(valueS.getValueAsString(), valueS2.getValueAsString());
112     assertEquals(valueVl.getValueAsString(), valueVl2.getValueAsString());
113     assertEquals(valueI.getValueAsString(), valueI2.getValueAsString());
114     assertEquals(valueB.getValueAsString(), valueB2.getValueAsString());
115     assertEquals(valueT.getValueAsString(), valueT2.getValueAsString());
116     assertEquals(valueSh.getValueAsString(), valueSh2.getValueAsString());
117     assertEquals(valueL.getValueAsString(), valueL2.getValueAsString());
118     assertEquals(valueF.getValueAsString(), valueF2.getValueAsString());
119     assertEquals(valueD.getValueAsString(), valueD2.getValueAsString());
120     assertEquals(valueA.getValueAsString(), valueA2.getValueAsString());
121     assertEquals(valueDa.getValueAsString(), valueDa2.getValueAsString());
122     assertEquals(valueTm.getValueAsString(), valueTm2.getValueAsString());
123     assertEquals(valueDaS.getValueAsString(), valueDaS2.getValueAsString());
124     assertEquals(valueClob.getValueAsString(), valueClob2.getValueAsString());
125     assertEquals(valueBlob.getValueAsString(), valueBlob2.getValueAsString());
126 
127 
128     valueS.setValue(valueS2.getValue());
129     valueS.setValue((String) valueS2.getValue());
130     valueVl.setValue((String) valueVl2.getValue());
131     valueI.setValue((Integer) valueI2.getValue());
132     valueB.setValue((Byte) valueB2.getValue());
133     valueT.setValue((Boolean) valueT2.getValue());
134     valueSh.setValue((Short) valueSh2.getValue());
135     valueL.setValue((Long) valueL2.getValue());
136     valueF.setValue((Float) valueF2.getValue());
137     valueD.setValue((Double) valueD2.getValue());
138     valueA.setValue((byte[]) valueA2.getValue());
139     valueDa.setValue((Date) valueDa2.getValue());
140     valueTm.setValue((Timestamp) valueTm2.getValue());
141     valueDaS.setValue((java.sql.Date) valueDaS2.getValue());
142     valueClob.setValue((Reader) valueClob2.getValue());
143     valueBlob.setValue((InputStream) valueBlob2.getValue());
144 
145     assertEquals(valueS.getValue(), valueS2.getValue());
146     assertEquals(valueVl.getValue(), valueVl2.getValue());
147     assertEquals(valueI.getValue(), valueI2.getValue());
148     assertEquals(valueB.getValue(), valueB2.getValue());
149     assertEquals(valueT.getValue(), valueT2.getValue());
150     assertEquals(valueSh.getValue(), valueSh2.getValue());
151     assertEquals(valueL.getValue(), valueL2.getValue());
152     assertEquals(valueF.getValue(), valueF2.getValue());
153     assertEquals(valueD.getValue(), valueD2.getValue());
154     assertEquals(valueA.getValue(), valueA2.getValue());
155     assertEquals(valueDa.getValue(), valueDa2.getValue());
156     assertEquals(valueTm.getValue(), valueTm2.getValue());
157     assertEquals(valueDaS.getValue(), valueDaS2.getValue());
158     assertEquals(valueClob.getValue(), valueClob2.getValue());
159     assertEquals(valueBlob.getValue(), valueBlob2.getValue());
160 
161     valueS.setValueFromString(valueS2.getValueAsString());
162     valueVl.setValueFromString(valueVl2.getValueAsString());
163     valueI.setValueFromString(valueI2.getValueAsString());
164     valueB.setValueFromString(valueB2.getValueAsString());
165     valueT.setValueFromString(valueT2.getValueAsString());
166     valueSh.setValueFromString(valueSh2.getValueAsString());
167     valueL.setValueFromString(valueL2.getValueAsString());
168     valueF.setValueFromString(valueF2.getValueAsString());
169     valueD.setValueFromString(valueD2.getValueAsString());
170     valueA.setValueFromString(valueA2.getValueAsString());
171     valueDa.setValueFromString(valueDa2.getValueAsString());
172     valueTm.setValueFromString(valueTm2.getValueAsString());
173     valueDaS.setValueFromString(valueDaS2.getValueAsString());
174     File file = new File("/tmp/clob.txt");
175     FileOutputStream outputStream = new FileOutputStream(file);
176     outputStream.write("content".getBytes());
177     outputStream.flush();
178     outputStream.close();
179     valueClob.setValueFromString(file.getAbsolutePath());
180     valueBlob.setValueFromString(file.getAbsolutePath());
181 
182     assertEquals(valueS.getValue(), valueS2.getValue());
183     assertEquals(valueVl.getValue(), valueVl2.getValue());
184     assertEquals(valueI.getValue(), valueI2.getValue());
185     assertEquals(valueB.getValue(), valueB2.getValue());
186     assertEquals(valueT.getValue(), valueT2.getValue());
187     assertEquals(valueSh.getValue(), valueSh2.getValue());
188     assertEquals(valueL.getValue(), valueL2.getValue());
189     assertEquals(valueF.getValue(), valueF2.getValue());
190     assertEquals(valueD.getValue(), valueD2.getValue());
191     assertArrayEquals((byte[]) valueA.getValue(), (byte[]) valueA2.getValue());
192     // Zone issue
193     //assertEquals(valueDa.getValue(), valueDa2.getValue());
194     //assertEquals(valueTm.getValue(), valueTm2.getValue());
195     //assertEquals(valueDaS.getValue(), valueDaS2.getValue());
