21  package org.waarp.common.lru;
22  
23  import org.junit.Before;
24  import org.junit.BeforeClass;
25  import org.junit.Rule;
26  import org.junit.Test;
27  import org.junit.rules.TestWatcher;
28  import org.waarp.common.utility.TestWatcherJunit4;
29  
30  import static org.junit.Assert.*;
31  
32  public class CapacityLruLinkedHashMapTest {
33    @Rule(order = Integer.MIN_VALUE)
34    public TestWatcher watchman = new TestWatcherJunit4();
35  
36    private static String[] keys;
37    private static String[] values;
38    private static int capacity;
39    private CapacityLruLinkedHashMap<String, String> map;
40  
41    @BeforeClass
42    public static void setUpClass() {
43      keys = new String[] { "aa", "bb", "cc", "dd", "ee", "ff", "gg", "hh" };
44      values = new String[] {
45          "aaa", "bbb", "ccc", "ddd", "eer", "fff", "ggg", "hhh"
46      };
47      capacity = 5;
48    }
49  
50    @Before
51    public void init() {
52      map = new CapacityLruLinkedHashMap<String, String>(capacity, 1, 0.75f);
53    }
54  
55    @Test(expected = IllegalArgumentException.class)
56    public void constructorZeroCapacityTest() {
57      map = new CapacityLruLinkedHashMap<String, String>(0, 1, 0.75f);
58    }
59  
60    @Test(expected = IllegalArgumentException.class)
61    public void constructorNegativeCapacityTest() {
62      map = new CapacityLruLinkedHashMap<String, String>(-5, 1, 0.75f);
63    }
64  
65    @Test
66    public void capacityTest() {
67      assertTrue(map.isEmpty());
68  
69      for (int i = 0; i < capacity; ++i) {
70        map.put(keys[i], values[i]);
71      }
72  
73      assertEquals(capacity, map.size());
74  
75      for (int i = capacity; i < keys.length; ++i) {
76        map.put(keys[i], values[i]);
77      }
78  
79      assertEquals(capacity, map.size());
80  
81      for (int i = 0; i < keys.length - capacity; ++i) {
82        assertFalse(map.containsKey(keys[i]));
83      }
84  
85      for (int i = keys.length - capacity; i < keys.length; ++i) {
86        assertTrue(map.containsKey(keys[i]));
87      }
88    }
89  }
