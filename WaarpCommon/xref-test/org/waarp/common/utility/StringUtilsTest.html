44    @Test
45    public void testRandom() {
46      final byte[] byteArray0 = StringUtils.getRandom(90);
47      assertNotNull(byteArray0);
48      final byte[] byteArray1 = StringUtils.getRandom(90);
49      assertFalse(Arrays.equals(byteArray0, byteArray1));
50      final byte[] byteArray2 = StringUtils.getBytesFromArraysToString(", ");
51      assertArrayEquals(BYTES_0_LENGTH, byteArray2);
52      final byte[] byteArray3 = StringUtils.getRandom(0);
53      assertArrayEquals(BYTES_0_LENGTH, byteArray3);
54      final byte[] byteArray4 = StringUtils.getRandom(-10);
55      assertArrayEquals(BYTES_0_LENGTH, byteArray4);
56    }

58    @Test
59    public void testBytesFromArrayToString() {
60      final byte[] byteArray0 = StringUtils.getBytesFromArraysToString("7");
61      assertArrayEquals(new byte[] { (byte) 7 }, byteArray0);
62      final byte[] byteArray1 = StringUtils.getRandom(90);
63      final String sbyte = Arrays.toString(byteArray1);
64      final byte[] byteArray2 = StringUtils.getBytesFromArraysToString(sbyte);
65      assertArrayEquals(byteArray1, byteArray2);
66  
67    }
