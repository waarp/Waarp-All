21  package org.waarp.common.utility;
22  
23  import org.junit.BeforeClass;
24  import org.junit.Test;
25  import org.waarp.common.future.WaarpFuture;
26  import org.waarp.common.utility.WaarpShutdownHook.ShutdownConfiguration;
27  import org.waarp.common.utility.WaarpShutdownHook.ShutdownTimerTask;
28  
29  import java.util.Timer;
30  
31  import static org.junit.Assert.*;
32  
33  public class WaarpShutdownHookTest {
34    private class WaarpShutdownHookForTest extends WaarpShutdownHook {
35  
36      public WaarpShutdownHookForTest(ShutdownConfiguration configuration) {
37        super(configuration);
38      }
39  
40      @Override
41      protected void exitService() {
42        // Do nothing
43      }
44    }
45  
46    @BeforeClass
47    public static void setup() {
48      DetectionUtils.setJunit(true);
49      WaarpShutdownHook.removeShutdownHook();
50    }
51  
52    @Test
53    public void addShutdownHook() throws InterruptedException {
54      WaarpShutdownHook.shutdownHook =
55          new WaarpShutdownHookForTest(new ShutdownConfiguration());
56      WaarpShutdownHook.addShutdownHook();
57      assertNotNull(WaarpShutdownHook.shutdownHook);
58      assertFalse(WaarpShutdownHook.isInShutdown());
59      assertFalse(WaarpShutdownHook.isShutdownStarting());
60      assertFalse(WaarpShutdownHook.isRestart());
61      WaarpShutdownHook.setRestart(true);
62      assertTrue(WaarpShutdownHook.isRestart());
63      WaarpShutdownHook.setRestart(false);
64      assertFalse(WaarpShutdownHook.isRestart());
65      WaarpShutdownHook.registerMain(FileConvert.class, new String[0]);
66      assertFalse(WaarpShutdownHook.isRestart());
67  
68      WaarpShutdownHook.shutdownWillStart();
69      assertTrue(WaarpShutdownHook.isShutdownStarting());
70      WaarpShutdownHook.terminate(false);
71      assertFalse(WaarpShutdownHook.isInShutdown());
72      assertNull(WaarpShutdownHook.shutdownHook);
73  
74      ShutdownConfiguration shutdownConfiguration = new ShutdownConfiguration();
75      shutdownConfiguration.timeout = 100;
76      shutdownConfiguration.serviceFuture = new WaarpFuture();
77      WaarpShutdownHook.shutdownHook =
78          new WaarpShutdownHookForTest(shutdownConfiguration);
79      new WaarpShutdownHookForTest(shutdownConfiguration);
80      WaarpShutdownHook.addShutdownHook();
81      assertNotNull(WaarpShutdownHook.shutdownHook);
82      final Timer timer = new Timer("WaarpFinalExit", true);
83      final ShutdownTimerTask timerTask = new ShutdownTimerTask();
84      timer.schedule(timerTask, 100);
85      shutdownConfiguration.serviceFuture.awaitOrInterruptible();
86      assertFalse(WaarpShutdownHook.isInShutdown());
87      WaarpShutdownHook.terminate(true);
88      assertFalse(WaarpShutdownHook.isInShutdown());
89      assertNull(WaarpShutdownHook.shutdownHook);
90  
91      shutdownConfiguration.timeout = 100;
92      shutdownConfiguration.serviceFuture = new WaarpFuture();
93      WaarpShutdownHook.shutdownHook =
94          new WaarpShutdownHookForTest(shutdownConfiguration);
95      WaarpShutdownHook.addShutdownHook();
96      assertNotNull(WaarpShutdownHook.shutdownHook);
97      assertFalse(WaarpShutdownHook.isInShutdown());
98      WaarpShutdownHook.shutdownHook.run();
99      assertFalse(WaarpShutdownHook.isInShutdown());
100     assertNull(WaarpShutdownHook.shutdownHook);
101   }
102 }
