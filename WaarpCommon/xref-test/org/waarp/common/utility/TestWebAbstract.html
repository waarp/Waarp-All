48    public static void initiateWebDriver(File file) {
49      File libdir = file.getParentFile().getParentFile().getParentFile();
50      // test-classes -> target -> WaarpR66 -> lib -> geckodriver (linux x64)
51      File libPhantomJS = new File("/tmp/phantomjs-2.1.1");
52      if (!libPhantomJS.canRead()) {
53        File libPhantomJSZip = new File(libdir, "lib/phantomjs-2.1.1.bz2");
54        if (libPhantomJSZip.canRead()) {
55          FileUtils.uncompressedBz2File(libPhantomJSZip, libPhantomJS);
56          libPhantomJS.setExecutable(true);
57        }
58      }
59      assertTrue(libPhantomJS.exists());
60      System.setProperty("phantomjs.binary.path", libPhantomJS.getAbsolutePath());
61  
62      File firefoxDriver = new File("/tmp/geckodriver");
63      if (!firefoxDriver.canRead()) {
64        File firefoxDriverZip = new File(libdir, "lib/geckodriver.bz2");
65        if (firefoxDriverZip.canRead()) {
66          FileUtils.uncompressedBz2File(firefoxDriverZip, firefoxDriver);
67          firefoxDriver.setExecutable(true);
68        } else {
69          fail("Don't find: " + firefoxDriverZip.getAbsolutePath());
70        }
71      }
72      assertTrue(firefoxDriver.exists());
73      System.setProperty("webdriver.gecko.driver", firefoxDriver.getAbsolutePath());
74      try {
75        driver = initializeDriver();
76      } catch (InterruptedException e) {//NOSONAR
77        e.printStackTrace();
78        fail(e.getMessage());
79      }
80    }
81  
82    public static void reloadDriver() throws InterruptedException {
83      if (driver != null) {
84        finalizeDriver();
85      }
86      driver = initializeDriver();
87    }

89    public static WebDriver initializeDriver() throws InterruptedException {
90      WebDriver driver;
91      switch (driverType) {
92        case PHANTOMJS:
93          driver = createPhantomJSDriver();
94          break;
95        case FIREFOX:
96        default:
97          driver = createFirefoxDriver();
98      }
99      driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(5));
100     driver.manage().window().setSize(new Dimension(1920, 1080));
101     driver.manage().timeouts().pageLoadTimeout(Duration.ofSeconds(10));
102     Thread.sleep(10);
103     return driver;
104   }

106   private static int createDriver = 0;
107   private static WebDriver createPhantomJSDriver() {
108     DesiredCapabilities desiredCapabilities = new DesiredCapabilities("phantomjs", "", Platform.ANY);
109     desiredCapabilities.setCapability(PhantomJSDriverService.PHANTOMJS_EXECUTABLE_PATH_PROPERTY,
110                                       System.getProperty("phantomjs.binary.path"));
111     desiredCapabilities.setCapability(CapabilityType.ELEMENT_SCROLL_BEHAVIOR, true);
112     desiredCapabilities.setCapability(CapabilityType.TAKES_SCREENSHOT, false);
113     desiredCapabilities.setCapability(CapabilityType.ENABLE_PROFILING_CAPABILITY, false);
114     LoggingPreferences logPrefs = new LoggingPreferences();
115     logPrefs.enable(LogType.BROWSER, Level.OFF);
116     logPrefs.enable(LogType.CLIENT, Level.OFF);
117     logPrefs.enable(LogType.DRIVER, Level.OFF);
118     logPrefs.enable(LogType.PERFORMANCE, Level.OFF);
119     logPrefs.enable(LogType.PROFILER, Level.OFF);
120     logPrefs.enable(LogType.SERVER, Level.OFF);
121     desiredCapabilities.setCapability(CapabilityType.LOGGING_PREFS, logPrefs);
122     desiredCapabilities.setCapability(CapabilityType.HAS_NATIVE_EVENTS, true);
123     desiredCapabilities.setCapability(PhantomJSDriverService.PHANTOMJS_GHOSTDRIVER_CLI_ARGS,
124                                       "--webdriver-loglevel=NONE");
125     desiredCapabilities.setJavascriptEnabled(true);
126 
127     ArrayList<String> cliArgs = new ArrayList<String>();
128     cliArgs.add("--web-security=true");
129     cliArgs.add("--ignore-ssl-errors=true");
130     cliArgs.add("--webdriver-loglevel=NONE");
131     desiredCapabilities.setCapability(PhantomJSDriverService.PHANTOMJS_CLI_ARGS, cliArgs);
132     int wait = 10;
133     createDriver++;
134     PhantomJSDriver phantomJSDriver = null;
135     for (int i = 0; i < 10; i++) {
136       try {
137         phantomJSDriver = new PhantomJSDriver(desiredCapabilities);
138         break;
139       } catch (Exception e) {
140         try {
141           Thread.sleep(wait);
142           wait += 10;
143         } catch (InterruptedException ex) {
144           // Ignore
145         }
146       }
147     }
148     if (phantomJSDriver == null) {
149       if (createDriver > 4) {
150         throw new RuntimeException("Cannot start Driver");
151       }
152       return createFirefoxDriver();
153     }
154     createDriver = 0;
155     phantomJSDriver.setLogLevel(Level.OFF);
156     return phantomJSDriver;
157   }

159   private static WebDriver createFirefoxDriver() {
160     FirefoxOptions options = new FirefoxOptions();
161     options.setHeadless(true);
162     options.setLogLevel(FirefoxDriverLogLevel.ERROR);
163     options.setAcceptInsecureCerts(true);
164     options.setCapability(CapabilityType.OVERLAPPING_CHECK_DISABLED, true);
165     options.setCapability(CapabilityType.ELEMENT_SCROLL_BEHAVIOR, true);
166     options.setCapability(CapabilityType.TAKES_SCREENSHOT, false);
167     options.setCapability(CapabilityType.ENABLE_PROFILING_CAPABILITY, false);
168     options.setCapability(CapabilityType.HAS_NATIVE_EVENTS, true);
169     options.setCapability(CapabilityType.SUPPORTS_JAVASCRIPT, true);
170     // options.setScriptTimeout(Duration.ofSeconds(10));
171     new File("/tmp/chromedir").mkdirs();
172     options.addArguments("--headless", "user-data-dir=/tmp/chromedir", "disable-infobars",
173                          "--disable-extensions", "--disable-dev-shm-usage", "--no-sandbox");
174     FirefoxDriver firefoxDriver = null;
175     int wait = 10;
176     createDriver++;
177     for (int i = 0; i < 10; i++) {
178       try {
179         firefoxDriver = new FirefoxDriver(options);
180         break;
181       } catch (Exception e) {
182         try {
183           Thread.sleep(wait);
184           wait += 10;
185         } catch (InterruptedException ex) {
186           // Ignore
187         }
188       }
189     }
190     if (firefoxDriver == null) {
191       if (createDriver > 4) {
192         throw new RuntimeException("Cannot start Driver");
193       }
194       return createPhantomJSDriver();
195     }
196     createDriver = 0;
197     // chromeDriver.manage().window().minimize();
198     // Specifying pageLoadTimeout and Implicit wait
199     firefoxDriver.manage().timeouts().pageLoadTimeout(Duration.ofSeconds(5));
200     firefoxDriver.manage().timeouts().implicitlyWait(Duration.ofSeconds(5));
201     return firefoxDriver;
202   }

204   public static void finalizeDriver() throws InterruptedException {
205     // 17 | close |  |  |
206     // driver.close();
207     if (driver != null) {
208       try {
209         driver.quit();
210       } catch (Exception e) {
211         // Ignore
212       }
213       driver = null;
214     }
215     Thread.sleep(10);
216   }
