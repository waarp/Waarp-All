75    public static synchronized int getPidLinux(Process p) {
76      int pid = -1;
77  
78      try {
79        if ("java.lang.UNIXProcess".equals(p.getClass().getName())) {
80          Field f = p.getClass().getDeclaredField("pid");
81          f.setAccessible(true);
82          pid = f.getInt(p);
83          f.setAccessible(false);
84        }
85      } catch (Exception e) {
86        throw new RuntimeException("Cannot find PID");
87      }
88      return pid;
89    }

91    public static int getPidOfRunnerCommandLinux(String filterByRunner,
92                                                 String filterByCommand) {
93      List<String> args = Arrays
94          .asList(("ps -C " + filterByRunner + " -e -o pid,args=").split(" +"));
95      // Example output:
96      // 22245 /opt/libreoffice5.4/program/soffice.bin --headless
97      // 22250 [soffice.bin] <defunct>
98  
99      try {
100       Process psAux = new ProcessBuilder(args).start();
101       try {
102         Thread.sleep(100);
103       } catch (InterruptedException ignored) {//NOSONAR
104       }
105       BufferedReader reader = new BufferedReader(
106           new InputStreamReader(psAux.getInputStream(), WaarpStringUtils.UTF8));
107       try {
108         String line;
109         while ((line = reader.readLine()) != null) {
110           if (!line.contains(filterByCommand)) {
111             continue;
112           }
113           String[] parts = line.split("[\\W]+");
114           if (parts.length < 2) {
115             throw new RuntimeException(
116                 "Unexpected format of the `ps` line, expected at least 2 " +
117                 "columns:\n\t" + line);
118           }
119           for (int i = 0; i < 2; i++) {
120             String pid = parts[i];
121             if (!pid.trim().isEmpty()) {
122               return Integer.parseInt(pid);
123             }
124           }
125         }
126       } finally {
127         FileUtils.close(reader);
128       }
129     } catch (IOException ex) {
130       throw new RuntimeException(
131           String.format("Failed executing %s: %s", args, ex.getMessage()), ex);
132     }
133     throw new RuntimeException("Cannot find PID");
134   }

136   public static int getPidOfRunnerCommandLinux(String filterByRunner,
137                                                String filterByCommand,
138                                                List<Integer> previousPids) {
139     List<String> args = Arrays
140         .asList(("ps -C " + filterByRunner + " -e -o pid,args=").split(" +"));
141     // Example output:
142     // 22245 /opt/libreoffice5.4/program/soffice.bin --headless
143     // 22250 [soffice.bin] <defunct>
144 
145     try {
146       Process psAux = new ProcessBuilder(args).start();
147       try {
148         Thread.sleep(100);
149       } catch (InterruptedException ignored) {//NOSONAR
150       }
151       BufferedReader reader = new BufferedReader(
152           new InputStreamReader(psAux.getInputStream(), WaarpStringUtils.UTF8));
153       try {
154         String line;
155         while ((line = reader.readLine()) != null) {
156           if (!line.contains(filterByCommand)) {
157             continue;
158           }
159           String[] parts = line.split("[\\W]+");
160           if (parts.length < 2) {
161             throw new RuntimeException(
162                 "Unexpected format of the `ps` line, expected at least 2 " +
163                 "columns:\n\t" + line);
164           }
165           for (int i = 0; i < 2; i++) {
166             String pid = parts[i];
167             if (!pid.trim().isEmpty()) {
168               boolean found = false;
169               int ipid = Integer.parseInt(pid);
170               for (Integer oldPid : previousPids) {
171                 if (ipid == oldPid) {
172                   found = true;
173                   break;
174                 }
175               }
176               if (found) {
177                 break;
178               }
179               return Integer.parseInt(pid);
180             }
181           }
182         }
183       } finally {
184         FileUtils.close(reader);
185       }
186     } catch (IOException ex) {
187       throw new RuntimeException(
188           String.format("Failed executing %s: %s", args, ex.getMessage()), ex);
189     }
190     throw new RuntimeException("Cannot find PID");
191   }

193   public static void kill(int pid, boolean graceful) {
194     if (System.getProperty("os.name").startsWith("Windows")) {
195       List<String> args = ImmutableList
196           .of("taskkill", graceful? "/t" : "/f", "/pid", Integer.toString(pid));
197       try {
198         Process process = new ProcessBuilder(args).start();
199         BufferedReader br = null;
200         try {
201           br = new BufferedReader(
202               new InputStreamReader(process.getInputStream()));
203           String line = null;
204           while ((line = br.readLine()) != null) {
205             // nothing
206           }
207         } finally {
208           FileUtils.close(br);
209         }
210         process.waitFor();
211       } catch (IOException ignored) {
212       } catch (InterruptedException ignored) {//NOSONAR
213       }
214     } else {
215       int signal = graceful? 15 : 9;
216       killLinux(pid, signal);
217     }
218     try {
219       Thread.sleep(100);
220     } catch (InterruptedException ignored) {//NOSONAR
221     }
222   }

224   public static boolean exists(int pid) {
225     if (System.getProperty("os.name").startsWith("Windows")) {
226       return existsWindows(pid);
227     } else {
228       return killLinux(pid, 0) == 0;
229     }

232   static int killLinux(int pid, int signal) {
233     if (pid < 100) {
234       SysErrLogger.FAKE_LOGGER
235           .syserr("Should never try to kill process with pid < 100");
236       throw new RuntimeException(
237           "Should never try to kill process with pid < 100");
238     }
239     if (posixHandler != null) {
240       return getPosix().kill(pid, signal);
241     } else {
242       List<String> args = new ArrayList<String>();
243       args.add("kill");
244       args.add("-s");
245       args.add(Integer.toString(signal));
246       args.add(Integer.toString(pid));
247       try {
248         Process process = new ProcessBuilder(args).start();
249         BufferedReader br = null;
250         try {
251           br = new BufferedReader(
252               new InputStreamReader(process.getInputStream()));
253           String line = null;
254           while ((line = br.readLine()) != null) {
255             // nothing
256           }
257         } finally {
258           FileUtils.close(br);
259         }
260         return process.waitFor();
261       } catch (IOException ignored) {
262       } catch (InterruptedException ignored) {//NOSONAR
263       }
264     }
265     throw new RuntimeException("Cannot find PID");
266   }

268   static boolean existsWindows(int pid) {
269     final AtomicBoolean exist = new AtomicBoolean(false);
270     try {
271       Process psAux =
272           new ProcessBuilder("tasklist.exe", "/fo", "csv", "/nh", "/fi",
273                              "PID eq " + pid).redirectErrorStream(true).start();
274       try {
275         Thread.sleep(100);
276       } catch (InterruptedException ignored) {//NOSONAR
277       }
278       BufferedReader reader = new BufferedReader(
279           new InputStreamReader(psAux.getInputStream(), WaarpStringUtils.UTF8));
280       try {
281         String line;
282         String spid = Integer.toString(pid);
283         while ((line = reader.readLine()) != null) {
284           if (line.contains(spid)) {
285             exist.set(true);
286           }
287         }
288       } finally {
289         FileUtils.close(reader);
290       }
291       psAux.waitFor();
292     } catch (Throwable ignored) {
293     }
294     return exist.get();
295   }

301   public static Process launchJavaProcess(String applArgs) throws IOException {
302     try {
303       // java binary
304       final String java = System.getProperty("java.home") + "/bin/java";
305       // vm arguments
306       final List<String> vmArguments =
307           ManagementFactory.getRuntimeMXBean().getInputArguments();
308       final StringBuilder vmArgsOneLine = new StringBuilder();
309       for (final String arg : vmArguments) {
310         // if it's the agent argument : we ignore it otherwise the
311         // address of the old application and the new one will be in conflict
312         if (!arg.contains("-agentlib")) {
313           vmArgsOneLine.append(arg).append(' ');
314         }
315       }
316       if (jvmArgsDefault != null) {
317         vmArgsOneLine.append(' ').append(jvmArgsDefault).append(' ');
318       }
319       // init the command to execute, add the vm args
320       final StringBuilder cmd;
321       if (DetectionUtils.isWindows()) {
322         cmd = new StringBuilder('"' + java + "\" " + vmArgsOneLine);
323       } else {
324         cmd = new StringBuilder(java + ' ' + vmArgsOneLine);
325       }
326       cmd.append(applArgs);
327       return Runtime.getRuntime().exec(cmd.toString());
328     } catch (final Throwable e) {
329       // something went wrong
330       throw new IOException("Error while trying to restart the application", e);
331     }
332   }

351   public static int executeJvm(Project project, File homeDir, Class<?> zclass,
352                                String[] args, boolean longTerm) {
353     int pid = 99999999;
354     try {
355       // initialize an java task
356       final Java javaTask = new Java();
357       javaTask.setNewenvironment(false);
358       javaTask.setTaskName(zclass.getSimpleName());
359       javaTask.setProject(project);
360       javaTask.setFork(true);
361       javaTask.setCloneVm(true);
362       javaTask.setSpawn(longTerm);
363       javaTask.setFailonerror(true);
364       javaTask.setClassname(zclass.getName());
365 
366       // add some vm args
367       final Argument jvmArgs = javaTask.createJvmarg();
368       if (jvmArgsDefault != null) {
369         jvmArgs.setLine("-Xms512m -Xmx1024m " + jvmArgsDefault);
370       } else {
371         jvmArgs.setLine("-Xms512m -Xmx1024m");
372       }
373 
374       // added some args for to class to launch
375       final Argument taskArgs = javaTask.createArg();
376       final StringBuilder builder = new StringBuilder();
377       for (final String string : args) {
378         builder.append(' ').append(string);
379       }
380       taskArgs.setLine(builder.toString());
381 
382       // set the class path
383       final String classpath = System.getProperty("java.class.path");
384       final Path classPath = javaTask.createClasspath();
385       classPath.setPath(classpath);
386       javaTask.setClasspath(classPath);
387 
388       javaTask.init();
389       final int ret = javaTask.executeJava();
390       if (longTerm) {
391         pid = getPidOfRunnerCommandLinux("java", zclass.getName());
392       }
393       SysErrLogger.FAKE_LOGGER.syserr(
394           zclass.getName() + ' ' + builder + " return code: " + ret + " pid: " +
395           pid);
396     } catch (final BuildException e) {
397       SysErrLogger.FAKE_LOGGER.syserr("While java task", e);
398     }
399     return pid;
400   }

415   private static final class OurPOSIXHandler implements POSIXHandler {
416     private final AtomicBoolean verbose = new AtomicBoolean(false);
417 
418     void setVerbose(boolean verbose) {
419       this.verbose.set(verbose);
420     }
421 
422     @Override
423     public void error(Errno error, String extraData) {
424       throw new RuntimeException(
425           "native error " + error.description() + ' ' + extraData);
426     }
427 
428     @Override
429     public void error(Errno errno, String extraData1, String extraData2) {
430       throw new RuntimeException(
431           "native error " + errno.description() + ' ' + extraData1 + ' ' +
432           extraData2);
433     }
434 
435     @Override
436     public void unimplementedError(String methodName) {
437       throw new IllegalStateException(
438           methodName + " is not implemented in jnr-posix");
439     }
440 
441     @Override
442     public void warn(WARNING_ID id, String message, Object... data) {
443       String msg;
444       try {
445         msg = String.format(message, data);
446       } catch (IllegalFormatException e) {
447         msg = message + ' ' + Arrays.toString(data);
448       }
449       Logger.getLogger("jnr-posix").log(Level.WARNING, msg);
450     }
451 
452     @Override
453     public boolean isVerbose() {
454       return verbose.get();
455     }
456 
457     @Override
458     public File getCurrentWorkingDirectory() {
459       return new File(".");
460     }
461 
462     @Override
463     public String[] getEnv() {
464       String[] envp = new String[System.getenv().size()];
465       int i = 0;
466       for (Map.Entry<String, String> pair : System.getenv().entrySet()) {
467         envp[i++] = pair.getKey() + '=' + pair.getValue();
468       }
469       return envp;
470 
471     }

473     @Override
474     public InputStream getInputStream() {
475       return System.in;
476     }
477 
478     @Override
479     public PrintStream getOutputStream() {
480       return System.out;
481     }
482 
483     @Override
484     public int getPID() {
485       throw new IllegalStateException("getPID is not implemented in jnr-posix");
486     }
487 
488     @Override
489     public PrintStream getErrorStream() {
490       return System.err;
491     }
