60    @Test
61    public void testEncodeBytesByteArray() throws Exception {
62      final String s = WaarpStringUtils.fillString('a', 100);
63      String sencoded = Base64.encodeBytes(s.getBytes());
64      byte[] src = Base64.decode(sencoded);
65      assertArrayEquals("Should be equals", s.getBytes(), src);
66      sencoded = Base64.encodeBytes(s.getBytes(), 0, s.getBytes().length);
67      src = Base64.decode(sencoded);
68      assertArrayEquals("Should be equals", s.getBytes(), src);
69  
70      ByteBuffer raw = ByteBuffer.wrap(src);
71      final ByteBuffer encoded =
72          ByteBuffer.allocate(sencoded.getBytes().length + 10);
73      Base64.encode(raw, encoded);
74      byte[] src2 = Base64.decode(encoded.array());
75      assertArrayEquals("Should be equals", src, src2);
76  
77      src2 = Base64.encodeBytesToBytes(s.getBytes());
78      assertArrayEquals("Should be equals", sencoded.getBytes(), src2);
79      src2 = Base64
80          .encodeBytesToBytes(s.getBytes(), 0, s.getBytes().length, Base64.GZIP);
81      final String sencoded2 = new String(src2);
82      assertArrayEquals("Should be equals", sencoded2.getBytes(), src2);
83  
84      raw = ByteBuffer.wrap(src);
85      final CharBuffer encode =
86          CharBuffer.allocate(sencoded.getBytes().length + 10);
87      Base64.encode(raw, encode);
88      final char[] chars = encode.array();
89      final byte[] src3 = new byte[chars.length];
90      for (int i = 0; i < chars.length; i++) {
91        src3[i] = (byte) chars[i];
92      }
93      src2 = Base64.decode(src3);
94      assertArrayEquals("Should be equals", src, src2);
95  
96      final SerializableObject serializableObject = new SerializableObject();
97      serializableObject.value = 2;
98      String sobj = Base64.encodeObject(serializableObject);
99      Object obj = Base64.decodeToObject(sobj);
100     assertTrue("Should be of correct class", obj instanceof SerializableObject);
101     assertEquals("Should be equals", serializableObject.value,
102                  ((SerializableObject) obj).value);
103 
104     sobj = Base64.encodeObject(serializableObject, Base64.GZIP);
105     obj = Base64.decodeToObject(sobj, Base64.GZIP, null);
106     assertTrue("Should be of correct class", obj instanceof SerializableObject);
107     assertEquals("Should be equals", serializableObject.value,
108                  ((SerializableObject) obj).value);
109 
110   }

112   @Test
113   public void testEncodeToFile() throws IOException {
114     final String s = WaarpStringUtils.fillString('a', 100);
115     Base64.encodeToFile(s.getBytes(), "/tmp/base64.txt");
116     final byte[] src = Base64.decodeFromFile("/tmp/base64.txt");
117     assertArrayEquals("Should be equals", s.getBytes(), src);
118 
119     final String s1 = WaarpStringUtils.readFile("/tmp/base64.txt");
120     final File file = new File("/tmp/source.txt");
121     Base64.decodeToFile(s1, "/tmp/source.txt");
122     final String s0 = WaarpStringUtils.readFile("/tmp/source.txt");
123     assertEquals("Should be equals", s, s0);
124 
125     Base64.encodeFileToFile(file.getAbsolutePath(), "/tmp/base64.txt");
126     String s2 = WaarpStringUtils.readFile("/tmp/base64.txt");
127     String s3 = Base64.encodeFromFile("/tmp/source.txt");
128     assertEquals("Should be equals", s2, s3);
129 
130     Base64.decodeFileToFile("/tmp/base64.txt", "/tmp/source2.txt");
131     s2 = WaarpStringUtils.readFile("/tmp/source.txt");
132     s3 = WaarpStringUtils.readFile("/tmp/source2.txt");
133     assertEquals("Should be equals", s2, s3);
134   }
