29  /**
30   * RegexFileFilterTest regroups unit tests for {@link RegexFileFilter}.
31   */
32  public class RegexFileFilterTest {
33    @Test
34    public void testRegexFilename() {
35      String rx = "test.csv";
36      RegexFileFilter rff = new RegexFileFilter(rx);
37      String[] shouldMatch = { "test.csv", "subdir/test.csv" };
38      String[] shouldNotMatch = { "foo.csv", "subdir/foo.csv" };
39  
40      for (String s : shouldMatch) {
41        assertTrue(
42            "RegexFileFilter with regex '" + rx + "' should match '" + s + "'",
43            doTestOnPath(rff, s));
44      }
45  
46      for (String s : shouldNotMatch) {
47        assertFalse(
48            "RegexFileFilter with regex '" + rx + "' should match '" + s + "'",
49            doTestOnPath(rff, s));
50      }
51    }

146   private boolean doTestOnPath(RegexFileFilter rff, String path) {
147     try {
148       setupStubPath(path);
149 
150       return rff.accept(new File(path));
151 
152     } catch (IOException e) {
153       fail("cannot create stub test file '" + path + "':" + e);
154     } finally {
155       tearDownStubFile(path);
156     }
157 
158     return false;
159   }
160 
161   private void setupStubPath(String path) throws IOException {
162     File f = new File(path);
163 
164     if (path.contains("/")) {
165       f.getParentFile().mkdirs();
166     }
167 
168     f.createNewFile();
169   }
170 
171   private void tearDownStubFile(String path) {
172     File tmpFile = new File(path);
173 
174     if (tmpFile.exists()) {
175       tmpFile.delete();
176     }
177 
178     if (path.contains("/")) {
179       tmpFile.getParentFile().delete();
180     }
181   }
