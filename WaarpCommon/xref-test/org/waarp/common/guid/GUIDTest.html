171     try {
172       final GUID parsed1 = new GUID(BASE32);
173       final GUID parsed2 = new GUID(BASE64);
174       final GUID parsed0 = new GUID(BASE16);
175       final GUID parsed8 = new GUID(BASEARK);
176       final GUID parsed9 = new GUID(BYTES);
177       assertEquals(parsed1, parsed2);
178       assertEquals(parsed1, parsed8);
179       assertEquals(parsed1, parsed9);
180       assertEquals(parsed1, parsed0);
181       final GUID parsed3 = new GUID(parsed9.getBytes());
182       final GUID parsed4 = new GUID(parsed9.toBase32());
183       final GUID parsed5 = new GUID(parsed9.toHex());
184       final GUID parsed6 = new GUID(parsed9.toString());
185       final GUID parsed7 = new GUID(parsed9.toBase64());
186       assertEquals(parsed9, parsed3);
187       assertEquals(parsed9, parsed4);
188       assertEquals(parsed9, parsed5);
189       assertEquals(parsed9, parsed6);
190       assertEquals(parsed9, parsed7);
191       final GUID generated = new GUID();
192       LOGGER.warn("{}", generated.getVersion());
193       assertEquals(1, generated.getVersion());
194     } catch (final InvalidArgumentException e) {
195       LOGGER.debug(e);
196       fail(e.getMessage());

273   @Test
274   public void concurrentGeneration() throws Exception {
275     final int numThreads = 10;
276     final Thread[] threads = new Thread[numThreads];
277     final int n = 1000000;
278     final int effectiveN = n / numThreads * numThreads;
279     final GUID[] uuids = new GUID[effectiveN];
280 
281     final long start = System.currentTimeMillis();
282     for (int i = 0; i < numThreads; i++) {
283       threads[i] = new Generator(n / numThreads, uuids, i);
284       threads[i].start();
285     }
286 
287     for (int i = 0; i < numThreads; i++) {
288       threads[i].join();
289     }
290     final long stop = System.currentTimeMillis();
291     System.out.println(
292         "Time = " + (stop - start) + " so " + (n / (stop - start)) * 1000 +
293         " Uuids/s");
294 
295     final Set<GUID> uuidSet = new HashSet<GUID>(effectiveN);
296     uuidSet.addAll(Arrays.asList(uuids));
297 
298     assertEquals(effectiveN, uuidSet.size());
299   }

301   private static class Generator extends Thread {
302     private final GUID[] uuids;
303     final int id;
304     final int n;
305 
306     public Generator(final int n, final GUID[] uuids, final int id) {
307       this.n = n;
308       this.uuids = uuids;
309       this.id = id * n;
310     }
311 
312     @Override
313     public void run() {
314       for (int i = 0; i < n; i++) {
315         uuids[id + i] = new GUID();
316       }
317     }
318   }
