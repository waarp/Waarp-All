34  import java.io.File;
35  import java.io.IOException;
36  import java.security.NoSuchAlgorithmException;

38  import static org.junit.Assert.*;
39  
40  public class FileUtilsTest {
41    @Rule(order = Integer.MIN_VALUE)
42    public TestWatcher watchman = new TestWatcherJunit4();
43  
44    private static File file1;
45    private static File file2;
46    private static File dirSource;
47  
48    @AfterClass
49    public static void terminate() {
50      FileUtils.forceDeleteRecursiveDir(dirSource);
51    }
52  
53    public static void generateFiles() throws IOException {
54      dirSource = new File("/tmp/testfile");
55      final File dir = new File(dirSource, "test");
56      dir.mkdirs();
57      file1 = new File(dir, "testFile2.txt");
58      file2 = new File(dir, "testFile.txt");
59      FileTestUtils.createTestFile(file1, 10);
60      FileTestUtils.createTestFile(file2, 10);
61    }
62  
63    @Test
64    public void copy() throws IOException {
65      generateFiles();
66      File dirTo = new File(dirSource, "test2");
67      dirTo.mkdirs();
68      try {
69        File[] files = FileUtils.copy(new File[] { file1, file2 }, dirTo, true);
70        assertNotNull(files);
71        assertEquals(2, files.length);
72      } catch (Reply550Exception e) {
73        e.printStackTrace();
74        fail(e.getMessage());
75      }
76    }
77  
78    @Test
79    public void deleteRecursiveDir() throws IOException {
80      generateFiles();
81      assertTrue(FileUtils.fileExist(file1.getName(), file1.getParent()));
82      assertFalse(FileUtils.deleteRecursiveDir(dirSource));
83      assertTrue(FileUtils.forceDeleteRecursiveDir(dirSource));
84      generateFiles();
85      file1.delete();
86      file2.delete();
87      assertTrue(FileUtils.deleteRecursiveDir(dirSource));
88    }
89  
90    @Test
91    public void computeGlobalHash() throws IOException, NoSuchAlgorithmException {
92      generateFiles();
93      FilesystemBasedDigest digest = new FilesystemBasedDigest(DigestAlgo.SHA512);
94      FileUtils.computeGlobalHash(digest, file1, (int) file1.length());
95      FilesystemBasedDigest digest2 =
96          new FilesystemBasedDigest(DigestAlgo.SHA512);
97      FileUtils.computeGlobalHash(digest2, file1, (int) file1.length() - 10);
98      assertFalse(
99          FilesystemBasedDigest.digestEquals(digest.Final(), digest2.Final()));
100   }
101 
102   @Test
103   public void uncompressBz2() throws Reply550Exception {
104     final ClassLoader classLoader = FileUtilsTest.class.getClassLoader();
105     File file = new File(classLoader.getResource("cert.jks").getFile());
106     file = file.getParentFile().getParentFile().getParentFile().getParentFile();
107     file = new File(file, "lib/phantomjs-2.1.1.bz2");
108     SysErrLogger.FAKE_LOGGER.sysout(file.getAbsolutePath());
109     assertTrue(file.canRead());
110     File to = new File("/tmp/phantomjs-2.1.1");
111     if (to.exists()) {
112       to.delete();
113     }
114     assertTrue(FileUtils.uncompressedBz2File(file, to) > 0);
115     assertTrue(to.canRead());
116     to.setExecutable(true);
117     SysErrLogger.FAKE_LOGGER.sysout(
118         to.getAbsolutePath() + " => " + to.length());
119   }
120 }
