20  package org.waarp.common.xml;
21  
22  /**
23   * XmlHash Hashtable for XmlValue utility. Hash all values (except subXml and
24   * root for Multiple)
25   */
26  public class XmlRootHash extends XmlHash {
27  
28    public XmlRootHash(final XmlValue[] values) {
29      for (final XmlValue xmlValue : values) {
30        hashtable.put(xmlValue.getName(), xmlValue);
31      }
32    }
33  }
