211   @Override
212   public void delete() throws WaarpDatabaseException {
213     if (dbSession == null) {
214       dbR66ConfigurationHashMap.remove(hostid);
215       isSaved = false;
216       return;
217     }
218     final DbPreparedStatement preparedStatement =
219         new DbPreparedStatement(dbSession);
220     try {
221       preparedStatement.createPrepareStatement(
222           "DELETE FROM " + table + WHERE + getWherePrimaryKey());
223       setPrimaryKey();
224       setValues(preparedStatement, primaryKey);
225       final int count = preparedStatement.executeUpdate();
226       if (count <= 0) {
227         throw new WaarpDatabaseNoDataException(NO_ROW_FOUND);
228       }
229       isSaved = false;
230     } finally {
231       preparedStatement.realClose();
232     }
233   }

235   @Override
236   public void insert() throws WaarpDatabaseException {
237     if (isSaved) {
238       return;
239     }
240     if (dbSession == null) {
241       dbR66ConfigurationHashMap.put(hostid, this);
242       isSaved = true;
243       return;
244     }
245     final DbPreparedStatement preparedStatement =
246         new DbPreparedStatement(dbSession);
247     try {
248       preparedStatement.createPrepareStatement(
249           "INSERT INTO " + table + " (" + selectAllFields + ") VALUES " +
250           insertAllValues);
251       setValues(preparedStatement, allFields);
252       final int count = preparedStatement.executeUpdate();
253       if (count <= 0) {
254         throw new WaarpDatabaseNoDataException(NO_ROW_FOUND);
255       }
256       isSaved = true;
257     } finally {
258       preparedStatement.realClose();
259     }
260   }

262   @Override
263   public boolean exist() throws WaarpDatabaseException {
264     if (dbSession == null) {
265       return dbR66ConfigurationHashMap.containsKey(hostid);
266     }
267     final DbPreparedStatement preparedStatement =
268         new DbPreparedStatement(dbSession);
269     try {
270       preparedStatement.createPrepareStatement(
271           SELECT + primaryKey[0].getColumn() + FROM + table + WHERE +
272           getWherePrimaryKey());
273       setPrimaryKey();
274       setValues(preparedStatement, primaryKey);
275       preparedStatement.executeQuery();
276       return preparedStatement.getNext();
277     } finally {
278       preparedStatement.realClose();
279     }
280   }

282   @Override
283   public void select() throws WaarpDatabaseException {
284     if (dbSession == null) {
285       final DbDataModel conf = dbR66ConfigurationHashMap.get(hostid);
286       if (conf == null) {
287         throw new WaarpDatabaseNoDataException(NO_ROW_FOUND);
288       } else {
289         // copy info
290         for (int i = 0; i < allFields.length; i++) {
291           allFields[i].setValue(conf.allFields[i].getValue());
292         }
293         setFromArray();
294         isSaved = true;
295         return;
296       }
297     }
298     final DbPreparedStatement preparedStatement =
299         new DbPreparedStatement(dbSession);
300     try {
301       preparedStatement.createPrepareStatement(
302           SELECT + selectAllFields + FROM + table + WHERE +
303           getWherePrimaryKey());
304       setPrimaryKey();
305       setValues(preparedStatement, primaryKey);
306       preparedStatement.executeQuery();
307       if (preparedStatement.getNext()) {
308         getValues(preparedStatement, allFields);
309         setFromArray();
310         isSaved = true;
311       } else {
312         throw new WaarpDatabaseNoDataException(NO_ROW_FOUND);
313       }
314     } finally {
315       preparedStatement.realClose();
316     }
317   }

342       preparedStatement.realClose();
343     }
