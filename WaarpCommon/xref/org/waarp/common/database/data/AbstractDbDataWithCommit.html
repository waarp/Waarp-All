20  package org.waarp.common.database.data;
21  
22  import org.waarp.common.database.DbPreparedStatement;
23  import org.waarp.common.database.DbSession;
24  import org.waarp.common.database.exception.WaarpDatabaseException;
25  import org.waarp.common.database.exception.WaarpDatabaseNoDataException;
26  
27  /**
28   * Abstract database table implementation with explicit COMMIT.<br>
29   * <br>
30   * <p>
31   * If the connection is in autocommit, this abstract should not be used.<br>
32   * If the connection is not in autocommit, one could use this implementation to
33   * implicitly commit when needed
34   * automatically as should do an autocommit connection.
35   */
36  public abstract class AbstractDbDataWithCommit extends AbstractDbData {
37    /**
38     * To be implemented
39     */
40    // public static String table
41    // public static final int NBPRKEY
42    // protected static String selectAllFields
43    // protected static String updateAllFields
44    // protected static String insertAllValues
45    // protected DbValue[] primaryKey
46    // protected DbValue[] otherFields
47    // protected DbValue[] allFields
48  
49    private static final String NO_ROW_FOUND = "No row found";
50  
51    /**
52     * Abstract constructor to set the DbSession to use
53     *
54     * @param dbSession
55     */
56    protected AbstractDbDataWithCommit(final DbSession dbSession) {
57      super(dbSession);
58      initObject();
59    }
60  
61    /**
62     * Insert object into table
63     *
64     * @throws WaarpDatabaseException
65     */
66    @Override
67    public void insert() throws WaarpDatabaseException {
68      if (isSaved) {
69        return;
70      }
71      if (dbSession == null) {
72        isSaved = true;
73        return;
74      }
75      setToArray();
76      final DbPreparedStatement preparedStatement =
77          new DbPreparedStatement(dbSession);
78      try {
79        preparedStatement.createPrepareStatement(
80            "INSERT INTO " + getTable() + " (" + getSelectAllFields() +
81            ") VALUES " + getInsertAllValues());
82        setValues(preparedStatement, allFields);
83        final int count = preparedStatement.executeUpdate();
84        if (count <= 0) {
85          throw new WaarpDatabaseNoDataException(NO_ROW_FOUND);
86        }
87        dbSession.commit();
88        isSaved = true;
89      } finally {
90        preparedStatement.realClose();
91      }
92    }
93  
94    /**
95     * Update object to table
96     *
97     * @throws WaarpDatabaseException
98     */
99    @Override
100   public void update() throws WaarpDatabaseException {
101     if (isSaved) {
102       return;
103     }
104     if (dbSession == null) {
105       isSaved = true;
106       return;
107     }
108     setToArray();
109     final DbPreparedStatement preparedStatement =
110         new DbPreparedStatement(dbSession);
111     try {
112       preparedStatement.createPrepareStatement(
113           "UPDATE " + getTable() + " SET " + getUpdateAllFields() + " WHERE " +
114           getWherePrimaryKey());
115       setValues(preparedStatement, allFields);
116       final int count = preparedStatement.executeUpdate();
117       if (count <= 0) {
118         throw new WaarpDatabaseNoDataException(NO_ROW_FOUND);
119       }
120       dbSession.commit();
121       isSaved = true;
122     } finally {
123       preparedStatement.realClose();
124     }
125   }
126 
127   /**
128    * Delete object from table
129    *
130    * @throws WaarpDatabaseException
131    */
132   @Override
133   public void delete() throws WaarpDatabaseException {
134     if (dbSession == null) {
135       return;
136     }
137     final DbPreparedStatement preparedStatement =
138         new DbPreparedStatement(dbSession);
139     try {
140       preparedStatement.createPrepareStatement(
141           "DELETE FROM " + getTable() + " WHERE " + getWherePrimaryKey());
142       setPrimaryKey();
143       setValues(preparedStatement, primaryKey);
144       final int count = preparedStatement.executeUpdate();
145       if (count <= 0) {
146         throw new WaarpDatabaseNoDataException(NO_ROW_FOUND);
147       }
148       dbSession.commit();
149       isSaved = false;
150     } finally {
151       preparedStatement.realClose();
152     }
153   }
154 }
