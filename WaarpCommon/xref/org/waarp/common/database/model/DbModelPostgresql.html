71    /**
72     * Create the object and initialize if necessary the driver
73     *
74     * @throws WaarpDatabaseNoConnectionException
75     */
76    protected DbModelPostgresql() throws WaarpDatabaseNoConnectionException {
77      if (DbModelFactory.classLoaded.contains(type.name())) {
78        return;
79      }
80      try {
81        DriverManager.registerDriver(new Driver());
82        DbModelFactory.classLoaded.add(type.name());
83      } catch (final SQLException e) {
84        // SQLException
85        logger.error(
86            "Cannot register Driver " + type.name() + ' ' + e.getMessage());
87        DbSession.error(e);
88        throw new WaarpDatabaseNoConnectionException(
89            "Cannot load database drive:" + type.name(), e);
90      }
91      // No pooling connection yet available through URL and not for production purpose
92      /*
93       * Quoting the PostgreSQL documentation: from http://jdbc.postgresql.org/documentation/head/ds-ds.html In
94       * general it is not recommended to use the PostgreSQLâ„¢ provided connection pool.
95       *
96       */
97      /*
98       * PGPoolingDataSource source = new PGPoolingDataSource(); source.setDataSourceName("A Data Source");
99       * source.setServerName("localhost"); source.setDatabaseName("test"); source.setUser("testuser");
100      * source.setPassword("testpassword"); source.setMaxConnections(10);
101      */
102   }

178   @Override
179   public long nextSequence(DbSession dbSession)
180       throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException,
181              WaarpDatabaseNoDataException {
182     long result = DbConstant.ILLEGALVALUE;
183     final String action = "SELECT NEXTVAL('" + DbDataModel.fieldseq + "')";
184     final DbPreparedStatement preparedStatement =
185         new DbPreparedStatement(dbSession);
186     try {
187       preparedStatement.createPrepareStatement(action);
188       // Limit the search
189       preparedStatement.executeQuery();
190       if (preparedStatement.getNext()) {
191         try {
192           result = preparedStatement.getResultSet().getLong(1);
193         } catch (final SQLException e) {
194           throw new WaarpDatabaseSqlException(e);
195         }
196         return result;
197       } else {
198         throw new WaarpDatabaseNoDataException(
199             "No sequence found. Must be initialized first");
200       }
201     } finally {
202       preparedStatement.realClose();
203     }
204   }
