20  package org.waarp.common.guid;
21  
22  import org.waarp.common.utility.Hexa;
23  import org.waarp.common.utility.StringUtils;
24  
25  import java.util.Arrays;
26  import java.util.concurrent.atomic.AtomicInteger;
27  
28  /**
29   * UUID Generator (also Global UUID Generator) but limited to 1 Long (64 bits)
30   * <br>
31   * <br>
32   * Inspired from com.groupon locality-uuid which used combination of internal
33   * counter value - process id - and
34   * Timestamp. see https://github.com/groupon/locality-uuid.java <br>
35   * <br>
36   * But force sequence and take care of errors and improves some performance
37   * issues
38   */
39  public final class LongUuid {
40    private static final Object SYNC_OBJECT = new Object();
41    /**
42     * Counter part
43     */
44    private static final AtomicInteger COUNTER =
45        new AtomicInteger(StringUtils.RANDOM.nextInt());
46    /**
47     * Byte size of UUID
48     */
49    private static final int UUIDSIZE = 8;
50  
51    /**
52     * real UUID
53     */
54    private final byte[] uuid = { 0, 0, 0, 0, 0, 0, 0, 0 };
55  
56    /**
57     * Constructor that generates a new UUID using the current process id, MAC
58     * address, and timestamp
59     */
60    public LongUuid() {
61      final long time = System.currentTimeMillis();
62      // atomically
63      final int count;
64      synchronized (SYNC_OBJECT) {
65        count = COUNTER.incrementAndGet();
66        if (count == Integer.MAX_VALUE) {
67          COUNTER.set(Integer.MIN_VALUE + 1);
68        }
69      }
70  
71      // copy pid to uuid
72      uuid[0] = (byte) (JvmProcessId.JVMPID >> 8);
73      uuid[1] = (byte) JvmProcessId.JVMPID;
74  
75      // copy timestamp into uuid (up to 2^36 s = 2 years rolling)
76      uuid[2] = (byte) (time >> 28);
77      uuid[3] = (byte) (time >> 20);
78      uuid[4] = (byte) (time >> 12);
79  
80      // Keep 4 first bytes, 4 bytes coming from Timestamp => 2^20 (at most 1M / 1/2s)
81      uuid[5] = (byte) (count >> 16 & 0x0F | time >> 4 & 0xF0);
82      uuid[6] = (byte) (count >> 8);
83      uuid[7] = (byte) count;
84    }
85  
86    /**
87     * Constructor that takes a byte array as this UUID's content
88     *
89     * @param bytes UUID content
90     */
91    public LongUuid(final byte[] bytes) {
92      if (bytes.length != UUIDSIZE) {
93        throw new RuntimeException(
94            "Attempted to parse malformed UUID: " + Arrays.toString(bytes));
95      }
96      System.arraycopy(bytes, 0, uuid, 0, UUIDSIZE);
97    }
98  
99    public LongUuid(final long value) {
100     uuid[0] = (byte) (value >> 56);
101     uuid[1] = (byte) (value >> 48);
102     uuid[2] = (byte) (value >> 40);
103     uuid[3] = (byte) (value >> 32);
104     uuid[4] = (byte) (value >> 24);
105     uuid[5] = (byte) (value >> 16);
106     uuid[6] = (byte) (value >> 8);
107     uuid[7] = (byte) value;
108   }
109 
110   public LongUuid(final String idsource) {
111     final String id = idsource.trim();
112 
113     if (id.length() != UUIDSIZE * 2) {
114       throw new RuntimeException("Attempted to parse malformed UUID: " + id);
115     }
116     System.arraycopy(Hexa.fromHex(id), 0, uuid, 0, UUIDSIZE);
117   }
118 
119   @Override
120   public String toString() {
121     return Hexa.toHex(uuid);
122   }
123 
124   /**
125    * copy the uuid of this UUID, so that it can't be changed, and return it
126    *
127    * @return raw byte array of UUID
128    */
129   public final byte[] getBytes() {
130     return Arrays.copyOf(uuid, UUIDSIZE);
131   }
132 
133   /**
134    * extract process id from raw UUID bytes and return as int
135    *
136    * @return id of process that generated the UUID
137    */
138   public final int getProcessId() {
139     return (uuid[0] & 0xFF) << 8 | uuid[1] & 0xFF;
140   }
141 
142   /**
143    * extract timestamp from raw UUID bytes and return as int
144    *
145    * @return millisecond UTC timestamp from generation of the UUID
146    */
147   public final long getTimestamp() {
148     long time;
149     time = ((long) uuid[2] & 0xFF) << 28;
150     time |= ((long) uuid[3] & 0xFF) << 20;
151     time |= ((long) uuid[4] & 0xFF) << 12;
152     time |= ((long) uuid[5] & 0xF0) << 4;
153     return time;
154   }
155 
156   @Override
157   public final boolean equals(final Object o) {
158     if (!(o instanceof LongUuid)) {
159       return false;
160     }
161     return this == o || Arrays.equals(uuid, ((LongUuid) o).uuid);
162   }
163 
164   @Override
165   public final int hashCode() {
166     return Arrays.hashCode(uuid);
167   }
168 
169   /**
170    * @return the equivalent UUID as long
171    */
172   public final long getLong() {
173     long value = ((long) uuid[0] & 0xFF) << 56;
174     value |= ((long) uuid[1] & 0xFF) << 48;
175     value |= ((long) uuid[2] & 0xFF) << 40;
176     value |= ((long) uuid[3] & 0xFF) << 32;
177     value |= ((long) uuid[4] & 0xFF) << 24;
178     value |= ((long) uuid[5] & 0xFF) << 16;
179     value |= ((long) uuid[6] & 0xFF) << 8;
180     value |= (long) uuid[7] & 0xFF;
181     return value;
182   }
183 }
