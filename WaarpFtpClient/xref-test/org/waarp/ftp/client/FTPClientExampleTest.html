45  /**
46   * This is the FTPClient example from Apache Commons-Net<br>
47   * <p>
48   * This is an example program demonstrating how to use the FTPClient class. This
49   * program connects to an FTP
50   * server and retrieves the specified file. If the -s flag is used, it stores
51   * the local file at the FTP
52   * server. Just so you can see what's happening, all reply strings are printed.
53   * If the -b flag is used, a
54   * binary transfer is assumed (default is ASCII). See below for further
55   * options.
56   */
57  public final class FTPClientExampleTest {
58    public static final String USAGE =
59        "Usage: ftp [options] <hostname> <username> <password> <account> [<remote file> [<local file>]]\n" +
60        "\nDefault behavior is to download a file and use ASCII transfer mode.\n" +
61        "\t-a - use local active mode (default is local passive)\n" +
62        "\t-b - use binary transfer mode\n" +
63        "\t-c cmd - issue arbitrary command (remote is used as a parameter if provided) \n" +
64        "\t-d - list directory details using MLSD (remote is used as the pathname if provided)\n" +
65        "\t-e - use EPSV with IPv4 (default false)\n" +
66        "\t-f - issue FEAT command (remote and local files are ignored)\n" +
67        "\t-h - list hidden files (applies to -l and -n only)\n" +
68        "\t-k secs - use keep-alive timer (setControlKeepAliveTimeout)\n" +
69        "\t-l - list files using LIST (remote is used as the pathname if provided)\n" +
70        "\t-L - use lenient future dates (server dates may be up to 1 day into future)\n" +
71        "\t-n - list file names using NLST (remote is used as the pathname if provided)\n" +
72        "\t-p true|false|protocol[,true|false] - use FTPSClient with the specified protocol and/or isImplicit setting\n" +
73        "\t-s - store file on server (upload)\n" +
74        "\t-t - list file details using MLST (remote is used as the pathname if provided)\n" +
75        "\t-w msec - wait time for keep-alive reply (setControlKeepAliveReplyTimeout)\n" +
76        "\t-T  all|valid|none - use one of the built-in TrustManager implementations (none = JVM default)\n" +
77        "\t-PrH server[:port] - HTTP Proxy host and optional port[80] \n" +
78        "\t-PrU user - HTTP Proxy server username\n" +
79        "\t-PrP password - HTTP Proxy server password\n" +
80        "\t-# - add hash display during transfers\n";
81    /**
82     * Internal Logger
83     */
84    private static final WaarpLogger logger =
85        WaarpLoggerFactory.getLogger(FTPClientExampleTest.class);
86  
87    public static final void main(String[] args) {
88      boolean storeFile = false, binaryTransfer = false, error = false,
89          listFiles = false, listNames = false, hidden = false;
90      boolean localActive = false, useEpsvWithIPv4 = false, feat = false,
91          printHash = false;
92      boolean mlst = false, mlsd = false;
93      boolean lenient = false;
94      long keepAliveTimeout = -1;
95      int controlKeepAliveReplyTimeout = -1;
96      int minParams = 5; // listings require 3 params
97      String protocol = null; // SSL protocol
98      String doCommand = null;
99      String trustmgr = null;
100     String proxyHost = null;
101     int proxyPort = 80;
102     String proxyUser = null;
103     String proxyPassword = null;
104 
105     int base = 0;
106     for (base = 0; base < args.length; base++) {
107       if ("-s".equals(args[base])) {
108         storeFile = true;
109       } else if ("-a".equals(args[base])) {
110         localActive = true;
111       } else if ("-b".equals(args[base])) {
112         binaryTransfer = true;
113       } else if ("-c".equals(args[base])) {
114         doCommand = args[++base];
115         minParams = 3;
116       } else if ("-d".equals(args[base])) {
117         mlsd = true;
118         minParams = 3;
119       } else if ("-e".equals(args[base])) {
120         useEpsvWithIPv4 = true;
121       } else if ("-f".equals(args[base])) {
122         feat = true;
123         minParams = 3;
124       } else if ("-h".equals(args[base])) {
125         hidden = true;
126       } else if ("-k".equals(args[base])) {
127         keepAliveTimeout = Long.parseLong(args[++base]);
128       } else if ("-l".equals(args[base])) {
129         listFiles = true;
130         minParams = 3;
131       } else if ("-L".equals(args[base])) {
132         lenient = true;
133       } else if ("-n".equals(args[base])) {
134         listNames = true;
135         minParams = 3;
136       } else if ("-p".equals(args[base])) {
137         protocol = args[++base];
138       } else if ("-t".equals(args[base])) {
139         mlst = true;
140         minParams = 3;
141       } else if ("-w".equals(args[base])) {
142         controlKeepAliveReplyTimeout = Integer.parseInt(args[++base]);
143       } else if ("-T".equals(args[base])) {
144         trustmgr = args[++base];
145       } else if ("-PrH".equals(args[base])) {
146         proxyHost = args[++base];
147         final String[] parts = proxyHost.split(":");
148         if (parts.length == 2) {
149           proxyHost = parts[0];
150           proxyPort = Integer.parseInt(parts[1]);
151         }
152       } else if ("-PrU".equals(args[base])) {
153         proxyUser = args[++base];
154       } else if ("-PrP".equals(args[base])) {
155         proxyPassword = args[++base];
156       } else if ("-#".equals(args[base])) {
157         printHash = true;
158       } else {
159         break;
160       }
161     }
162 
163     final int remain = args.length - base;
164     if (remain < minParams) // server, user, pass, remote, local [protocol]
165     {
166       System.err.println(USAGE);
167       DetectionUtils.systemExit(1);
168       return;
169     }
170 
171     String server = args[base++];
172     int port = 0;
173     final String[] parts = server.split(":");
174     if (parts.length == 2) {
175       server = parts[0];
176       port = Integer.parseInt(parts[1]);
177     }
178     final String username = args[base++];
179     final String password = args[base++];
180     final String account = args[base++];
181 
182     String remote = null;
183     if (args.length - base > 0) {
184       remote = args[base++];
185     }
186 
187     String local = null;
188     if (args.length - base > 0) {
189       local = args[base++];
190     }
191 
192     boolean mustCallProtP = false;
193     final FTPClient ftp;
194     if (protocol == null) {
195       if (proxyHost != null) {
196         System.out.println("Using HTTP proxy server: " + proxyHost);
197         ftp = new FTPHTTPClient(proxyHost, proxyPort, proxyUser, proxyPassword);
198       } else {
199         ftp = new FTPClient();
200       }
201     } else {
202       FTPSClient ftps;
203       if ("true".equals(protocol)) {
204         System.out.println("Implicit FTPS");
205         ftps = new FTPSClient(true);
206       } else if ("false".equals(protocol)) {
207         System.out.println("Explicit FTPS");
208         ftps = new FTPSClient(false);
209         mustCallProtP = true;
210       } else {
211         final String[] prot = protocol.split(",");
212         System.out.println("Protocl FTPS: " + protocol);
213 
214         if (prot.length == 1) { // Just protocol
215           ftps = new FTPSClient(protocol);
216         } else { // protocol,true|false
217           mustCallProtP = !Boolean.parseBoolean(prot[1]);
218           ftps = new FTPSClient(prot[0], !mustCallProtP);
219         }
220       }
221       ftp = ftps;
222       if ("all".equals(trustmgr)) {
223         System.out.println("Accept all");
224         ftps.setTrustManager(TrustManagerUtils.getAcceptAllTrustManager());
225       } else if ("valid".equals(trustmgr)) {
226         System.out.println("Accept after valid");
227         ftps.setTrustManager(
228             TrustManagerUtils.getValidateServerCertificateTrustManager());
229       } else if ("none".equals(trustmgr)) {
230         System.out.println("Accept none");
231         ftps.setTrustManager(null);
232       }
233     }
234 
235     if (printHash) {
236       ftp.setCopyStreamListener(createListener());
237     }
238     if (keepAliveTimeout >= 0) {
239       ftp.setControlKeepAliveTimeout(keepAliveTimeout);
240     }
241     if (controlKeepAliveReplyTimeout >= 0) {
242       ftp.setControlKeepAliveReplyTimeout(controlKeepAliveReplyTimeout);
243     }
244     ftp.setListHiddenFiles(hidden);
245 
246     // suppress login details
247     ftp.addProtocolCommandListener(
248         new PrintCommandListener(new PrintWriter(System.out), true));
249 
250     try {
251       int reply;
252       if (port > 0) {
253         ftp.connect(server, port);
254       } else {
255         ftp.connect(server);
256       }
257       System.out.println("Connected to " + server + " on " +
258                          (port > 0? port : ftp.getDefaultPort()));
259 
260       // After connection attempt, you should check the reply code to verify
261       // success.
262       reply = ftp.getReplyCode();
263 
264       if (!FTPReply.isPositiveCompletion(reply)) {
265         ftp.disconnect();
266         System.err.println("FTP server refused connection.");
267         DetectionUtils.systemExit(1);
268         return;
269       }
270     } catch (final IOException e) {
271       if (ftp.getDataConnectionMode() ==
272           FTPClient.ACTIVE_LOCAL_DATA_CONNECTION_MODE) {
273         try {
274           ftp.disconnect();
275         } catch (final IOException f) {
276           // do nothing
277         }
278       }
279       System.err.println("Could not connect to server.");
280       e.printStackTrace();
281       DetectionUtils.systemExit(1);
282       return;
283     }
284 
285     __main:
286     try {
287       if (account == null) {
288         if (!ftp.login(username, password)) {
289           ftp.logout();
290           error = true;
291           break __main;
292         }
293       } else {
294         if (!ftp.login(username, password, account)) {
295           ftp.logout();
296           error = true;
297           break __main;
298         }
299       }
300       System.out.println("Remote system is " + ftp.getSystemType());
301 
302       if (binaryTransfer) {
303         ftp.setFileType(FTP.BINARY_FILE_TYPE);
304       }
305 
306       // Use passive mode as default because most of us are
307       // behind firewalls these days.
308       if (localActive) {
309         ftp.enterLocalActiveMode();
310       } else {
311         ftp.enterLocalPassiveMode();
312       }
313 
314       ftp.setUseEPSVwithIPv4(useEpsvWithIPv4);
315 
316       if (mustCallProtP) {
317         ((FTPSClient) ftp).execPBSZ(0);
318         ((FTPSClient) ftp).execPROT("P");
319       }
320 
321       if (storeFile) {
322         InputStream input;
323 
324         input = new FileInputStream(local);
325 
326         ftp.storeFile(remote, input);
327 
328         input.close();
329       } else if (listFiles) {
330         if (lenient) {
331           final FTPClientConfig config = new FTPClientConfig();
332           config.setLenientFutureDates(true);
333           ftp.configure(config);
334         }
335 
336         for (final FTPFile f : ftp.listFiles(remote)) {
337           System.out.println(f.getRawListing());
338           System.out.println(f.toFormattedString());
339         }
340       } else if (mlsd) {
341         for (final FTPFile f : ftp.mlistDir(remote)) {
342           System.out.println(f.getRawListing());
343           System.out.println(f.toFormattedString());
344         }
345       } else if (mlst) {
346         final FTPFile f = ftp.mlistFile(remote);
347         if (f != null) {
348           System.out.println(f.toFormattedString());
349         }
350       } else if (listNames) {
351         for (final String s : ftp.listNames(remote)) {
352           System.out.println(s);
353         }
354       } else if (feat) {
355         // boolean feature check
356         if (remote != null) { // See if the command is present
357           if (ftp.hasFeature(remote)) {
358             System.out.println("Has feature: " + remote);
359           } else {
360             if (FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
361               System.out.println("FEAT " + remote + " was not detected");
362             } else {
363               System.out.println("Command failed: " + ftp.getReplyString());
364             }
365           }
366 
367           // Strings feature check
368           final String[] features = ftp.featureValues(remote);
369           if (features != null) {
370             for (final String f : features) {
371               System.out.println("FEAT " + remote + '=' + f + '.');
372             }
373           } else {
374             if (FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
375               System.out.println("FEAT " + remote + " is not present");
376             } else {
377               System.out.println("Command failed: " + ftp.getReplyString());
378             }
379           }
380         } else {
381           if (ftp.features()) {
382             // Command listener has already printed the output
383           } else {
384             System.out.println("Failed: " + ftp.getReplyString());
385           }
386         }
387       } else if (doCommand != null) {
388         if (ftp.doCommand(doCommand, remote)) {
389           // Command listener has already printed the output
390           // for(String s : ftp.getReplyStrings()) {
391           // System.out.println(s);
392           // }
393         } else {
394           System.out.println("Failed: " + ftp.getReplyString());
395         }
396       } else {
397         OutputStream output;
398 
399         output = new FileOutputStream(local);
400 
401         ftp.retrieveFile(remote, output);
402 
403         output.close();
404       }
405 
406       ftp.noop(); // check that control connection is working OK
407 
408       ftp.logout();
409     } catch (final FTPConnectionClosedException e) {
410       error = true;
411       System.err.println("Server closed connection.");
412       e.printStackTrace();
413     } catch (final IOException e) {
414       error = true;
415       e.printStackTrace();
416     } finally {
417       if (ftp.getDataConnectionMode() ==
418           FTPClient.ACTIVE_LOCAL_DATA_CONNECTION_MODE) {
419         try {
420           ftp.disconnect();
421         } catch (final IOException f) {
422           // do nothing
423         }
424       }
425     }
426 
427     DetectionUtils.systemExit(error? 1 : 0);
428   } // end main

430   private static CopyStreamListener createListener() {
431     return new CopyStreamListener() {
432       private long megsTotal;
433 
434       @Override
435       public void bytesTransferred(CopyStreamEvent event) {
436         bytesTransferred(event.getTotalBytesTransferred(),
437                          event.getBytesTransferred(), event.getStreamSize());
438       }
439 
440       @Override
441       public void bytesTransferred(long totalBytesTransferred,
442                                    int bytesTransferred, long streamSize) {
443         final long megs = totalBytesTransferred / 1000000;
444         for (long l = megsTotal; l < megs; l++) {
445           System.err.print("#");
446         }
447         megsTotal = megs;
448       }
449     };
