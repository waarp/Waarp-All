174   @Test
175   public void test0_0Ftp4JZ() throws IOException {
176     FtpClientTest.numberKO.set(0);
177     FtpClientTest.numberOK.set(0);
178     final File localFilename = new File("/tmp/ftpfileZ.bin");
179     final Ftp4JClientTransactionTest client =
180         new Ftp4JClientTransactionTest("127.0.0.1", port, "fred", "fred2", "a",
181                                        SSL_MODE);
182     client.setReportActiveExternalIPAddress("127.0.0.1");
183     logger.warn("First connexion");
184     if (!client.connect()) {
185       logger.error("Can't connect");
186       if (AbstractFtpClient.versionJavaCompatible()) {
187         numberKO.incrementAndGet();
188       }
189       Assert.assertEquals("No KO", 0, numberKO.get());
190       return;
191     }
192     try {
193       logger.warn("Try Mode Z");
194       client.compressionMode(true);
195       String remoteFilename = localFilename.getName();
196       logger.warn("Try Store Mode Z");
197       if (!client.transferFile(localFilename.getAbsolutePath(), remoteFilename,
198                                true)) {
199         logger.warn("Cant Z store file active mode " + 0);
200         FtpClientTest.numberKO.incrementAndGet();
201         return;
202       } else {
203         FtpClientTest.numberOK.incrementAndGet();
204       }
205       try {
206         Thread.sleep(10);
207       } catch (InterruptedException e) {
208         // Ignore
209       }
210       assertEquals(0, FtpClientTest.numberKO.get());
211       logger.warn("Try Retrieve Mode Z");
212       client.compressionMode(true);
213       if (!client.transferFile(null, remoteFilename, false)) {
214         logger.warn("Cant Z retrieve file active mode " + 0);
215         FtpClientTest.numberKO.incrementAndGet();
216       } else {
217         FtpClientTest.numberOK.incrementAndGet();
218       }
219     } finally {
220       client.compressionMode(false);
221       logger.warn("Logout");
222       client.logout();
223     }
224     assertEquals(0, FtpClientTest.numberKO.get());
225   }

227   public void testFtp4J(String server, int port, String username, String passwd,
228                         String account, int isSSL, String localFilename,
229                         int type, int delay, boolean shutdown, int numberThread,
230                         int numberIteration) {
231     // initiate Directories
232     final Ftp4JClientTransactionTest client =
233         new Ftp4JClientTransactionTest(server, port, username, passwd, account,
234                                        isSSL);
235     client.setReportActiveExternalIPAddress("127.0.0.1");
236     client.setActiveDataTransferPortRange(57000, 63000);
237     logger.warn("First connexion");
238     if (!client.connect()) {
239       logger.error("Can't connect");
240       if (AbstractFtpClient.versionJavaCompatible()) {
241         numberKO.incrementAndGet();
242       }
243       Assert.assertEquals("No KO", 0, numberKO.get());
244       return;
245     }
246     try {
247       logger.warn("Create Dirs");
248       for (int i = 0; i < numberThread; i++) {
249         System.out.print('.');
250         client.makeDir("T" + i);
251       }
252       try {
253         Thread.sleep(10);
254       } catch (InterruptedException e) {
255         // Ignore
256       }
257       System.out.println();
258       logger.warn("Feature commands");
259       System.out.println("SITE: " + client.featureEnabled("SITE"));
260       System.out.println("SITE CRC: " + client.featureEnabled("SITE XCRC"));
261       System.out.println("CRC: " + client.featureEnabled("XCRC"));
262       System.out.println("MD5: " + client.featureEnabled("XMD5"));
263       System.out.println("SHA1: " + client.featureEnabled("XSHA1"));
264       System.out.println("DIGEST: " + client.featureEnabled("XDIGEST"));
265     } finally {
266       logger.warn("Logout");
267       client.logout();
268     }
269     final ExecutorService executorService = Executors.newCachedThreadPool();
270     logger.warn("Will start {} Threads", numberThread);
271     final long date1 = System.currentTimeMillis();
272     for (int i = 0; i < numberThread; i++) {
273       executorService.execute(
274           new FtpClientThread("T" + i, server, port, username, passwd, account,
275                               localFilename, numberIteration, type, delay,
276                               isSSL));
277       Thread.yield();
278     }
279     try {
280       Thread.sleep(100);
281     } catch (final InterruptedException e1) {//NOSONAR
282       e1.printStackTrace();
283     }
284     executorService.shutdown();
285     long date2 = 0;
286     try {
287       if (!executorService.awaitTermination(120000, TimeUnit.MILLISECONDS)) {
288         executorService.shutdownNow();
289         if (!executorService.awaitTermination(60, TimeUnit.SECONDS)) {
290           logger.error("Really not shutdown normally");
291         }
292       }
293     } catch (final InterruptedException e) {//NOSONAR
294       e.printStackTrace();
295       executorService.shutdownNow();
296     }
297     date2 = System.currentTimeMillis();
298 
299     logger.warn(
300         localFilename + ' ' + numberThread + ' ' + numberIteration + ' ' +
301         type + " Real: " + (date2 - date1) + " OK: " +
302         FtpClientTest.numberOK.get() + " KO: " + FtpClientTest.numberKO.get() +
303         " Trf/s: " + FtpClientTest.numberOK.get() * 1000 / (date2 - date1));
304     assertEquals("No KO", 0, FtpClientTest.numberKO.get());
305   }

307   public static void launchFtpClient(String server, int port, String username,
308                                      String password, String account,
309                                      boolean localActive, String local,
310                                      String remote) {
311     final WaarpFtpClient waarpFtpClient =
312         new WaarpFtpClient(server, port, username, password, account,
313                            !localActive, SSL_MODE, 30000, 30000, true);
314     boolean error = false;
315 
316     boolean connected = waarpFtpClient.connect();
317     // After connection attempt, you should check the reply code to verify
318     // success.
319     if (!connected) {
320       logger.error("FTP server refused connection.");
321       fail("Can't connect");
322       return;
323     }
324 
325     try {
326       if (!waarpFtpClient.store(local, remote)) {
327         error = true;
328         return;
329       }
330       if (!waarpFtpClient.retrieve(local, remote)) {
331         error = true;
332         return;
333       }
334 
335       for (final String f : waarpFtpClient.listFiles(remote)) {
336         System.out.println(f);
337       }
338       for (final String f : waarpFtpClient.mlistFiles(remote)) {
339         System.out.println(f);
340       }
341       if (!waarpFtpClient.deleteFile(remote)) {
342         error = true;
343         logger.error("Failed delete");
344       }
345       // boolean feature check
346       String[] features = waarpFtpClient.features();
347       if (features != null) {
348         // Command listener has already printed the output
349       } else {
350         logger.error("Failed: " + waarpFtpClient.getResult());
351         error = true;
352       }
353 
354       waarpFtpClient.noop(); // check that control connection is working OK
355     } finally {
356       waarpFtpClient.disconnect();
357       assertFalse("Error occurs", error);
358     }
359   }
