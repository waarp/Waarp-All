145     if (!isDataNetworkHandlerReady) {
146       Thread.sleep(10);

149         throw new InterruptedException("Bad initialization");
150       }

154   /**
155    * Set the new opened Channel (from channelConnected of {@link
156    * DataNetworkHandler})
157    *
158    * @param channel
159    * @param dataNetworkHandler
160    */
161   public void setOpenedDataChannel(Channel channel,
162                                    DataNetworkHandler dataNetworkHandler) {
163     logger.debug("SetOpenedDataChannel: " +
164                  (channel != null? channel.remoteAddress() : "no channel"));
165     if (channel != null) {
166       session.getDataConn().setDataNetworkHandler(dataNetworkHandler);
167       waitForOpenedDataChannel.setChannel(channel);
168       waitForOpenedDataChannel.setSuccess();
169     } else {
170       waitForOpenedDataChannel.cancel();
171     }
172   }

174   /**
175    * Wait that the new opened connection is ready (same method in {@link
176    * FtpDataAsyncConn} from openConnection)
177    *
178    * @return the new opened Channel
179    *
180    * @throws InterruptedException
181    */
182   public Channel waitForOpenedDataChannel() throws InterruptedException {
183     Channel channel = null;
184     if (waitForOpenedDataChannel.awaitOrInterruptible()) {
185       if (waitForOpenedDataChannel.isSuccess()) {
186         channel = waitForOpenedDataChannel.channel();
187       } else {
188         logger.warn("data connection is in error");
189       }
190     } else {
191       logger.warn("Timeout occurs during data connection");
192       throw new InterruptedException("Cannot get data connection");
193     }
194     return channel;
195   }

204     waitForOpenedDataChannel = new WaarpChannelFuture(true);

207   /**
208    * Wait for the client to be connected (Passive) or Wait for the server to
209    * be
210    * connected to the client (Active)
211    *
212    * @return True if the connection is OK
213    *
214    * @throws Reply425Exception
215    */
216   public synchronized boolean openDataConnection() throws Reply425Exception {
217     // Prepare this Data channel to be closed ;-)
218     // In fact, prepare the future close op which should occur since it is
219     // now opened
220     if (commandSetup != null) {
221       commandSetup.cancel();
222     }
223     commandSetup = new WaarpFuture(true);
224     final FtpDataAsyncConn dataAsyncConn = session.getDataConn();
225     if (!dataAsyncConn.isStreamFile()) {
226       // FIXME isActive or isDNHReady ?
227       if (dataAsyncConn.isActive()) {
228         // Already connected
229         logger.debug("Connection already open");
230         session.setReplyCode(ReplyCode.REPLY_125_DATA_CONNECTION_ALREADY_OPEN,
231                              dataAsyncConn.getType().name() +
232                              " mode data connection already open");
233         return true;
234       }
235     } else {
236       // Stream, Data Connection should not be opened
237       if (dataAsyncConn.isActive()) {
238         logger.error(
239             "Connection already open but should not since in Stream mode");
240         setTransferAbortedFromInternal(false);
241         throw new Reply425Exception(
242             "Connection already open but should not since in Stream mode");
243       }
244     }
245     // Need to open connection
246     session.setReplyCode(ReplyCode.REPLY_150_FILE_STATUS_OKAY,
247                          "Opening " + dataAsyncConn.getType().name() +
248                          " mode data connection");
249     if (dataAsyncConn.isPassiveMode()) {
250       if (!dataAsyncConn.isBind()) {
251         // No passive connection prepared
252         throw new Reply425Exception("No passive data connection prepared");
253       }
254       // Wait for the connection to be done by the client
255       logger.debug("Passive mode standby");
256       try {
257         dataChannel = waitForOpenedDataChannel();
258         dataAsyncConn.setNewOpenedDataChannel(dataChannel);
259       } catch (final InterruptedException e) {//NOSONAR
260         logger.warn("Connection abort in passive mode", e);
261         // Cannot open connection
262         throw new Reply425Exception("Cannot open passive data connection");
263       }
264       logger.debug("Passive mode connected");
265     } else {
266       // Wait for the server to be connected to the client
267       final InetAddress inetAddress =
268           dataAsyncConn.getLocalAddress().getAddress();
269       final InetSocketAddress inetSocketAddress =
270           dataAsyncConn.getRemoteAddress();
271       if (session.getConfiguration().getFtpInternalConfiguration()
272                  .hasFtpSession(inetAddress, inetSocketAddress)) {
273         throw new Reply425Exception(
274             "Cannot open active data connection since remote address is already in use: " +
275             inetSocketAddress);
276       }
277       logger.debug("Active mode standby");
278       final Bootstrap bootstrap =
279           session.getConfiguration().getFtpInternalConfiguration()
280                  .getActiveBootstrap(session.isDataSsl());
281       session.getConfiguration()
282              .setNewFtpSession(inetAddress, inetSocketAddress, session);
283       // Set the session for the future dataChannel
284       final String mylog = session.toString();
285       logger.debug(
286           "DataConn for: " + session.getCurrentCommand().getCommand() + " to " +
287           inetSocketAddress);
288       final ChannelFuture future =
289           bootstrap.connect(inetSocketAddress, dataAsyncConn.getLocalAddress());
290       try {
291         future.await(session.getConfiguration().getTimeoutCon());
292       } catch (final InterruptedException e1) {//NOSONAR
293         SysErrLogger.FAKE_LOGGER.ignoreLog(e1);
294       }
295       if (!future.isSuccess()) {
296         logger.warn(
297             "Connection abort in active mode from future while session: " +
298             session + "\nTrying connect to: " + inetSocketAddress + " From: " +
299             dataAsyncConn.getLocalAddress() + "\nWas: " + mylog,
300             future.cause());
301         // Cannot open connection
302         session.getConfiguration()
303                .delFtpSession(inetAddress, inetSocketAddress);
304         throw new Reply425Exception("Cannot open active data connection");
305       }
306       try {
307         dataChannel = waitForOpenedDataChannel();
308         dataAsyncConn.setNewOpenedDataChannel(dataChannel);
309       } catch (final InterruptedException e) {//NOSONAR
310         logger.warn("Connection abort in active mode", e);
311         // Cannot open connection
312         session.getConfiguration()
313                .delFtpSession(inetAddress, inetSocketAddress);
314         throw new Reply425Exception("Cannot open active data connection");
315       }
316       logger.debug("Active mode connected");
317     }
318     if (dataChannel == null) {
319       // Cannot have a new Data connection since shutdown
320       if (!dataAsyncConn.isPassiveMode()) {
321         session.getConfiguration().getFtpInternalConfiguration()
322                .delFtpSession(dataAsyncConn.getLocalAddress().getAddress(),
323                               dataAsyncConn.getRemoteAddress());
324       }
325       throw new Reply425Exception("Cannot open data connection, shuting down");
326     }
327     return true;
328   }

332   /**
333    * Run the command from an executor
334    */
335   private void runExecutor() {
336     endOfCommand = new WaarpFuture(true);
337     try {
338       session.getDataConn().getDataNetworkHandler()
339              .setFtpTransfer(executingCommand);
340     } catch (final FtpNoConnectionException ignored) {
341       // nothing
342     }
343 
344     if (!waitForOpenedDataChannel.awaitOrInterruptible()) {
345       commandFinishing.cancel();
346       endOfCommand.cancel();
347       return;
348     }
349     waitForOpenedDataChannel.channel().config().setAutoRead(true);
350     // Run the command
351     if (executorService == null) {
352       executorService = Executors.newSingleThreadExecutor();
353     }
354     executorService.execute(new FtpTransferExecutor(session, executingCommand));
355     commandFinishing.awaitOrInterruptible();
356     if (commandFinishing.isFailed()) {
357       endOfCommand.cancel();
358     }
359   }

361   /**
362    * Add a new transfer to be executed. This is to be called from Command
363    * after
364    * connection is opened and before
365    * answering to the client that command is ready to be executed (for Store
366    * or
367    * Retrieve like operations).
368    *
369    * @param command
370    * @param file
371    */
372   public void setNewFtpTransfer(FtpCommandCode command, FtpFile file) {
373     isExecutingCommandFinished = false;
374     commandFinishing = new WaarpFuture(true);
375     logger.debug("setNewCommand: {}", command);
376     setDataNetworkHandlerReady();
377     executingCommand = new FtpTransfer(command, file);
378     runExecutor();
379     commandFinishing = null;
380     commandSetup.setSuccess();
381     if (!session.getDataConn().isStreamFile()) {
382       waitForOpenedDataChannel.channel().config().setAutoRead(false);
383     }
384   }

387    * Add a new transfer to be executed. This is to be called from Command
388    * after
389    * connection is opened and before
390    * answering to the client that command is ready to be executed (for List
391    * like
392    * operations).
393    *
394    * @param command
395    * @param list
396    * @param path as Original Path
397    */
398   public void setNewFtpTransfer(FtpCommandCode command, List<String> list,
399                                 String path) {
400     isExecutingCommandFinished = false;
401     commandFinishing = new WaarpFuture(true);
402     logger.debug("setNewCommand: {}", command);
403     setDataNetworkHandlerReady();
404     executingCommand = new FtpTransfer(command, list, path);
405     runExecutor();
406     commandFinishing = null;
407     commandSetup.setSuccess();
408     if (!session.getDataConn().isStreamFile()) {
409       waitForOpenedDataChannel.channel().config().setAutoRead(false);
410     }
411     try {
412       session.getDataConn().getDataNetworkHandler().setFtpTransfer(null);
413     } catch (final FtpNoConnectionException ignored) {
414       // nothing
415     }
416   }

418   public boolean waitFtpTransferExecuting() {
419     boolean notFinished = true;
420     for (int i = 0; i < FtpInternalConfiguration.RETRYNB * 100; i++) {
421       if (isExecutingCommandFinished || commandFinishing == null ||
422           session.isCurrentCommandFinished() || commandFinishing != null &&
423                                                 commandFinishing
424                                                     .awaitOrInterruptible(
425                                                         FtpInternalConfiguration.RETRYINMS)) {
426         notFinished = false;
427         break;
428       }
429     }
430     return notFinished;
431   }

457   /**
458    * @return True if the current FtpTransfer is a Retrieve like transfer
459    *
460    * @throws FtpNoTransferException
461    * @throws CommandAbstractException
462    * @throws FtpNoFileException
463    */
464   boolean isExecutingRetrLikeTransfer()
465       throws FtpNoTransferException, CommandAbstractException,
466              FtpNoFileException {
467     return !session.isCurrentCommandFinished() && FtpCommandCode
468         .isRetrLikeCommand(getExecutingFtpTransfer().getCommand()) &&
469            getExecutingFtpTransfer().getFtpFile().isInReading();
470   }

472   /**
473    * Called when a transfer is finished from setEndOfTransfer
474    *
475    * @return True if it was already called before
476    *
477    * @throws FtpNoTransferException
478    */
479   private boolean checkFtpTransferStatus() throws FtpNoTransferException {
480     if (isCheckAlreadyCalled) {
481       logger.warn("Check: ALREADY CALLED");
482       return true;
483     }
484     if (isExecutingCommandFinished) {
485       // already done
486       logger.warn("Check: already Finished");
487       if (commandFinishing != null) {
488         commandFinishing.cancel();
489       }
490       throw new FtpNoTransferException("No transfer running");
491     }
492     if (!isDataNetworkHandlerReady) {
493       // already done
494       logger.warn("Check: already DNH not ready");
495       throw new FtpNoTransferException("No connection");
496     }
497     isCheckAlreadyCalled = true;
498     final FtpTransfer executedTransfer = getExecutingFtpTransfer();
499     logger.debug("Check: command {}", executedTransfer.getCommand());
500     // DNH is ready and Transfer is running
501     if (FtpCommandCode.isListLikeCommand(executedTransfer.getCommand())) {
502       if (executedTransfer.getStatus()) {
503         // Special status for List Like command
504         logger.debug("Check: List OK");
505         closeTransfer();
506         return false;
507       }
508       logger.debug("Check: List Ko");
509       abortTransfer();
510       return false;
511     } else if (FtpCommandCode
512         .isRetrLikeCommand(executedTransfer.getCommand())) {
513       FtpFile file;
514       try {
515         file = executedTransfer.getFtpFile();
516       } catch (final FtpNoFileException e) {
517         logger.debug("Check: Retr no FtpFile for Retr");
518         abortTransfer();
519         return false;
520       }
521       try {
522         if (file.isInReading()) {
523           logger.debug("Check: Retr FtpFile still in reading KO");
524           abortTransfer();
525         } else {
526           logger.debug("Check: Retr FtpFile no more in reading OK");
527           closeTransfer();
528         }
529       } catch (final CommandAbstractException e) {
530         logger.warn("Retr Test is in Reading problem", e);
531         closeTransfer();
532       }
533       return false;
534     } else if (FtpCommandCode
535         .isStoreLikeCommand(executedTransfer.getCommand())) {
536       // logger.debug("Check: Store OK")
537       closeTransfer();
538       return false;
539     } else {
540       logger.warn("Check: Unknown command");
541       abortTransfer();
542     }
543     return false;
544   }

546   /**
547    * Abort the current transfer
548    */
549   private void abortTransfer() {
550     logger
551         .debug("Will abort transfer and write: ", new Exception("trace only"));
552     FtpFile file;
553     FtpTransfer current = null;
554     try {
555       current = getExecutingFtpTransfer();
556       file = current.getFtpFile();
557       file.abortFile();
558     } catch (final FtpNoTransferException e) {
559       logger.warn("Abort problem", e);
560     } catch (final FtpNoFileException ignored) {
561       // nothing
562     } catch (final CommandAbstractException e) {
563       logger.warn("Abort problem", e);
564     }
565     if (current != null) {
566       current.setStatus(false);
567     }
568     endDataConnection();
569     session.setReplyCode(ReplyCode.REPLY_426_CONNECTION_CLOSED_TRANSFER_ABORTED,
570                          "Transfer aborted for " +
571                          (current == null? "Unknown command" :
572                              current.toString()));
573     if (current != null &&
574         !FtpCommandCode.isListLikeCommand(current.getCommand())) {
575       try {
576         session.getBusinessHandler().afterTransferDoneBeforeAnswer(current);
577       } catch (final CommandAbstractException e) {
578         session.setReplyCode(e);
579       }
580     }
581     finalizeExecution();
582   }

584   /**
585    * Finish correctly a transfer
586    */
587   private void closeTransfer() {
588     logger.debug("Will close transfer");
589     FtpFile file;
590     FtpTransfer current = null;
591     try {
592       current = getExecutingFtpTransfer();
593       file = current.getFtpFile();
594       file.closeFile();
595     } catch (final FtpNoTransferException e) {
596       logger.warn("Close problem", e);
597     } catch (final FtpNoFileException ignored) {
598       // nothing
599     } catch (final CommandAbstractException e) {
600       logger.warn("Close problem", e);
601     }
602     if (current != null) {
603       current.setStatus(true);
604     }
605     if (session.getDataConn().isStreamFile()) {
606       endDataConnection();
607     }
608     session.setReplyCode(ReplyCode.REPLY_226_CLOSING_DATA_CONNECTION,
609                          "Transfer complete for " +
610                          (current == null? "Unknown command" :
611                              current.toString()));
612     if (current != null) {
613       if (!FtpCommandCode.isListLikeCommand(current.getCommand())) {
614         try {
615           session.getBusinessHandler().afterTransferDoneBeforeAnswer(current);
616         } catch (final CommandAbstractException e) {
617           session.setReplyCode(e);
618         }
619       } else {
620         // Special wait to prevent fast LIST following by STOR or RETR command
621         try {
622           Thread.sleep(FtpInternalConfiguration.RETRYINMS);
623         } catch (final InterruptedException e) {//NOSONAR
624           SysErrLogger.FAKE_LOGGER.ignoreLog(e);
625         }
626       }
627     }
628     finalizeExecution();
629   }

643   /**
644    * To enable abort from internal error
645    *
646    * @param write True means the message is write back to the control
647    *     command, false it is only prepared
648    */
649   public void setTransferAbortedFromInternal(boolean write) {
650     logger.debug("Set transfer aborted internal {}", write);
651     abortTransfer();
652     if (write) {
653       session.getNetworkHandler().writeIntermediateAnswer();
654     }
655     if (endOfCommand != null) {
656       endOfCommand.cancel();
657     }
658   }

679   public void waitForEndOfTransfer() throws InterruptedException {

689   /**
690    * Finalize execution
691    */
692   private void finalizeExecution() {
693     // logger.debug("Finalize execution")
694     if (commandFinishing != null) {
695       commandFinishing.setSuccess();
696     }
697     isExecutingCommandFinished = true;
698     executingCommand = null;
699     resetWaitForOpenedDataChannel();
700   }
701 
702   /**
703    * End the data connection if any
704    */
705   private synchronized void endDataConnection() {
706     logger.debug("End Data connection");
707     if (isDataNetworkHandlerReady && dataChannel != null) {
708       WaarpNettyUtil
709           .awaitOrInterrupted(WaarpSslUtility.closingSslChannel(dataChannel));
710       isDataNetworkHandlerReady = false;
711       // logger.debug("waitForClosedDataChannel over")
712       dataChannel = null;
713     }
714   }

716   /**
717    * Clear the FtpTransferControl (called when the data connection must be
718    * over
719    * like from clear of
720    * {@link FtpDataAsyncConn}, abort from {@link ABOR} or ending control
721    * connection from {@link NetworkHandler}.
722    */
723   public void clear() {
724     // logger.debug("Clear Ftp Transfer Control")
725     endDataConnection();
726     finalizeExecution();
727     if (endOfCommand != null) {
728       endOfCommand.cancel();
729     }
730     if (waitForOpenedDataChannel != null) {
731       waitForOpenedDataChannel.cancel();
732     }
733     if (commandSetup != null) {
734       commandSetup.cancel();
735     }
736     if (executorService != null) {
737       executorService.shutdownNow();
738       executorService = null;
739     }
740   }
