20  package org.waarp.ftp.core.command.parameter;
21  
22  import org.waarp.common.command.ReplyCode;
23  import org.waarp.common.command.exception.Reply425Exception;
24  import org.waarp.common.command.exception.Reply501Exception;
25  import org.waarp.common.logging.WaarpLogger;
26  import org.waarp.common.logging.WaarpLoggerFactory;
27  import org.waarp.ftp.core.command.AbstractCommand;
28  import org.waarp.ftp.core.config.FtpInternalConfiguration;
29  import org.waarp.ftp.core.data.FtpDataAsyncConn;
30  import org.waarp.ftp.core.utils.FtpChannelUtils;
31  
32  import java.net.InetAddress;
33  import java.net.InetSocketAddress;
34  
35  /**
36   * PASV command
37   */
38  public class PASV extends AbstractCommand {
39    /**
40     * Internal Logger
41     */
42    private static final WaarpLogger logger =
43        WaarpLoggerFactory.getLogger(PASV.class);
44  
45    @Override
46    public void exec() throws Reply425Exception, Reply501Exception {
47      // First Check if any argument
48      if (hasArg()) {
49        throw new Reply501Exception("No argument allowed");
50      }
51      // Take a new port: 3 attempts
52      boolean isInit = false;
53      for (int i = 1; i <= FtpInternalConfiguration.RETRYNB; i++) {
54        final int newport =
55            FtpDataAsyncConn.getNewPassivePort(getConfiguration());
56        if (newport == -1) {
57          throw new Reply425Exception("No port available");
58        }
59        if (getSession().getDataConn().isPassiveMode()) {
60          // Previous mode was Passive so remove the current configuration
61          final InetSocketAddress local =
62              getSession().getDataConn().getLocalAddress();
63          final InetAddress remote =
64              getSession().getDataConn().getRemoteAddress().getAddress();
65          getConfiguration().delFtpSession(remote, local);
66        }
67        logger.info("PASV: set Passive Port {}", newport);
68        getSession().getDataConn().setLocalPort(newport);
69        getSession().getDataConn().setPassive();
70        // Init the connection
71        try {
72          if (getSession().getDataConn().initPassiveConnection()) {
73            isInit = true;
74            break;
75          }
76        } catch (final Reply425Exception e) {
77          logger.warn(
78              "Pasv refused at try: " + i + " with port:  since {}" + newport,
79              e.getMessage());
80        }
81      }
82      if (!isInit) {
83        throw new Reply425Exception("Passive mode not started");
84      }
85      // Return the address in Ftp format
86      final InetSocketAddress local =
87          getSession().getDataConn().getLocalAddress();
88      final int servPort = local.getPort();
89      String address = getSession().getConfiguration().getServerAddress();
90      if (address == null) {
91        address = local.getAddress().getHostAddress();
92      }
93      final String slocal = "Entering Passive Mode (" +
94                            FtpChannelUtils.getAddress(address, servPort) + ')';
95      final InetAddress remote =
96          getSession().getDataConn().getRemoteAddress().getAddress();
97      // Add the current FtpSession into the reference of session since the
98      // client will open the connection
99      getConfiguration().setNewFtpSession(remote, local, getSession());
100     // prepare the validation of the next connection
101     getSession().getDataConn().getFtpTransferControl()
102                 .resetWaitForOpenedDataChannel();
103     getSession()
104         .setReplyCode(ReplyCode.REPLY_227_ENTERING_PASSIVE_MODE, slocal);
105     logger.info("PASV: answer ready on {}", slocal);
106   }
107 
108 }
