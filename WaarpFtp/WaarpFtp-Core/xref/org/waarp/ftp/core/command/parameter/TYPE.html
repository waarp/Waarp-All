20  package org.waarp.ftp.core.command.parameter;
21  
22  import org.waarp.common.command.ReplyCode;
23  import org.waarp.common.command.exception.Reply501Exception;
24  import org.waarp.common.command.exception.Reply504Exception;
25  import org.waarp.common.exception.InvalidArgumentException;
26  import org.waarp.ftp.core.command.AbstractCommand;
27  import org.waarp.ftp.core.command.FtpArgumentCode;
28  import org.waarp.ftp.core.command.FtpArgumentCode.TransferSubType;
29  
30  /**
31   * TYPE command
32   */
33  public class TYPE extends AbstractCommand {
34    @Override
35    public void exec() throws Reply501Exception, Reply504Exception {
36      // First Check if any argument
37      if (!hasArg()) {
38        getSession().getDataConn().setType(FtpArgumentCode.TransferType.ASCII);
39        getSession().getDataConn().setSubType(TransferSubType.NONPRINT);
40        getSession().setReplyCode(ReplyCode.REPLY_200_COMMAND_OKAY,
41                                  "Type set to " +
42                                  FtpArgumentCode.TransferType.ASCII.name() +
43                                  ' ' + TransferSubType.NONPRINT);
44        return;
45      }
46      FtpArgumentCode.TransferType transferType;
47      final String[] types = getArgs();
48      try {
49        transferType = FtpArgumentCode.getTransferType(types[0].charAt(0));
50      } catch (final InvalidArgumentException e) {
51        throw new Reply501Exception("Unrecognize Type: " + getArg());
52      }
53      if (transferType == FtpArgumentCode.TransferType.ASCII) {
54        getSession().getDataConn().setType(FtpArgumentCode.TransferType.ASCII);
55      } else if (transferType == FtpArgumentCode.TransferType.IMAGE) {
56        getSession().getDataConn().setType(FtpArgumentCode.TransferType.IMAGE);
57      } else {
58        throw new Reply504Exception(
59            "Type not implemented: " + transferType.name());
60      }
61      // Look at the subtype or format control
62      if (types.length > 2) {
63        TransferSubType transferSubType;
64        for (int i = 1; i < types.length; i++) {
65          if (types[i].length() != 0) {
66            try {
67              transferSubType =
68                  FtpArgumentCode.getTransferSubType(types[i].charAt(0));
69            } catch (final InvalidArgumentException e) {
70              throw new Reply501Exception(
71                  "Unrecognize Format Control: " + types[i]);
72            }
73            if (transferSubType != TransferSubType.NONPRINT) {
74              throw new Reply504Exception(
75                  "Format Control not implemented: " + transferSubType.name());
76            }
77          }
78        }
79      }
80      getSession().getDataConn().setSubType(TransferSubType.NONPRINT);
81      getSession().setReplyCode(ReplyCode.REPLY_200_COMMAND_OKAY,
82                                "Type set to " + transferType.name() + ' ' +
83                                TransferSubType.NONPRINT);
84    }
85  
86  }
