package org.waarp.compress;

public class MalformedInputException extends RuntimeException {
  private static final long serialVersionUID = 2770594310659926623L;
  private final long offset;

  public MalformedInputException(final long offset) {
    this(offset, "Malformed input");
  }

  public MalformedInputException(final long offset, final String reason) {
    super(reason + ": offset=" + offset);
    this.offset = offset;
  }

  public MalformedInputException(final Exception e) {
    super(e);
    this.offset = 0;
  }

  public final long getOffset() {
    return offset;
  }
}
