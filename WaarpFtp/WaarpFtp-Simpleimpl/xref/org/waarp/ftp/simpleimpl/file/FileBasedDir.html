20  package org.waarp.ftp.simpleimpl.file;
21  
22  import org.waarp.common.command.exception.CommandAbstractException;
23  import org.waarp.ftp.core.file.FtpFile;
24  import org.waarp.ftp.core.session.FtpSession;
25  import org.waarp.ftp.filesystembased.FilesystemBasedFtpDir;
26  
27  /**
28   * FtpFile implementation based on true directories and files
29   */
30  public class FileBasedDir extends FilesystemBasedFtpDir {
31    /**
32     * @param session
33     */
34    public FileBasedDir(FtpSession session) {
35      super(session);
36    }
37  
38    @Override
39    public FtpFile newFile(String path, boolean append)
40        throws CommandAbstractException {
41      return new FileBasedFile((FtpSession) getSession(), this, path, append);
42    }
43  }
