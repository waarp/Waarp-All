package org.waarp.common.logging;

import java.util.logging.Level;
import java.util.logging.Logger;

/**
 * Logger factory which creates a
 * <a href="http://docs.oracle.com/javase/7/docs/technotes/guides/logging/">java.util.logging</a>
 * logger.
 */
public class WaarpJdkLoggerFactory extends WaarpLoggerFactory {

  /**
   * @param level
   */
  public WaarpJdkLoggerFactory(final WaarpLogLevel level) {
    super(level);
    seLevelSpecific(currentLevel);
  }

  @Override
  public WaarpLogger newInstance(final String name) {
    return new WaarpJdkLogger(Logger.getLogger(name));
  }

  @Override
  protected final void seLevelSpecific(final WaarpLogLevel level) {
    final Logger logger = Logger.getLogger(Logger.GLOBAL_LOGGER_NAME);
    switch (level) {
      case TRACE:
        logger.setLevel(Level.FINEST); //NOSONAR
        break;
      case DEBUG:
        logger.setLevel(Level.FINE); //NOSONAR
        break;
      case INFO:
        logger.setLevel(Level.INFO); //NOSONAR
        break;
      case ERROR:
        logger.setLevel(Level.SEVERE); //NOSONAR
        break;
      case WARN:
        logger.setLevel(Level.WARNING); //NOSONAR
        break;
      default:
        logger.setLevel(Level.OFF); //NOSONAR
        break;
    }
  }

  @Override
  protected final WaarpLogLevel getLevelSpecific() {
    final Logger logger = Logger.getLogger(Logger.GLOBAL_LOGGER_NAME);
    if (logger.isLoggable(Level.FINEST)) {
      return WaarpLogLevel.TRACE;
    } else if (logger.isLoggable(Level.FINE)) {
      return WaarpLogLevel.DEBUG;
    } else if (logger.isLoggable(Level.INFO)) {
      return WaarpLogLevel.INFO;
    } else if (logger.isLoggable(Level.WARNING)) {
      return WaarpLogLevel.WARN;
    } else if (logger.isLoggable(Level.SEVERE)) {
      return WaarpLogLevel.ERROR;
    }
    return WaarpLogLevel.NONE;
  }
}
