21  package org.waarp.ftp;
22  
23  import org.waarp.common.file.FileUtils;
24  import org.waarp.common.file.filesystembased.FilesystemBasedFileParameterImpl;
25  import org.waarp.common.logging.SysErrLogger;
26  import org.waarp.common.logging.WaarpLogger;
27  import org.waarp.common.logging.WaarpLoggerFactory;
28  import org.waarp.common.logging.WaarpSlf4JLoggerFactory;
29  import org.waarp.ftp.core.exception.FtpNoConnectionException;
30  import org.waarp.ftp.simpleimpl.SimpleGatewayFtpServer;
31  import org.waarp.ftp.simpleimpl.config.FileBasedConfiguration;
32  import org.waarp.ftp.simpleimpl.config.FileBasedSslConfiguration;
33  import org.waarp.ftp.simpleimpl.control.SimpleBusinessHandler;
34  import org.waarp.ftp.simpleimpl.data.FileSystemBasedDataBusinessHandler;
35  
36  import java.io.File;
37  
38  public class FtpServer {
39    /**
40     * Internal Logger
41     */
42    protected static WaarpLogger logger;
43    protected static File dir;
44    protected static FileBasedConfiguration configuration;
45  
46    public static void startFtpServer(String config, String sslconfig,
47                                      boolean useSsl, boolean useNative) {
48      WaarpLoggerFactory.setDefaultFactoryIfNotSame(
49          new WaarpSlf4JLoggerFactory(null));
50      if (logger == null) {
51        logger = WaarpLoggerFactory.getLogger(FtpServer.class);
52      }
53      final ClassLoader classLoader = FtpServer.class.getClassLoader();
54      final File file = new File(classLoader.getResource(config).getFile());
55      if (file.exists()) {
56        dir = file.getParentFile();
57        final File base = new File("/tmp/GGFTP");
58        base.mkdirs();
59        FileUtils.forceDeleteRecursiveDir(base);
60        base.mkdirs();
61        configuration = new FileBasedConfiguration(SimpleGatewayFtpServer.class,
62                                                   SimpleBusinessHandler.class,
63                                                   FileSystemBasedDataBusinessHandler.class,
64                                                   new FilesystemBasedFileParameterImpl());
65        if (!configuration.setConfigurationFromXml(file.getAbsolutePath())) {
66          logger.error("Bad configuration");
67          return;
68        }
69        configuration.setTimeoutCon(1000);
70        if (useSsl) {
71          if (!FileBasedSslConfiguration.setConfigurationServerFromXml(
72              configuration, sslconfig)) {
73            SysErrLogger.FAKE_LOGGER.syserr("Bad Ssl configuration");
74            return;
75          }
76          if (useNative) {
77            // native SSL support
78            configuration.getFtpInternalConfiguration().setUsingNativeSsl(true);
79            configuration.getFtpInternalConfiguration().setAcceptAuthProt(false);
80          } else {
81            // AUTH, PROT, ... support
82            configuration.getFtpInternalConfiguration().setUsingNativeSsl(false);
83            configuration.getFtpInternalConfiguration().setAcceptAuthProt(true);
84          }
85        }
86        // Start server.
87        try {
88          configuration.serverStartup();
89        } catch (final FtpNoConnectionException e) {
90          logger.error("FTP not started", e);
91        }
92        logger.info("FTP started");
93      }
94    }
95  
96    public static void stopFtpServer() {
97      configuration.setShutdown(true);
98      configuration.getFtpInternalConfiguration().releaseResources();
99    }
100 }
