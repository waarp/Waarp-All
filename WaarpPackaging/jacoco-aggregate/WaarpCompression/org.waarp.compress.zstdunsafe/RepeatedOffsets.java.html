package org.waarp.compress.zstdunsafe;

class RepeatedOffsets {
  private int offset0 = 1;
  private int offset1 = 4;

  private int tempOffset0;
  private int tempOffset1;

  public int getOffset0() {
    return offset0;
  }

  public int getOffset1() {
    return offset1;
  }

  public void saveOffset0(final int offset) {
    tempOffset0 = offset;
  }

  public void saveOffset1(final int offset) {
    tempOffset1 = offset;
  }

  public void commit() {
    offset0 = tempOffset0;
    offset1 = tempOffset1;
  }
}
