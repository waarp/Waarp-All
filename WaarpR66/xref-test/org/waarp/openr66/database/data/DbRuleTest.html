40  public class DbRuleTest {
41      @Test
42      public void testJsonSerialisation() {
43          List<String> hosts = Arrays.asList("sup1");
44  
45          List<RuleTask> spretasks = new ArrayList<RuleTask>();
46          spretasks.add(new RuleTask("LOG", "spretasks", 0));
47  
48          List<RuleTask> sposttasks = new ArrayList<RuleTask>();
49          sposttasks.add(new RuleTask("LOG", "sposttasks", 0));
50  
51          List<RuleTask> serrortasks = new ArrayList<RuleTask>();
52          serrortasks.add(new RuleTask("LOG", "serrortasks", 0));
53  
54          List<RuleTask> rpretasks = new ArrayList<RuleTask>();
55          rpretasks.add(new RuleTask("LOG", "rpretasks", 0));
56  
57          List<RuleTask> rposttasks = new ArrayList<RuleTask>();
58          rposttasks.add(new RuleTask("LOG", "rposttasks", 0));
59  
60          List<RuleTask> rerrortasks = new ArrayList<RuleTask>();
61          rerrortasks.add(new RuleTask("LOG", "rerrortasks", 0));
62  
63  		Rule tested = new Rule("rulename", 3, hosts, "recv-path", "send-path",
64              "arch-path", "work-path", rpretasks, rposttasks, rerrortasks,
65              spretasks, sposttasks, serrortasks, UpdatedInfo.TOSUBMIT);
66          DbRule dbRule = new DbRule(tested);
67  
68          String expected = "{" +
69              "\"@model\":\"DbRule\"," +
70              "\"IDRULE\":\"rulename\"," +
71              "\"MODETRANS\":3," +
72              "\"RECVPATH\":\"/recv-path\"," +
73              "\"SENDPATH\":\"/send-path\"," +
74              "\"ARCHIVEPATH\":\"/arch-path\"," +
75              "\"WORKPATH\":\"/work-path\"," +
76              "\"UPDATEDINFO\":3," +
77              "\"HOSTIDS\":\"<hostids><hostid>sup1</hostid></hostids>\"," +
78              "\"SPRETASKS\":\"<tasks><task><type>LOG</type><path>spretasks</path><delay>0</delay></task></tasks>\"," +
79              "\"SERRORTASKS\":\"<tasks><task><type>LOG</type><path>serrortasks</path><delay>0</delay></task></tasks>\"," +
80              "\"RPRETASKS\":\"<tasks><task><type>LOG</type><path>rpretasks</path><delay>0</delay></task></tasks>\"," +
81              "\"RPOSTTASKS\":\"<tasks><task><type>LOG</type><path>rposttasks</path><delay>0</delay></task></tasks>\"," +
82              "\"RERRORTASKS\":\"<tasks><task><type>LOG</type><path>rerrortasks</path><delay>0</delay></task></tasks>\"," +
83              "\"SPOSTTASKS\":\"<tasks><task><type>LOG</type><path>sposttasks</path><delay>0</delay></task></tasks>\"" +
84              "}";
85  
86  
87          ObjectNode asJson = dbRule.getJson();
88          String got = JsonHandler.writeAsString(asJson);
89  
90          assertEquals(expected, got);
91      }
