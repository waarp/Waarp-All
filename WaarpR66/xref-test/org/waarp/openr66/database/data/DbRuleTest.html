24  import com.fasterxml.jackson.databind.node.ObjectNode;
25  import org.junit.Test;
26  import org.junit.rules.TestWatcher;
27  import org.waarp.common.database.exception.WaarpDatabaseSqlException;
28  import org.waarp.common.json.JsonHandler;
29  import org.waarp.common.utility.TestWatcherJunit4;
30  import org.waarp.openr66.pojo.Rule;
31  import org.waarp.openr66.pojo.RuleTask;
32  import org.waarp.openr66.pojo.UpdatedInfo;
33  
34  import java.util.ArrayList;
35  import java.util.Arrays;
36  import java.util.List;
37  
38  import static org.junit.Assert.*;
39  
40  public class DbRuleTest {
41    @org.junit.Rule(order = Integer.MIN_VALUE)
42    public TestWatcher watchman = new TestWatcherJunit4();
43  
44    @Test
45    public void testJsonSerialisation() throws WaarpDatabaseSqlException {
46      List<String> hosts = Arrays.asList("sup1");
47  
48      List<RuleTask> spretasks = new ArrayList<RuleTask>();
49      spretasks.add(new RuleTask("LOG", "spretasks", 0));
50  
51      List<RuleTask> sposttasks = new ArrayList<RuleTask>();
52      sposttasks.add(new RuleTask("LOG", "sposttasks", 0));
53  
54      List<RuleTask> serrortasks = new ArrayList<RuleTask>();
55      serrortasks.add(new RuleTask("LOG", "serrortasks", 0));
56  
57      List<RuleTask> rpretasks = new ArrayList<RuleTask>();
58      rpretasks.add(new RuleTask("LOG", "rpretasks", 0));
59  
60      List<RuleTask> rposttasks = new ArrayList<RuleTask>();
61      rposttasks.add(new RuleTask("LOG", "rposttasks", 0));
62  
63      List<RuleTask> rerrortasks = new ArrayList<RuleTask>();
64      rerrortasks.add(new RuleTask("LOG", "rerrortasks", 0));
65  
66      Rule tested =
67          new Rule("rulename", 3, hosts, "recv-path", "send-path", "arch-path",
68                   "work-path", rpretasks, rposttasks, rerrortasks, spretasks,
69                   sposttasks, serrortasks, UpdatedInfo.TOSUBMIT);
70      DbRule dbRule = new DbRule(tested);
71  
72      String expected =
73          "{" + "\"@model\":\"DbRule\"," + "\"IDRULE\":\"rulename\"," +
74          "\"MODETRANS\":3," + "\"RECVPATH\":\"/recv-path\"," +
75          "\"SENDPATH\":\"/send-path\"," + "\"ARCHIVEPATH\":\"/arch-path\"," +
76          "\"WORKPATH\":\"/work-path\"," + "\"UPDATEDINFO\":3," +
77          "\"HOSTIDS\":\"<hostids><hostid>sup1</hostid></hostids>\"," +
78          "\"SPRETASKS\":\"<tasks><task><type>LOG</type><path>spretasks</path><delay>0</delay></task></tasks>\"," +
79          "\"SERRORTASKS\":\"<tasks><task><type>LOG</type><path>serrortasks</path><delay>0</delay></task></tasks>\"," +
80          "\"RPRETASKS\":\"<tasks><task><type>LOG</type><path>rpretasks</path><delay>0</delay></task></tasks>\"," +
81          "\"RPOSTTASKS\":\"<tasks><task><type>LOG</type><path>rposttasks</path><delay>0</delay></task></tasks>\"," +
82          "\"RERRORTASKS\":\"<tasks><task><type>LOG</type><path>rerrortasks</path><delay>0</delay></task></tasks>\"," +
83          "\"SPOSTTASKS\":\"<tasks><task><type>LOG</type><path>sposttasks</path><delay>0</delay></task></tasks>\"" +
84          "}";
85  
86  
87      ObjectNode asJson = dbRule.getJson();
88      String got = JsonHandler.writeAsString(asJson);
89  
90      assertEquals(expected, got);
91    }
92  }
