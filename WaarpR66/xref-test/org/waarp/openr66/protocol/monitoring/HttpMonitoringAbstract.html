86    @AfterClass
87    public static void tearDownAfterClass() throws Exception {
88      WaarpSystemUtil.stopLogger(true);
89      Thread.sleep(100);
90      for (final DbTaskRunner dbTaskRunner : dbTaskRunners) {
91        try {
92          dbTaskRunner.delete();
93        } catch (final WaarpDatabaseException e) {
94          logger.warn("Cannot apply nolog to " + dbTaskRunner + " : {}",
95                      e.getMessage());
96        }
97      }
98      final DbHostAuth host = new DbHostAuth("hostas");
99      final SocketAddress socketServerAddress;
100     try {
101       socketServerAddress = host.getSocketAddress();
102     } catch (final IllegalArgumentException e) {
103       logger.error("Needs a correct configuration file as first argument");
104       return;
105     }
106     final byte scode = -1;
107 
108     // Shutdown server
109     logger.warn("Shutdown Server");
110     Configuration.configuration.setTimeoutCon(100);
111     final R66Future future = new R66Future(true);
112     final ShutdownOrBlockJsonPacket node8 = new ShutdownOrBlockJsonPacket();
113     node8.setRestartOrBlock(false);
114     node8.setShutdownOrBlock(true);
115     node8.setKey(FilesystemBasedDigest.passwdCrypt(
116         Configuration.configuration.getServerAdminKey()));
117     final AbstractLocalPacket valid =
118         new JsonCommandPacket(node8, LocalPacketFactory.BLOCKREQUESTPACKET);
119     sendInformation(valid, socketServerAddress, future, scode, false,
120                     R66FiniteDualStates.SHUTDOWN, true);
121     Thread.sleep(200);
122 
123     // Stop monitoring shall be implicit
124     tearDownAfterClassClient();
125     tearDownAfterClassMinimal();
126     tearDownAfterClassServer();
127     // Stop Fake REST Server
128     HttpServerExample.group.close().awaitUninterruptibly();
129   }

140     final File totest = generateOutFile("/tmp/R66/out/testTask.txt", 10);
141     final ExecutorService executorService = Executors.newCachedThreadPool();
142     final int nb = nbThread;
143     final R66Future[] arrayFuture = new R66Future[nb];
144     logger.warn("Start Test of DirectTransfer");
145     final long time1 = System.currentTimeMillis();
146     for (int i = 0; i < nb; i++) {
147       arrayFuture[i] = new R66Future(true);
148       final TestTransferNoDb transaction =

164         error++;
165       }

167     final long time2 = System.currentTimeMillis();

170     executorService.shutdown();
171     totest.delete();
172     numberTransfer += nb;
173     assertEquals("Success should be total", nb, success);
174     assertEquals("Errors should be 0", 0, error);
175   }

177   private void waitForAllDone(DbTaskRunner runner) {
178     while (true) {
179       try {
180         runner.select();
181         if (runner.isAllDone()) {
182           logger.warn("DbTaskRunner done");
183           return;
184         } else if (runner.isInError()) {
185           logger.error("DbTaskRunner in error");
186           return;
187         }
188         Thread.sleep(100);
189       } catch (InterruptedException e) {//NOSONAR
190         logger.error("Interrupted", e);
191         return;
192       } catch (WaarpDatabaseException e) {
193         logger.error("Cannot found DbTaskRunner: {}", e.getMessage());
194         return;
195       }
196     }
197   }

200   public void test5_SubmitTransfer() throws Exception {
201     final File totest = generateOutFile("/tmp/R66/out/testTask.txt", 10);
202     final int nb = nbThread;
203     final R66Future[] arrayFuture = new R66Future[nb];
204     logger.warn("Start Test of SubmitTransfer");
205 
206     for (int i = 0; i < nb; i++) {
207       arrayFuture[i] = new R66Future(true);
208       final Timestamp newstart =
209           new Timestamp(System.currentTimeMillis() + i * 10);
210       final SubmitTransfer transaction =
211           new SubmitTransfer(arrayFuture[i], "hostas", "testTask.txt", "rule3",
212                              "Test Send Submit Small", true, 8192,
213                              DbConstantR66.ILLEGALVALUE, newstart);
214       transaction.run();
215     }
216     int success = 0;
217     int error = 0;
218     for (int i = 0; i < nb; i++) {
219       arrayFuture[i].awaitOrInterruptible();
220       if (arrayFuture[i].isSuccess()) {
221         success++;
222       } else {
223         error++;
224       }
225     }
226     logger.warn("Prepare transfer Success: " + success + " Error: " + error);
227     assertEquals("Success should be total", nb, success);
228     assertEquals("Errors should be 0", 0, error);
229     // Now wait for all transfers done
230     for (int i = 0; i < nb; i++) {
231       if (arrayFuture[i].isSuccess()) {
232         final DbTaskRunner runner = arrayFuture[i].getResult().getRunner();
233         if (runner != null) {
234           waitForAllDone(runner);
235           dbTaskRunners.add(runner);
236         }
237       }
238     }
239     numberTransfer += nb;
240     Thread.sleep(100);
241     totest.delete();
242   }

303   private void test_Spooled(final SpooledThread spooledThread)
304       throws IOException, InterruptedException {
305     Assume.assumeTrue(spooledThread.spooledDirectoryTransfer != null);
306     final int size = 200;
307     Configuration.configuration.changeNetworkLimit(0, 0, 0, 0, 1000);
308     File directory = new File(SpooledThread.TMP_R_66_TEST_OUT_EXAMPLE);
309     directory.mkdirs();
310     File stop = new File(SpooledThread.TMP_R_66_TEST_STOPOUT_TXT);
311     stop.delete();
312     final R66Future future = new R66Future(true);
313     spooledThread.future = future;
314     spooledThread.start();
315     Thread.sleep(200);
316     int cpt = 0;
317     logger.warn("1rst file");
318     final File totestBig = generateOutFile(
319         SpooledThread.TMP_R_66_TEST_OUT_EXAMPLE + "/testTaskBig.txt", size);
320     cpt++;
321     for (int i = 0; i < 100; i++) {
322       if (spooledThread.spooledDirectoryTransfer.getSent() != cpt) {
323         Thread.sleep(200);
324       } else {
325         break;
326       }
327     }
328     logger.warn("1rst file delete");
329     totestBig.delete();
330     Thread.sleep(2000);
331     logger.warn("Second file");
332     generateOutFile(
333         SpooledThread.TMP_R_66_TEST_OUT_EXAMPLE + "/testTaskBig.txt", size);
334     cpt++;
335     for (int i = 0; i < 100; i++) {
336       if (spooledThread.spooledDirectoryTransfer.getSent() != cpt) {
337         Thread.sleep(200);
338       } else {
339         break;
340       }
341     }
342     logger.warn("Third file modifying second file");
343     generateOutFile(
344         SpooledThread.TMP_R_66_TEST_OUT_EXAMPLE + "/testTaskBig.txt", size + 1,
345         "abcdefghij");
346     if (spooledThread.ignoreAlreadyUsed) {
347       Thread.sleep(2000);
348     } else {
349       cpt++;
350       for (int i = 0; i < 100; i++) {
351         if (spooledThread.spooledDirectoryTransfer.getSent() != cpt) {
352           Thread.sleep(200);
353         } else {
354           break;
355         }
356       }
357     }
358     logger.warn("Third file deleted");
359     totestBig.delete();
360     Thread.sleep(1000);
361     generateOutFile(stop.getAbsolutePath(), 10);
362     future.awaitOrInterruptible();
363     assertTrue(future.isSuccess());
364     stop.delete();
365     File all = new File(SpooledThread.SPOOLED_ROOT);
366     FileUtils.forceDeleteRecursiveDir(all);
367     logger.warn("Launched {}",
368                 spooledThread.spooledDirectoryTransfer.getSent());
369     logger.warn("Error {}", spooledThread.spooledDirectoryTransfer.getError());
370     numberTransfer += 3;
371   }

473   private static class SpooledThread extends Thread {
474     private static final String TMP_R_66_TEST_OUT_EXAMPLE =
475         "/tmp/R66/test/out/example";
476     private static final String TMP_R_66_TEST_STOPOUT_TXT =
477         "/tmp/R66/test/stopout.txt";
478     private static final String SPOOLED_ROOT = "/tmp/R66/test";
479     R66Future future;
480     private boolean ignoreAlreadyUsed = false;
481     private boolean submit = false;
482     private String host = "hostas";
483     private final String waarpHost = "hostas";
484     private SpooledDirectoryTransfer spooledDirectoryTransfer;
485 
486     @Override
487     public void run() {
488       Arguments arguments = new Arguments();
489       arguments.setName("SpooledClient");
490       arguments.getLocalDirectory().add(TMP_R_66_TEST_OUT_EXAMPLE);
491       arguments.setStatusFile("/tmp/R66/test/statusoutdirect1.json");
492       arguments.setStopFile(TMP_R_66_TEST_STOPOUT_TXT);
493       arguments.setRule("rule3del");
494       arguments.setFileInfo("fileInfo");
495       arguments.setMd5(true);
496       arguments.getRemoteHosts().add(host);
497       if (waarpHost != null) {
498         arguments.getWaarpHosts().add(waarpHost);
499       }
500       arguments.setBlock(5000);
501       arguments.setRegex(null);
502       arguments.setElapsed(100);
503       arguments.setToSubmit(submit);
504       arguments.setNoLog(false);
505       arguments.setRecursive(false);
506       arguments.setElapsedWaarp(100);
507       arguments.setParallel(false);
508       arguments.setLimitParallel(1);
509       arguments.setMinimalSize(100);
510       arguments.setLogWarn(false);
511       arguments.setIgnoreAlreadyUsed(ignoreAlreadyUsed);
512 
513       spooledDirectoryTransfer =
514           new SpooledDirectoryTransfer(future, arguments, networkTransaction);
515       spooledDirectoryTransfer.run();
516     }
517   }

519   private void checkFinalResult(R66Future future, R66Result result, long delay,
520                                 long size) {
521     if (future.isSuccess()) {
522       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
523         logger.warn("Warning with Id: " + result.getRunner().getSpecialId() +
524                     " on file: " +
525                     (result.getFile() != null? result.getFile().toString() :
526                         "no file") + " delay: " + delay + " kbps: " +
527                     size * 8 / delay);
528       } else {
529         logger.warn("Success with Id: " + result.getRunner().getSpecialId() +
530                     " on Final file: " +
531                     (result.getFile() != null? result.getFile().toString() :
532                         "no file") + " delay: " + delay + " kbps: " +
533                     size * 8 / delay);
534       }
535       // In case of success, delete the runner
536       dbTaskRunners.add(result.getRunner());
537     } else {
538       if (result == null || result.getRunner() == null) {
539         logger.warn("Transfer in Error with no Id" + " : {}",
540                     future.getCause() != null? future.getCause().getMessage() :
541                         "");
542         assertTrue("Result should not be null, neither runner", false);
543       }
544       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
545         logger.warn("Transfer in Warning with Id: " +
546                     result.getRunner().getSpecialId() + " : {}",
547                     future.getCause() != null? future.getCause().getMessage() :
548                         "");
549         assertTrue("Transfer in Warning", false);
550       } else {
551         logger.error(
552             "Transfer in Error with Id: " + result.getRunner().getSpecialId() +
553             " : {}",
554             future.getCause() != null? future.getCause().getMessage() : "");
555         assertTrue("Transfer in Error", false);
556       }
557     }
558   }
