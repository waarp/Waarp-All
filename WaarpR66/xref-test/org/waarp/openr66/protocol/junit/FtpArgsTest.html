21  package org.waarp.openr66.protocol.junit;
22  
23  import org.junit.Rule;
24  import org.junit.Test;
25  import org.junit.rules.TestWatcher;
26  import org.waarp.common.digest.FilesystemBasedDigest.DigestAlgo;
27  import org.waarp.common.utility.TestWatcherJunit4;
28  import org.waarp.openr66.context.task.FtpArgs;
29  import org.waarp.openr66.context.task.exception.OpenR66RunnerErrorException;
30  
31  import static org.junit.Assert.*;
32  
33  /**
34   * FtpArgs Tester.
35   */
36  public class FtpArgsTest {
37    @Rule(order = Integer.MIN_VALUE)
38    public TestWatcher watchman = new TestWatcherJunit4();
39  
40    /**
41     * Method: getFtpArgs(String[] args)
42     */
43    @Test
44    public void testGetFtpArgs() throws Exception {
45      String[] args0 = {
46          "-file", "filesource", "-to", "host", "-port", "a1234"
47      };
48      FtpArgs ftpArgs;
49      try {
50        ftpArgs = FtpArgs.getFtpArgs(args0);
51        fail("Should raised an exception");
52      } catch (OpenR66RunnerErrorException e) {
53        //ignore
54      }
55      String[] args01 = {
56          "-file", "filesource", "-to", "host", "-port", "a1234", "-user",
57          "userftp", "-pwd", "userpwd"
58      };
59      try {
60        ftpArgs = FtpArgs.getFtpArgs(args01);
61        fail("Should raised an exception");
62      } catch (OpenR66RunnerErrorException e) {
63        //ignore
64      }
65      String[] args02 = {
66          "-file", "filesource", "-to", "host", "-port", "1234", "-user",
67          "userftp", "-pwd", "userpwd", "-command", "fake"
68      };
69      try {
70        ftpArgs = FtpArgs.getFtpArgs(args02);
71        fail("Should raised an exception");
72      } catch (OpenR66RunnerErrorException e) {
73        //ignore
74      }
75      String[] args = {
76          "-file", "filesource", "-to", "host", "-port", "1234", "-user",
77          "userftp", "-pwd", "userpwd", "-account", "accountftp", "-mode",
78          "active", "-ssl", "no", "-cwd", "remotepath", "-digest", "sha256",
79          "-pre", "torun,new,arg", "-command", "get", "-post",
80          "torunafter,new2,arg2"
81      };
82      ftpArgs = FtpArgs.getFtpArgs(args);
83      assertEquals(ftpArgs.getFilepath(), "filesource");
84      assertEquals(ftpArgs.getRequested(), "host");
85      assertEquals(ftpArgs.getUser(), "userftp");
86      assertEquals(ftpArgs.getPwd(), "userpwd");
87      assertEquals(ftpArgs.getAcct(), "accountftp");
88      assertEquals(ftpArgs.getCwd(), "remotepath");
89      assertEquals(ftpArgs.getPreArgs(), "torun new arg");
90      assertEquals(ftpArgs.getPostArgs(), "torunafter new2 arg2");
91      assertEquals(ftpArgs.getCommand(), "get");
92      assertEquals(ftpArgs.getDigest(), DigestAlgo.SHA256);
93      assertEquals(ftpArgs.getPort(), 1234);
94      assertEquals(ftpArgs.getSsl(), 0);
95      assertEquals(ftpArgs.getCodeCommand(), -1);
96      assertTrue(!ftpArgs.isPassive());
97      String[] args2 = {
98          "-file", "filesource", "-to", "host", "-port", "1234", "-user",
99          "userftp", "-pwd", "userpwd", "-account", "accountftp", "-mode",
100         "passive", "-ssl", "implicit", "-cwd", "remotepath", "-digest",
101         "sha512", "-pre", "torun,new,arg", "-command", "put", "-post",
102         "torunafter,new2,arg2"
103     };
104     ftpArgs = FtpArgs.getFtpArgs(args2);
105     assertEquals(ftpArgs.getFilepath(), "filesource");
106     assertEquals(ftpArgs.getRequested(), "host");
107     assertEquals(ftpArgs.getUser(), "userftp");
108     assertEquals(ftpArgs.getPwd(), "userpwd");
109     assertEquals(ftpArgs.getAcct(), "accountftp");
110     assertEquals(ftpArgs.getCwd(), "remotepath");
111     assertEquals(ftpArgs.getPreArgs(), "torun new arg");
112     assertEquals(ftpArgs.getPostArgs(), "torunafter new2 arg2");
113     assertEquals(ftpArgs.getCommand(), "put");
114     assertEquals(ftpArgs.getDigest(), DigestAlgo.SHA512);
115     assertEquals(ftpArgs.getPort(), 1234);
116     assertEquals(ftpArgs.getSsl(), -1);
117     assertEquals(ftpArgs.getCodeCommand(), 1);
118     assertTrue(ftpArgs.isPassive());
119     String[] args3 = {
120         "-file", "filesource", "-to", "host", "-port", "1234", "-user",
121         "userftp", "-pwd", "userpwd", "-command", "append", "-ssl", "explicit"
122     };
123     ftpArgs = FtpArgs.getFtpArgs(args3);
124     assertEquals(ftpArgs.getFilepath(), "filesource");
125     assertEquals(ftpArgs.getRequested(), "host");
126     assertEquals(ftpArgs.getUser(), "userftp");
127     assertEquals(ftpArgs.getPwd(), "userpwd");
128     assertEquals(ftpArgs.getAcct(), null);
129     assertEquals(ftpArgs.getCwd(), null);
130     assertEquals(ftpArgs.getPreArgs(), null);
131     assertEquals(ftpArgs.getPostArgs(), null);
132     assertEquals(ftpArgs.getCommand(), "append");
133     assertEquals(ftpArgs.getDigest(), null);
134     assertEquals(ftpArgs.getPort(), 1234);
135     assertEquals(ftpArgs.getSsl(), 1);
136     assertEquals(ftpArgs.getCodeCommand(), 2);
137     assertTrue(ftpArgs.isPassive());
138 
139   }
