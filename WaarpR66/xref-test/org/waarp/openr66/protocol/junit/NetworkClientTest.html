307   private void checkFinalResult(R66Future future, R66Result result, long delay,
308                                 long size) {
309     if (future.isSuccess()) {
310       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
311         logger.warn("Warning with Id: " + result.getRunner().getSpecialId() +
312                     " on file: " +
313                     (result.getFile() != null? result.getFile().toString() :
314                         "no file") + " delay: " + delay + " kbps: " +
315                     size * 8 / delay);
316       } else {
317         logger.warn("Success with Id: " + result.getRunner().getSpecialId() +
318                     " on Final file: " +
319                     (result.getFile() != null? result.getFile().toString() :
320                         "no file") + " delay: " + delay + " kbps: " +
321                     size * 8 / delay);
322       }
323       if (result.getRunner().shallIgnoreSave()) {
324         // In case of success, delete the runner
325         dbTaskRunners.add(result.getRunner());
326       }
327     } else {
328       if (result == null || result.getRunner() == null) {
329         logger.warn("Transfer in Error with no Id", future.getCause());
330         assertTrue("Result should not be null, neither runner", false);
331       }
332       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
333         logger.warn(
334             "Transfer in Warning with Id: " + result.getRunner().getSpecialId(),
335             future.getCause());
336         assertTrue("Transfer in Warning", false);
337       } else {
338         logger.error(
339             "Transfer in Error with Id: " + result.getRunner().getSpecialId(),
340             future.getCause());
341         assertTrue("Transfer in Error", false);
342       }
343     }
344   }
