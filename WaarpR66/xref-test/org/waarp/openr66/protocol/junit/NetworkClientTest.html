342   private void checkFinalResult(R66Future future, R66Result result, long delay,
343                                 long size) {
344     if (future.isSuccess()) {
345       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
346         logger.warn("Warning with Id: " + result.getRunner().getSpecialId() +
347                     " on file: " +
348                     (result.getFile() != null? result.getFile().toString() :
349                         "no file") + " delay: " + delay + " kbps: " +
350                     size * 8 / delay);
351       } else {
352         logger.warn("Success with Id: " + result.getRunner().getSpecialId() +
353                     " on Final file: " +
354                     (result.getFile() != null? result.getFile().toString() :
355                         "no file") + " delay: " + delay + " kbps: " +
356                     size * 8 / delay);
357       }
358       if (result.getRunner().shallIgnoreSave()) {
359         // In case of success, delete the runner
360         dbTaskRunners.add(result.getRunner());
361       }
362     } else {
363       if (result == null || result.getRunner() == null) {
364         logger.warn("Transfer in Error with no Id", future.getCause());
365         assertTrue("Result should not be null, neither runner", false);
366       }
367       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
368         logger.warn(
369             "Transfer in Warning with Id: " + result.getRunner().getSpecialId(),
370             future.getCause());
371         assertTrue("Transfer in Warning", false);
372       } else {
373         logger.error(
374             "Transfer in Error with Id: " + result.getRunner().getSpecialId(),
375             future.getCause());
376         assertTrue("Transfer in Error", false);
377       }
378     }
379   }
