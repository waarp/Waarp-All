126   private static void get(final String url) throws NoSuchElementException {
127     Exception error = null;
128     for (int i = 0; i < RETRY; i++) {
129       try {
130         TestWebAbstract.driver.get(url);
131         return;
132       } catch (final Exception e) {
133         error = e;
134       }
135     }
136     throw new NoSuchElementException("Get Error", error);
137   }
138 
139   private static boolean currentUrlEquals(final String url) {
140     for (int i = 0; i < RETRY; i++) {
141       try {
142         return TestWebAbstract.driver.getCurrentUrl().equals(url);
143       } catch (final Exception e) {
144         // ignore
145       }
146     }
147     return false;
148   }
149 
150   private static WebElement findElement(final By by)
151       throws NoSuchElementException {
152     Exception error = null;
153     for (int i = 0; i < RETRY; i++) {
154       try {
155         WebElement element = TestWebAbstract.driver.findElement(by);
156         return element;
157       } catch (final Exception e) {
158         error = e;
159       }
160     }
161     throw new NoSuchElementException("Find Element Error", error);
162   }
