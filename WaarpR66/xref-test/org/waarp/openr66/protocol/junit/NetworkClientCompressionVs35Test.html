127   public static int startServer(String serverConfig) throws Exception {
128     final File file2;
129     if (serverConfig.charAt(0) == '/') {

134     if (file2.exists()) {
135       System.err.println("Find server file: " + file2.getAbsolutePath());
136       final String[] argsServer = {
137           file2.getAbsolutePath()
138       };
139       // global ant project settings
140       // First download previous version
141       URL url = new URL(URL_3_5_2_WAARP_R66);
142       InputStream stream = url.openStream();
143       File jar = new File(JAR352);
144       FileOutputStream outputStream = new FileOutputStream(jar);
145       FileUtils.copy(stream, outputStream);
146       if (!jar.exists()) {
147         logger.error("CANNOT FIND version 3.5.2");
148         fail("CANNOT FIND version 3.5.2");
149         return 9999;
150       }
151       project = Processes.getProject(homeDir);
152       Processes.executeJvmSpecificClasspath(project, jar, R66Server.class,
153                                             argsServer, true);
154       int pid = Processes.getPidOfRunnerCommandLinux("java",
155                                                      R66Server.class.getName(),
156                                                      PIDS);
157       PIDS.add(pid);
158       logger.warn("Start Done: {}", pid);
159       return pid;
160     } else {
161       System.err.println("Cannot find server file: " + file2.getAbsolutePath());
162       fail("Cannot find server file");
163       return 999999;
164     }

236   private boolean isCheckCompressionOk(final DbTaskRunner taskRunner) {
237     if (taskRunner == null) {
238       return true;
239     }
240     try {
241       taskRunner.select();
242       return SHOULD_COMPRESS == taskRunner.isBlockCompression();
243     } catch (final WaarpDatabaseException e) {
244       logger.error(e.getMessage());
245       return true;
246     }
247   }

249   private void checkCompression(final DbTaskRunner taskRunner) {
250     if (taskRunner == null) {
251       return;
252     }
253     try {
254       taskRunner.select();
255       assertEquals(SHOULD_COMPRESS, taskRunner.isBlockCompression());
256     } catch (final WaarpDatabaseException e) {
257       fail(e.getMessage());
258     }

377   private void checkFinalResult(R66Future future, R66Result result, long delay,
378                                 long size) {
379     if (future.isSuccess()) {
380       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
381         logger.warn("Warning with Id: " + result.getRunner().getSpecialId() +
382                     " on file: " +
383                     (result.getFile() != null? result.getFile().toString() :
384                         "no file") + " delay: " + delay + " kbps: " +
385                     size * 8 / delay);
386       } else {
387         logger.warn("Success with Id: " + result.getRunner().getSpecialId() +
388                     " on Final file: " +
389                     (result.getFile() != null? result.getFile().toString() :
390                         "no file") + " delay: " + delay + " kbps: " +
391                     size * 8 / delay);
392       }
393       // Check is compression correct
394       checkCompression(result.getRunner());
395       // In case of success, delete the runner
396       dbTaskRunners.add(result.getRunner());
397     } else {
398       if (result == null || result.getRunner() == null) {
399         logger.warn("Transfer in Error with no Id", future.getCause());
400         assertTrue("Result should not be null, neither runner", false);
401       }
402       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
403         logger.warn(
404             "Transfer in Warning with Id: " + result.getRunner().getSpecialId(),
405             future.getCause());
406         assertTrue("Transfer in Warning", false);
407       } else {
408         logger.error(
409             "Transfer in Error with Id: " + result.getRunner().getSpecialId(),
410             future.getCause());
411         assertTrue("Transfer in Error", false);
412       }
413     }
414   }

716   private void waitForAllDone(DbTaskRunner runner) {
717     while (true) {
718       try {
719         runner.select();
720         if (runner.isAllDone()) {
721           logger.info("DbTaskRunner done");
722           if (!isCheckCompressionOk(runner)) {
723             logger.error("DbTaskRunner in error for compression: {}", runner);
724             fail("Compression shall be " + SHOULD_COMPRESS + " but is " +
725                  runner.isBlockCompression());
726           }
727           return;
728         } else if (runner.isInError()) {
729           logger.error("DbTaskRunner in error");
730           return;
731         }
732         Thread.sleep(100);
733       } catch (InterruptedException e) {//NOSONAR
734         logger.error("Interrupted", e);
735         return;
736       } catch (WaarpDatabaseException e) {
737         logger.error("Cannot found DbTaskRunner", e);
738         return;
739       }
