117   protected static void sendInformation(AbstractLocalPacket informationPacket,
118                                         final SocketAddress socketServerAddress,
119                                         R66Future future, byte scode,
120                                         boolean check,
121                                         R66FiniteDualStates state,
122                                         boolean isSsl)
123       throws OpenR66ProtocolPacketException {
124     logger.warn("Start connection for Extra commands {} \n\t{}",
125                 informationPacket.getClass().getSimpleName(),
126                 informationPacket);
127     final LocalChannelReference localChannelReference = networkTransaction
128         .createConnectionWithRetry(socketServerAddress, isSsl, future);
129     if (localChannelReference == null) {
130       if (state != R66FiniteDualStates.SHUTDOWN) {
131         fail("Connection not OK");
132       } else {
133         return;
134       }
135     }
136     localChannelReference.sessionNewState(state);
137     logger.warn("Send {}", informationPacket);
138     ChannelUtils
139         .writeAbstractLocalPacket(localChannelReference, informationPacket,
140                                   false);
141     if (informationPacket instanceof KeepAlivePacket ||
142         informationPacket instanceof NoOpPacket) {
143       // do no await
144       localChannelReference.getFutureValidRequest().setSuccess();
145       localChannelReference.close();
146       return;
147     } else {
148       localChannelReference.getFutureRequest().awaitOrInterruptible();
149       future.awaitOrInterruptible();
150       if (!(localChannelReference.getFutureRequest().isDone() &&
151             future.isDone())) {
152         fail("Cannot send information");
153       }
154     }
155     final R66Result r66result = future.getResult();
156     if (state == R66FiniteDualStates.VALIDOTHER) {
157       logger.warn("feedback: {}", r66result);
158     } else {
159       final ValidPacket info = (ValidPacket) r66result.getOther();
160       if (info != null && scode != -1) {
161         logger.warn("nb: {}", Integer.parseInt(info.getSmiddle()));
162         final String[] files = info.getSheader().split("\n");
163         int i = 0;
164         for (final String file : files) {
165           i++;
166           logger.warn("file: {} {}", i, file);
167         }
168       } else {
169         if (info != null && info.getSheader() != null) {
170           try {
171             final DbTaskRunner runner =
172                 DbTaskRunner.fromStringXml(info.getSheader(), false);
173             logger.warn("{}", runner.getJson());
174           } catch (final OpenR66ProtocolBusinessException e) {
175             logger.warn("{}: {}", FIELDS.transfer.name(), info.getSheader());
176           }
177         }
178       }
179     }
180     localChannelReference.close();
181     if (check) {
182       assertTrue("Information not OK", future.isSuccess());
183     } else {
184       logger.warn("Get {} {}", future.isSuccess(), future.isSuccess()? "OK" :
185           future.getCause() != null? future.getCause().getMessage() :
186               "No Cause");
187     }
188   }

190   public static void initiateWebDriver(File file) {
191     File libdir = file.getParentFile().getParentFile().getParentFile();
192     // test-classes -> target -> WaarpR66 -> lib -> geckodriver (linux x64)
193     File libPhantomJS = new File("/tmp/phantomjs-2.1.1");
194     if (!libPhantomJS.canRead()) {
195       File libPhantomJSZip = new File(libdir, "lib/phantomjs-2.1.1.bz2");
196       if (libPhantomJSZip.canRead()) {
197         try {
198           FileUtils.uncompressedBz2File(libPhantomJSZip, libPhantomJS);
199           libPhantomJS.setExecutable(true);
200         } catch (Reply550Exception e) {
201           fail(e.getMessage());
202         }
203       }
204     }
205     assertTrue(libPhantomJS.exists());
206     System.setProperty("phantomjs.binary.path", libPhantomJS.getAbsolutePath());
207     try {
208       driver = initializeDriver();
209     } catch (InterruptedException e) {//NOSONAR
210       e.printStackTrace();
211       fail(e.getMessage());
212     }
213   }

222   public static WebDriver initializeDriver() throws InterruptedException {
223     WebDriver driver;
224     switch (driverType) {
225       case PHANTOMJS:
226       default:
227         driver = createPhantomJSDriver();
228     }
229     driver.manage().timeouts().implicitlyWait(5, TimeUnit.SECONDS);
230     //driver.manage().timeouts().pageLoadTimeout(10, TimeUnit.SECONDS);
231     Thread.sleep(10);
232     return driver;
233   }

235   private static WebDriver createPhantomJSDriver() {
236     DesiredCapabilities desiredCapabilities =
237         new DesiredCapabilities("phantomjs", "", Platform.ANY);
238     desiredCapabilities.setCapability(
239         PhantomJSDriverService.PHANTOMJS_EXECUTABLE_PATH_PROPERTY,
240         System.getProperty("phantomjs.binary.path"));
241     desiredCapabilities
242         .setCapability(CapabilityType.ELEMENT_SCROLL_BEHAVIOR, true);
243     desiredCapabilities.setCapability(CapabilityType.TAKES_SCREENSHOT, false);
244     desiredCapabilities
245         .setCapability(CapabilityType.ENABLE_PROFILING_CAPABILITY, false);
246     LoggingPreferences logPrefs = new LoggingPreferences();
247     logPrefs.enable(LogType.BROWSER, Level.OFF);
248     logPrefs.enable(LogType.CLIENT, Level.OFF);
249     logPrefs.enable(LogType.DRIVER, Level.OFF);
250     logPrefs.enable(LogType.PERFORMANCE, Level.OFF);
251     logPrefs.enable(LogType.PROFILER, Level.OFF);
252     logPrefs.enable(LogType.SERVER, Level.OFF);
253     desiredCapabilities.setCapability(CapabilityType.LOGGING_PREFS, logPrefs);
254     desiredCapabilities.setCapability(CapabilityType.HAS_NATIVE_EVENTS, true);
255     desiredCapabilities
256         .setCapability(PhantomJSDriverService.PHANTOMJS_GHOSTDRIVER_CLI_ARGS,
257                        "--webdriver-loglevel=NONE");
258     desiredCapabilities.setJavascriptEnabled(true);
259 
260     ArrayList<String> cliArgs = new ArrayList<String>();
261     cliArgs.add("--web-security=true");
262     cliArgs.add("--ignore-ssl-errors=true");
263     cliArgs.add("--webdriver-loglevel=NONE");
264     desiredCapabilities
265         .setCapability(PhantomJSDriverService.PHANTOMJS_CLI_ARGS, cliArgs);
266 
267     PhantomJSDriver phantomJSDriver = new PhantomJSDriver(desiredCapabilities);
268     phantomJSDriver.setLogLevel(Level.OFF);
269     return phantomJSDriver;
270   }

301   public static void setUpBeforeClassServer(String serverInit,
302                                             String serverConfig, boolean start)
303       throws Exception {
304 
305     if (Configuration.configuration != null &&
306         Configuration.configuration.isConfigured()) {
307       // Ensure we redo configuration
308       Configuration.configuration.setConfigured(false);
309       Configuration.configuration = new Configuration();
310     }
311     final ClassLoader classLoader = TestAbstract.class.getClassLoader();
312     final File file;
313     if (serverInit.startsWith("/")) {
314       file = new File(serverInit);
315     } else {
316       file = new File(classLoader.getResource(serverInit).getFile());
317     }
318     if (file.exists()) {
319       dir = file.getParentFile();
320       System.err.println("Find serverInit file: " + file.getAbsolutePath());
321       ServerInitDatabase.main(new String[] {
322           file.getAbsolutePath(), "-initdb", "-dir", dir.getAbsolutePath(),
323           "-auth", new File(dir, "OpenR66-authent-A.xml").getAbsolutePath(),
324           "-limit", new File(dir, "limitConfiga.xml").getAbsolutePath()
325       });
326       Configuration.configuration.setTimeoutCon(100);
327       logger.warn("Init Done");
328       if (start) {
329         final File file2;
330         if (serverConfig.startsWith("/")) {
331           file2 = new File(serverConfig);
332         } else {
333           file2 = new File(dir, serverConfig);
334         }
335         if (file2.exists()) {
336           System.err.println("Find server file: " + file2.getAbsolutePath());
337           R66Server.main(new String[] { file2.getAbsolutePath() });
338           logger.warn("Start Done");
339         } else {
340           System.err
341               .println("Cannot find server file: " + file2.getAbsolutePath());
342         }
343       }
344     } else {
345       System.err
346           .println("Cannot find serverInit file: " + file.getAbsolutePath());
347     }
348   }

369   public static void setUpBeforeClassClient(String clientConfig)
370       throws Exception {
371     final File clientConfigFile;
372     if (clientConfig.startsWith("/")) {
373       clientConfigFile = new File(clientConfig);
374     } else {
375       clientConfigFile = new File(dir, clientConfig);
376     }
377     if (clientConfigFile.isFile()) {
378       System.err.println(
379           "Find serverInit file: " + clientConfigFile.getAbsolutePath());
380       if (!FileBasedConfiguration
381           .setClientConfigurationFromXml(Configuration.configuration,
382                                          clientConfigFile.getAbsolutePath())) {
383         logger.error("Needs a correct configuration file as first argument");
384         return;
385       }
386     } else {
387       logger.error("Needs a correct configuration file as first argument: {}",
388                    clientConfigFile.getAbsolutePath());
389       return;
390     }
391     Configuration.configuration.setTimeoutCon(100);
392     Configuration.configuration.pipelineInit();
393     networkTransaction = new NetworkTransaction();
394     DbTaskRunner.clearCache();
395     TransferDAO transferAccess = null;
396     try {
397       transferAccess = DAOFactory.getInstance().getTransferDAO();
398       transferAccess.deleteAll();
399     } catch (final DAOConnectionException e) {
400       throw new WaarpDatabaseException(e);
401     } finally {
402       if (transferAccess != null) {
403         transferAccess.close();
404       }
405     }
406   }
