20  package org.waarp.openr66.protocol.test;
21  
22  import org.apache.thrift.TException;
23  import org.apache.thrift.protocol.TBinaryProtocol;
24  import org.apache.thrift.protocol.TProtocol;
25  import org.apache.thrift.transport.TSocket;
26  import org.apache.thrift.transport.TTransport;
27  import org.apache.thrift.transport.TTransportException;
28  import org.waarp.thrift.r66.Action;
29  import org.waarp.thrift.r66.R66Request;
30  import org.waarp.thrift.r66.R66Result;
31  import org.waarp.thrift.r66.R66Service;
32  import org.waarp.thrift.r66.RequestMode;
33  
34  import java.util.List;
35  
36  /**
37   * Example of Java class to interact with the Thrift R66 service
38   */
39  public class TestThriftClientExample {
40    private static final int PORT = 4266;
41    private static final int tries = 10000;
42  
43    public static void main(String[] args) {
44      TTransport transport = null;
45      try {
46        transport = new TSocket("localhost", PORT);
47        final TProtocol protocol = new TBinaryProtocol(transport);
48        final R66Service.Client client = new R66Service.Client(protocol);
49        transport.open();
50        R66Request request = new R66Request(RequestMode.INFOFILE);
51        request.setDestuid("hostas");
52        request.setFromuid("tests");
53        request.setRule("rule3");
54        request.setAction(Action.List);
55  
56        System.out.println("REQUEST1: " + request);
57        List<String> list = client.infoListQuery(request);
58        System.out.println("RESULT1: " + list.size());
59        for (final String slist : list) {
60          System.out.println(slist);
61        }
62  
63        final long start = System.currentTimeMillis();
64        for (int i = 0; i < tries; i++) {
65          list = client.infoListQuery(request);
66        }
67        final long end = System.currentTimeMillis();
68        System.out.println(
69            "Delay: " + (end - start) + " : " + (tries * 1000) / (end - start));
70  
71        final long startEx = System.currentTimeMillis();
72        boolean dontknow = false;
73        for (int i = 0; i < tries; i++) {
74          dontknow = client.isStillRunning("tests", "hostas", 1346080633424L);
75        }
76        final long endEx = System.currentTimeMillis();
77        System.out.println("StillRunning: " + dontknow);
78        System.out.println("Delay: " + (endEx - startEx) + " : " +
79                           (tries * 1000) / (endEx - startEx));
80  
81        request.setMode(RequestMode.INFOREQUEST);
82        request.setTid(1346080633424L);
83        request.setAction(Action.Detail);
84        R66Result result = client.infoTransferQuery(request);
85        System.out.println("RESULT2: " + result);
86        final long startQu = System.currentTimeMillis();
87        for (int i = 0; i < tries; i++) {
88          result = client.infoTransferQuery(request);
89        }
90        final long endQu = System.currentTimeMillis();
91        System.out.println("Delay: " + (endQu - startQu) + " : " +
92                           (tries * 1000) / (endQu - startQu));
93  
94        System.out.println("Exist: " + client.isStillRunning(request.getFromuid(),
95                                                             request.getDestuid(),
96                                                             request.getTid()));
97  
98        request.setMode(RequestMode.INFOFILE);
99        request.setAction(Action.Mlsx);
100       list = client.infoListQuery(request);
101       System.out.println("RESULT3: " + list.size());
102       for (final String slist : list) {
103         System.out.println(slist);
104       }
105 
106       request = new R66Request(RequestMode.ASYNCTRANSFER);
107       request.setDestuid("hostbs");
108       request.setRule("rule3");
109       request.setFile("Documents2.rar");
110       request.setInfo("Submitted from Thrift");
111       result = client.transferRequestQuery(request);
112       System.out.println("RESULT4: " + result);
113 
114       request = new R66Request(RequestMode.SYNCTRANSFER);
115       request.setDestuid("hostbs");
116       request.setRule("rule3");
117       request.setFile("Documents2.rar");
118       request.setInfo("Submitted from Thrift");
119       result = client.transferRequestQuery(request);
120       System.out.println("RESULT5: " + result);
121 
122       // Wrong request
123       request = new R66Request(RequestMode.INFOFILE);
124 
125       System.out.println("WRONG REQUEST: " + request);
126       list = client.infoListQuery(request);
127       System.out.println("RESULT of Wrong Request: " + list.size());
128       for (final String slist : list) {
129         System.out.println(slist);
130       }
131 
132     } catch (final TTransportException e) {
133       e.printStackTrace();
134     } catch (final TException e) {
135       e.printStackTrace();
136     }
137     transport.close();
138   }
139 
140 }
