36  /**
37   * Test class for multiple SubmitTransfer
38   */
39  public class TestSubmitTransfer extends SubmitTransfer {
40    static int nb = 100;
41  
42    public TestSubmitTransfer(R66Future future, String remoteHost,
43                              String filename, String rulename, String fileinfo,
44                              boolean isMD5, int blocksize, Timestamp starttime) {
45      super(future, remoteHost, filename, rulename, fileinfo, isMD5, blocksize,
46            DbConstantR66.ILLEGALVALUE, starttime);
47      if (!fileinfo.contains("-nofollow")) {
48        TransferArgs.forceAnalyzeFollow(this);
49      }
50    }
51  
52    public static void main(String[] args) {
53      WaarpLoggerFactory.setDefaultFactoryIfNotSame(
54          new WaarpSlf4JLoggerFactory(null));
55      ResourceLeakDetector.setLevel(Level.PARANOID);
56      if (logger == null) {
57        logger = WaarpLoggerFactory.getLogger(SubmitTransfer.class);
58      }
59      if (!getParams(args, true)) {
60        logger.error("Wrong initialization");
61        if (DbConstantR66.admin != null) {
62          DbConstantR66.admin.close();
63        }
64        WaarpSystemUtil.systemExit(1);
65        return;
66      }
67      getSpecialParams(args, 1);
68  
69      final ExecutorService executorService = Executors.newCachedThreadPool();
70      final R66Future[] arrayFuture = new R66Future[nb];
71  
72      logger.warn("Start Test Submit");
73      for (int i = 0; i < nb; i++) {
74        arrayFuture[i] = new R66Future(true);
75        Timestamp newstart = ttimestart;
76        if (newstart != null) {
77          // delay of 10 ms between each
78          newstart = new Timestamp(newstart.getTime() + i * 10);
79        }
80        final TestSubmitTransfer transaction =
81            new TestSubmitTransfer(arrayFuture[i], rhost, localFilename, rule,
82                                   transferInfo, ismd5, block, newstart);
83        transaction.normalInfoAsWarn = snormalInfoAsWarn;
84        // executorService.execute(transaction);
85        transaction.run();
86      }
87      int success = 0;
88      int error = 0;
89      for (int i = 0; i < nb; i++) {
90        arrayFuture[i].awaitOrInterruptible();
91        if (arrayFuture[i].isSuccess()) {
92          success++;
93        } else {
94          error++;
95        }
96      }
97      executorService.shutdown();
98      logger.warn("Prepare transfer Success: " + success + " Error: " + error);
99    }

101   /**
102    * @param args
103    * @param rank
104    *
105    * @return True if OK
106    */
107   protected static boolean getSpecialParams(String[] args, int rank) {
108     for (int i = rank; i < args.length; i++) {
109       if ("-nb".equalsIgnoreCase(args[i])) {
110         i++;
111         nb = Integer.parseInt(args[i]);
112       } else if ("-md5".equalsIgnoreCase(args[i])) {
113       } else if (args[i].charAt(0) == '-') {
114         i++;// jump one
115       }
116     }
117     return true;
118   }
