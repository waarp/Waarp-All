436   private static void waitForAllDone(DbTaskRunner runner) {
437     while (true) {
438       try {
439         DbTaskRunner checkedRunner = DbTaskRunner.reloadFromDatabase(runner);
440         if (checkedRunner.isAllDone()) {
441           SysErrLogger.FAKE_LOGGER.sysout("DbTaskRunner done");
442           return;
443         } else if (checkedRunner.isInError()) {
444           SysErrLogger.FAKE_LOGGER.syserr("DbTaskRunner in error");
445           return;
446         }
447         Thread.sleep(100);
448       } catch (InterruptedException e) {//NOSONAR
449         SysErrLogger.FAKE_LOGGER.syserr("Interrupted", e);
450         return;
451       } catch (WaarpDatabaseException e) {
452         SysErrLogger.FAKE_LOGGER.syserr("Cannot found DbTaskRunner", e);
453         return;
454       }
455     }
456   }
