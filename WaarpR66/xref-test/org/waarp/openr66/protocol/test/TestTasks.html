462   private static void waitForAllDone(DbTaskRunner runner) {
463     while (true) {
464       try {
465         runner.select();
466         if (runner.isAllDone()) {
467           SysErrLogger.FAKE_LOGGER.sysout("DbTaskRunner done");
468           return;
469         } else if (runner.isInError()) {
470           SysErrLogger.FAKE_LOGGER.syserr("DbTaskRunner in error");
471           return;
472         }
473         Thread.sleep(100);
474       } catch (InterruptedException e) {//NOSONAR
475         SysErrLogger.FAKE_LOGGER.syserr("Interrupted", e);
476         return;
477       } catch (WaarpDatabaseException e) {
478         SysErrLogger.FAKE_LOGGER.syserr("Cannot found DbTaskRunner", e);
479         return;
480       }
481     }
