154   public String getServerConfigFile() {
155     if (configFile != null) {
156       logger.warn("ConfigFile already set to {]", configFile);
157       return configFile.getAbsolutePath();
158     }
159     logger.warn("Build configFile");
160     ClassLoader classLoader = ScenarioBaseLoop.class.getClassLoader();
161     File file =
162         new File(classLoader.getResource(RESOURCES_SERVER_1_XML).getFile());
163     if (!file.exists()) {
164       SysErrLogger.FAKE_LOGGER.syserr(
165           "Cannot find in  " + file.getAbsolutePath());
166       fail("Cannot find " + file.getAbsolutePath());
167     }
168     String content = WaarpStringUtils.readFile(file.getAbsolutePath());
169     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getJdbcUrl());
170     String driver = getJDC().getDriverClassName();
171     String target = "notfound";
172     String jdbcUrl = getJDC().getJdbcUrl();
173     if (driver.equalsIgnoreCase("org.mariadb.jdbc.Driver")) {
174       target = "mariadb";
175     } else if (driver.equalsIgnoreCase("org.h2.Driver")) {
176       target = "h2";
177     } else if (driver.equalsIgnoreCase("oracle.jdbc.OracleDriver")) {
178       target = "oracle";
179       jdbcUrl = "jdbc:oracle:thin:@//localhost:1521/test";
180       SysErrLogger.FAKE_LOGGER.syserr(
181           jdbcUrl + " while should be something like " + jdbcUrl);
182       throw new UnsupportedOperationException(
183           "Unsupported Test for Oracle since wrong JDBC driver");
184     } else if (driver.equalsIgnoreCase("org.postgresql.Driver")) {
185       target = "postgresql";
186     } else if (DbModelMysql.MYSQL_DRIVER_JRE6.equalsIgnoreCase(driver) ||
187                DbModelMysql.MYSQL_DRIVER_JRE8.equalsIgnoreCase(driver)) {
188       target = "mysql";
189     } else {
190       SysErrLogger.FAKE_LOGGER.syserr("Cannot find driver for " + driver);
191     }
192     content = content.replace("XXXJDBCXXX", jdbcUrl);
193     content = content.replace("XXXDRIVERXXX", target);
194     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getDriverClassName());
195     SysErrLogger.FAKE_LOGGER.sysout(target);
196     File fileTo = new File(TMP_R66_CONFIG_R1);
197     fileTo.getParentFile().mkdirs();
198     FileWriter writer = null;
199     try {
200       writer = new FileWriter(fileTo);
201       writer.write(content);
202       writer.flush();
203     } catch (IOException e) {
204       e.printStackTrace();
205     } finally {
206       if (writer != null) {
207         FileUtils.close(writer);
208       }
209     }
210     configFile = fileTo;
211     logger.warn("Config file created at {}", fileTo);
212     File copy = new File(TMP_CONFIG_XML);
213     try {
214       FileUtils.copy(configFile, copy, false, false);
215       logger.warn("Copy from {} to {}", configFile, copy);
216     } catch (Reply550Exception e) {
217       e.printStackTrace();
218     }
219     logger.warn("Copy from {} to {} and return {}", configFile, copy,
220                 TMP_R66_CONFIG_R1);
221     return TMP_R66_CONFIG_R1;
222   }

329   @AfterClass
330   public static void tearDownAfterClass() throws Exception {
331     if (NUMBER_FILES == -1) {
332       Configuration.configuration.setTimeoutCon(100);
333       for (int pid : PIDS) {
334         Processes.kill(pid, true);
335       }
336       tearDownAfterClassClient();
337       tearDownAfterClassMinimal();
338       tearDownAfterClassServer();
339       return;
340     }
341     CloseableHttpClient httpClient = null;
342     int max = SystemPropertyUtil.get(IT_LONG_TEST, false)? 60 : 20;
343     if (max < NUMBER_FILES) {
344       max = NUMBER_FILES * 3;
345     }
346     int totalTransfers = max;
347     int nb = 0;
348     int every10sec = 10;
349     HttpGet request =
350         new HttpGet("http://127.0.0.1:8098/v2/transfers?countOrder=true");
351     final long startTime = System.currentTimeMillis();
352     try {
353       httpClient = HttpClientBuilder.create().setConnectionManagerShared(true)
354                                     .disableAutomaticRetries().build();
355       CloseableHttpResponse response = null;
356       try {
357         response = httpClient.execute(request);
358         String content = EntityUtils.toString(response.getEntity());
359         ObjectNode node = JsonHandler.getFromString(content);
360         if (node != null) {
361           JsonNode number = node.findValue("totalResults");
362           if (number != null) {
363             long newNb = number.asLong();
364             max += newNb;
365             totalTransfers = (int) newNb;
366             logger.warn("Found {} transfers", newNb);
367           }
368         }
369       } finally {
370         if (response != null) {
371           response.close();
372         }
373       }
374       while (nb < max) {
375         try {
376           response = httpClient.execute(request);
377           if (400 <= response.getStatusLine().getStatusCode()) {
378             logger.warn("Bad Code {}",
379                         response.getStatusLine().getStatusCode());
380           }
381           String content = EntityUtils.toString(response.getEntity());
382           ObjectNode node = JsonHandler.getFromString(content);
383           if (node != null) {
384             JsonNode number = node.findValue("totalResults");
385             if (number != null) {
386               int newNb = number.asInt();
387               if (newNb != nb || every10sec == 0) {
388                 every10sec = 10;
389                 nb = newNb;
390                 logger.warn("Found {} transfers", nb);
391               }
392             }
393             every10sec--;
394           }
395         } finally {
396           if (response != null) {
397             response.close();
398           }
399         }
400         Thread.sleep(1000);
401       }
402     } catch (ExecuteException e) {
403       // ignore
404     } finally {
405       if (httpClient != null) {
406         httpClient.close();
407       }
408     }
409     final long stopTime = System.currentTimeMillis();
410     totalTransfers -= nb;
411     logger.warn("Duration {} for {} item so {} items/s",
412                 (stopTime - startTime) / 1000.0, totalTransfers,
413                 totalTransfers / ((stopTime - startTime) / 1000.0));
414     WaarpSystemUtil.stopLogger(true);
415     Configuration.configuration.setTimeoutCon(100);
416     for (int pid : PIDS) {
417       Processes.kill(pid, true);
418     }
419     tearDownAfterClassClient();
420     tearDownAfterClassMinimal();
421     tearDownAfterClassServer();
422   }

445   private void testBigTransfer(boolean limit, String serverName, boolean direct,
446                                boolean recv, int size) throws IOException {
447     if (!SERVER1_IN_JUNIT && !direct) {
448       logger.warn("Only Direct is enabled in Client mode");
449       return;
450     }
451     Assume.assumeNotNull(networkTransaction);
452     if (limit) {
453       Configuration.configuration.changeNetworkLimit(BANDWIDTH * 2,
454                                                      BANDWIDTH * 2, BANDWIDTH,
455                                                      BANDWIDTH, 1000);
456     } else {
457       Configuration.configuration.changeNetworkLimit(0, 0, 0, 0, 1000);
458     }
459     File baseDir;
460     if (recv && !"server1".equals(serverName)) {
461       baseDir = new File("/tmp/R66/scenario_big_file_limitbdw/R2/out/");
462     } else {
463       baseDir = new File("/tmp/R66/scenario_big_file_limitbdw/R1/out/");
464     }
465     File fileOut = new File(baseDir, "hello" + size);
466     final File outHello = generateOutFile(fileOut.getAbsolutePath(), size);
467     long timestart = System.currentTimeMillis();
468     final R66Future future = new R66Future(true);
469     if (direct) {
470       if (recv) {
471         final TestRecvThroughClient transaction =
472             new TestRecvThroughClient(future, handler, serverName,
473                                       "hello" + size, "recvthrough",
474                                       "Test Big RecvThrough", true, BLOCK_SIZE,
475                                       networkTransaction);
476         transaction.setNormalInfoAsWarn(false);
477         transaction.run();
478       } else {
479         final TestTransferNoDb transaction =
480             new TestTransferNoDb(future, serverName, "hello" + size, "loop",
481                                  "Test Loop Send", true, BLOCK_SIZE,
482                                  DbConstantR66.ILLEGALVALUE,
483                                  networkTransaction);
484         transaction.setNormalInfoAsWarn(false);
485         transaction.run();
486       }
487       future.awaitOrInterruptible();
488       assertTrue(future.isSuccess());
489       logger.info("Runner: {}", future.getRunner());
490     } else {
491       String ruleName = recv? "recv" : "sendany";
492       final SubmitTransfer transaction =
493           new SubmitTransfer(future, serverName, "hello" + size, ruleName,
494                              "Test Big " + ruleName, true, BLOCK_SIZE,
495                              DbConstantR66.ILLEGALVALUE, null);
496       TransferArgs.forceAnalyzeFollow(transaction);
497       transaction.run();
498       future.awaitOrInterruptible();
499       assertTrue(future.isSuccess());
500       final DbTaskRunner runner = future.getResult().getRunner();
501       if (runner != null) {
502         logger.info("Runner: {}", runner.toString());
503         waitForAllDone(runner);
504         dbTaskRunners.add(runner);
505       }
506     }
507     long timestop = System.currentTimeMillis();
508     logger.warn("Direct {}, Recv {}, LimitBandwidth {} " +
509                 "({} seconds,  {} MBPS vs {} " +
510                 "and {}) of size {} with block size {}", direct, recv, limit,
511                 (timestop - timestart) / 1000,
512                 size / 1000.0 / (timestop - timestart),
513                 Configuration.configuration.getServerGlobalReadLimit() /
514                 1000000.0,
515                 Configuration.configuration.getServerChannelReadLimit() /
516                 1000000.0, size, BLOCK_SIZE);
517     checkMemory();
518   }

669   private void waitForAllDone(DbTaskRunner runner) {
670     while (true) {
671       try {
672         runner.select();
673         if (runner.isAllDone()) {
674           logger.warn("DbTaskRunner done");
675           return;
676         } else if (runner.isInError()) {
677           logger.error("DbTaskRunner in error");
678           fail("DbTaskRunner in error");
679           return;
680         }
681         Thread.sleep(100);
682       } catch (InterruptedException e) {//NOSONAR
683         logger.error("Interrupted", e);
684         return;
685       } catch (WaarpDatabaseException e) {
686         logger.error("Cannot found DbTaskRunner", e);
687         return;
688       }
689     }
690   }
