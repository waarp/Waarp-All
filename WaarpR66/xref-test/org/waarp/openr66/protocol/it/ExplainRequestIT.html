199   public String getServerConfigFile() {
200     if (configFile != null) {
201       logger.warn("ConfigFile already set to {]", configFile);
202       return configFile.getAbsolutePath();
203     }
204     logger.warn("Build configFile");
205     ClassLoader classLoader = ExplainRequestIT.class.getClassLoader();
206     File file =
207         new File(classLoader.getResource(RESOURCES_SERVER_1_XML).getFile());
208     if (!file.exists()) {
209       SysErrLogger.FAKE_LOGGER.syserr(
210           "Cannot find in  " + file.getAbsolutePath());
211       fail("Cannot find " + file.getAbsolutePath());
212     }
213     String content = WaarpStringUtils.readFile(file.getAbsolutePath());
214     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getJdbcUrl());
215     String driver = getJDC().getDriverClassName();
216     String target = "notfound";
217     String jdbcUrl = getJDC().getJdbcUrl();
218     if (driver.equalsIgnoreCase("org.mariadb.jdbc.Driver")) {
219       target = "mariadb";
220     } else if (driver.equalsIgnoreCase("org.h2.Driver")) {
221       target = "h2";
222     } else if (driver.equalsIgnoreCase("oracle.jdbc.OracleDriver")) {
223       target = "oracle";
224       jdbcUrl = "jdbc:oracle:thin:@//localhost:1521/test";
225       SysErrLogger.FAKE_LOGGER.syserr(
226           jdbcUrl + " while should be something like " + jdbcUrl);
227       throw new UnsupportedOperationException(
228           "Unsupported Test for Oracle since wrong JDBC driver");
229     } else if (driver.equalsIgnoreCase("org.postgresql.Driver")) {
230       target = "postgresql";
231     } else if (DbModelMysql.MYSQL_DRIVER_JRE6.equalsIgnoreCase(driver) ||
232                DbModelMysql.MYSQL_DRIVER_JRE8.equalsIgnoreCase(driver)) {
233       target = "mysql";
234     } else {
235       SysErrLogger.FAKE_LOGGER.syserr("Cannot find driver for " + driver);
236     }
237     content = content.replace("XXXJDBCXXX", jdbcUrl);
238     content = content.replace("XXXDRIVERXXX", target);
239     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getDriverClassName());
240     SysErrLogger.FAKE_LOGGER.sysout(target);
241     File fileTo = new File(TMP_R66_CONFIG_R1);
242     fileTo.getParentFile().mkdirs();
243     FileWriter writer = null;
244     try {
245       writer = new FileWriter(fileTo);
246       writer.write(content);
247       writer.flush();
248     } catch (IOException e) {
249       e.printStackTrace();
250     } finally {
251       if (writer != null) {
252         FileUtils.close(writer);
253       }
254     }
255     configFile = fileTo;
256     logger.warn("Config file created at {}", fileTo);
257     File copy = new File(TMP_CONFIG_XML);
258     try {
259       FileUtils.copy(configFile, copy, false, false);
260       logger.warn("Copy from {} to {}", configFile, copy);
261     } catch (Reply550Exception e) {
262       e.printStackTrace();
263     }
264     logger.warn("Copy from {} to {} and return {}", configFile, copy,
265                 TMP_R66_CONFIG_R1);
266     return TMP_R66_CONFIG_R1;
267   }

383   private static final class LoadDb implements Runnable {
384     @Override
385     public void run() {
386       final R66Future[] arrayFuture = new R66Future[NB_FILES];
387       for (int i = 0; i < NB_FILES; i++) {
388         final R66Future future2 = new R66Future(true);
389         arrayFuture[i] = future2;
390         final SubmitTransfer transaction =
391             new SubmitTransfer(future2, "server2", "hello", "recvthrough",
392                                "Fill RecvThrough " + i, true, BLOCK_SIZE,
393                                DbConstantR66.ILLEGALVALUE, null);
394         transaction.setNormalInfoAsWarn(false);
395         transaction.run();
396       }
397       for (int i = 0; i < NB_FILES; i++) {
398         arrayFuture[i].awaitOrInterruptible();
399       }
400       for (int i = 0; i < NB_FILES; i++) {
401         if (arrayFuture[i].isSuccess()) {
402           final DbTaskRunner runner = arrayFuture[i].getResult().getRunner();
403           if (runner != null) {
404             waitForAllDone(runner);
405           }
406         }
407       }
408     }
409 
410     private void waitForAllDone(DbTaskRunner runner) {
411       while (true) {
412         try {
413           runner.select();
414           if (runner.isAllDone()) {
415             return;
416           } else if (runner.isInError()) {
417             logger.error("DbTaskRunner in error");
418             return;
419           }
420           Thread.sleep(100);
421         } catch (InterruptedException e) {//NOSONAR
422           logger.error("Interrupted", e);
423           return;
424         } catch (WaarpDatabaseException e) {
425           logger.error("Cannot found DbTaskRunner", e);
426           return;
427         }
428       }
429     }
430   }
