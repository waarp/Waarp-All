21  package org.waarp.openr66.protocol.it;
22  
23  import org.junit.BeforeClass;
24  import org.junit.ClassRule;
25  import org.junit.FixMethodOrder;
26  import org.junit.Rule;
27  import org.junit.rules.TestWatcher;
28  import org.junit.runners.MethodSorters;
29  import org.testcontainers.containers.JdbcDatabaseContainer;
30  import org.testcontainers.containers.PostgreSQLContainer;
31  import org.waarp.common.utility.TestWatcherJunit4;
32  
33  import java.util.HashMap;
34  import java.util.Map;
35  
36  @FixMethodOrder(MethodSorters.NAME_ASCENDING)
37  public class ScenarioDualServerMultipleFilesBenchmarkPostGreSqlIT
38      extends ScenarioBaseBenchmarkDualServerMultipleFiles {
39    @Rule(order = Integer.MIN_VALUE)
40    public TestWatcher watchman = new TestWatcherJunit4();
41  
42    protected static final Map<String, String> TMPFSMAP =
43        new HashMap<String, String>();
44  
45    static {
46      TMPFSMAP.clear();
47      TMPFSMAP.put("/tmp/postgresql/data", "rw");
48    }
49  
50    @ClassRule
51    public static PostgreSQLContainer db =
52        (PostgreSQLContainer) new PostgreSQLContainer().withCommand(
53            "postgres -c fsync=false -c synchronous_commit=off -c " +
54            "full_page_writes=false -c wal_level=minimal -c " +
55            "max_wal_senders=0 -c enable_seqscan=off").withTmpFs(TMPFSMAP);
56  
57    public JdbcDatabaseContainer getJDC() {
58      return db;
59    }
60  
61    @BeforeClass
62    public static void setup() throws Exception {
63      logger.warn("START PostGreSQL IT TEST");
64      scenarioBase = new ScenarioDualServerMultipleFilesBenchmarkPostGreSqlIT();
65      setUpBeforeClass();
66    }
67  
68  }
