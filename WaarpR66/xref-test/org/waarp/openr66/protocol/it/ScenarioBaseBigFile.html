139   public String getServerConfigFile() {
140     if (configFile != null) {
141       logger.warn("ConfigFile already set to {]", configFile);
142       return configFile.getAbsolutePath();
143     }
144     logger.warn("Build configFile");
145     ClassLoader classLoader = ScenarioBaseBigFile.class.getClassLoader();
146     File file =
147         new File(classLoader.getResource(RESOURCES_SERVER_1_XML).getFile());
148     if (!file.exists()) {
149       SysErrLogger.FAKE_LOGGER.syserr(
150           "Cannot find in  " + file.getAbsolutePath());
151       fail("Cannot find " + file.getAbsolutePath());
152     }
153     String content = WaarpStringUtils.readFile(file.getAbsolutePath());
154     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getJdbcUrl());
155     String driver = getJDC().getDriverClassName();
156     String target = "notfound";
157     String jdbcUrl = getJDC().getJdbcUrl();
158     if (driver.equalsIgnoreCase("org.mariadb.jdbc.Driver")) {
159       target = "mariadb";
160     } else if (driver.equalsIgnoreCase("org.h2.Driver")) {
161       target = "h2";
162     } else if (driver.equalsIgnoreCase("oracle.jdbc.OracleDriver")) {
163       target = "oracle";
164       jdbcUrl = "jdbc:oracle:thin:@//localhost:1521/test";
165       SysErrLogger.FAKE_LOGGER.syserr(
166           jdbcUrl + " while should be something like " + jdbcUrl);
167       throw new UnsupportedOperationException(
168           "Unsupported Test for Oracle since wrong JDBC driver");
169     } else if (driver.equalsIgnoreCase("org.postgresql.Driver")) {
170       target = "postgresql";
171     } else if (DbModelMysql.MYSQL_DRIVER_JRE6.equalsIgnoreCase(driver) ||
172                DbModelMysql.MYSQL_DRIVER_JRE8.equalsIgnoreCase(driver)) {
173       target = "mysql";
174     } else {
175       SysErrLogger.FAKE_LOGGER.syserr("Cannot find driver for " + driver);
176     }
177     content = content.replace("XXXJDBCXXX", jdbcUrl);
178     content = content.replace("XXXDRIVERXXX", target);
179     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getDriverClassName());
180     SysErrLogger.FAKE_LOGGER.sysout(target);
181     File fileTo = new File(TMP_R66_CONFIG_R1);
182     fileTo.getParentFile().mkdirs();
183     FileWriter writer = null;
184     try {
185       writer = new FileWriter(fileTo);
186       writer.write(content);
187       writer.flush();
188     } catch (IOException e) {
189       e.printStackTrace();
190     } finally {
191       if (writer != null) {
192         FileUtils.close(writer);
193       }
194     }
195     configFile = fileTo;
196     logger.warn("Config file created at {}", fileTo);
197     File copy = new File(TMP_CONFIG_XML);
198     try {
199       FileUtils.copy(configFile, copy, false, false);
200       logger.warn("Copy from {} to {}", configFile, copy);
201     } catch (Reply550Exception e) {
202       e.printStackTrace();
203     }
204     logger.warn("Copy from {} to {} and return {}", configFile, copy,
205                 TMP_R66_CONFIG_R1);
206     return TMP_R66_CONFIG_R1;
207   }

283   public static int startServer(String serverConfig) throws Exception {
284     final File file2;
285     if (serverConfig.charAt(0) == '/') {
286       file2 = new File(serverConfig);
287     } else {
288       file2 = new File(dirResources, serverConfig);
289     }
290     if (file2.exists()) {
291       System.err.println("Find server file: " + file2.getAbsolutePath());
292       final String[] argsServer = {
293           file2.getAbsolutePath()
294       };
295       // global ant project settings
296       project = Processes.getProject(homeDir);
297       Processes.executeJvm(project, R66Server.class, argsServer, true);
298       int pid = Processes.getPidOfRunnerCommandLinux("java",
299                                                      R66Server.class.getName(),
300                                                      PIDS);
301       PIDS.add(pid);
302       logger.warn("Start Done: {}", pid);
303       return pid;
304     } else {
305       System.err.println("Cannot find server file: " + file2.getAbsolutePath());
306       fail("Cannot find server file");
307       return 999999;
308     }

356   private void testBigTransfer(boolean limit, String serverName, boolean direct,
357                                boolean recv) throws IOException {
358     if (!SERVER1_IN_JUNIT && !direct) {
359       logger.warn("Only Direct is enabled in Client mode");
360       return;
361     }
362     Assume.assumeNotNull(networkTransaction);
363     if (limit) {
364       Configuration.configuration.changeNetworkLimit(BANDWIDTH * 2,
365                                                      BANDWIDTH * 2, BANDWIDTH,
366                                                      BANDWIDTH, 1000);
367     } else {
368       Configuration.configuration.changeNetworkLimit(0, 0, 0, 0, 100000);
369     }
370     File baseDir;
371     if (recv && !"server1".equals(serverName)) {
372       baseDir = new File("/tmp/R66/scenario_big_file_limitbdw/R2/out/");
373     } else {
374       baseDir = new File("/tmp/R66/scenario_big_file_limitbdw/R1/out/");
375     }
376     File fileOut = new File(baseDir, "hello");
377     final File outHello =
378         generateOutFile(fileOut.getAbsolutePath(), LARGE_SIZE);
379     long timestart = System.currentTimeMillis();
380     final R66Future future = new R66Future(true);
381     if (direct) {
382       if (recv) {
383         final NoOpRecvThroughHandler handler = new NoOpRecvThroughHandler();
384         final TestRecvThroughClient transaction =
385             new TestRecvThroughClient(future, handler, serverName, "hello",
386                                       "recvthrough", "Test Big RecvThrough",
387                                       true, BLOCK_SIZE, networkTransaction);
388         transaction.setNormalInfoAsWarn(false);
389         transaction.run();
390       } else {
391         final TestTransferNoDb transaction =
392             new TestTransferNoDb(future, serverName, "hello", "sendany",
393                                  "Test Big Send", true, BLOCK_SIZE,
394                                  DbConstantR66.ILLEGALVALUE,
395                                  networkTransaction);
396         transaction.setNormalInfoAsWarn(false);
397         transaction.run();
398       }
399       future.awaitOrInterruptible();
400       assertTrue(future.isSuccess());
401     } else {
402       String ruleName = recv? "recv" : "sendany";
403       final SubmitTransfer transaction =
404           new SubmitTransfer(future, serverName, "hello", ruleName,
405                              "Test Big " + ruleName, true, BLOCK_SIZE,
406                              DbConstantR66.ILLEGALVALUE, null);
407       transaction.run();
408       future.awaitOrInterruptible();
409       assertTrue(future.isSuccess());
410       final DbTaskRunner runner = future.getResult().getRunner();
411       if (runner != null) {
412         logger.info("Runner: {}", runner.toString());
413         waitForAllDone(runner);
414         dbTaskRunners.add(runner);
415       }
416     }
417     long timestop = System.currentTimeMillis();
418     logger.warn("Direct {}, Recv {}, LimitBandwidth {} " +
419                 "({} seconds,  {} MBPS vs {} " +
420                 "and {}) of size {} with block size {}", direct, recv, limit,
421                 (timestop - timestart) / 1000,
422                 LARGE_SIZE / 1000.0 / (timestop - timestart),
423                 Configuration.configuration.getServerGlobalReadLimit() /
424                 1000000.0,
425                 Configuration.configuration.getServerChannelReadLimit() /
426                 1000000.0, LARGE_SIZE, BLOCK_SIZE);
427     outHello.delete();
428     FileUtils.forceDeleteRecursiveDir(baseDir);
429     checkMemory();
430   }

657   private void waitForAllDone(DbTaskRunner runner) {
658     while (true) {
659       try {
660         runner.select();
661         if (runner.isAllDone()) {
662           logger.warn("DbTaskRunner done");
663           return;
664         } else if (runner.isInError()) {
665           logger.error("DbTaskRunner in error");
666           fail("DbTaskRunner in error");
667           return;
668         }
669         Thread.sleep(100);
670       } catch (InterruptedException e) {//NOSONAR
671         logger.error("Interrupted", e);
672         return;
673       } catch (WaarpDatabaseException e) {
674         logger.error("Cannot found DbTaskRunner", e);
675         return;
676       }
677     }
678   }
