32    @BeforeClass
33    public static void beforeClass() {
34      WaarpLoggerFactory.setDefaultFactoryIfNotSame(
35          new WaarpSlf4JLoggerFactory(WaarpLogLevel.WARN));
36      ResourceLeakDetector.setLevel(Level.PARANOID);
37    }

39    public static final class RestHandlerHookForTest extends RestHandlerHook {
40      public RestHandlerHookForTest(final boolean authenticated,
41                                    final HmacSha256 hmac, final long delay) {
42        super(authenticated, hmac, delay);
43      }
44  
45      public void testValidateHMACredentials(Host host, String authDate,
46                                             String authUser, String authKey)
47          throws InternalServerErrorException {
48        validateHMACCredentials(host, authDate, authUser, authKey);
49      }
50    }
51  
52    @Test
53    public void testCheckCredentialsWithHMAC() throws Exception {
54      final HmacSha256 hmac = new HmacSha256();
55      hmac.generateKey();
56  
57      final Des dyn = new Des();
58      dyn.generateKey();
59      Des oldKey = Configuration.configuration.getCryptoKey();
60      Configuration.configuration.setCryptoKey(dyn);
61  
62      final String user = "user";
63      final String password = "mypassword";
64      final String timestamp =
65          new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssXX").format(new Date());
66  
67      try {
68  
69        final String hostkey = dyn.cryptToHex(password);
70        final String sig = hmac.cryptToHex(timestamp + user + password);
71  
72        final RestHandlerHookForTest hook =
73            new RestHandlerHookForTest(true, hmac, 10000);
74  
75        try {
76          final Host host = new Host(user, "127.0.0.1", 1,
77                                     hostkey.getBytes(WaarpStringUtils.UTF8),
78                                     false, true);
79  
80          hook.testValidateHMACredentials(host, timestamp, user, sig);
81        } catch (InternalServerErrorException e) {
82          System.out.println(e);
83          fail("credentials validation failed, it should have succeeded");
84        }
85  
86      } finally {
87        Configuration.configuration.setCryptoKey(oldKey);
88      }
89  
90    }
