65  /**
66   * Test Rest client response handler
67   * <p>
68   * Note that for testing, result is only the last "json" command, and therefore
69   * future is only validated once
70   * all items are passed in a chain. In normal condition, each step should
71   * produce: setting the RestArgument to
72   * the RestFuture and validating (error or ok) to RestFuture.
73   */
74  public class HttpTestResponseHandler extends HttpRestR66ClientResponseHandler {
75    /**
76     * @param channel
77     *
78     * @throws HttpInvalidAuthenticationException
79     */
80  
81    @Override
82    protected boolean afterError(Channel channel, RestArgument ra) {
83      HttpTestRestR66Client.count.incrementAndGet();
84      WaarpSslUtility.closingSslChannel(channel);
85      return false;
86    }

88    @Override
89    protected boolean afterDbGet(Channel channel, RestArgument ra)
90        throws HttpInvalidAuthenticationException {
91      HttpTestRestR66Client.count.incrementAndGet();
92      // Update
93      HttpTestRestR66Client.updateData(channel, ra);
94      return true;
95    }

97    @Override
98    protected boolean afterDbPost(Channel channel, RestArgument ra)
99        throws HttpInvalidAuthenticationException {
100     HttpTestRestR66Client.count.incrementAndGet();
101     if (ra.getAnswer().path(DbHostAuth.Columns.ADMINROLE.name()).asBoolean()) {
102       WaarpSslUtility.closingSslChannel(channel);
103       return false;
104     }
105     // Select 1
106     HttpTestRestR66Client.readData(channel, ra);
107     return true;
108   }
109 
110   @Override
111   protected boolean afterDbPut(Channel channel, RestArgument ra)
112       throws HttpInvalidAuthenticationException {
113     HttpTestRestR66Client.count.incrementAndGet();
114     if ("hosta".equals(
115         ra.getAnswer().path(DbHostConfiguration.Columns.HOSTID.name())
116           .asText())) {
117       WaarpSslUtility.closingSslChannel(channel);
118       return false;
119     }
120     // Delete 1
121     HttpTestRestR66Client.deleteData(channel, ra);
122     return true;
123   }

139   @Override
140   protected boolean afterDbOptions(Channel channel, RestArgument ra)
141       throws HttpInvalidAuthenticationException {
142     HttpTestRestR66Client.count.incrementAndGet();
143     boolean newMessage = false;
144     AtomicInteger counter = null;
145     final RestFuture future =
146         channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT).get();
147     if (future.getOtherObject() == null) {
148       counter = new AtomicInteger();
149       future.setOtherObject(counter);
150       final JsonNode node = ra.getDetailedAllowOption();
151       if (!node.isMissingNode()) {
152         for (final JsonNode jsonNode : node) {
153           final Iterator<String> iterator = jsonNode.fieldNames();
154           while (iterator.hasNext()) {
155             final String name = iterator.next();
156             if (!jsonNode.path(name)
157                          .path(RestArgument.REST_FIELD.JSON_PATH.field)
158                          .isMissingNode()) {
159               break;
160             }
161             if (name.equals(RootOptionsRestMethodHandler.ROOT)) {
162               continue;
163             }
164             counter.incrementAndGet();
165             HttpTestRestR66Client.options(channel, name);
166             newMessage = true;
167           }
168         }
169       }
170     }
171     if (!newMessage) {
172       counter = (AtomicInteger) future.getOtherObject();
173       newMessage = counter.decrementAndGet() > 0;
174       if (!newMessage) {
175         future.setOtherObject(null);
176       }
177     }
178     if (!newMessage) {
179       WaarpSslUtility.closingSslChannel(channel);
180     }
181     return newMessage;
182   }
183 
184   @Override
185   protected boolean action(Channel channel, RestArgument ra, ACTIONS_TYPE act) {
186     HttpTestRestR66Client.count.incrementAndGet();
187     boolean newMessage = false;
188     switch (act) {
189       case CreateTransfer: {
190         // Continue with GetTransferInformation
191         TransferRequestJsonPacket recv;
192         try {
193           recv = (TransferRequestJsonPacket) JsonPacket.createFromBuffer(
194               JsonHandler.writeAsString(ra.getResults().get(0)));
195         } catch (final Exception e) {
196           e.printStackTrace();
197           return newMessage;
198         }
199         final InformationJsonPacket node =
200             new InformationJsonPacket(recv.getSpecialId(), false,
201                                       recv.getRequested());
202         HttpTestRestR66Client.action(channel, HttpMethod.GET,
203                                      RESTHANDLERS.Control.uri, node);
204         newMessage = true;
205         break;
206       }
207       case ExecuteBusiness:
208         // End
209         break;
210       case ExportConfig:
211         // no Import in automatic test
212         break;
213       case GetBandwidth:
214         // End
215         break;
216       case GetInformation:
217         // End
218         break;
219       case GetLog:
220         // End
221         break;
222       case GetTransferInformation: {
223         // Continue with Stop in StopOrCancelTransfer
224         final ObjectNode answer = (ObjectNode) ra.getResults().get(0);
225         final StopOrCancelJsonPacket node = new StopOrCancelJsonPacket();
226         node.setRequestUserPacket();
227         node.setStop();
228         node.setRequested(
229             answer.path(DbTaskRunner.Columns.REQUESTED.name()).asText());
230         node.setRequester(
231             answer.path(DbTaskRunner.Columns.REQUESTER.name()).asText());
232         node.setSpecialid(
233             answer.path(DbTaskRunner.Columns.SPECIALID.name()).asLong());
234         HttpTestRestR66Client.action(channel, HttpMethod.PUT,
235                                      RESTHANDLERS.Control.uri, node);
236         newMessage = true;
237         break;
238       }
239       case ImportConfig:
240         // End
241         break;
242       case OPTIONS:
243         break;
244       case RestartTransfer: {
245         // Continue with delete transfer
246         RestartTransferJsonPacket recv;
247         try {
248           recv = (RestartTransferJsonPacket) JsonPacket.createFromBuffer(
249               JsonHandler.writeAsString(ra.getResults().get(0)));
250         } catch (final Exception e) {
251           e.printStackTrace();
252           return newMessage;
253         }
254         try {
255           HttpTestRestR66Client.deleteData(channel, recv.getRequested(),
256                                            recv.getRequester(),
257                                            recv.getSpecialid());
258         } catch (final HttpInvalidAuthenticationException e) {
259           e.printStackTrace();
260         }
261         newMessage = true;
262         break;
263       }
264       case SetBandwidth: {
265         // Continue with GetBandwidth
266         BandwidthJsonPacket recv;
267         try {
268           recv = (BandwidthJsonPacket) JsonPacket.createFromBuffer(
269               JsonHandler.writeAsString(ra.getResults().get(0)));
270         } catch (final Exception e) {
271           e.printStackTrace();
272           return newMessage;
273         }
274         recv.setSetter(false);
275         HttpTestRestR66Client.action(channel, HttpMethod.GET,
276                                      RESTHANDLERS.Bandwidth.uri, recv);
277         newMessage = true;
278         break;
279       }
280       case ShutdownOrBlock:
281         // End
282         break;
283       case StopOrCancelTransfer: {
284         // Continue with RestartTransfer
285         StopOrCancelJsonPacket recv;
286         try {
287           recv = (StopOrCancelJsonPacket) JsonPacket.createFromBuffer(
288               JsonHandler.writeAsString(ra.getResults().get(0)));
289         } catch (final Exception e) {
290           e.printStackTrace();
291           return newMessage;
292         }
293         final RestartTransferJsonPacket node = new RestartTransferJsonPacket();
294         node.setRequestUserPacket();
295         node.setRequested(recv.getRequested());
296         node.setRequester(recv.getRequester());
297         node.setSpecialid(recv.getSpecialid());
298         HttpTestRestR66Client.action(channel, HttpMethod.PUT,
299                                      RESTHANDLERS.Control.uri, node);
300         newMessage = true;
301         break;
302       }
303       case GetStatus:
304         break;
305       default:
306         break;
307 
308     }
309     if (!newMessage) {
310       WaarpSslUtility.closingSslChannel(channel);
311     }
312     return newMessage;
313   }
