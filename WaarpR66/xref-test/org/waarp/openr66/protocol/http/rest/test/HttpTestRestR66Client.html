536   public static void options(String uri)
537       throws HttpInvalidAuthenticationException {
538     final Channel channel = clientHelper.getChannel(host,
539                                                     HttpTestR66PseudoMain.config.getRestPort());
540     if (channel == null) {
541       logger.warn("Cannot connect to: " + host + ':' +
542                   HttpTestR66PseudoMain.config.getRestPort());
543       fail("Cant connect");
544       return;
545     }
546     final RestFuture future = options(channel, uri);
547     future.awaitOrInterruptible();
548     logger.debug("Closing Channel");
549     WaarpSslUtility.closingSslChannel(channel);
550     assertTrue("Action should be ok", future.isSuccess());
551     logger.debug("Channel closed");
552   }
