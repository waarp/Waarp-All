36  package org.waarp.openr66.protocol.http.rest.test;
37  
38  import io.netty.channel.ChannelInitializer;
39  import io.netty.channel.ChannelPipeline;
40  import io.netty.channel.socket.SocketChannel;
41  import io.netty.handler.codec.http.HttpClientCodec;
42  import io.netty.handler.codec.http.HttpContentDecompressor;
43  import io.netty.handler.ssl.SslHandler;
44  import io.netty.handler.stream.ChunkedWriteHandler;
45  import org.waarp.common.crypto.ssl.WaarpSslContextFactory;
46  
47  /**
48   * Test Rest client pipeline factory
49   */
50  public class HttpTestRestClientInitializer
51      extends ChannelInitializer<SocketChannel> {
52    private final WaarpSslContextFactory waarpSslContextFactory;
53  
54    public HttpTestRestClientInitializer(
55        WaarpSslContextFactory waarpSslContextFactory) {
56      this.waarpSslContextFactory = waarpSslContextFactory;
57    }
58  
59    @Override
60    protected void initChannel(SocketChannel ch) throws Exception {
61      // Create a default pipeline implementation.
62      final ChannelPipeline pipeline = ch.pipeline();
63  
64      // Enable HTTPS if necessary.
65      if (waarpSslContextFactory != null) {
66        final SslHandler handler = waarpSslContextFactory.createHandlerClient(ch);
67        pipeline.addLast("ssl", handler);
68      }
69  
70      pipeline.addLast("codec", new HttpClientCodec());
71      // Remove the following line if you don't want automatic content
72      // decompression.
73      pipeline.addLast("inflater", new HttpContentDecompressor());
74  
75      // to be used since huge file transfer
76      pipeline.addLast("streamer", new ChunkedWriteHandler());
77  
78      pipeline.addLast("handler", new HttpTestResponseHandler());
79    }
80  }
