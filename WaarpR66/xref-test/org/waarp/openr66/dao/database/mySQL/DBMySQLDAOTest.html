21  package org.waarp.openr66.dao.database.mySQL;
22  
23  import org.junit.ClassRule;
24  import org.junit.Rule;
25  import org.junit.rules.TestWatcher;
26  import org.testcontainers.containers.JdbcDatabaseContainer;
27  import org.testcontainers.containers.MySQLContainer;
28  import org.waarp.common.utility.TestWatcherJunit4;
29  import org.waarp.openr66.dao.database.DBAllDAOTest;
30  import org.waarp.openr66.dao.database.DBTransferDAO;
31  import org.waarp.openr66.dao.database.mariadb.MariaDBTransferDAO;
32  import org.waarp.openr66.dao.exception.DAOConnectionException;
33  
34  import java.sql.Connection;
35  import java.sql.DriverManager;
36  import java.sql.SQLException;
37  
38  public class DBMySQLDAOTest extends DBAllDAOTest {
39    @Rule(order = Integer.MIN_VALUE)
40    public TestWatcher watchman = new TestWatcherJunit4();
41  
42    static {
43      TMPFSMAP.clear();
44      TMPFSMAP.put("/var/lib/mysql/data", "rw");
45    }
46  
47    @ClassRule
48    public static MySQLContainer db = (MySQLContainer) new MySQLContainer();
49    // No TMPFS
50    private final String createScript = "mysql/create.sql";
51    private final String populateScript = "mysql/populate.sql";
52    private final String cleanScript = "mysql/clean.sql";
53  
54    @Override
55    public DBTransferDAO getDAO(Connection con) throws DAOConnectionException {
56      return new MariaDBTransferDAO(con);
57    }
58  
59    @Override
60    public Connection getConnection() throws SQLException {
61      return DriverManager
62          .getConnection(db.getJdbcUrl(), db.getUsername(), db.getPassword());
63    }
64  
65    @Override
66    public void initDB() {
67      runScript(createScript);
68      runScript(populateScript);
69    }
70  
71    @Override
72    public void cleanDB() {
73      runScript(cleanScript);
74    }
75  
76    @Override
77    public JdbcDatabaseContainer getJDC() {
78      return db;
79    }
80  
81  }
