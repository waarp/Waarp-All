12  public class DbTransferDAOStatistic extends DBTransferDAO {
13    protected static WaarpLogger logger =
14        WaarpLoggerFactory.getLogger(DbTransferDAOStatistic.class);
15    private final static String PG_EXCLUDE = "pg_";
16    private final DBTransferDAO dbTransferDAO;
17  
18    public DbTransferDAOStatistic(final DBTransferDAO transferDAO) {
19      super(transferDAO.connection);
20      dbTransferDAO = transferDAO;
21    }
22  
23    @Override
24    protected long getNextId() throws DAOConnectionException {
25      return 0;
26    }
27  
28    public void initStatistics() throws SQLException {
29      try {
30        final Statement statement = connection.createStatement();
31        statement.execute("CREATE EXTENSION IF NOT EXISTS pg_stat_statements");
32        statement.close();
33      } catch (final SQLException e) {
34        logger.warn(e);
35      }
36    }
37  
38    public void executeStatistics() throws SQLException {
39      final String hotQueries = "select \n" + "\tquery, \n" + "\tcalls,\n" +
40                                "\ttotal_exec_time::integer, \n" +
41                                "\t(calls/total_exec_time)::integer as " +
42                                "ms_per_call, \n" + "\tshared_blks_hit, \n" +
43                                "\tshared_blks_read \n" +
44                                "from pg_stat_statements pss\n" +
45                                "order by calls desc\n" + "limit 100";
46      final String slowQueries =
47          "select query, calls, (total_exec_time/calls)::integer as avg_time_ms" +
48          " \n" + "from pg_stat_statements\n" + "where calls > 1000\n" +
49          "order by avg_time_ms desc\n" + "limit 100";
50      final String slowLowQueries =
51          "select query, calls, (total_exec_time/calls)::integer as avg_time_ms" +
52          " \n" + "from pg_stat_statements\n" + "where calls > 1000\n" +
53          "and rows < 50\n" + "order by avg_time_ms desc\n" + "limit 100";
54      final String topQueries = "SELECT query,\n" + "      calls,\n" +
55                                "      round(total_exec_time::numeric, 2) AS total_time,\n" +
56                                "      round(mean_exec_time::numeric, 2) AS mean_time,\n" +
57                                "      round((100 * total_exec_time / sum(total_exec_time) OVER ())::numeric, 2) AS percentage\n" +
58                                "FROM pg_stat_statements\n" +
59                                "ORDER BY total_exec_time DESC\n" + "LIMIT 10";
60      try {
61        final Statement statement = connection.createStatement();
62        final ResultSet resultSet = statement.executeQuery(slowQueries);
63        SysErrLogger.FAKE_LOGGER.sysout("Calls\tAvgTime\tAll Query");
64        while (resultSet.next()) {
65          final String query = resultSet.getString(1);
66          if (!query.contains(PG_EXCLUDE)) {
67            SysErrLogger.FAKE_LOGGER.sysout(
68                resultSet.getLong(2) + "\t" + resultSet.getLong(3) + "\t" +
69                query);
70          }
71        }
72        resultSet.close();
73        final ResultSet resultSet2 = statement.executeQuery(slowLowQueries);
74        SysErrLogger.FAKE_LOGGER.sysout("Calls\tAvgTime\tSmall Query");
75        while (resultSet2.next()) {
76          final String query = resultSet2.getString(1);
77          if (!query.contains(PG_EXCLUDE)) {
78            SysErrLogger.FAKE_LOGGER.sysout(
79                resultSet2.getLong(2) + "\t" + resultSet2.getLong(3) + "\t" +
80                query);
81          }
82        }
83        resultSet2.close();
84        final ResultSet resultSet4 = statement.executeQuery(topQueries);
85        SysErrLogger.FAKE_LOGGER.sysout(
86            "Calls\tTotalTime\tAverage\tPercent\tTop Query");
87        while (resultSet4.next()) {
88          final String query = resultSet4.getString(1);
89          if (!query.contains(PG_EXCLUDE)) {
90            SysErrLogger.FAKE_LOGGER.sysout(
91                resultSet4.getLong(2) + "\t" + resultSet4.getLong(3) + "\t" +
92                resultSet4.getLong(4) + "\t" + resultSet4.getLong(5) + "\t" +
93                query);
94          }
95        }
96        resultSet4.close();
97        final ResultSet resultSet3 = statement.executeQuery(hotQueries);
98        SysErrLogger.FAKE_LOGGER.sysout(
99            "Calls\tTotalTime\tMsPerCall\tHit\tRead\tHot Query");
100       while (resultSet3.next()) {
101         final String query = resultSet3.getString(1);
102         if (!query.contains(PG_EXCLUDE)) {
103           SysErrLogger.FAKE_LOGGER.sysout(
104               resultSet3.getLong(2) + "\t" + resultSet3.getLong(3) + "\t" +
105               resultSet3.getLong(4) + "\t" + resultSet3.getLong(5) + "\t" +
106               resultSet3.getLong(6) + "\t" + query);
107         }
108       }
109       resultSet3.close();
110       statement.close();
111     } catch (final SQLException e) {
112       e.printStackTrace();
113     }
114   }
