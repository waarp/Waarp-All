105   @Test
106   public void testMapCapture() {
107     String testNoMap = "test1 test 2";
108     Map<String, Object> map = DbTaskRunner.getMapFromString(testNoMap);
109     String nomap = DbTaskRunner.getOutOfMapFromString(testNoMap);
110     assertEquals(testNoMap, nomap);
111     assertEquals(0, map.size());
112     logger.warn("{} = {} + {}", testNoMap, map, nomap);
113 
114     String testSingleMap = "{}";
115     map = DbTaskRunner.getMapFromString(testSingleMap);
116     nomap = DbTaskRunner.getOutOfMapFromString(testSingleMap);
117     assertEquals("", nomap);
118     assertEquals(0, map.size());
119     logger.warn("{} = {} + {}", testSingleMap, map, nomap);
120 
121     String testSingleMapKey = "{'key': 'value'}";
122     map = DbTaskRunner.getMapFromString(testSingleMapKey);
123     nomap = DbTaskRunner.getOutOfMapFromString(testSingleMapKey);
124     assertEquals("", nomap);
125     assertEquals(1, map.size());
126     logger.warn("{} = {} + {}", testSingleMapKey, map, nomap);
127 
128     String testMultipleMap = "{} {}";
129     map = DbTaskRunner.getMapFromString(testMultipleMap);
130     nomap = DbTaskRunner.getOutOfMapFromString(testMultipleMap);
131     assertEquals(" ", nomap);
132     assertEquals(0, map.size());
133     logger.warn("{} = {} + {}", testMultipleMap, map, nomap);
134 
135     String testMultipleMapKey = "{'key': 'value'} {}";
136     map = DbTaskRunner.getMapFromString(testMultipleMapKey);
137     nomap = DbTaskRunner.getOutOfMapFromString(testMultipleMapKey);
138     assertEquals(" ", nomap);
139     assertEquals(1, map.size());
140     logger.warn("{} = {} + {}", testMultipleMapKey, map, nomap);
141 
142     String testMultipleMapKeyReverse = "{} {'key': 'value'} ";
143     map = DbTaskRunner.getMapFromString(testMultipleMapKeyReverse);
144     nomap = DbTaskRunner.getOutOfMapFromString(testMultipleMapKeyReverse);
145     assertEquals("  ", nomap);
146     assertEquals(1, map.size());
147     logger.warn("{} = {} + {}", testMultipleMapKeyReverse, map, nomap);
148 
149     String testMultipleMapKeys =
150         "{} {'key': 'value'}{'key2': 'value'}{'key3': 'value'}";
151     map = DbTaskRunner.getMapFromString(testMultipleMapKeys);
152     nomap = DbTaskRunner.getOutOfMapFromString(testMultipleMapKeys);
153     assertEquals(" ", nomap);
154     assertEquals(3, map.size());
155     logger.warn("{} = {} + {}", testMultipleMapKeys, map, nomap);
156 
157     String testMultipleMap2Key = "{'key': 'value'} {'key2': 'value2'}";
158     map = DbTaskRunner.getMapFromString(testMultipleMap2Key);
159     nomap = DbTaskRunner.getOutOfMapFromString(testMultipleMap2Key);
160     assertEquals(" ", nomap);
161     assertEquals(2, map.size());
162     logger.warn("{} = {} + {}", testMultipleMap2Key, map, nomap);
163 
164     String testMultipleMap2SameKey = "{'key': 'value'} {'key': 'value2'}";
165     map = DbTaskRunner.getMapFromString(testMultipleMap2SameKey);
166     nomap = DbTaskRunner.getOutOfMapFromString(testMultipleMap2SameKey);
167     assertEquals(" ", nomap);
168     assertEquals(1, map.size());
169     logger.warn("{} = {} + {}", testMultipleMap2SameKey, map, nomap);
170 
171     String testMultipleMap2SameKeyAndNotMap =
172         "extra information {'key': 'value'} other" +
173         "{'key2': 'value2'}and another one";
174     map = DbTaskRunner.getMapFromString(testMultipleMap2SameKeyAndNotMap);
175     nomap =
176         DbTaskRunner.getOutOfMapFromString(testMultipleMap2SameKeyAndNotMap);
177     assertEquals("extra information  other" + "and another one", nomap);
178     assertEquals(2, map.size());
179     logger.warn("{} = {} + {}", testMultipleMap2SameKeyAndNotMap, map, nomap);
180 
181 
182     String testMultipleMap2SameKeyAndNotMapEnding =
183         "extra information {'key': 'value'} other" + "{'key2': 'value2'}";
184     map = DbTaskRunner.getMapFromString(testMultipleMap2SameKeyAndNotMapEnding);
185     nomap = DbTaskRunner.getOutOfMapFromString(
186         testMultipleMap2SameKeyAndNotMapEnding);
187     assertEquals("extra information  other", nomap);
188     assertEquals(2, map.size());
189     logger.warn("{} = {} + {}", testMultipleMap2SameKeyAndNotMapEnding, map,
190                 nomap);
191 
192     String testMultipleMap2SameKeyAndTrueNotMapEnding =
193         "extra information {'key': 'value'} other" + " {'key2': 'value2'";
194     map = DbTaskRunner.getMapFromString(
195         testMultipleMap2SameKeyAndTrueNotMapEnding);
196     nomap = DbTaskRunner.getOutOfMapFromString(
197         testMultipleMap2SameKeyAndTrueNotMapEnding);
198     assertEquals("extra information  other {'key2': 'value2'", nomap);
199     assertEquals(1, map.size());
200     logger.warn("{} = {} + {}", testMultipleMap2SameKeyAndTrueNotMapEnding, map,
201                 nomap);
202   }
