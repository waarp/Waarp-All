20  package org.waarp.openr66.commander;
21  
22  import org.waarp.common.database.exception.WaarpDatabaseNoConnectionException;
23  import org.waarp.common.database.exception.WaarpDatabaseSqlException;
24  import org.waarp.common.logging.WaarpLogger;
25  import org.waarp.common.logging.WaarpLoggerFactory;
26  import org.waarp.common.utility.WaarpThreadFactory;
27  import org.waarp.openr66.database.data.DbTaskRunner;
28  import org.waarp.openr66.protocol.configuration.Configuration;
29  import org.waarp.openr66.protocol.networkhandler.NetworkTransaction;
30  
31  import java.util.concurrent.BlockingQueue;
32  import java.util.concurrent.Executors;
33  import java.util.concurrent.RejectedExecutionHandler;
34  import java.util.concurrent.ScheduledExecutorService;
35  import java.util.concurrent.ScheduledFuture;
36  import java.util.concurrent.SynchronousQueue;
37  import java.util.concurrent.ThreadPoolExecutor;
38  import java.util.concurrent.TimeUnit;
39  
40  /**
41   * This class launch and control the Commander and enable TaskRunner job
42   * submissions
43   */
44  public class InternalRunner {
45    /**
46     * Internal Logger
47     */
48    private static final WaarpLogger logger =
49        WaarpLoggerFactory.getLogger(InternalRunner.class);
50  
51    private final ScheduledExecutorService scheduledExecutorService;
52    private ScheduledFuture<?> scheduledFuture;
53    private CommanderInterface commander;
54    private volatile boolean isRunning = true;
55    private final ThreadPoolExecutor threadPoolExecutor;
56    private final NetworkTransaction networkTransaction;
57  
58    /**
59     * Create the structure to enable submission by database
60     *
61     * @throws WaarpDatabaseNoConnectionException
62     * @throws WaarpDatabaseSqlException
63     */
64    public InternalRunner()
65        throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
66      commander = new Commander(this, true);
67      // This is not daemon intentionally
68      scheduledExecutorService = Executors.newSingleThreadScheduledExecutor(
69          new WaarpThreadFactory("InternalRunner", false));
70      isRunning = true;
71      final BlockingQueue<Runnable> workQueue = new SynchronousQueue<Runnable>();
72      threadPoolExecutor =
73          new ThreadPoolExecutor(Configuration.configuration.getRunnerThread(),
74                                 Configuration.configuration.getRunnerThread() *
75                                 2, 1000, TimeUnit.MILLISECONDS, workQueue,
76                                 new WaarpThreadFactory("ClientRunner"),
77                                 new RejectedExecutionHandler() {
78                                   @Override
79                                   public void rejectedExecution(
80                                       final Runnable runnable,
81                                       final ThreadPoolExecutor threadPoolExecutor) {
82                                     logger.debug("Task rescheduled");
83                                   }
84                                 });
85      scheduledFuture = scheduledExecutorService.scheduleWithFixedDelay(commander,
86                                                                        Configuration.configuration
87                                                                            .getDelayCommander(),
88                                                                        Configuration.configuration
89                                                                            .getDelayCommander(),
90                                                                        TimeUnit.MILLISECONDS);
91      networkTransaction = new NetworkTransaction();
92    }
93  
94    public NetworkTransaction getNetworkTransaction() {
95      return networkTransaction;
96    }
97  
98    public int allowedToSubmit() {
99      final int active = threadPoolExecutor.getActiveCount();
100     if ((isRunning || !Configuration.configuration.isShutdown()) &&
101         (active < Configuration.configuration.getRunnerThread())) {
102       return Configuration.configuration.getRunnerThread() - active;
103     }
104     return 0;
105   }
106 
107   /**
108    * Submit a task
109    *
110    * @param taskRunner
111    */
112   public void submitTaskRunner(final DbTaskRunner taskRunner) {
113     if (isRunning || !Configuration.configuration.isShutdown()) {
114       // last check: number can have raised up since Commander checks
115       if (threadPoolExecutor.getActiveCount() <
116           Configuration.configuration.getRunnerThread()) {
117         logger.debug("Will run {}", taskRunner);
118         final ClientRunner runner =
119             new ClientRunner(networkTransaction, taskRunner, null);
120         if (taskRunner.isSendThrough() && (taskRunner.isRescheduledTransfer() ||
121                                            taskRunner.isPreTaskStarting())) {
122           runner.setSendThroughMode();
123           taskRunner.checkThroughMode();
124         }
125         // create the client, connect and run
126         threadPoolExecutor.execute(runner);
127       } else {
128         // too many current active threads
129         logger.debug("Task rescheduled {}", taskRunner);
130       }
131     }
132   }
133 
134   /**
135    * First step while shutting down the service
136    */
137   public void prepareStopInternalRunner() {
138     isRunning = false;
139     scheduledFuture.cancel(false);
140     if (commander != null) {
141       commander.finalizeCommander();
142     }
143     scheduledExecutorService.shutdown();
144     threadPoolExecutor.shutdown();
145   }
146 
147   /**
148    * This should be called when the server is shutting down, after stopping
149    * active requests if possible.
150    */
151   public void stopInternalRunner() {
152     isRunning = false;
153     logger.info("Stopping Commander and Runner Tasks");
154     scheduledFuture.cancel(true);
155     if (commander != null) {
156       commander.finalizeCommander();
157     }
158     scheduledExecutorService.shutdownNow();
159     threadPoolExecutor.shutdownNow();
160     networkTransaction.closeAll(false);
161   }
162 
163   public int nbInternalRunner() {
164     return threadPoolExecutor.getActiveCount();
165   }
166 
167   public void reloadInternalRunner()
168       throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
169     scheduledFuture.cancel(false);
170     if (commander != null) {
171       commander.finalizeCommander();
172     }
173     commander = new Commander(this);
174     scheduledFuture = scheduledExecutorService.scheduleWithFixedDelay(commander,
175                                                                       2 *
176                                                                       Configuration.configuration
177                                                                           .getDelayCommander(),
178                                                                       Configuration.configuration
179                                                                           .getDelayCommander(),
180                                                                       TimeUnit.MILLISECONDS);
181   }
182 }
