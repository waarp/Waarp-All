20  package org.waarp.openr66.server;

22  import org.waarp.common.logging.SysErrLogger;
23  import org.waarp.common.logging.WaarpLogger;
24  import org.waarp.common.logging.WaarpLoggerFactory;
25  import org.waarp.common.logging.WaarpSlf4JLoggerFactory;
26  import org.waarp.common.utility.DetectionUtils;
27  import org.waarp.openr66.configuration.FileBasedConfiguration;
28  import org.waarp.openr66.database.data.DbTaskRunner;
29  import org.waarp.openr66.protocol.configuration.Configuration;
30  import org.waarp.openr66.protocol.exception.OpenR66ProtocolBusinessException;
31  import org.waarp.openr66.protocol.utils.ChannelUtils;
32  
33  import java.io.File;
34  
35  import static org.waarp.common.database.DbConstant.*;
36  
37  /**
38   * To import logs that were exported (one should not try to import on the same
39   * server to prevent strange
40   * behavior).
41   */
42  public class LogImport {
43    /**
44     * Internal Logger
45     */
46    private static WaarpLogger logger;
47  
48    /**
49     * @param args
50     */
51    public static void main(final String[] args) {
52      WaarpLoggerFactory
53          .setDefaultFactoryIfNotSame(new WaarpSlf4JLoggerFactory(null));
54      if (logger == null) {
55        logger = WaarpLoggerFactory.getLogger(LogImport.class);
56      }
57      if (args.length < 2) {
58        SysErrLogger.FAKE_LOGGER
59            .syserr("Need configuration file and the logfile to import");
60        System.exit(1);//NOSONAR
61      }
62      try {
63        if (!FileBasedConfiguration
64            .setConfigurationServerMinimalFromXml(Configuration.configuration,
65                                                  args[0])) {
66          logger.error("Needs a correct configuration file as first argument");
67          if (admin != null) {
68            admin.close();
69          }
70          if (DetectionUtils.isJunit()) {
71            return;
72          }
73          ChannelUtils.stopLogger();
74          System.exit(1);//NOSONAR
75          return;
76        }
77        final long time1 = System.currentTimeMillis();
78        final File logsFile = new File(args[1]);
79        try {
80          DbTaskRunner.loadXml(logsFile);
81        } catch (final OpenR66ProtocolBusinessException e) {
82          logger.error("Cannot load the logs from " + logsFile.getAbsolutePath() +
83                       " since: " + e.getMessage(), e);
84          if (admin != null) {
85            admin.close();
86          }
87          if (DetectionUtils.isJunit()) {
88            return;
89          }
90          ChannelUtils.stopLogger();
91          System.exit(1);//NOSONAR
92          return;
93        }
94        final long time2 = System.currentTimeMillis();
95        final long delay = time2 - time1;
96        logger.warn("LogFile imported in " + delay + " ms");
97      } finally {
98        if (admin != null) {
99          admin.close();
100       }
101       System.exit(0);//NOSONAR
102     }
103   }
104 }
