106   protected static boolean getParams(String[] args) {
107     if (logger == null) {
108       logger = WaarpLoggerFactory.getLogger(RequestTransfer.class);
109     }
110     _infoArgs = Messages.getString("RequestTransfer.0") +
111                 Messages.getString("Message.OutputFormat"); //$NON-NLS-1$
112     if (args.length < 5) {
113       logger.error(_infoArgs);
114       return false;
115     }
116     if (!FileBasedConfiguration
117         .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
118       logger.error(
119           Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
120       return false;
121     }
122     for (int i = 1; i < args.length; i++) {
123       if ("-id".equalsIgnoreCase(args[i])) {
124         i++;
125         try {
126           sspecialId = Long.parseLong(args[i]);
127         } catch (final NumberFormatException e) {
128           logger.error(Messages.getString("RequestTransfer.1") + args[i],
129                        e); //$NON-NLS-1$
130           return false;
131         }
132       } else if ("-to".equalsIgnoreCase(args[i])) {
133         i++;
134         srequested = args[i];
135         if (Configuration.configuration.getAliases().containsKey(srequested)) {
136           srequested = Configuration.configuration.getAliases().get(srequested);
137         }
138         rhost = srequested;
139         try {
140           srequester = Configuration.configuration.getHostId(srequested);
141         } catch (final WaarpDatabaseException e) {
142           logger.error(Messages.getString("RequestTransfer.5") + srequested,
143                        e); //$NON-NLS-1$
144           return false;
145         }
146       } else if ("-from".equalsIgnoreCase(args[i])) {
147         i++;
148         srequester = args[i];
149         if (Configuration.configuration.getAliases().containsKey(srequester)) {
150           srequester = Configuration.configuration.getAliases().get(srequester);
151         }
152         rhost = srequester;
153         try {
154           srequested = Configuration.configuration.getHostId(srequester);
155         } catch (final WaarpDatabaseException e) {
156           logger.error(Messages.getString("RequestTransfer.5") + srequester,
157                        e); //$NON-NLS-1$
158           return false;
159         }
160       } else if ("-cancel".equalsIgnoreCase(args[i])) {
161         scancel = true;
162       } else if ("-stop".equalsIgnoreCase(args[i])) {
163         sstop = true;
164       } else if ("-restart".equalsIgnoreCase(args[i])) {
165         srestart = true;
166       } else if ("-start".equalsIgnoreCase(args[i])) {
167         i++;
168         srestarttime = args[i];
169       } else if ("-logWarn".equalsIgnoreCase(args[i])) {
170         snormalInfoAsWarn = true;
171       } else if ("-notlogWarn".equalsIgnoreCase(args[i])) {
172         snormalInfoAsWarn = false;
173       } else if ("-delay".equalsIgnoreCase(args[i])) {
174         i++;
175         final SimpleDateFormat dateFormat =
176             new SimpleDateFormat("yyyyMMddHHmmss");
177         if (args[i].charAt(0) == '+') {
178           final Date date = new Date(System.currentTimeMillis() +
179                                      Long.parseLong(args[i].substring(1)));
180           srestarttime = dateFormat.format(date);
181         } else {
182           final Date date = new Date(Long.parseLong(args[i]));
183           srestarttime = dateFormat.format(date);
184         }
185       }
186       OutputFormat.getParams(args);
187     }
188     if (scancel && srestart || scancel && sstop || srestart && sstop) {
189       logger.error(
190           Messages.getString("RequestTransfer.15") + _infoArgs); //$NON-NLS-1$
191       return false;
192     }
193     if (sspecialId == ILLEGALVALUE || srequested == null) {
194       logger.error(
195           Messages.getString("RequestTransfer.16") + _infoArgs); //$NON-NLS-1$
196       return false;
197     }
198 
199     return true;
200   }

419               break;
420           }
421         }
422       } else {
423         // Only request

426         future.setResult(
427             new R66Result(null, true, runner.getErrorInfo(), runner));

429       }

431       if (!future.isDone()) {
432         if (runner != null) {
433           // Default set to success

435               new R66Result(null, true, runner.getErrorInfo(), runner));
436           future.setSuccess();
437         } else {
438           future.setFailure(new WaarpDatabaseException(
439               "Transfer requested but internal error"));
440         }
441       }
442     }

467       future.setFailure(e);
468       return ErrorCode.Internal;
469     }
470     // check if requester is "client" so no connect from him but direct action
471     logger.debug("Requester Host isClient: " + host.isClient());
472     if (host.isClient()) {
473       if (code == LocalPacketFactory.VALIDPACKET) {
474         logger.info(Messages.getString("RequestTransfer.42") + //$NON-NLS-1$

478             new DirectTransfer(transfer, runner.getRequested(),
479                                runner.getOriginalFilename(), runner.getRuleId(),
480                                runner.getFileInformation(), false,
481                                runner.getBlocksize(), runner.getSpecialId(),

483         transaction.normalInfoAsWarn = normalInfoAsWarn;
484         transaction.run();
485         transfer.awaitOrInterruptible();

487             "Request done with " + (transfer.isSuccess()? "success" : "error"));

491           future.getResult().setRunner(runner);
492           future.setSuccess();
