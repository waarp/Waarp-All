67    private DbTaskRunner initRequest(R66Request request) {
68      Timestamp ttimestart = null;
69      if (request.isSetStart()) {
70        Date date;
71        try {
72          final SimpleDateFormat dateFormat =
73              new SimpleDateFormat("yyyyMMddHHmmss");
74          date = dateFormat.parse(request.getStart());
75          ttimestart = new Timestamp(date.getTime());
76        } catch (final ParseException ignored) {
77          // nothing
78        }
79      } else if (request.isSetDelay()) {
80        if (request.getDelay().charAt(0) == '+') {
81          ttimestart = new Timestamp(System.currentTimeMillis() + Long.parseLong(
82              request.getDelay().substring(1)));
83        } else {
84          ttimestart = new Timestamp(Long.parseLong(request.getDelay()));
85        }
86      }
87      DbRule rule;
88      try {
89        rule = new DbRule(request.getRule());
90      } catch (final WaarpDatabaseException e) {
91        logger.warn("Cannot get Rule: " + request.getRule(), e);
92        return null;
93      }
94      int mode = rule.getMode();
95      if (request.isMd5()) {
96        mode = RequestPacket.getModeMD5(mode);
97      }
98      DbTaskRunner taskRunner;
99      long tid = ILLEGALVALUE;
100     if (request.isSetTid()) {
101       tid = request.getTid();
102     }
103     if (tid != ILLEGALVALUE) {
104       try {
105         taskRunner = new DbTaskRunner(tid, request.getDestuid());
106         // requested
107         taskRunner.setSenderByRequestToValidate(true);
108       } catch (final WaarpDatabaseException e) {
109         logger.warn("Cannot get task", e);
110         return null;
111       }
112     } else {
113       final String sep =
114           PartnerConfiguration.getSeparator(request.getDestuid());
115       final RequestPacket requestPacket =
116           new RequestPacket(request.getRule(), mode, request.getFile(),
117                             request.getBlocksize(), 0, tid, request.getInfo(),
118                             -1, sep);
119       // Not isRecv since it is the requester, so send => isRetrieve is true
120       final boolean isRetrieve =
121           !RequestPacket.isRecvMode(requestPacket.getMode());
122       try {
123         taskRunner = new DbTaskRunner(rule, isRetrieve, requestPacket,
124                                       request.getDestuid(), ttimestart);
125       } catch (final WaarpDatabaseException e) {
126         logger.warn("Cannot get task", e);
127         return null;
128       }
129     }
130     return taskRunner;
131   }

133   @Override
134   public R66Result transferRequestQuery(R66Request request) throws TException {
135     final DbTaskRunner runner = initRequest(request);
136     if (runner != null) {
137       runner.changeUpdatedInfo(AbstractDbData.UpdatedInfo.TOSUBMIT);
138       final boolean isSender = runner.isSender();
139       if (!runner.forceSaveStatus()) {
140         logger.warn("Cannot prepare task");
141         return new R66Result(request.getMode(), ErrorCode.CommandNotFound,
142                              "ERROR: Cannot prepare transfer");
143       }
144       final R66Result result =
145           new R66Result(request.getMode(), ErrorCode.InitOk,
146                         "Transfer Scheduled");
147       if (request.getMode() == RequestMode.SYNCTRANSFER) {
148         // now need to wait but first, reload the runner
149         try {
150           runner.select();
151           while (!runner.isFinished()) {
152             try {
153               Thread.sleep(1000);
154               runner.select();
155             } catch (final InterruptedException e) {//NOSONAR
156               SysErrLogger.FAKE_LOGGER.ignoreLog(e);
157               break;
158             }
159           }
160           runner.setSender(isSender);
161         } catch (final WaarpDatabaseException ignored) {
162           // nothing
163         }
164         setResultFromRunner(runner, result);
165         if (runner.isAllDone()) {
166           result.setCode(ErrorCode.CompleteOk);
167           result.setResultinfo("Transfer Done");
168         } else {
169           result.setCode(ErrorCode.valueOf(runner.getErrorInfo().name()));
170           result.setResultinfo(runner.getErrorInfo().getMesg());
171         }
172       } else {
173         try {
174           runner.select();
175         } catch (final WaarpDatabaseException ignored) {
176           // nothing
177         }
178         runner.setSender(isSender);
179         setResultFromRunner(runner, result);
180       }
181       return result;
182     } else {
183       logger.warn("ERROR: Transfer NOT scheduled");
184       return new R66Result(request.getMode(), ErrorCode.Internal,
185                            "ERROR: Transfer NOT scheduled");
186     }
187   }

189   private void setResultFromRunner(DbTaskRunner runner, R66Result result) {
190     result.setDestuid(runner.getRequested());
191     result.setFromuid(runner.getRequester());
192     result.setTid(runner.getSpecialId());
193     result.setRule(runner.getRuleId());
194     result.setBlocksize(runner.getBlocksize());
195     result.setFile(runner.getFilename());
196     result.setOriginalfilename(runner.getOriginalFilename());
197     result.setIsmoved(runner.isFileMoved());
198     result.setModetransfer(runner.getMode());
199     result.setRetrievemode(runner.isSender());
200     result.setStep(runner.getStep());
201     result.setGloballaststep(runner.getGloballaststep());
202     result.setRank(runner.getRank());
203     result.setStart(runner.getStart().toString());
204     result.setStop(runner.getStop().toString());
205     result.setResultinfo(runner.getFileInformation());
206   }

208   private void setResultFromLCR(LocalChannelReference lcr, R66Result result) {
209     final R66Session session = lcr.getSession();
210     DbTaskRunner runner = null;
211     if (session != null) {
212       runner = session.getRunner();
213     } else {
214       final ClientRunner run = lcr.getClientRunner();
215       if (run != null) {
216         runner = run.getTaskRunner();
217       }
218     }
219     if (runner != null) {
220       setResultFromRunner(runner, result);
221     }
222   }

224   private int stopOrCancelRunner(long id, String reqd, String reqr,
225                                  org.waarp.openr66.context.ErrorCode code) {
226     try {
227       final DbTaskRunner taskRunner =
228           new DbTaskRunner(null, null, id, reqr, reqd);
229       return taskRunner.stopOrCancelRunner(code)? 1 : 0;
230     } catch (final WaarpDatabaseException ignored) {
231       // nothing
232     }
233     logger
234         .warn("Cannot accomplished action on task: " + id + ' ' + code.name());
235     return -1;
236   }

238   private R66Result stopOrCancel(R66Request request, LocalChannelReference lcr,
239                                  org.waarp.openr66.context.ErrorCode r66code) {
240     // stop the current transfer
241     R66Result resulttest;
242     if (lcr != null) {
243       int rank = 0;
244       if (r66code == org.waarp.openr66.context.ErrorCode.StoppedTransfer &&
245           lcr.getSession() != null) {
246         final DbTaskRunner taskRunner = lcr.getSession().getRunner();
247         if (taskRunner != null) {
248           rank = taskRunner.getRank();
249         }
250       }
251       final ErrorPacket error =
252           new ErrorPacket(r66code.name() + ' ' + rank, r66code.getCode(),
253                           ErrorPacket.FORWARDCLOSECODE);
254       try {
255         // inform local instead of remote
256         LocalServerHandler.channelRead0(lcr, error);
257       } catch (final Exception ignored) {
258         // nothing
259       }
260       resulttest = new R66Result(request.getMode(), ErrorCode.CompleteOk,
261                                  r66code.name());
262       setResultFromLCR(lcr, resulttest);
263     } else {
264       // Transfer is not running
265       // but maybe need action on database
266       final int test =
267           stopOrCancelRunner(request.getTid(), request.getDestuid(),
268                              request.getFromuid(), r66code);
269       if (test > 0) {
270         resulttest = new R66Result(request.getMode(), ErrorCode.CompleteOk,
271                                    r66code.name());
272       } else if (test == 0) {
273         resulttest = new R66Result(request.getMode(), ErrorCode.TransferOk,
274                                    r66code.name());
275       } else {
276         resulttest = new R66Result(request.getMode(), ErrorCode.CommandNotFound,
277                                    "Error: cannot accomplished task on transfer");
278       }
279     }
280     return resulttest;
281   }

283   private R66Result restart(R66Request request, LocalChannelReference lcr) {
284     // Try to validate a restarting transfer
285     // validLimit on requested side
286     if (Configuration.configuration.getConstraintLimitHandler()
287                                    .checkConstraints()) {
288       logger
289           .warn("Limit exceeded {} while asking to relaunch a task " + request,
290                 Configuration.configuration
291                     .getConstraintLimitHandler().lastAlert);
292       return new R66Result(request.getMode(), ErrorCode.ServerOverloaded,
293                            "Limit exceeded while asking to relaunch a task");
294     }
295     // Try to validate a restarting transfer
296     // header = ?; middle = requested+blank+requester+blank+specialId
297     DbTaskRunner taskRunner;
298     try {
299       taskRunner =
300           new DbTaskRunner(null, null, request.getTid(), request.getFromuid(),
301                            request.getDestuid());
302       final org.waarp.openr66.context.R66Result resulttest =
303           TransferUtils.restartTransfer(taskRunner, lcr);
304       return new R66Result(request.getMode(),
305                            ErrorCode.valueOf(resulttest.getCode().name()),
306                            resulttest.getMessage());
307     } catch (final WaarpDatabaseException e1) {
308       logger.warn("Exception while trying to restart transfer", e1);
309       return new R66Result(request.getMode(), ErrorCode.Internal,
310                            "Exception while trying to restart transfer");
311     }
312   }

314   @Override
315   public R66Result infoTransferQuery(R66Request request) throws TException {
316     final RequestMode mode = request.getMode();
317     if (mode != RequestMode.INFOREQUEST) {
318       // error
319       logger.warn("Mode is uncompatible with infoTransferQuery");
320       return new R66Result(request.getMode(), ErrorCode.Unimplemented,
321                            "Mode is uncompatible with infoTransferQuery");
322     }
323     // now check if enough arguments are provided
324     if (!request.isSetTid() ||
325         !request.isSetDestuid() && !request.isSetFromuid() ||
326         !request.isSetAction()) {
327       // error
328       logger.warn("Not enough arguments");
329       return new R66Result(request.getMode(), ErrorCode.RemoteError,
330                            "Not enough arguments");
331     }
332     // requested+blank+requester+blank+specialId
333     final LocalChannelReference lcr =
334         Configuration.configuration.getLocalTransaction().getFromRequest(
335             request.getDestuid() + ' ' + request.getFromuid() + ' ' +
336             request.getTid());
337     org.waarp.openr66.context.ErrorCode r66code;
338     switch (request.getAction()) {
339       case Detail: {
340         final R66Result result =
341             new R66Result(request.getMode(), ErrorCode.CompleteOk,
342                           "Existence test OK");
343         result.setAction(Action.Exist);
344         result.setDestuid(request.getDestuid());
345         result.setFromuid(request.getFromuid());
346         result.setTid(request.getTid());
347         if (lcr != null) {
348           setResultFromLCR(lcr, result);
349         } else {
350           try {
351             final DbTaskRunner runner =
352                 new DbTaskRunner(null, null, request.getTid(),
353                                  request.getFromuid(), request.getDestuid());
354             if (runner != null) {
355               setResultFromRunner(runner, result);
356             }
357           } catch (final WaarpDatabaseException e) {
358             result.setCode(ErrorCode.FileNotFound);
359           }
360         }
361         return result;
362       }
363       case Restart:
364         return restart(request, lcr);
365       case Cancel:
366         r66code = org.waarp.openr66.context.ErrorCode.CanceledTransfer;
367         return stopOrCancel(request, lcr, r66code);
368       case Stop:
369         r66code = org.waarp.openr66.context.ErrorCode.StoppedTransfer;
370         return stopOrCancel(request, lcr, r66code);
371       default:
372         logger.warn("Uncompatible with " + request.getAction().name());
373         return new R66Result(request.getMode(), ErrorCode.Unimplemented,
374                              "Uncompatible with " + request.getAction().name());
375     }
376   }

394   @Override
395   public List<String> infoListQuery(R66Request request) throws TException {
396     List<String> list = new ArrayList<String>();
397     final RequestMode mode = request.getMode();
398     if (mode != RequestMode.INFOFILE) {
399       // error
400       logger.warn("Not correct mode for infoListQuery");
401       list.add("Not correct mode for infoListQuery");
402       return list;
403     }
404     // now check if enough arguments are provided
405     if (!request.isSetRule() || !request.isSetAction()) {
406       // error
407       logger.warn("Not enough arguments");
408       list.add("Not enough arguments");
409       return list;
410     }
411     final R66Sessionion.html#R66Session">R66Session session = new R66Session();
412     session.getAuth().specialNoSessionAuth(false, Configuration.configuration
413         .getHostId());
414     DbRule rule;
415     try {
416       rule = new DbRule(request.getRule());
417     } catch (final WaarpDatabaseException e) {
418       logger.warn("Rule is unknown: " + request.getRule());
419       list.add("Rule is unknown: " + request.getRule());
420       return list;
421     }
422     try {
423       if (RequestPacket.isRecvMode(rule.getMode())) {
424         session.getDir().changeDirectory(rule.getWorkPath());
425       } else {
426         session.getDir().changeDirectory(rule.getSendPath());
427       }
428 
429       if (request.getAction() == Action.List ||
430           request.getAction() == Action.Mlsx) {
431         // ls or mls from current directory
432         if (request.getAction() == Action.List) {
433           list = session.getDir().list(request.getFile());
434         } else {
435           list = session.getDir().listFull(request.getFile(), false);
436         }
437       } else {
438         // ls pr mls from current directory and filename
439         if (!request.isSetFile()) {
440           logger.warn("File missing");
441           list.add("File missing");
442           return list;
443         }
444         final R66File file =
445             (R66File) session.getDir().setFile(request.getFile(), false);
446         String sresult;
447         if (request.getAction() == Action.Exist) {
448           sresult = String.valueOf(file.exists());
449           list.add(sresult);
450         } else if (request.getAction() == Action.Detail) {
451           sresult = session.getDir().fileFull(request.getFile(), false);
452           final String[] slist = sresult.split("\n");
453           sresult = slist[1];
454           list.add(sresult);
455         }
456       }
457       return list;
458     } catch (final CommandAbstractException e) {
459       logger.warn("Error occurs during: " + request, e);
460       list.add("Error occurs during: " + request);
461       return list;
462     }
