57    public enum TASKSTEP {
58      NOTASK(0), PRETASK(1), TRANSFERTASK(2), POSTTASK(3), ALLDONETASK(4),
59      ERRORTASK(5);
60  
61      private final int taskNo;
62  
63      private static final Map<Integer, TASKSTEP> map =
64          new HashMap<Integer, TASKSTEP>();
65  
66      static {
67        for (final TASKSTEP task : TASKSTEP.values()) {
68          map.put(task.taskNo, task);
69        }
70      }
71  
72      TASKSTEP(final int task) {
73        taskNo = task;
74      }
75  
76      public static TASKSTEP valueOf(final int taskStep) {
77        return map.get(taskStep);
78      }
79  
80      public DbTaskRunner.TASKSTEP toLegacy() {
81        return DbTaskRunner.TASKSTEP.valueOf(name());
82      }
83  
84      @JsonValue
85      public int getTaskNo() {
86        return taskNo;
87      }
88    }

298     this.infoStatus = infoStatus;
299     this.rank = rank;
300     this.start = start;
301     this.stop = stop;

304   /**
305    * Constructor to create transfer locally with delayed start time
306    *
307    * @param rule
308    * @param retrieveMode
309    * @param file
310    * @param fileInfo
311    * @param blockSize
312    */
313   public Transfer(final String remote, final String rule, final int ruleMode,
314                   final boolean retrieveMode, final String file,
315                   final String fileInfo, final int blockSize,
316                   final Timestamp start) {
317     ownerRequest = Configuration.configuration.getHostId();
318     requester = Configuration.configuration.getHostId();
319     requested = remote;
320     this.rule = rule;
321     transferMode = ruleMode;
322     this.retrieveMode = retrieveMode;
323     filename = file;
324     originalName = file;
325     this.fileInfo = fileInfo;
326     this.blockSize = blockSize;
327     this.start = start;
328   }
