176               String sendPath, String archivePath, String workPath,
177               List<RuleTask> rPre, List<RuleTask> rPost, List<RuleTask> rError,
178               List<RuleTask> sPre, List<RuleTask> sPost,
179               List<RuleTask> sError) {

184     this.sendPath = sendPath;
185     this.archivePath = archivePath;
186     this.workPath = workPath;
187     rPreTasks = rPre;
188     rPostTasks = rPost;
189     rErrorTasks = rError;
190     sPreTasks = sPre;
191     sPostTasks = sPost;
192     sErrorTasks = sError;
193   }

215   @JsonProperty("HOSTIDS")
216   public String getXMLHostids() {
217     StringBuilder res = new StringBuilder("<hostids>");
218     for (final String hostid : hostids) {
219       res.append("<hostid>").append(hostid).append("</hostid>");
220     }
221     return res.append("</hostids>").toString();
222   }

254   private String getXMLTasks(List<RuleTask> tasks) {
255     StringBuilder res = new StringBuilder("<tasks>");
256     for (final RuleTask task : tasks) {
257       res.append(task.getXML());
258     }
259     return res.append("</tasks>").toString();
260   }
