21  package org.waarp.openr66.dao.database.postgres;
22  
23  import org.waarp.openr66.dao.database.DBTransferDAO;
24  import org.waarp.openr66.dao.exception.DAOConnectionException;
25  
26  import java.sql.Connection;
27  import java.sql.PreparedStatement;
28  import java.sql.ResultSet;
29  import java.sql.SQLException;
30  
31  public class PostgreSQLTransferDAO extends DBTransferDAO {
32  
33    protected static final String SQL_GET_ID = "SELECT NEXTVAL('runseq')";
34  
35    public PostgreSQLTransferDAO(final Connection con)
36        throws DAOConnectionException {
37      super(con);
38    }
39  
40    @Override
41    protected final long getNextId() throws DAOConnectionException {
42      PreparedStatement ps = null;
43      ResultSet rs = null;
44      try {
45        ps = connection.prepareStatement(SQL_GET_ID);
46        rs = ps.executeQuery();
47        if (rs.next()) {
48          return rs.getLong(1);
49        } else {
50          throw new DAOConnectionException(
51              "Error no id available, you should purge the database.");
52        }
53      } catch (final SQLException e) {
54        throw new DAOConnectionException(e);
55      } finally {
56        try {
57          if (rs != null) {
58            rs.close();
59          }
60        } catch (final SQLException e) {
61          // ignore
62        }
63        closeStatement(ps);
64      }
65    }
66  }
