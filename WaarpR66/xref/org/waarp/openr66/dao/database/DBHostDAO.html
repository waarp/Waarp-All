21  package org.waarp.openr66.dao.database;
22  
23  import org.waarp.common.logging.WaarpLogger;
24  import org.waarp.common.logging.WaarpLoggerFactory;
25  import org.waarp.openr66.dao.HostDAO;
26  import org.waarp.openr66.dao.exception.DAOConnectionException;
27  import org.waarp.openr66.pojo.Host;
28  import org.waarp.openr66.pojo.UpdatedInfo;
29  
30  import java.sql.Connection;
31  import java.sql.ResultSet;
32  import java.sql.SQLException;
33  
34  /**
35   * Implementation of HostDAO for standard SQL databases
36   */
37  public class DBHostDAO extends StatementExecutor<Host> implements HostDAO {
38  
39    private static final WaarpLogger logger =
40        WaarpLoggerFactory.getLogger(DBHostDAO.class);
41  
42    protected static final String TABLE = "hosts";
43  
44    public static final String HOSTID_FIELD = "hostid";
45    public static final String ADDRESS_FIELD = "address";
46    public static final String PORT_FIELD = "port";
47    public static final String IS_SSL_FIELD = "isssl";
48    public static final String IS_CLIENT_FIELD = "isclient";
49    public static final String IS_ACTIVE_FIELD = "isactive";
50    public static final String IS_PROXIFIED_FIELD = "isproxified";
51    public static final String HOSTKEY_FIELD = "hostkey";
52    public static final String ADMINROLE_FIELD = "adminrole";
53    public static final String UPDATED_INFO_FIELD = "updatedinfo";
54  
55    protected static final String SQL_DELETE_ALL = "DELETE FROM " + TABLE;
56    protected static final String SQL_DELETE =
57        "DELETE FROM " + TABLE + " WHERE " + HOSTID_FIELD + " = ?";
58    protected static final String SQL_GET_ALL = "SELECT * FROM " + TABLE;
59    protected static final String SQL_EXIST =
60        "SELECT 1 FROM " + TABLE + " WHERE " + HOSTID_FIELD + " = ?";
61    protected static final String SQL_SELECT =
62        "SELECT * FROM " + TABLE + " WHERE " + HOSTID_FIELD + " = ?";
63    protected static final String SQL_INSERT =
64        "INSERT INTO " + TABLE + " (" + HOSTID_FIELD + ", " + ADDRESS_FIELD +
65        ", " + PORT_FIELD + ", " + IS_SSL_FIELD + ", " + IS_CLIENT_FIELD + ", " +
66        IS_ACTIVE_FIELD + ", " + IS_PROXIFIED_FIELD + ", " + HOSTKEY_FIELD +
67        ", " + ADMINROLE_FIELD + ", " + UPDATED_INFO_FIELD +
68        ") VALUES (?,?,?,?,?,?,?,?,?,?)";
69    protected static final String SQL_UPDATE =
70        "UPDATE " + TABLE + " SET " + HOSTID_FIELD + " = ?, " + ADDRESS_FIELD +
71        " = ?, " + PORT_FIELD + " = ?, " + IS_SSL_FIELD + " = ?, " +
72        IS_CLIENT_FIELD + " = ?, " + IS_ACTIVE_FIELD + " = ?, " +
73        IS_PROXIFIED_FIELD + " = ?, " + HOSTKEY_FIELD + " = ?, " +
74        ADMINROLE_FIELD + " = ?, " + UPDATED_INFO_FIELD + " = ? WHERE " +
75        HOSTID_FIELD + " = ?";
76  
77  
78    public DBHostDAO(Connection con) throws DAOConnectionException {
79      super(con);
80    }
81  
82    @Override
83    protected String getId(final Host e1) {
84      return e1.getHostid();
85    }
86  
87    @Override
88    protected String getSelectRequest() {
89      return SQL_SELECT;
90    }
91  
92    @Override
93    protected String getGetAllRequest() {
94      return SQL_GET_ALL;
95    }
96  
97    @Override
98    protected String getExistRequest() {
99      return SQL_EXIST;
100   }
101 
102   @Override
103   protected Object[] getInsertValues(final Host host) {
104     return new Object[] {
105         host.getHostid(), host.getAddress(), host.getPort(), host.isSSL(),
106         host.isClient(), host.isActive(), host.isProxified(), host.getHostkey(),
107         host.isAdmin(), host.getUpdatedInfo().ordinal()
108     };
109   }
110 
111   @Override
112   protected String getInsertRequest() {
113     return SQL_INSERT;
114   }
115 
116   @Override
117   protected Object[] getUpdateValues(final Host host) {
118     return new Object[] {
119         host.getHostid(), host.getAddress(), host.getPort(), host.isSSL(),
120         host.isClient(), host.isActive(), host.isProxified(), host.getHostkey(),
121         host.isAdmin(), host.getUpdatedInfo().ordinal(), host.getHostid()
122     };
123   }
124 
125   @Override
126   protected String getUpdateRequest() {
127     return SQL_UPDATE;
128   }
129 
130   @Override
131   protected String getDeleteRequest() {
132     return SQL_DELETE;
133   }
134 
135   @Override
136   protected String getDeleteAllRequest() {
137     return SQL_DELETE_ALL;
138   }
139 
140   @Override
141   public Host getFromResultSet(ResultSet set) throws SQLException {
142     return new Host(set.getString(HOSTID_FIELD), set.getString(ADDRESS_FIELD),
143                     set.getInt(PORT_FIELD), set.getBytes(HOSTKEY_FIELD),
144                     set.getBoolean(IS_SSL_FIELD),
145                     set.getBoolean(IS_CLIENT_FIELD),
146                     set.getBoolean(IS_PROXIFIED_FIELD),
147                     set.getBoolean(ADMINROLE_FIELD),
148                     set.getBoolean(IS_ACTIVE_FIELD),
149                     UpdatedInfo.valueOf(set.getInt(UPDATED_INFO_FIELD)));
150   }
151 }
