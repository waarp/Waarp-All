20  package org.waarp.openr66.protocol.configuration;
21  
22  /**
23   * This class is only intend to concentrate all SystemProperty definitions
24   */
25  public final class R66SystemProperties {
26  
27    /**
28     * Used to set en/fr or other locale defined (default en)
29     */
30    public static final String OPENR66_LOCALE = "openr66.locale";
31    /**
32     * True if this host is proxyfied for other partners (default false)
33     */
34    public static final String OPENR66_ISHOSTPROXYFIED =
35        "openr66.ishostproxyfied";
36    /**
37     * Shall Waarp use space as separator (buggy, should be false)
38     */
39    public static final String OPENR66_USESPACESEPARATOR =
40        "openr66.usespaceseparator";
41    /**
42     * Shall Waarp allow to execute error actions in case no transfer at all
43     * occurs but an error in "pre-task"
44     * occurs: default true
45     */
46    public static final String OPENR66_EXECUTEBEFORETRANSFERRED =
47        "openr66.executebeforetransferred";
48    /**
49     * Use in Windows Service mode to specify the location of the configuration
50     * file
51     */
52    public static final String OPENR66_CONFIGFILE = "org.waarp.r66.config.file";
53    /**
54     * Shall we print info of startup in Warning mode (true, default) or Info
55     * mode
56     * (false)
57     */
58    public static final String OPENR66_STARTUP_WARNING =
59        "openr66.startup.warning";
60    /**
61     * Shall the database being check at startup (default is false)
62     *
63     * @deprecated
64     */
65    @Deprecated
66    public static final String OPENR66_STARTUP_DATABASE_CHECK =
67        "openr66.startup.dbcheck";
68    /**
69     * Shall the database being check at startup (default is false)
70     */
71    public static final String OPENR66_STARTUP_DATABASE_AUTOUPGRADE =
72        "openr66.startup.autoUpgrade";
73    /**
74     * Shall we allow or not a request as RECV that will access to file outside
75     * Rule Out directory (default =
76     * false)
77     */
78    public static final String OPENR66_CHROOT_CHECKED = "openr66.chroot.checked";
79    /**
80     * Shall we blacklist badly authenticated servers (default = true, if
81     * OPENR66_ISHOSTPROXYFIED is true, then is
82     * mandatory false). Note that this must not be true if several partners
83     * might
84     * have the same IP (proxyfied for
85     * instance), since they will all be banned.
86     */
87    public static final String OPENR66_BLACKLIST_BADAUTHENT =
88        "openr66.blacklist.badauthent";
89    /**
90     * Filename max length (default = 255), only for Basename (not full path).
91     * to
92     * choose the default max filename
93     * length used when receiving a file (for the temporary filename and final
94     * filename). This does not prevent to
95     * change the filename after (and #ORIGINALFILENAME# does still contain the
96     * full filename, not truncated).
97     */
98    public static final String OPENR66_FILENAME_MAXLENGTH =
99        "openr66.filename.maxlength";
100   /**
101    * Debug by making a trace of consumption every x s, where x is specify as
102    * value in the definition. If 0 or
103    * less, means not activated.
104    */
105   public static final String OPENR66_TRACE_STATS = "openr66.trace.stats";
106   /**
107    * Maximum number of DbTaskRunners to keep in LRU cache (used in self
108    * request
109    * and for instance without
110    * database). Minimal value is 100.
111    */
112   public static final String OPENR66_CACHE_LIMIT = "openr66.cache.limit";
113   /**
114    * Maximum time in ms of valid element once created, used or updated (used
115    * in
116    * self request and for instance
117    * without database). Minimal value is 1000 ms (1s). If set to 1000, the
118    * value
119    * will not be regularly deleted.
120    */
121   public static final String OPENR66_CACHE_TIMELIMIT =
122       "openr66.cache.timelimit";
123 
124   /**
125    * Shall the server or client try to reuse already authenticated
126    * partner when using the very same network link (multiplexing)?
127    * If 0 (default), it will reuse the first successful authentication.
128    * If 1, it will always use a new authentication for each command.
129    */
130   public static final String OPENR66_AUTHENT_NO_REUSE =
131       "openr66.authent.noreuse";
132 
133   /**
134    * Shall the server or client try to use GUID as transfer ID
135    * instead of SQL nextLong Id (default)
136    */
137   public static final String OPENR66_TRANSFER_GUID = "openr66.transfer.guid";
138 
139   /**
140    * Test only: apply a NoOpRecvThroughHandler to each recv operation
141    * if rule is RecvThrough
142    */
143   public static final String OPENR66_JUNIT_RECV_THROUGH =
144       "openr66.junit.recvthrough";
145 
146   private R66SystemProperties() {
147   }
148 }
