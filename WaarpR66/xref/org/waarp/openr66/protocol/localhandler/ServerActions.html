183    * Receive a request of information
184    *
185    * @param packet
186    *
187    * @throws CommandAbstractException
188    * @throws OpenR66ProtocolNotAuthenticatedException
189    * @throws OpenR66ProtocolNoDataException
190    * @throws OpenR66ProtocolPacketException
191    */
192   public void information(InformationPacket packet)
193       throws OpenR66ProtocolNotAuthenticatedException,
194              OpenR66ProtocolNoDataException, OpenR66ProtocolPacketException {
195     final byte request = packet.getRequest();
196     final String rulename = packet.getRulename();
197     final String filename = packet.getFilename();
198     packet.clear();
199     long id = ILLEGALVALUE;
200     if (request == REQUEST_CHECK) {
201       try {
202         id = Long.parseLong(rulename);
203       } catch (final NumberFormatException e) {
204         logger.error("Incorrect Transfer ID", e);
205         throw new OpenR66ProtocolNoDataException("Incorrect Transfer ID", e);
206       }
207     }
208     final boolean isTo = "1".equals(filename);
209     ValidPacket validPacket;
210     if (request == REQUEST_CHECK) {
211       validPacket = informationRequest(id, isTo, rulename, false);
212     } else {
213       validPacket = informationFile(request, rulename, filename, false);
214     }
215     if (validPacket != null) {
216       ChannelUtils
217           .writeAbstractLocalPacket(localChannelReference, validPacket, true);
218       localChannelReference.close();
219     } else {
220       session.newState(ERROR);
221       final ErrorPacket error =
222           new ErrorPacket("Error while Request " + request,
223                           ErrorCode.Internal.getCode(),
224                           ErrorPacket.FORWARDCLOSECODE);
225       ChannelUtils.writeAbstractLocalPacket(localChannelReference, error, true);
226       ChannelCloseTimer
227           .closeFutureTransaction(localChannelReference.getServerHandler());
228     }
229   }

241   public void valid(ValidPacket packet)
242       throws OpenR66ProtocolNotAuthenticatedException,
243              OpenR66RunnerErrorException, OpenR66ProtocolSystemException,
244              OpenR66ProtocolBusinessException {
245     // SHUTDOWNPACKET does not need authentication
246     if (packet.getTypeValid() != LocalPacketFactory.SHUTDOWNPACKET &&
247         !session.isAuthenticated()) {
248       logger
249           .warn("Valid packet received while not authenticated: {} {}", packet,
250                 session);
251       session.newState(ERROR);
252       packet.clear();
253       throw new OpenR66ProtocolNotAuthenticatedException(
254           "Not authenticated while Valid received");
255     }
256     switch (packet.getTypeValid()) {
257       case LocalPacketFactory.SHUTDOWNPACKET: {
258         int rank = -1;
259         if (session.getRunner() != null && session.getRunner().isInTransfer()) {
260           final String srank = packet.getSmiddle();
261           if (srank != null && !srank.isEmpty()) {
262             // Save last rank from remote point of view
263             try {
264               rank = Integer.parseInt(srank);
265             } catch (final NumberFormatException e) {
266               // ignore
267             }
268           }
269         }
270         final R66Result result =
271             new R66Result(new OpenR66ProtocolShutdownException(), session, true,
272                           ErrorCode.Shutdown, session.getRunner());
273         result.setOther(packet);
274         rank = shutdownRequest(result, rank);
275         if (rank >= 0) {
276           packet.setSmiddle(Integer.toString(rank));
277           try {
278             ChannelUtils
279                 .writeAbstractLocalPacket(localChannelReference, packet, true);
280           } catch (final OpenR66ProtocolPacketException ignored) {
281             // ignore
282           }
283         }
284         shutdownLocalChannel();
285         break;
286       }
287       case LocalPacketFactory.STOPPACKET:
288       case LocalPacketFactory.CANCELPACKET: {
289         final String[] keys = packet.getSmiddle().split(" ");
290         final long id = Long.parseLong(keys[2]);
291         session.newState(VALIDOTHER);
292         final R66Result resulttest =
293             stopOrCancel(packet.getTypeValid(), keys[0], keys[1], id);
294         // inform back the requester
295         final ValidPacket valid =
296             new ValidPacket(packet.getSmiddle(), resulttest.getCode().getCode(),
297                             LocalPacketFactory.REQUESTUSERPACKET);
298         resulttest.setOther(packet);
299         localChannelReference.validateRequest(resulttest);
300         try {
301           ChannelUtils
302               .writeAbstractLocalPacket(localChannelReference, valid, true);
303         } catch (final OpenR66ProtocolPacketException ignored) {
304           // ignore
305         }
306         session.setStatus(27);
307         localChannelReference.close();
308         break;
309       }
310       case LocalPacketFactory.VALIDPACKET: {
311         // header = ?; middle = requested+blank+requester+blank+specialId
312         // note: might contains one more argument = time to reschedule in yyyyMMddHHmmss format
313         final String[] keys = packet.getSmiddle().split(" ");
314         ValidPacket valid;
315         if (keys.length < 3) {
316           // not enough args
317           valid = new ValidPacket(packet.getSmiddle(),
318                                   ErrorCode.IncorrectCommand.getCode(),
319                                   LocalPacketFactory.REQUESTUSERPACKET);
320           final R66Resultult.html#R66Result">R66Result resulttest = new R66Result(
321               new OpenR66ProtocolBusinessRemoteFileNotFoundException(
322                   "Not enough arguments"), session, true,
323               ErrorCode.IncorrectCommand, null);
324           resulttest.setOther(packet);
325           localChannelReference.invalidateRequest(resulttest);
326         } else {
327           final long id = Long.parseLong(keys[2]);
328           Date date = null;
329           if (keys.length > 3) {
330             // time to reschedule in yyyyMMddHHmmss format
331             logger.debug("Debug: restart with " + keys[3]);
332             final SimpleDateFormat dateFormat =
333                 new SimpleDateFormat("yyyyMMddHHmmss");
334             try {
335               date = dateFormat.parse(keys[3]);
336             } catch (final ParseException ignored) {
337               // ignore
338             }
339           }
340           session.newState(VALIDOTHER);
341           final R66Result result = requestRestart(keys[0], keys[1], id, date);
342           valid =
343               new ValidPacket(packet.getSmiddle(), result.getCode().getCode(),
344                               LocalPacketFactory.REQUESTUSERPACKET);
345           result.setOther(packet);
346           if (isCodeValid(result.getCode())) {
347             localChannelReference.validateRequest(result);
348           } else {
349             localChannelReference.invalidateRequest(result);
350           }
351         }
352         // inform back the requester
353         try {
354           ChannelUtils
355               .writeAbstractLocalPacket(localChannelReference, valid, true);
356         } catch (final OpenR66ProtocolPacketException ignored) {
357           // ignore
358         }
359         localChannelReference.close();
360         break;
361       }
362       case LocalPacketFactory.REQUESTUSERPACKET: {
363         session.newState(VALIDOTHER);
364         // Validate user request
365         final R66Resultult.html#R66Result">R66Result resulttest = new R66Result(session, true, ErrorCode
366             .getFromCode(packet.getSmiddle()), null);
367         resulttest.setOther(packet);
368         switch (resulttest.getCode()) {
369           case CompleteOk:
370           case InitOk:
371           case PostProcessingOk:
372           case PreProcessingOk:
373           case QueryAlreadyFinished:
374           case QueryStillRunning:
375           case Running:
376           case TransferOk:
377             break;
378           default:
379             localChannelReference.invalidateRequest(resulttest);
380             session.setStatus(102);
381             localChannelReference.close();
382             return;
383         }
384         localChannelReference.validateRequest(resulttest);
385         session.setStatus(28);
386         localChannelReference.close();
387         break;
388       }
389       case LocalPacketFactory.LOGPACKET:
390       case LocalPacketFactory.LOGPURGEPACKET: {
391         session.newState(VALIDOTHER);
392         // should be from the local server or from an authorized hosts: LOGCONTROL
393         try {
394           if (!session.getAuth().getUser().equals(Configuration.configuration
395                                                       .getHostId(
396                                                           session.getAuth()
397                                                                  .isSsl())) &&
398               !session.getAuth().isValidRole(ROLE.LOGCONTROL)) {
399             throw new OpenR66ProtocolNotAuthenticatedException(
400                 NOT_CORRECTLY_AUTHENTICATED);
401           }
402         } catch (final OpenR66ProtocolNoSslException e1) {
403           throw new OpenR66ProtocolNotAuthenticatedException(
404               NOT_CORRECTLY_AUTHENTICATED_SINCE_SSL_IS_NOT_SUPPORTED, e1);
405         }
406         final String sstart = packet.getSheader();
407         final String sstop = packet.getSmiddle();
408         final boolean isPurge =
409             packet.getTypeValid() == LocalPacketFactory.LOGPURGEPACKET;
410         final Timestamp start = sstart == null || sstart.isEmpty()? null :
411             Timestamp.valueOf(sstart);
412         final Timestamp stop =
413             sstop == null || sstop.isEmpty()? null : Timestamp.valueOf(sstop);
414         packet.clear();
415         // create export of log and optionally purge them from database
416         final String filename = Configuration.configuration.getBaseDirectory() +
417                                 Configuration.configuration.getArchivePath() +
418                                 DirInterface.SEPARATOR +
419                                 Configuration.configuration.getHostId() + '_' +
420                                 System.currentTimeMillis() + "_runners.xml";
421         DbPreparedStatement statement = null;
422         try {
423           statement = DbTaskRunner
424               .getLogPrepareStatement(localChannelReference.getDbSession(),
425                                       start, stop);
426           DbTaskRunner.writeXMLWriter(statement, filename);
427         } catch (final WaarpDatabaseNoConnectionException e) {
428           throw new OpenR66ProtocolBusinessException(e);
429         } catch (final WaarpDatabaseSqlException e) {
430           throw new OpenR66ProtocolBusinessException(e);
431         } finally {
432           if (statement != null) {
433             statement.realClose();
434           }
435         }
436         // in case of purge
437         int nb = 0;
438         if (isPurge) {
439           // purge in same interval all runners with globallaststep
440           // as ALLDONETASK or ERRORTASK
441           if (Configuration.configuration.getR66Mib() != null) {
442             Configuration.configuration.getR66Mib().notifyWarning(
443                 "Purge Log Order received", session.getAuth().getUser());
444           }
445           try {
446             nb = DbTaskRunner
447                 .purgeLogPrepareStatement(localChannelReference.getDbSession(),
448                                           start, stop);
449           } catch (final WaarpDatabaseNoConnectionException e) {
450             throw new OpenR66ProtocolBusinessException(e);
451           } catch (final WaarpDatabaseSqlException e) {
452             throw new OpenR66ProtocolBusinessException(e);
453           }
454         }
455         final R66Result result =
456             new R66Result(session, true, ErrorCode.CompleteOk, null);
457         // Now answer
458         final ValidPacket valid =
459             new ValidPacket(filename + ' ' + nb, result.getCode().getCode(),
460                             LocalPacketFactory.REQUESTUSERPACKET);
461         localChannelReference.validateRequest(result);
462         try {
463           ChannelUtils
464               .writeAbstractLocalPacket(localChannelReference, valid, true);
465         } catch (final OpenR66ProtocolPacketException ignored) {
466           // ignore
467         }
468         localChannelReference.close();
469         break;
470       }
471       case LocalPacketFactory.CONFEXPORTPACKET: {
472         final String shost = packet.getSheader();
473         final String srule = packet.getSmiddle();
474         final boolean bhost = Boolean.parseBoolean(shost);
475         final boolean brule = Boolean.parseBoolean(srule);
476         packet.clear();
477         session.newState(VALIDOTHER);
478         final String[] sresult =
479             configExport(bhost, brule, false, false, false);
480         R66Result result;
481         if (sresult[0] != null || sresult[1] != null) {
482           result = new R66Result(session, true, ErrorCode.CompleteOk, null);
483         } else {
484           result = new R66Result(session, true, ErrorCode.TransferError, null);
485         }
486         // Now answer
487         final ValidPacket valid =
488             new ValidPacket(shost + ' ' + srule, result.getCode().getCode(),
489                             LocalPacketFactory.REQUESTUSERPACKET);
490         localChannelReference.validateRequest(result);
491         try {
492           ChannelUtils
493               .writeAbstractLocalPacket(localChannelReference, valid, true);
494         } catch (final OpenR66ProtocolPacketException ignored) {
495           // ignore
496         }
497         localChannelReference.close();
498         break;
499       }
500       case LocalPacketFactory.CONFIMPORTPACKET: {
501         session.newState(VALIDOTHER);
502         // Authentication must be the local server or CONFIGADMIN authorization
503         try {
504           if (!session.getAuth().getUser().equals(Configuration.configuration
505                                                       .getHostId(
506                                                           session.getAuth()
507                                                                  .isSsl())) &&
508               !session.getAuth().isValidRole(ROLE.CONFIGADMIN)) {
509             throw new OpenR66ProtocolNotAuthenticatedException(
510                 NOT_CORRECTLY_AUTHENTICATED);
511           }
512         } catch (final OpenR66ProtocolNoSslException e1) {
513           throw new OpenR66ProtocolNotAuthenticatedException(
514               NOT_CORRECTLY_AUTHENTICATED_SINCE_SSL_IS_NOT_SUPPORTED, e1);
515         }
516         if (Configuration.configuration.getR66Mib() != null) {
517           Configuration.configuration.getR66Mib().notifyWarning(
518               "Import Configuration Order received",
519               session.getAuth().getUser());
520         }
521         String shost = packet.getSheader();
522         String srule = packet.getSmiddle();
523         final boolean bhostPurge = shost.startsWith("1 ");
524         shost = shost.substring(2);
525         final boolean brulePurge = srule.startsWith("1 ");
526         srule = srule.substring(2);
527         boolean bhost = !shost.isEmpty();
528         boolean brule = !srule.isEmpty();
529         packet.clear();
530         if (bhost) {
531           DbHostAuth[] oldHosts = null;
532           if (bhostPurge) {
533             // Need to first delete all entries
534             try {
535               oldHosts = DbHostAuth.deleteAll();
536             } catch (final WaarpDatabaseException e) {
537               // ignore
538             }
539           }
540           final String filename = shost;
541           if (AuthenticationFileBasedConfiguration
542               .loadAuthentication(Configuration.configuration, filename)) {
543             shost = "Host:OK";
544           } else {
545             logger.error("Error in Load Hosts");
546             shost = "Host:KO";
547             bhost = false;
548           }
549           if (!bhost && oldHosts != null) {
550             for (final DbHostAuth dbHost : oldHosts) {
551               try {
552                 if (!dbHost.exist()) {
553                   dbHost.insert();
554                 }
555               } catch (final WaarpDatabaseException e1) {
556                 // ignore
557               }
558             }
559           }
560         }
561         if (brule) {
562           DbRule[] oldRules = null;
563           if (brulePurge) {
564             // Need to first delete all entries
565             try {
566               oldRules = DbRule.deleteAll();
567             } catch (final WaarpDatabaseException e) {
568               // ignore
569             }
570           }
571           final File file = new File(srule);
572           try {
573             RuleFileBasedConfiguration.getMultipleFromFile(file);
574             srule = "Rule:OK";
575             brule = true;
576           } catch (final WaarpDatabaseNoConnectionException e) {
577             logger.error("Error", e);
578             srule = "Rule:KO";
579             brule = false;
580           } catch (final WaarpDatabaseSqlException e) {
581             logger.error("Error", e);
582             srule = "Rule:KO";
583             brule = false;
584           } catch (final WaarpDatabaseNoDataException e) {
585             logger.error("Error", e);
586             srule = "Rule:KO";
587             brule = false;
588           } catch (final WaarpDatabaseException e) {
589             logger.error("Error", e);
590             srule = "Rule:KO";
591             brule = false;
592           }
593           if (!brule && oldRules != null) {
594             for (final DbRule dbRule : oldRules) {
595               try {
596                 if (!dbRule.exist()) {
597                   dbRule.insert();
598                 }
599               } catch (final WaarpDatabaseException e1) {
600                 // ignore
601               }
602             }
603           }
604         }
605         R66Result result;
606         if (brule || bhost) {
607           result = new R66Result(session, true, ErrorCode.CompleteOk, null);
608         } else {
609           result = new R66Result(session, true, ErrorCode.TransferError, null);
