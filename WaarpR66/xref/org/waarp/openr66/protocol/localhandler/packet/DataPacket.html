31  import org.waarp.openr66.protocol.utils.FileUtils;
32  
33  import java.util.Arrays;

35  /**
36   * Data packet
37   * <p>
38   * header = packetRank middle = data end = key
39   */
40  public class DataPacket extends AbstractLocalPacket {
41    private static final WaarpLogger logger =
42        WaarpLoggerFactory.getLogger(DataPacket.class);
43  
44    private final int packetRank;
45  
46    private final int lengthPacket;
47  
48    private byte[] data;
49    private ByteBuf dataRecv;
50  
51    private byte[] key;
52  
53    /**
54     * @param headerLength
55     * @param middleLength
56     * @param endLength
57     * @param buf
58     *
59     * @return the new DataPacket from buffer
60     *
61     * @throws OpenR66ProtocolPacketException
62     */
63    public static DataPacket createFromBuffer(final int headerLength,
64                                              final int middleLength,
65                                              final int endLength,
66                                              final ByteBuf buf)
67        throws OpenR66ProtocolPacketException {
68      if (headerLength - 1 <= 0) {
69        throw new OpenR66ProtocolPacketException("Not enough data");
70      }
71      if (middleLength <= 0) {
72        throw new OpenR66ProtocolPacketException("Not enough data");
73      }
74      final int packetRank = buf.readInt();
75      final int index = buf.readerIndex();
76      final ByteBuf recvData = buf.retainedSlice(index, middleLength);
77      buf.skipBytes(middleLength);
78      final byte[] key = endLength > 0? new byte[endLength] : EMPTY_ARRAY;
79      if (endLength > 0) {
80        buf.readBytes(key);
81      }
82      return new DataPacket(packetRank, recvData, key);
83    }
84  
85    /**
86     * @param packetRank
87     * @param data
88     * @param key
89     */
90    private DataPacket(final int packetRank, final ByteBuf data,
91                       final byte[] key) {
92      this.packetRank = packetRank;
93      this.dataRecv = data;
94      this.data = null;
95      this.key = key == null? EMPTY_ARRAY : key;
96      lengthPacket = dataRecv.readableBytes();
97    }
98  
99    /**
100    * @param packetRank
101    * @param data
102    * @param key
103    */
104   public DataPacket(final int packetRank, final byte[] data, final byte[] key) {
105     this.packetRank = packetRank;
106     this.data = data;
107     this.dataRecv = null;
108     this.key = key == null? EMPTY_ARRAY : key;
109     lengthPacket = data.length;
110   }
111 
112   @Override
113   public boolean hasGlobalBuffer() {
114     return false;
115   }
116 
117   @Override
118   public void createAllBuffers(final LocalChannelReference lcr,
119                                final int networkHeader) {
120     throw new IllegalStateException("Should not be called");
121   }
122 
123   @Override
124   public void createEnd(final LocalChannelReference lcr) {
125     end = WaarpNettyUtil.wrappedBuffer(key);
126   }
127 
128   @Override
129   public void createHeader(final LocalChannelReference lcr) {
130     header = ByteBufAllocator.DEFAULT.buffer(4, 4);
131     header.writeInt(packetRank);
132   }
133 
134   @Override
135   public void createMiddle(final LocalChannelReference lcr) {
136     if (dataRecv != null) {
137       middle = dataRecv;
138     } else {
139       middle = WaarpNettyUtil.wrappedBuffer(data);
140     }
141   }
142 
143   @Override
144   public byte getType() {
145     return LocalPacketFactory.DATAPACKET;
146   }
147 
148   @Override
149   public String toString() {
150     return "DataPacket: " + packetRank + ':' + lengthPacket;
151   }
152 
153   /**
154    * @return the packetRank
155    */
156   public int getPacketRank() {
157     return packetRank;
158   }
159 
160   /**
161    * @return the lengthPacket
162    */
163   public int getLengthPacket() {
164     return lengthPacket;
165   }
166 
167   /**
168    * Only for Network incoming buffer
169    *
170    * @return the Data ByteBuf
171    */
172   public ByteBuf getRecvData() {
173     return dataRecv;
174   }
175 
176   public void createByteBufFromRecv(final byte[] buffer) {
177     dataRecv.getBytes(dataRecv.readerIndex(), buffer);
178     data = buffer;
179   }
180 
181   /**
182    * @return the data
183    */
184   public byte[] getData() {
185     ParametersChecker
186         .checkParameter("Data is not setup correctly", data, logger);
187     return data;
188   }
189 
190   /**
191    * @return the key
192    */
193   public byte[] getKey() {
194     return key;
195   }
196 
197   /**
198    * @return True if the Hashed key is valid (or no key is set)
199    */
200   public boolean isKeyValid(final FilesystemBasedDigest digestBlock,
201                             final FilesystemBasedDigest digestGlobal,
202                             final FilesystemBasedDigest digestLocal) {
203     ParametersChecker
204         .checkParameter("Data is not setup correctly", data, logger);
205     if (key == null || key.length == 0) {
206       if (digestGlobal != null || digestLocal != null) {
207         FileUtils.computeGlobalHash(digestGlobal, digestLocal, data);
208       }
209       logger.error("Should received a Digest but don't");
210       return false;
211     }
212     digestBlock.Update(data, 0, data.length);
213     final byte[] newkey = digestBlock.Final();
214     FileUtils.computeGlobalHash(digestGlobal, digestLocal, data);
215     final boolean equal = Arrays.equals(key, newkey);
216     if (!equal) {
217       logger.error("DIGEST {} != {} for {} bytes using {} at rank {}",
218                    FilesystemBasedDigest.getHex(key),
219                    FilesystemBasedDigest.getHex(newkey), data.length,
220                    digestBlock.getAlgo(), packetRank);
221     }
222     return equal;
223   }
224 
225   @Override
226   public void clear() {
227     super.clear();
228     WaarpNettyUtil.release(dataRecv);
229     dataRecv = null;
230     data = null;
231     key = null;
232   }
233 }
