20  package org.waarp.openr66.protocol.localhandler.packet;
21  
22  import io.netty.buffer.ByteBuf;
23  import io.netty.buffer.Unpooled;
24  import org.waarp.openr66.protocol.exception.OpenR66ProtocolPacketException;
25  import org.waarp.openr66.protocol.localhandler.LocalChannelReference;
26  
27  /**
28   * This class represents Abstract Packet with its header, middle and end parts.
29   * A Packet is composed of one
30   * Header part, one Middle part (data), and one End part. Header: length field
31   * (4 bytes) = Middle length field
32   * (4 bytes), End length field (4 bytes), type field (1 byte), ...<br>
33   * Middle: (Middle length field bytes)<br>
34   * End: (End length field bytes) = code status field (4 bytes), ...<br>
35   */
36  public abstract class AbstractLocalPacket {
37    protected ByteBuf header;
38  
39    protected ByteBuf middle;
40  
41    protected ByteBuf end;
42  
43    protected AbstractLocalPacket(ByteBuf header, ByteBuf middle, ByteBuf end) {
44      this.header = header;
45      this.middle = middle;
46      this.end = end;
47    }
48  
49    protected AbstractLocalPacket() {
50      header = null;
51      middle = null;
52      end = null;
53    }
54  
55    /**
56     * Prepare the Header buffer
57     *
58     * @throws OpenR66ProtocolPacketException
59     */
60    public abstract void createHeader(LocalChannelReference lcr)
61        throws OpenR66ProtocolPacketException;
62  
63    /**
64     * Prepare the Middle buffer
65     *
66     * @throws OpenR66ProtocolPacketException
67     */
68    public abstract void createMiddle(LocalChannelReference lcr)
69        throws OpenR66ProtocolPacketException;
70  
71    /**
72     * Prepare the End buffer
73     *
74     * @throws OpenR66ProtocolPacketException
75     */
76    public abstract void createEnd(LocalChannelReference lcr)
77        throws OpenR66ProtocolPacketException;
78  
79    /**
80     * @return the type of Packet
81     */
82    public abstract byte getType();
83  
84    @Override
85    public abstract String toString();
86  
87    /**
88     * @param lcr the LocalChannelReference in use
89     *
90     * @return the ByteBuf as LocalPacket
91     *
92     * @throws OpenR66ProtocolPacketException
93     */
94    public ByteBuf getLocalPacket(LocalChannelReference lcr)
95        throws OpenR66ProtocolPacketException {
96      final ByteBuf buf = Unpooled.buffer(4 * 3 + 1);// 3 header
97      // lengths+type
98      if (header == null) {
99        createHeader(lcr);
100     }
101     final ByteBuf newHeader = header != null? header : Unpooled.EMPTY_BUFFER;
102     final int headerLength = 4 * 2 + 1 + newHeader.readableBytes();
103     if (middle == null) {
104       createMiddle(lcr);
105     }
106     final ByteBuf newMiddle = middle != null? middle : Unpooled.EMPTY_BUFFER;
107     final int middleLength = newMiddle.readableBytes();
108     if (end == null) {
109       createEnd(lcr);
110     }
111     final ByteBuf newEnd = end != null? end : Unpooled.EMPTY_BUFFER;
112     final int endLength = newEnd.readableBytes();
113     buf.writeInt(headerLength);
114     buf.writeInt(middleLength);
115     buf.writeInt(endLength);
116     buf.writeByte(getType());
117     return Unpooled.wrappedBuffer(buf, newHeader, newMiddle, newEnd);
118   }
119 
120   public void clear() {
121     if (header != null && header.release()) {
122       header = null;
123     }
124     if (middle != null && middle.release()) {
125       middle = null;
126     }
127     if (end != null && end.release()) {
128       end = null;
129     }
130   }
131 
132   public void retain() {
133     if (header != null) {
134       header.retain();
135     }
136     if (middle != null) {
137       middle.retain();
138     }
139     if (end != null) {
140       end.retain();
141     }
142   }
143 }
