20  package org.waarp.openr66.protocol.localhandler.packet;
21  
22  import io.netty.buffer.ByteBuf;
23  import io.netty.buffer.Unpooled;
24  import org.waarp.common.digest.FilesystemBasedDigest;
25  import org.waarp.openr66.database.data.DbHostAuth;
26  import org.waarp.openr66.protocol.configuration.Configuration;
27  import org.waarp.openr66.protocol.configuration.PartnerConfiguration;
28  import org.waarp.openr66.protocol.exception.OpenR66ProtocolNoSslException;
29  import org.waarp.openr66.protocol.exception.OpenR66ProtocolPacketException;
30  import org.waarp.openr66.protocol.localhandler.LocalChannelReference;
31  import org.waarp.openr66.protocol.utils.R66Versions;

33  /**
34   * Request Authentication class
35   * <p>
36   * header = "hostId" middle = "key bytes" end = localId + way + (optional
37   * version: could be a JSON on the form
38   * version.{})
39   */
40  public class AuthentPacket extends AbstractLocalPacket {
41    private static final String NOT_ENOUGH_DATA = "Not enough data";
42  
43    private static final byte ASKVALIDATE = 0;
44  
45    private static final byte ANSWERVALIDATE = 1;
46  
47    private final Integer localId;
48  
49    private byte way;
50  
51    private String version;
52  
53    private String hostId;
54  
55    private byte[] key;
56  
57    /**
58     * @param headerLength
59     * @param middleLength
60     * @param endLength
61     * @param buf
62     *
63     * @return the new AuthentPacket from buffer
64     *
65     * @throws OpenR66ProtocolPacketException
66     */
67    public static AuthentPacket createFromBuffer(int headerLength,
68                                                 int middleLength, int endLength,
69                                                 ByteBuf buf)
70        throws OpenR66ProtocolPacketException {
71      if (headerLength - 1 <= 0) {
72        throw new OpenR66ProtocolPacketException(NOT_ENOUGH_DATA);
73      }
74      if (middleLength <= 0) {
75        throw new OpenR66ProtocolPacketException(NOT_ENOUGH_DATA);
76      }
77      if (endLength < 5) {
78        throw new OpenR66ProtocolPacketException(NOT_ENOUGH_DATA);
79      }
80      final byte[] bheader = new byte[headerLength - 1];
81      final byte[] bmiddle = new byte[middleLength];
82      if (headerLength - 1 > 0) {
83        buf.readBytes(bheader);
84      }
85      if (middleLength > 0) {
86        buf.readBytes(bmiddle);
87      }
88      // end part
89      final Integer newId = buf.readInt();
90      final byte valid = buf.readByte();
91      String version = R66Versions.V2_4_12
92          .getVersion(); // first base reference where it is unacceptable
93      if (endLength > 5) {
94        // version
95        final byte[] bversion = new byte[endLength - 5];
96        buf.readBytes(bversion);
97        version = new String(bversion);
98      }
99      final String sheader = new String(bheader);
100     return new AuthentPacket(sheader, bmiddle, newId, valid, version);
101   }
102 
103   /**
104    * @param hostId
105    * @param key
106    * @param newId
107    * @param valid
108    * @param version
109    */
110   private AuthentPacket(String hostId, byte[] key, Integer newId, byte valid,
111                         String version) {
112     this.hostId = hostId;
113     this.key = key;
114     localId = newId;
115     way = valid;
116     Configuration.configuration.getVersions().put(hostId,
117                                                   new PartnerConfiguration(
118                                                       hostId, version));
119     this.version = version;
120   }
121 
122   /**
123    * @param hostId
124    * @param key
125    * @param newId
126    */
127   public AuthentPacket(String hostId, byte[] key, Integer newId) {
128     this.hostId = hostId;
129     this.key = key;
130     localId = newId;
131     way = ASKVALIDATE;
132     Configuration.configuration.getVersions().putIfAbsent(hostId,
133                                                           new PartnerConfiguration(
134                                                               hostId));
135     version = Configuration.configuration.getVersions().get(hostId).toString();
136   }
137 
138   @Override
139   public void createEnd(LocalChannelReference lcr)
140       throws OpenR66ProtocolPacketException {
141     final byte[] bversion = version != null? version.getBytes() : null;
142     end = Unpooled.buffer(5 + (version != null? bversion.length : 0));
143     end.writeInt(localId);
144     end.writeByte(way);
145     if (version != null) {
146       end.writeBytes(bversion);
147     }
148   }
149 
150   @Override
151   public void createHeader(LocalChannelReference lcr)
152       throws OpenR66ProtocolPacketException {
153     if (hostId == null) {
154       throw new OpenR66ProtocolPacketException(NOT_ENOUGH_DATA);
155     }
156     header = Unpooled.wrappedBuffer(hostId.getBytes());
157   }
158 
159   @Override
160   public void createMiddle(LocalChannelReference lcr)
161       throws OpenR66ProtocolPacketException {
162     if (key == null) {
163       throw new OpenR66ProtocolPacketException(NOT_ENOUGH_DATA);
164     }
165     middle = Unpooled.wrappedBuffer(key);
166   }
167 
168   @Override
169   public byte getType() {
170     return LocalPacketFactory.AUTHENTPACKET;
171   }
172 
173   @Override
174   public String toString() {
175     return "AuthentPacket: " + hostId + ' ' + localId + ' ' + way + ' ' +
176            version;
177   }
178 
179   /**
180    * @return the hostId
181    */
182   public String getHostId() {
183     return hostId;
184   }
185 
186   /**
187    * @return the key
188    */
189   public byte[] getKey() {
190     return key;
191   }
192 
193   /**
194    * @return the localId
195    */
196   public Integer getLocalId() {
197     return localId;
198   }
199 
200   /**
201    * @return True if this packet is to be validated
202    */
203   public boolean isToValidate() {
204     return way == ASKVALIDATE;
205   }
206 
207   /**
208    * Validate the connection
209    */
210   public void validate(boolean isSSL) {
211     way = ANSWERVALIDATE;
212     DbHostAuth auth = isSSL? Configuration.configuration.getHostSslAuth() :
213         Configuration.configuration.getHostAuth();
214     try {
215       hostId = Configuration.configuration.getHostId(isSSL);
216     } catch (final OpenR66ProtocolNoSslException e) {
217       hostId = Configuration.configuration.getHostId();
218       auth = Configuration.configuration.getHostAuth();
219     }
220     key = FilesystemBasedDigest.passwdCrypt(auth.getHostkey());
221     Configuration.configuration.getVersions().putIfAbsent(hostId,
222                                                           new PartnerConfiguration(
223                                                               hostId));
224     version = Configuration.configuration.getVersions().get(hostId).toString();
225     header = null;
226     middle = null;
227     end = null;
228   }
