64    public HttpRestServerR66Handler(RestConfiguration config, METHOD... methods) {
65      super(BASEURI, config, METHOD.OPTIONS);

69    @Override
70    public void endParsingRequest(HttpRestHandler handler, RestArgument arguments,
71                                  RestArgument result, Object body)
72        throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
73      logger.debug("debug: {} ### {}", arguments, result);
74      if (body != null) {
75        logger.debug("Obj: {}", body);
76      }
77      handler.setWillClose(false);
78      final ServerActions serverHandler =
79          ((HttpRestR66Handler) handler).getServerHandler();
80      if (arguments.getMethod() == METHOD.GET) {
81        // status Json
82        result.setCommand(ACTIONS_TYPE.GetStatus.name());
83        Configuration.configuration.getMonitoring().run(0, true);
84        final ObjectNode node =
85            Configuration.configuration.getMonitoring().exportAsJson(true);
86        result.setDetail("Current Waarp R66 status");
87        result.addResult(node);
88        setOk(handler, result, null, HttpResponseStatus.OK);
89        return;
90      }
91      // now action according to body
92      final JsonPacket/../../../../../../org/waarp/openr66/protocol/localhandler/packet/json/JsonPacket.html#JsonPacket">JsonPacket json = (JsonPacket) body;
93      if (json == null) {
94        result.setDetail("not enough information");
95        setError(handler, result, HttpResponseStatus.BAD_REQUEST);
96        return;
97      }
98      result.getAnswer()
99            .put(AbstractDbData.JSON_MODEL, RESTHANDLERS.Server.name());
100     try {
101       if (json instanceof ShutdownOrBlockJsonPacket) {//
102         final ShutdownOrBlockJsonPacket/../org/waarp/openr66/protocol/localhandler/packet/json/ShutdownOrBlockJsonPacket.html#ShutdownOrBlockJsonPacket">ShutdownOrBlockJsonPacket node = (ShutdownOrBlockJsonPacket) json;
103         result.setCommand(ACTIONS_TYPE.ShutdownOrBlock.name());
104         final byte[] key = node.getKey();
105         if (node.isShutdownOrBlock()) {
106           // Shutdown
107           serverHandler.shutdown(key, node.isRestartOrBlock());
108           result.setDetail("Shutdown on going");
109         } else {
110           // Block
111           serverHandler.blockRequest(key, node.isRestartOrBlock());
112           node.setComment(
113               (node.isRestartOrBlock()? "Block" : "Unblock") + " new request");
114           result.setDetail(
115               (node.isRestartOrBlock()? "Block" : "Unblock") + " new request");
116         }
117         setOk(handler, result, json, HttpResponseStatus.OK);
118       } else {
119         logger.info("Validation is ignored: " + json);
120         result.setDetail("Unknown command");
121         setError(handler, result, json, HttpResponseStatus.PRECONDITION_FAILED);
122       }
123     } catch (final OpenR66ProtocolNotAuthenticatedException e) {
124       throw new HttpInvalidAuthenticationException(e);
125     } catch (final OpenR66ProtocolBusinessException e) {
126       throw new HttpIncorrectRequestException(e);
127     } catch (final OpenR66ProtocolShutdownException e) {
128       result.setDetail("Shutdown on going");
129       setOk(handler, result, json, HttpResponseStatus.OK);
130       WaarpShutdownHook.shutdownWillStart();
131       logger.warn(Messages.getString("LocalServerHandler.0") + //$NON-NLS-1$
132                   serverHandler.getSession().getAuth().getUser());
133       // dont'close, thread will do
134       ChannelUtils.startShutdown();
135     }
136   }
