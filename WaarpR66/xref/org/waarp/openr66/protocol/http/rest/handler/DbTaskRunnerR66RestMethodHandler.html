68      INTRANSFER("boolean"), INERROR("boolean"), DONE("boolean"),
69      ALLSTATUS("boolean"), STARTTRANS("Date in ISO 8601 format or ms"),
70      STOPTRANS("Date in ISO 8601 format or ms");

75        this.type = type;

88    @Override
89    protected DbTaskRunner getItem(final HttpRestHandler handler,
90                                   final RestArgument arguments,
91                                   final RestArgument result, final Object body)
92        throws HttpIncorrectRequestException, HttpInvalidAuthenticationException,
93               HttpNotFoundRequestException {
94      final ObjectNode arg = arguments.getUriArgs().deepCopy();
95      arg.setAll(arguments.getBody());
96      try {
97        final JsonNode node = RestArgument.getId(arg);
98        final long id;
99        if (node.isMissingNode()) {
100         // shall not be but continue however
101         id = arg.path(DbTaskRunner.Columns.SPECIALID.name()).asLong();
102       } else {
103         id = node.asLong();
104       }
105       return new DbTaskRunner(id,
106                               arg.path(DbTaskRunner.Columns.REQUESTER.name())
107                                  .asText(),
108                               arg.path(DbTaskRunner.Columns.REQUESTED.name())
109                                  .asText(),
110                               arg.path(DbTaskRunner.Columns.OWNERREQ.name())
111                                  .asText());
112     } catch (final WaarpDatabaseException e) {
113       throw new HttpNotFoundRequestException(
114           "Issue while reading from database " + arg, e);
115     }
116   }

134   @Override
135   protected DbPreparedStatement getPreparedStatement(
136       final HttpRestHandler handler, final RestArgument arguments,
137       final RestArgument result, final Object body)
138       throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
139     final ObjectNode arg = arguments.getUriArgs().deepCopy();
140     arg.setAll(arguments.getBody());
141     final int limit = arg.path(FILTER_ARGS.LIMIT.name()).asInt(0);
142     final boolean orderBySpecialId =
143         arg.path(FILTER_ARGS.ORDERBYID.name()).asBoolean(false);
144     JsonNode node = arg.path(FILTER_ARGS.STARTID.name());
145     String startid = null;
146     if (!node.isMissingNode()) {
147       startid = node.asText();
148     }
149     if (startid == null || startid.isEmpty()) {
150       startid = null;
151     }
152     node = arg.path(FILTER_ARGS.STOPID.name());
153     String stopid = null;
154     if (!node.isMissingNode()) {
155       stopid = node.asText();
156     }
157     if (stopid == null || stopid.isEmpty()) {
158       stopid = null;
159     }
160     String rule = arg.path(FILTER_ARGS.IDRULE.name()).asText();
161     if (rule == null || rule.isEmpty()) {
162       rule = null;
163     }
164     String req = arg.path(FILTER_ARGS.PARTNER.name()).asText();
165     if (req == null || req.isEmpty()) {
166       req = null;
167     }
168     String owner = arg.path(DbTaskRunner.Columns.OWNERREQ.name()).asText();
169     if (owner == null || owner.isEmpty()) {
170       owner = null;
171     }
172     final boolean pending =
173         arg.path(FILTER_ARGS.PENDING.name()).asBoolean(false);
174     final boolean transfer =
175         arg.path(FILTER_ARGS.INTRANSFER.name()).asBoolean(false);
176     final boolean error = arg.path(FILTER_ARGS.INERROR.name()).asBoolean(false);
177     final boolean done = arg.path(FILTER_ARGS.DONE.name()).asBoolean(false);
178     final boolean all = arg.path(FILTER_ARGS.ALLSTATUS.name()).asBoolean(false);
179     Timestamp start = null;
180     node = arg.path(FILTER_ARGS.STARTTRANS.name());
181     if (!node.isMissingNode()) {
182       long val = node.asLong();
183       if (val == 0) {
184         final DateTime received = DateTime.parse(node.asText());
185         val = received.getMillis();
186       }
187       start = new Timestamp(val);
188     }
189     Timestamp stop = null;
190     node = arg.path(FILTER_ARGS.STOPTRANS.name());
191     if (!node.isMissingNode()) {
192       long val = node.asLong();
193       if (val == 0) {
194         final DateTime received = DateTime.parse(node.asText());
195         val = received.getMillis();
196       }
197       stop = new Timestamp(val);
198     }
199     try {
200       return DbTaskRunner
201           .getFilterPrepareStatement(handler.getDbSession(), limit,
202                                      orderBySpecialId, startid, stopid, start,
203                                      stop, rule, req, pending, transfer, error,
204                                      done, all, owner);
205     } catch (final WaarpDatabaseNoConnectionException e) {
206       throw new HttpIncorrectRequestException(
207           "Issue while reading from database", e);
208     } catch (final WaarpDatabaseSqlException e) {
209       throw new HttpIncorrectRequestException(
210           "Issue while reading from database", e);
211     }
212   }

229   @Override
230   protected ArrayNode getDetailedAllow() {
231     final ArrayNode node = JsonHandler.createArrayNode();
232 
233     final ObjectNode node1 = JsonHandler.createObjectNode();
234     node1.put(AbstractDbData.JSON_MODEL, DbTaskRunner.class.getSimpleName());
235     for (final DbTaskRunner.Columns column : DbTaskRunner.Columns.values()) {
236       node1.put(column.name(), DbTaskRunner.dbTypes[column.ordinal()]);
237     }
238 
239     ObjectNode node2;
240     ObjectNode node3 = JsonHandler.createObjectNode();
241     if (methods.contains(METHOD.GET)) {
242       node3.put(DbTaskRunner.Columns.SPECIALID.name(),
243                 SPECIAL_ID_AS_LONG_IN_URI_AS + path + "/id");
244       node3.put(DbTaskRunner.Columns.REQUESTER.name(),
245                 PARTNER_AS_REQUESTER_AS_VARCHAR);
246       node3.put(DbTaskRunner.Columns.REQUESTED.name(),
247                 PARTNER_AS_REQUESTED_AS_VARCHAR);
248       node3.put(DbTaskRunner.Columns.OWNERREQ.name(),
249                 OWNER_OF_THIS_REQUEST_OPTIONAL_AS_VARCHAR);
250       node2 = RestArgument
251           .fillDetailedAllow(METHOD.GET, path + "/id", COMMAND_TYPE.GET.name(),
252                              node3, node1);
253       node.add(node2);
254 
255       node3 = JsonHandler.createObjectNode();
256       for (final FILTER_ARGS arg : FILTER_ARGS.values()) {
257         node3.put(arg.name(), arg.type);
258       }
259       node3.put(DbTaskRunner.Columns.OWNERREQ.name(),
260                 OWNER_OF_THIS_REQUEST_OPTIONAL_AS_VARCHAR);
261       node2 = RestArgument
262           .fillDetailedAllow(METHOD.GET, path, COMMAND_TYPE.MULTIGET.name(),
263                              node3, JsonHandler.createArrayNode().add(node1));
264       node.add(node2);
265     }
266     if (methods.contains(METHOD.PUT)) {
267       node3 = JsonHandler.createObjectNode();
268       node3.put(DbTaskRunner.Columns.SPECIALID.name(),
269                 SPECIAL_ID_AS_LONG_IN_URI_AS + path + "/id");
270       node3.put(DbTaskRunner.Columns.REQUESTER.name(),
271                 PARTNER_AS_REQUESTER_AS_VARCHAR);
272       node3.put(DbTaskRunner.Columns.REQUESTED.name(),
273                 PARTNER_AS_REQUESTED_AS_VARCHAR);
274       node3.put(DbTaskRunner.Columns.OWNERREQ.name(),
275                 OWNER_OF_THIS_REQUEST_OPTIONAL_AS_VARCHAR);
276       for (final DbTaskRunner.Columns column : DbTaskRunner.Columns.values()) {
277         if (column.name()
278                   .equalsIgnoreCase(DbTaskRunner.Columns.IDRULE.name())) {
279           continue;
280         }
281         node3.put(column.name(), DbTaskRunner.dbTypes[column.ordinal()]);
282       }
283       node2 = RestArgument.fillDetailedAllow(METHOD.PUT, path + "/id",
284                                              COMMAND_TYPE.UPDATE.name(), node3,
285                                              node1);
286       node.add(node2);
287     }
288     if (methods.contains(METHOD.DELETE)) {
289       node3 = JsonHandler.createObjectNode();
290       node3.put(DbTaskRunner.Columns.SPECIALID.name(),
291                 SPECIAL_ID_AS_LONG_IN_URI_AS + path + "/id");
292       node3.put(DbTaskRunner.Columns.REQUESTER.name(),
293                 PARTNER_AS_REQUESTER_AS_VARCHAR);
294       node3.put(DbTaskRunner.Columns.REQUESTED.name(),
295                 PARTNER_AS_REQUESTED_AS_VARCHAR);
296       node3.put(DbTaskRunner.Columns.OWNERREQ.name(),
297                 OWNER_OF_THIS_REQUEST_OPTIONAL_AS_VARCHAR);
298       node2 = RestArgument.fillDetailedAllow(METHOD.DELETE, path + "/id",
299                                              COMMAND_TYPE.DELETE.name(), node3,
300                                              node1);
301       node.add(node2);
302     }
303     if (methods.contains(METHOD.POST)) {
304       node3 = JsonHandler.createObjectNode();
305       for (final DbTaskRunner.Columns column : DbTaskRunner.Columns.values()) {
306         node3.put(column.name(), DbTaskRunner.dbTypes[column.ordinal()]);
307       }
308       node2 = RestArgument
309           .fillDetailedAllow(METHOD.POST, path, COMMAND_TYPE.CREATE.name(),
310                              node3, node1);
311       node.add(node2);
312     }
313     node2 = RestArgument
314         .fillDetailedAllow(METHOD.OPTIONS, path, COMMAND_TYPE.OPTIONS.name(),
315                            null, null);
316     node.add(node2);
317 
318     return node;
319   }
