21  package org.waarp.openr66.protocol.http.restv2.dbhandlers;
22  
23  import io.cdap.http.AbstractHttpHandler;
24  import io.netty.handler.codec.http.HttpMethod;
25  import io.netty.handler.codec.http.HttpRequest;
26  import org.waarp.gateway.kernel.rest.RestConfiguration.CRUD;
27  
28  import static io.netty.handler.codec.http.HttpMethod.*;
29  
30  /**
31   * This abstract class represents a handler for the WaarpR66 REST API. A handler
32   * can handle one or multiple
33   * entry points with multiple methods each.
34   * <p>
35   * To assign an entry point to a handler, use the {@link javax.ws.rs.Path}
36   * annotation with the URI as value.
37   * This annotation can also be used on methods of the handler to create entry
38   * points below the handler's one.
39   * <p>
40   * For each entry point, all the desired HTTP methods must be assigned a
41   * corresponding Java method using the
42   * appropriate {@link javax.ws.rs.HttpMethod}. These methods must all have an
43   * {@link HttpRequest} and a
44   * {@link io.cdap.http.HttpResponder} as their first 2 parameters, which
45   * correspond respectively to the
46   * request and its reply. The methods may also have other parameters marked with
47   * the
48   * {@link javax.ws.rs.PathParam} annotation, meaning that these parameters
49   * correspond to the {id} given in the
50   * URI of the request. Some parameters may also be marked with the {@link
51   * javax.ws.rs.QueryParam} an
52   * annotation, in which case the parameter marked is referring to the query
53   * parameter whose name is given in
54   * the annotation's parameter. A query parameter can be given a default value
55   * with the
56   * {@link javax.ws.rs.DefaultValue} which will be assigned to the parameter when
57   * the request does not contain
58   * a value for it.
59   * <p>
60   * These methods should also be annotated with the {@link javax.ws.rs.Consumes}
61   * annotation to indicate the
62   * expected content type of the request. Finally, each method should also be
63   * annotated with the
64   * {@link RequiredRole} annotation to make the method only accessible to users
65   * with the specified
66   * {@link org.waarp.common.role.RoleDefault.ROLE} or higher.
67   */
68  public abstract class AbstractRestDbHandler extends AbstractHttpHandler {
69  
70    /**
71     * A byte mask defining which HTTP methods are allowed on the handler.
72     */
73    protected final byte crud;
74  
75    /**
76     * Initializes the handler with the given CRUD mask.
77     *
78     * @param crud the CRUD mask for this handler
79     */
80    protected AbstractRestDbHandler(final byte crud) {
81      this.crud = crud;
82    }
83  
84    /**
85     * Checks whether the {@link HttpRequest} given can be made on the handler
86     * in accordance with the handler's
87     * CRUD configuration.
88     *
89     * @param request the HTTP request to check
90     *
91     * @return {@code true} if the request is active, {@code false} otherwise.
92     */
93    public boolean checkCRUD(final HttpRequest request) {
94      final HttpMethod method = request.method();
95      if (method.equals(GET)) {
96        return CRUD.READ.isValid(crud);
97      } else if (method.equals(POST)) {
98        return CRUD.CREATE.isValid(crud);
99      } else if (method.equals(DELETE)) {
100       return CRUD.DELETE.isValid(crud);
101     } else if (method.equals(PUT)) {
102       return CRUD.UPDATE.isValid(crud);
103     } else {
104       return method.equals(OPTIONS);
105     }
106   }
107 }
