95    /**
96     * Method called to get a list of host entries from the server's database,
97     * with optional filters applied.
98     *
99     * @param request The {@link HttpRequest} made on the resource.
100    * @param responder The {@link HttpResponder} which sends the reply
101    *     to the request.
102    * @param limit_str HTTP query parameter, maximum number of entries
103    *     allowed in the response.
104    * @param offset_str HTTP query parameter, index of the first
105    *     accepted entry in the list of all valid
106    *     answers.
107    * @param order_str HTTP query parameter, the criteria used to sort
108    *     the entries and the way of ordering.
109    * @param address HTTP query parameter, filter only hosts with this
110    *     address.
111    * @param isSSL_str HTTP query parameter, filter only hosts that use
112    *     SSL, or those that don't. Leave empty
113    *     to get both.
114    * @param isActive_str HTTP query parameter, filter hosts that are
115    *     active, or those that aren't. Leave empty
116    *     to get both.
117    */
118   @GET
119   @Consumes(APPLICATION_FORM_URLENCODED)
120   @RequiredRole(ROLE.READONLY)
121   public void filterHosts(HttpRequest request, HttpResponder responder,
122                           @QueryParam(LIMIT) @DefaultValue("20")
123                               String limit_str,
124                           @QueryParam(OFFSET) @DefaultValue("0")
125                               String offset_str,
126                           @QueryParam(ORDER) @DefaultValue("ascId")
127                               String order_str,
128                           @QueryParam(ADDRESS) String address,
129                           @QueryParam(IS_SSL) String isSSL_str,
130                           @QueryParam(IS_ACTIVE) String isActive_str) {
131 
132     final List<RestError> errors = new ArrayList<RestError>();
133 
134     int limit = 20;
135     int offset = 0;
136     HostConverter.Order order = HostConverter.Order.ascId;
137     try {
138       limit = Integer.parseInt(limit_str);
139     } catch (final NumberFormatException e) {
140       errors.add(ILLEGAL_PARAMETER_VALUE(LIMIT, limit_str));
141     }
142     try {
143       order = HostConverter.Order.valueOf(order_str);
144     } catch (final IllegalArgumentException e) {
145       errors.add(ILLEGAL_PARAMETER_VALUE(ORDER, order_str));
146     }
147     try {
148       offset = Integer.parseInt(offset_str);
149     } catch (final NumberFormatException e) {
150       errors.add(ILLEGAL_PARAMETER_VALUE(OFFSET, offset_str));
151     }
152 
153     if (limit < 0) {
154       errors.add(ILLEGAL_PARAMETER_VALUE(LIMIT, limit_str));
155     } else if (offset < 0) {
156       errors.add(ILLEGAL_PARAMETER_VALUE(OFFSET, offset_str));
157     }
158 
159     boolean isSSL = false;
160     boolean isActive = false;
161     if (isSSL_str != null) {
162       try {
163         isSSL = RestUtils.stringToBoolean(isSSL_str);
164       } catch (final IllegalArgumentException e) {
165         errors.add(ILLEGAL_PARAMETER_VALUE(IS_SSL, isSSL_str));
166       }
167     }
168     if (isActive_str != null) {
169       try {
170         isActive = RestUtils.stringToBoolean(isActive_str);
171       } catch (final IllegalArgumentException e) {
172         errors.add(ILLEGAL_PARAMETER_VALUE(IS_ACTIVE, isActive_str));
173       }
174     }
175 
176     if (!errors.isEmpty()) {
177       throw new RestErrorException(errors);
178     }
179 
180     final List<Filter> filters = new ArrayList<Filter>();
181     if (address != null) {
182       filters.add(new Filter(ADDRESS_FIELD, "=", address));
183     }
184     if (isSSL_str != null) {
185       filters.add(new Filter(IS_SSL_FIELD, "=", isSSL));
186     }
187     if (isActive_str != null) {
188       filters.add(new Filter(IS_ACTIVE_FIELD, "=", isActive));
189     }
190     List<Host> hosts;
191     HostDAO hostDAO = null;
192     try {
193       hostDAO = DAO_FACTORY.getHostDAO();
194       hosts = hostDAO.find(filters);
195     } catch (final DAOConnectionException e) {
196       throw new InternalServerErrorException(e);
197     } finally {
198       DAOFactory.closeDAO(hostDAO);
199     }
200 
201     final int totalResults = hosts.size();
202     Collections.sort(hosts, order.comparator);
203 
204     final ArrayNode results = new ArrayNode(JsonNodeFactory.instance);
205     for (int i = offset; i < offset + limit && i < hosts.size(); i++) {
206       results.add(HostConverter.hostToNode(hosts.get(i)));
207     }
208 
209     final ObjectNode responseObject = new ObjectNode(JsonNodeFactory.instance);
210     responseObject.put("totalResults", totalResults);
211     responseObject.set("results", results);
212 
213     final String responseText = JsonUtils.nodeToString(responseObject);
214     responder.sendJson(OK, responseText);
215   }

217   /**
218    * Method called to add a new host authentication entry to the server
219    * database.
220    *
221    * @param request The {@link HttpRequest} made on the resource.
222    * @param responder The {@link HttpResponder} which sends the reply
223    *     to the request.
224    */
225   @POST
226   @Consumes(APPLICATION_JSON)
227   @RequiredRole(ROLE.HOST)
228   public void addHost(HttpRequest request, HttpResponder responder) {
229 
230     final ObjectNode requestObject = JsonUtils.deserializeRequest(request);
231     final Host host = HostConverter.nodeToNewHost(requestObject);
232 
233     HostDAO hostDAO = null;
234     try {
235       hostDAO = DAO_FACTORY.getHostDAO();
236 
237       if (!hostDAO.exist(host.getHostid())) {
238         throw new RestErrorException(ALREADY_EXISTING(host.getHostid()));
239       }
240 
241       hostDAO.insert(host);
242     } catch (final DAOConnectionException e) {
243       throw new InternalServerErrorException(e);
244     } finally {
245       DAOFactory.closeDAO(hostDAO);
246     }
247 
248     final ObjectNode responseObject = HostConverter.hostToNode(host);
249     final String responseText = JsonUtils.nodeToString(responseObject);
250 
251     final DefaultHttpHeaders headers = new DefaultHttpHeaders();
252     headers.add(CONTENT_TYPE, APPLICATION_JSON);
253     headers.add("host-uri", HOSTS_HANDLER_URI + host.getHostid());
254 
255     responder.sendString(CREATED, responseText, headers);
256   }
