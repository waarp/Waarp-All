78    /**
79     * Method called to obtain the information on the transfer whose id was
80     * given in the request's URI. The
81     * requested transfer is sent back in JSON format, unless an unexpected
82     * error prevents it or if the request id
83     * does not exist.
84     * <p>
85     * **NOTE:** The {@code uri} parameter refers to the concatenation of the
86     * transfer's id, and the name of the
87     * host to which the transfer was requested, separated by an underscore
88     * character.
89     *
90     * @param request the HttpRequest made to the resource
91     * @param responder the HttpResponder which sends the reply to the
92     *     request
93     * @param uri the transfer's unique identifier
94     */
95    @GET
96    @Consumes(WILDCARD)
97    @RequiredRole(READONLY)
98    public void getTransfer(HttpRequest request, HttpResponder responder,
99                            @PathParam(URI_ID) String uri)
100       throws UnsupportedEncodingException {
101 
102     final String key = URLDecoder.decode(uri, UTF8_CHARSET.name());
103     final Pattern pattern = Pattern.compile("(-?\\d+)_(.+)");
104     final Matcher matcher = pattern.matcher(key);
105     if (!matcher.find()) {
106       responder.sendStatus(NOT_FOUND);
107       return;
108     }
109     final String id = matcher.group(1);
110     final String requested = matcher.group(2);
111 
112     TransferDAO transferDAO = null;
113     try {
114       final long transID = Long.parseLong(id);
115       transferDAO = DAO_FACTORY.getTransferDAO();
116       if (!transferDAO.exist(transID, SERVER_NAME, requested, SERVER_NAME)) {
117         responder.sendStatus(NOT_FOUND);
118       } else {
119         final Transfer transfer =
120             transferDAO.select(transID, SERVER_NAME, requested, SERVER_NAME);
121         final ObjectNode response = TransferConverter.transferToNode(transfer);
122         final String responseText = JsonUtils.nodeToString(response);
123         responder.sendJson(OK, responseText);
124       }
125     } catch (final NumberFormatException e) {
126       responder.sendStatus(NOT_FOUND);
127     } catch (final DAOConnectionException e) {
128       throw new InternalServerErrorException(e);
129     } catch (final DAONoDataException e) {
130       responder.sendStatus(NOT_FOUND);
131     } finally {
132       DAOFactory.closeDAO(transferDAO);
133     }
134   }

323    * **NOTE:** The {@code uri} parameter refers to the concatenation of the

327    *

329    * @param responder the HttpResponder which sends the reply to the

333   @OPTIONS
334   @Consumes(WILDCARD)
335   @RequiredRole(NOACCESS)
336   public void options(HttpRequest request, HttpResponder responder,
337                       @PathParam(URI_ID) String uri) {
338     final HttpHeaders allow = new DefaultHttpHeaders();
339     allow.add(ALLOW, HttpMethod.OPTIONS);
340     responder.sendStatus(OK, allow);
341   }

343   @Path("{ep}")
344   @OPTIONS
345   @Consumes(WILDCARD)
346   @RequiredRole(NOACCESS)
347   public void subOptions(HttpRequest request, HttpResponder responder,
348                          @PathParam(URI_ID) String uri,
349                          @PathParam("ep") String ep) {
350 
351     final HttpHeaders allow = new DefaultHttpHeaders();
352     final List<HttpMethod> methods = new ArrayList<HttpMethod>();
353 
354     if (ep.equals(TransferCommandsURI.RESTART_URI)) {
355       methods.add(HttpMethod.PUT);
356     } else if (ep.equals(TransferCommandsURI.STOP_URI)) {
357       methods.add(HttpMethod.PUT);
358     } else if (ep.equals(TransferCommandsURI.CANCEL_URI)) {
359       methods.add(HttpMethod.PUT);
360     } else {
361       responder.sendStatus(NOT_FOUND);
362       return;
363     }
364     methods.add(HttpMethod.OPTIONS);
365     allow.add(ALLOW, methods);
366     responder.sendStatus(OK, allow);
367   }
