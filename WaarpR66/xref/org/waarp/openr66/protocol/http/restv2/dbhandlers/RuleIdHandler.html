54  /**
55   * This is the {@link AbstractRestDbHandler} handling all requests made on a
56   * single rule REST entry point.
57   */
58  @Path(RULE_ID_HANDLER_URI)
59  public class RuleIdHandler extends AbstractRestDbHandler {
60  
61    /**
62     * The content of the 'Allow' header sent when an 'OPTIONS' request is made
63     * on the handler.
64     */
65    private static final HttpHeaders OPTIONS_HEADERS;
66  
67    static {
68      OPTIONS_HEADERS = new DefaultHttpHeaders();
69      final List<HttpMethod> allow = new ArrayList<HttpMethod>();
70      allow.add(HttpMethod.GET);
71      allow.add(HttpMethod.PUT);
72      allow.add(HttpMethod.DELETE);
73      allow.add(HttpMethod.OPTIONS);
74      OPTIONS_HEADERS.add(ALLOW, allow);
75    }
76  
77    /**
78     * Initializes the handler with the given CRUD mask.
79     *
80     * @param crud the CRUD mask for this handler
81     */
82    public RuleIdHandler(byte crud) {
83      super(crud);
84    }
85  
86    /**
87     * Method called to retrieve a transfer rule with the id given in the
88     * request URI.
89     *
90     * @param request the HttpRequest made on the resource
91     * @param responder the HttpResponder which sends the reply to the
92     *     request
93     * @param id the requested rule's name
94     */
95    @GET
96    @Consumes(WILDCARD)
97    @RequiredRole(NOACCESS)
98    public void getRule(HttpRequest request, HttpResponder responder,
99                        @PathParam(URI_ID) String id) {
100 
101     RuleDAO ruleDAO = null;
102     try {
103       ruleDAO = DAO_FACTORY.getRuleDAO();
104       final Rule rule = ruleDAO.select(id);
105 
106       if (rule == null) {
107         responder.sendStatus(NOT_FOUND);
108         return;
109       }
110       final ObjectNode responseObject = RuleConverter.ruleToNode(rule);
111       final String responseText = JsonUtils.nodeToString(responseObject);
112       responder.sendJson(OK, responseText);
113     } catch (final DAOConnectionException e) {
114       throw new InternalServerErrorException(e);
115     } catch (final DAONoDataException e) {
116       responder.sendStatus(NOT_FOUND);
117     } finally {
118       DAOFactory.closeDAO(ruleDAO);
119     }
120   }
121 
122   /**
123    * Method called to update a transfer rule.
124    *
125    * @param request the HttpRequest made on the resource
126    * @param responder the HttpResponder which sends the reply to the
127    *     request
128    * @param id the requested rule's name
129    */
130   @PUT
131   @Consumes(APPLICATION_JSON)
132   @RequiredRole(RULE)
133   public void updateRule(HttpRequest request, HttpResponder responder,
134                          @PathParam(URI_ID) String id) {
135 
136     RuleDAO ruleDAO = null;
137     try {
138       ruleDAO = DAO_FACTORY.getRuleDAO();
139       final Rule oldRule = ruleDAO.select(id);
140 
141       if (oldRule == null) {
142         responder.sendStatus(NOT_FOUND);
143         return;
144       }
145 
146       final ObjectNode requestObject = JsonUtils.deserializeRequest(request);
147       final Rule newRule =
148           RuleConverter.nodeToUpdatedRule(requestObject, oldRule);
149 
150       ruleDAO.update(newRule);
151 
152       final ObjectNode responseObject = RuleConverter.ruleToNode(newRule);
153       final String responseText = JsonUtils.nodeToString(responseObject);
154       responder.sendJson(CREATED, responseText);
155     } catch (final DAOConnectionException e) {
156       throw new InternalServerErrorException(e);
157     } catch (final DAONoDataException e) {
158       responder.sendStatus(NOT_FOUND);
159     } finally {
160       DAOFactory.closeDAO(ruleDAO);
161     }
162   }
163 
164   /**
165    * Method called to delete a transfer rule from the database.
166    *
167    * @param request the HttpRequest made on the resource
168    * @param responder the HttpResponder which sends the reply to the
169    *     request
170    * @param id the requested rule's name
171    */
172   @DELETE
173   @Consumes(WILDCARD)
174   @RequiredRole(RULE)
175   public void deleteRule(HttpRequest request, HttpResponder responder,
176                          @PathParam(URI_ID) String id) {
177 
178     RuleDAO ruleDAO = null;
179     try {
180       ruleDAO = DAO_FACTORY.getRuleDAO();
181       final Rule rule = ruleDAO.select(id);
182       if (rule == null) {
183         responder.sendStatus(NOT_FOUND);
184       } else {
185         ruleDAO.delete(rule);
186         responder.sendStatus(NO_CONTENT);
187       }
188     } catch (final DAOConnectionException e) {
189       throw new InternalServerErrorException(e);
190     } catch (final DAONoDataException e) {
191       responder.sendStatus(NOT_FOUND);
192     } finally {
193       DAOFactory.closeDAO(ruleDAO);
194     }
195   }
196 
197   /**
198    * Method called to get a list of all allowed HTTP methods on this entry
199    * point. The HTTP methods are sent as
200    * an array in the reply's headers.
201    *
202    * @param request the HttpRequest made on the resource
203    * @param responder the HttpResponder which sends the reply to the
204    *     request
205    * @param id the requested rule's name
206    */
207   @OPTIONS
208   @Consumes(WILDCARD)
209   @RequiredRole(NOACCESS)
210   public void options(HttpRequest request, HttpResponder responder,
211                       @PathParam(URI_ID) String id) {
212     responder.sendStatus(OK, OPTIONS_HEADERS);
213   }
