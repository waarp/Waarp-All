108   @Override
109   public void channelInactive(ChannelHandlerContext ctx) throws Exception {
110     try {
111       if (networkChannelReference != null) {
112         if (networkChannelReference.nbLocalChannels() > 0) {
113           logger.info("Network Channel Closed: {} LocalChannels Left: {}",
114                       ctx.channel().id(),
115                       networkChannelReference.nbLocalChannels());
116           // Give an extra time if necessary to let the local channel being closed
117           try {
118             Thread.sleep(Configuration.RETRYINMS * 2);
119           } catch (final InterruptedException e1) {//NOSONAR
120             SysErrLogger.FAKE_LOGGER.ignoreLog(e1);
121           }
122         }
123         try {
124           NetworkTransaction.closedNetworkChannel(networkChannelReference);
125         } catch (RejectedExecutionException e) {
126           logger.debug(e);
127         }
128       } else {
129         if (remoteAddress == null) {
130           remoteAddress = ctx.channel().remoteAddress();
131         }
132         try {
133           NetworkTransaction.closedNetworkChannel(remoteAddress);
134         } catch (RejectedExecutionException e) {
135           logger.debug(e);
136         }
137       }
138       // Now force the close of the database after a wait
139       if (dbSession != null && admin != null && admin.getSession() != null &&
140           !dbSession.equals(admin.getSession())) {
141         dbSession.forceDisconnect();
142         dbSession = null;
143       }
144     } catch (RejectedExecutionException e) {
145       logger.debug(e);
146     }
147   }

149   @Override
150   public void channelActive(ChannelHandlerContext ctx) throws Exception {
151     final Channel netChannel = ctx.channel();
152     remoteAddress = netChannel.remoteAddress();
153     logger.debug(
154         "Will the Connection be refused if Partner is BlackListed from " +
155         remoteAddress);
156     if (NetworkTransaction.isBlacklisted(netChannel)) {
157       logger.warn("Connection refused since Partner is BlackListed from " +
158                   remoteAddress);
159       isBlackListed = true;
160       if (Configuration.configuration.getR66Mib() != null) {
161         Configuration.configuration.getR66Mib().notifyError(
162             "Black Listed connection temptative", "During connection");
163       }
164       // close immediately the connection
165       WaarpSslUtility.closingSslChannel(netChannel);
166       return;
167     }
168     try {
169       networkChannelReference =
170           NetworkTransaction.addNetworkChannel(netChannel, isSSL);
171     } catch (final OpenR66ProtocolRemoteShutdownException e2) {
172       logger.warn("Connection refused since Partner is in Shutdown from " +
173                   remoteAddress + " : {}", e2.getMessage());
174       isBlackListed = true;
175       // close immediately the connection
176       WaarpSslUtility.closingSslChannel(netChannel);
177       return;
178     }
179     try {
180       if (admin.isCompatibleWithThreadSharedConnexion()) {
181         dbSession = new DbSession(admin, false);
182         dbSession.useConnection();
183       } else {
184         logger.debug("DbSession will be adjusted on LocalChannelReference");
185         dbSession = admin.getSession();
186       }
187     } catch (final WaarpDatabaseNoConnectionException e1) {
188       // Cannot connect so use default connection
189       logger.warn("Use default database connection");
190       dbSession = admin.getSession();
191     }
192     logger.debug("Network Channel Connected: {} ", ctx.channel().id());
193   }

195   @Override
196   public void userEventTriggered(ChannelHandlerContext ctx, Object evt)
197       throws Exception {
198     if (Configuration.configuration.isShutdown()) {
199       return;
200     }
201     if (evt instanceof IdleStateEvent) {
202       if (networkChannelReference != null && networkChannelReference
203                                                  .checkLastTime(
204                                                      Configuration.configuration
205                                                          .getTimeoutCon() *
206                                                      2) <= 0) {
207         keepAlivedSent.set(0);
208         return;
209       }
210       if (keepAlivedSent.get() > 0) {
211         if (networkChannelReference != null &&
212             networkChannelReference.nbLocalChannels() > 0 &&
213             keepAlivedSent.get() < 5) {
214           // ignore this time
215           keepAlivedSent.getAndIncrement();
216           return;
217         }
218         logger.error("Not getting KAlive: closing channel");
219         if (Configuration.configuration.getR66Mib() != null) {
220           Configuration.configuration.getR66Mib()
221                                      .notifyWarning("KeepAlive get no answer",
222                                                     "Closing network connection");
223         }
224         ChannelCloseTimer.closeFutureChannel(ctx.channel());
225       } else {
226         keepAlivedSent.set(1);
227         final KeepAlivePacketket/KeepAlivePacket.html#KeepAlivePacket">KeepAlivePacket keepAlivePacket = new KeepAlivePacket();
228         final NetworkPacket response =
229             new NetworkPacket(ChannelUtils.NOCHANNEL, ChannelUtils.NOCHANNEL,
230                               keepAlivePacket, null);
231         logger.info("Write KAlive");
232         ctx.channel().writeAndFlush(response);
233       }
234     }
235   }

241   @Override
242   public void channelRead0(ChannelHandlerContext ctx, NetworkPacket msg)
243       throws Exception {
244     if (isBlackListed) {
245       // ignore message since close on going
246       msg.clear();
247       return;
248     }
249     final Channel channel = ctx.channel();
250     if (msg.getCode() == LocalPacketFactory.NOOPPACKET) {
251       if (networkChannelReference != null) {
252         networkChannelReference.useIfUsed();
253       }
254       msg.clear();
255       // Do nothing
256       return;
257     } else if (msg.getCode() == LocalPacketFactory.CONNECTERRORPACKET) {
258       logger.debug("NetworkRecv: {}", msg);
259       // Special code to STOP here
260       if (msg.getLocalId() == ChannelUtils.NOCHANNEL) {
261         final int nb = networkChannelReference.nbLocalChannels();
262         if (nb > 0) {
263           logger.warn(
264               "Temptative of connection failed but still some connection are there so not closing the server channel immediately: " +
265               nb);
266           NetworkTransaction
267               .shuttingDownNetworkChannel(networkChannelReference);
268           return;
269         }
270         // No way to know what is wrong: close all connections with
271         // remote host
272         logger.error(
273             "Will close NETWORK channel, Cannot continue connection with remote Host: " +
274             msg + " : " + channel.remoteAddress() + " : " + nb);
275         WaarpSslUtility.closingSslChannel(channel);
276         msg.clear();
277         return;
278       }
279     } else if (msg.getCode() == LocalPacketFactory.KEEPALIVEPACKET) {
280       if (networkChannelReference != null) {
281         networkChannelReference.useIfUsed();
282       }
283       keepAlivedSent.set(0);
284       try {
285         final KeepAlivePacket keepAlivePacket =
286             (KeepAlivePacket) LocalPacketCodec
287                 .decodeNetworkPacket(msg.getBuffer());
288         if (keepAlivePacket.isToValidate()) {
289           keepAlivePacket.validate();
290           final NetworkPacket response =
291               new NetworkPacket(ChannelUtils.NOCHANNEL, ChannelUtils.NOCHANNEL,
292                                 keepAlivePacket, null);
293           logger.info("Answer KAlive");
294           ctx.writeAndFlush(response);
295         } else {
296           logger.info("Get KAlive");
297         }
298       } catch (final OpenR66ProtocolPacketException ignored) {
299         // nothing
300       }
301       msg.clear();
302       return;
303     }
304     logger.trace("TRACE ID GET MSG: {}", msg);
305     networkChannelReference.use();
306     LocalChannelReference localChannelReference;
307     if (msg.getLocalId() == ChannelUtils.NOCHANNEL) {
308       localChannelReference = NetworkTransaction
309           .createConnectionFromNetworkChannelStartup(networkChannelReference,
310                                                      msg, isSSL);
311     } else {
312       if (msg.getCode() == LocalPacketFactory.ENDREQUESTPACKET) {
313         // Coming from remote
314         try {
315           localChannelReference =
316               Configuration.configuration.getLocalTransaction()
317                                          .getClient(msg.getRemoteId(),
318                                                     msg.getLocalId());
319         } catch (final OpenR66ProtocolSystemException e1) {
320           // do not send anything since the packet is external
321           try {
322             logger.debug(
323                 "Cannot get LocalChannel while an end of request comes: {}",
324                 LocalPacketCodec.decodeNetworkPacket(msg.getBuffer()));
325           } catch (final OpenR66ProtocolPacketException e2) {
326             logger.debug(
327                 "Cannot get LocalChannel while an end of request comes: {}",
328                 msg.toString());
329           }
330           msg.clear();
331           return;
332         }
333         // OK continue and send to the local channel
334       } else if (msg.getCode() == LocalPacketFactory.CONNECTERRORPACKET) {
335         // Not a local error but a remote one
336         try {
337           localChannelReference =
338               Configuration.configuration.getLocalTransaction()
339                                          .getClient(msg.getRemoteId(),
340                                                     msg.getLocalId());
341         } catch (final OpenR66ProtocolSystemException e1) {
342           // do not send anything since the packet is external
343           try {
344             logger.debug(
345                 "Cannot get LocalChannel while an external error comes: {}",
346                 LocalPacketCodec.decodeNetworkPacket(msg.getBuffer()));
347           } catch (final OpenR66ProtocolPacketException e2) {
348             logger.debug(
349                 "Cannot get LocalChannel while an external error comes: {}",
350                 msg.toString());
351           }
352           msg.clear();
353           return;
354         }
355         // OK continue and send to the local channel
356       } else {
357         try {
358           localChannelReference =
359               Configuration.configuration.getLocalTransaction()
360                                          .getClient(msg.getRemoteId(),
361                                                     msg.getLocalId());
362         } catch (final OpenR66ProtocolSystemException e1) {
363           if (remoteAddress == null) {
364             remoteAddress = channel.remoteAddress();
365           }
366           if (NetworkTransaction.isShuttingdownNetworkChannel(remoteAddress) ||
367               WaarpShutdownHook.isShutdownStarting()) {
368             // ignore
369             msg.clear();
370             return;
371           }
372           // try to send later
373           logger.debug(
374               "Cannot get LocalChannel: " + msg + " due to " + e1.getMessage());
375           final ConnectionErrorPacket/packet/ConnectionErrorPacket.html#ConnectionErrorPacket">ConnectionErrorPacket error = new ConnectionErrorPacket(
376               "Cannot get localChannel since localId is not found anymore",
377               String.valueOf(msg.getLocalId()));
378           writeError(channel, msg.getRemoteId(), msg.getLocalId(), error);
379           msg.clear();
380           return;
381         }
382       }
383     }
384     // check if not already in shutdown or closed
385     if (NetworkTransaction.isShuttingdownNetworkChannel(remoteAddress) ||
386         WaarpShutdownHook.isShutdownStarting()) {
387       logger.debug("Cannot use LocalChannel since already in shutdown: " + msg);
388       // ignore
389       msg.clear();
390       return;
391     }
392     LocalServerHandler.channelRead0(localChannelReference, msg);
393   }

395   @Override
396   public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
397     final Channel channel = ctx.channel();
398     if (isBlackListed) {
399       logger.info("While partner is blacklisted, Network Channel Exception: {}",
400                   channel.id(), cause.getClass().getName() + " : " + cause);
401       // ignore
402       return;
403     }
404     logger.debug("Network Channel Exception: {}", channel.id(), cause);
405     if (cause instanceof ReadTimeoutException) {
406       final ReadTimeoutException exception = (ReadTimeoutException) cause;
407       // No read for too long
408       logger.error("ReadTimeout so Will close NETWORK channel {}",
409                    exception.getClass().getName() + " : " +
410                    exception.getMessage());
411       ChannelCloseTimer.closeFutureChannel(channel);
412       return;
413     }
414     if (cause instanceof BindException) {
415       // received when not yet connected
416       logger.debug("BindException");
417       ChannelCloseTimer.closeFutureChannel(channel);
418       return;
419     }
420     final OpenR66Exception exception = OpenR66ExceptionTrappedFactory
421         .getExceptionFromTrappedException(channel, cause);
422     if (exception != null) {
423       if (exception instanceof OpenR66ProtocolBusinessNoWriteBackException) {
424         if (networkChannelReference != null &&
425             networkChannelReference.nbLocalChannels() > 0) {
426           logger.debug("Network Channel Exception: {} {}", channel.id(),
427                        exception.getClass().getName() + " : " +
428                        exception.getMessage());
429         }
430         logger.debug("Will close NETWORK channel");
431         ChannelCloseTimer.closeFutureChannel(channel);
432         return;
433       } else if (exception instanceof OpenR66ProtocolNoConnectionException) {
434         logger.debug("Connection impossible with NETWORK channel {}",
435                      exception.getClass().getName() + " : " +
436                      exception.getMessage());
437         channel.close();
438         return;
439       } else {
440         logger.debug("Network Channel Exception: {} {}", channel.id(),
441                      exception.getClass().getName() + " : " +
442                      exception.getMessage());
443       }
444       final ConnectionErrorPackett/ConnectionErrorPacket.html#ConnectionErrorPacket">ConnectionErrorPacket errorPacket = new ConnectionErrorPacket(
445           exception.getClass().getName() + " : " + exception.getMessage(),
446           null);
447       writeError(channel, ChannelUtils.NOCHANNEL, ChannelUtils.NOCHANNEL,
448                  errorPacket);
449       logger.debug("Will close NETWORK channel: {}",
450                    exception.getClass().getName() + " : " +
451                    exception.getMessage());
452       ChannelCloseTimer.closeFutureChannel(channel);
453     } else {
454       // Nothing to do
455     }
456   }

458   /**
459    * Write error back to remote client
460    *
461    * @param channel
462    * @param remoteId
463    * @param localId
464    * @param error
465    */
466   public static void writeError(Channel channel, Integer remoteId,
467                                 Integer localId, AbstractLocalPacket error) {
468     NetworkPacket networkPacket = null;
469     try {
470       networkPacket = new NetworkPacket(localId, remoteId, error, null);
471     } catch (final OpenR66ProtocolPacketException ignored) {
472       // nothing
473     }
474     try {
475       if (channel.isActive()) {
476         channel.writeAndFlush(networkPacket).await(Configuration.WAITFORNETOP);
477       }
478     } catch (final InterruptedException e) {//NOSONAR
479       SysErrLogger.FAKE_LOGGER.ignoreLog(e);
480     }
481   }
