40   */
41  public class NetworkPacketCodec extends ByteToMessageCodec<NetworkPacket> {
42    /**
43     * Internal Logger
44     */
45    private static final WaarpLogger logger =
46        WaarpLoggerFactory.getLogger(NetworkPacketCodec.class);
47  
48    @Override
49    protected void decode(ChannelHandlerContext ctx, ByteBuf buf,
50                          List<Object> out) throws Exception {
51      // Make sure if the length field was received.
52      if (buf.readableBytes() < 4) {
53        // The length field was not received yet - return null.
54        // This method will be invoked again when more packets are
55        // received and appended to the buffer.
56        return;
57      }
58      // Mark the current buffer position
59      buf.markReaderIndex();
60      // Read the length field
61      final int length = buf.readInt();
62      if (length < 9) {
63        throw new OpenR66ProtocolPacketException(
64            "Incorrect decode first field in Network Packet: " + length + " < 9");
65      }
66      if (buf.readableBytes() < length) {
67        buf.resetReaderIndex();
68        return;
69      }
70      // Now we can read the two Ids
71      final int localId = buf.readInt();
72      final int remoteId = buf.readInt();
73      final byte code = buf.readByte();
74      final int readerInder = buf.readerIndex();
75      final ByteBuf buffer = buf.slice(readerInder, length - 9);
76      buffer.retain();
77      buf.skipBytes(length - 9);
78      NetworkPacket networkPacket =
79          new NetworkPacket(localId, remoteId, code, buffer);
80      if (code == LocalPacketFactory.KEEPALIVEPACKET) {
81        final KeepAlivePacketorg/waarp/openr66/protocol/localhandler/packet/KeepAlivePacket.html#KeepAlivePacket">KeepAlivePacket keepAlivePacket = (KeepAlivePacket) LocalPacketCodec
82            .decodeNetworkPacket(networkPacket.getBuffer());
83        if (keepAlivePacket.isToValidate()) {
84          keepAlivePacket.validate();
85          final NetworkPacket response =
86              new NetworkPacket(ChannelUtils.NOCHANNEL, ChannelUtils.NOCHANNEL,
87                                keepAlivePacket, null);
88          final NetworkChannelReference nc =
89              NetworkTransaction.getImmediateNetworkChannel(ctx.channel());
90          if (nc != null) {
91            nc.useIfUsed();
92          }
93          ctx.writeAndFlush(response.getNetworkPacket());
94        }
95        buffer.release();
96        final NetworkServerHandler nsh =
97            (NetworkServerHandler) ctx.pipeline().last();
98        nsh.setKeepAlivedSent();
99        return;
100     }
101     out.add(networkPacket);
102   }
103 
104   @Override
105   protected void encode(ChannelHandlerContext ctx, NetworkPacket msg,
106                         ByteBuf out) throws Exception {
107     logger.trace("TRACE ID sending network packet {}", msg);
108     final ByteBuf finalBuf = msg.getNetworkPacket();
109     out.writeBytes(finalBuf);
110     finalBuf.release();
111   }

113 }
