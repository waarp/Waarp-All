58    /**
59     * Same as setUnique() except that File will be prefixed by id and postfixed
60     * by filename
61     *
62     * @param prefix
63     * @param filename
64     *
65     * @return the R66File with a unique filename and a temporary extension
66     *
67     * @throws CommandAbstractException
68     */
69    public synchronized R66File setUniqueFile(long prefix, String filename)
70        throws CommandAbstractException {
71      checkIdentify();
72      File file;
73      String prename = prefix + "_";
74      if (prename.length() < 3) {
75        prename = "xx_" + prename;
76      }
77      String basename = R66File.getBasename(filename);
78      if (basename.length() >
79          Configuration.configuration.getMaxfilenamelength() - 55) {
80        basename = basename.substring(basename.length() -
81                                      Configuration.configuration
82                                          .getMaxfilenamelength() + 55);
83      }
84      try {
85        file =
86            File.createTempFile(prename, '_' + basename + Configuration.EXT_R66,
87                                getFileFromPath(currentDir));
88      } catch (final IOException e) {
89        throw new Reply550Exception("Cannot create unique file from " + basename);
90      }
91      final String currentFile = getRelativePath(file);
92      return newFile(normalizePath(currentFile), false);
93    }

95    /**
96     * @param file
97     *
98     * @return the final unique basename without the temporary extension
99     */
100   public static String getFinalUniqueFilename(R66File file) {
101     String finalpath = file.getBasename();
102     final int pos = finalpath.lastIndexOf(Configuration.EXT_R66);
103     if (pos > 0) {
104       finalpath = finalpath.substring(0, pos);
105     }
106     return finalpath;
107   }

109   /**
110    * Finds all files matching a wildcard expression (based on '?', '~' or '*')
111    * but without checking
112    * BusinessPath, thus returning absolute path.
113    *
114    * @param pathWithWildcard The wildcard expression with a business
115    *     path.
116    *
117    * @return List of String as relative paths matching the wildcard
118    *     expression.
119    *     Those files are tested as valid
120    *     from business point of view. If Wildcard support is not active,
121    *     if the
122    *     path contains any wildcards,
123    *     it will throw an error.
124    *
125    * @throws CommandAbstractException
126    */
127   protected List<String> wildcardFilesNoCheck(String pathWithWildcard)
128       throws CommandAbstractException {
129     final List<String> resultPaths = new ArrayList<String>();
130     // First check if pathWithWildcard contains wildcards
131     if (!(pathWithWildcard.contains("*") || pathWithWildcard.contains("?") ||
132           pathWithWildcard.contains("~"))) {
133       // No so simply return the list containing this path
134       resultPaths.add(pathWithWildcard);
135       return resultPaths;
136     }
137     // Do we support Wildcard path
138     if (!FilesystemBasedDirJdkAbstract.ueApacheCommonsIo) {
139       throw new Reply553Exception("Wildcards in pathname is not allowed");
140     }
141     File wildcardFile = new File(pathWithWildcard);
142     File rootFile;
143     if (ISUNIX) {
144       rootFile = new File("/");
145     } else {
146       rootFile = getCorrespondingRoot(wildcardFile);
147     }
148     // Split wildcard path into subdirectories.
149     final List<String> subdirs = new ArrayList<String>();
150     while (wildcardFile != null) {
151       final File parent = wildcardFile.getParentFile();
152       if (parent == null) {
153         subdirs.add(0, wildcardFile.getPath());
154         break;
155       }
156       subdirs.add(0, wildcardFile.getName());
157       if (parent.equals(rootFile)) {
158         // End of wildcard path
159         subdirs.add(0, parent.getPath());
160         break;
161       }
162       wildcardFile = parent;
163     }
164     List<File> basedPaths = new ArrayList<File>();
165     // First set root
166     basedPaths.add(new File(subdirs.get(0)));
167     int i = 1;
168     // For each wilcard subdirectory
169     while (i < subdirs.size()) {
170       // Set current filter
171       final FileFilter fileFilter =
172           FilesystemBasedCommonsIo.getWildcardFileFilter(subdirs.get(i));
173       final List<File> newBasedPaths = new ArrayList<File>();
174       // Look for matches in all the current search paths
175       for (final File dir : basedPaths) {
176         if (dir.isDirectory()) {
177           Collections.addAll(newBasedPaths, dir.listFiles(fileFilter));
178         }
179       }
180       // base Search Path changes now
181       basedPaths = newBasedPaths;
182       i++;
183     }
184     // Valid each file first
185     for (final File file : basedPaths) {
186       resultPaths.add(file.getAbsolutePath());
187     }
188     return resultPaths;
189   }

191   /**
192    * Create a new file according to the path without checking BusinessPath, so
193    * as external File.
194    *
195    * @param path
196    *
197    * @return the File created
198    *
199    * @throws CommandAbstractException
200    */
201   public R66File setFileNoCheck(String path) throws CommandAbstractException {
202     checkIdentify();
203     final String newpath = consolidatePath(path);
204     final List<String> paths = wildcardFilesNoCheck(newpath);
205     if (paths.size() != 1) {
206       throw new Reply550Exception(
207           "File not found from: " + newpath + " and " + paths.size() +
208           " founds");
209     }
210     final String extDir = paths.get(0);
211     return new R66File((R66Session) getSession(), this, extDir);
212   }

214   /**
215    * This method returns the Full path for the current directory
216    *
217    * @return the full path associated with the current Dir
218    */
219   public String getFullPath() {
220     if (session.getAuth() == null) {
221       return currentDir;
222     }
223     if (isAbsolute(currentDir)) {
224       return currentDir;
225     }
226     return ((R66Auth) session.getAuth()).getAbsolutePath(currentDir);
227   }

229   @Override

231     return "Dir: " + currentDir;
