35  /**
36   * This enum class references all available tasks.
37   * <p>
38   * If a new task type is to be created, this is the place where it should be
39   * referenced.
40   */
41  public enum TaskType {
42    LOG, MOVE, MOVERENAME, COPY, COPYRENAME, EXEC, EXECMOVE, LINKRENAME, TRANSFER,
43    VALIDFILEPATH, DELETE, TAR, ZIP, EXECOUTPUT, RESCHEDULE, EXECJAVA, TRANSCODE,
44    SNMP, FTP, RENAME, RESTART, UNZEROED, CHMOD, CHKFILE, ICAP, EXTENDED,
45    COMPRESS;
46  
47    public static final String UNVALID_TASK = "Unvalid Task: ";
48  
49    public static final String NAME_UNKNOWN = "name unknown: ";
50  
51    private static final Map<String, AbstractExtendedTaskFactory>
52        extendedTaskTypeMap = new HashMap<String, AbstractExtendedTaskFactory>();
53  
54    final int type;
55  
56    final String name;
57    /**
58     * Internal Logger
59     */
60    private static final WaarpLogger logger =
61        WaarpLoggerFactory.getLogger(TaskType.class);
62  
63    TaskType() {
64      type = ordinal();
65      name = name();
66    }
67  
68    /**
69     * Allow to add extended task that will be denoted as EXTENDED
70     *
71     * @param associatedTaskNames if a previously TASK name was already set, it is replaced by the new Factory, including standard TASK_TYPE
72     * @param extendedTaskFactory the Factory to use to create the associated Task with its name
73     */
74    public static void addExtendedTaskFactory(
75        final Set<String> associatedTaskNames,
76        final AbstractExtendedTaskFactory extendedTaskFactory) {
77      for (final String associatedName : associatedTaskNames) {
78        logger.debug("Add {} with {}", associatedName,
79                     extendedTaskFactory.getName());
80        extendedTaskTypeMap.put(associatedName, extendedTaskFactory);
81      }
82    }
83  
84    public static final String S_3_TASK_FACTORY =
85        "org.waarp.openr66.s3.taskfactory.S3TaskFactory";
86  
87    static {
88      // Try to load S3 Tasks if S3TaskFactory exists
89      try {
90        final Class s3TaskFactoryClass = Class.forName(S_3_TASK_FACTORY);
91        WaarpSystemUtil.newInstance(s3TaskFactoryClass);
92      } catch (final Exception ignore) {
93        // Not found and ignore
94        SysErrLogger.FAKE_LOGGER.ignoreLog(ignore);
95      }
96    }

98    /**
99     * @return the current Set of ExtendedTaskFactories
100    */
101   public static Collection<AbstractExtendedTaskFactory> getExtendedFactories() {
102     return extendedTaskTypeMap.values();
103   }

105   /**
106    * Check if the given Task exists
107    *
108    * @param task
109    *
110    * @return True if this task is known
111    */
112   public static boolean isValidTask(final String task) {
113     try {
114       TaskType.valueOf(task);
115       return true;
116     } catch (final IllegalArgumentException e) {
117       logger.debug("{} is in keys: {}", task,
118                    extendedTaskTypeMap.containsKey(task));
119       return extendedTaskTypeMap.containsKey(task);
120     }
121   }

123   /**
124    * @param type
125    * @param argRule
126    * @param delay
127    * @param session
128    *
129    * @return the corresponding AbstractTask
130    *
131    * @throws OpenR66RunnerErrorException
132    */
133   private static AbstractTask getTaskFromId(final TaskType type,
134                                             final String argRule,
135                                             final int delay,
136                                             final R66Session session)
137       throws OpenR66RunnerErrorException {
138     switch (type) {
139       case LOG:
140         return new LogTask(argRule, delay,
141                            session.getRunner().getFileInformation(), session);
142       case MOVE:
143         return new MoveTask(argRule, delay,
144                             session.getRunner().getFileInformation(), session);
145       case MOVERENAME:
146         return new MoveRenameTask(argRule, delay,
147                                   session.getRunner().getFileInformation(),
148                                   session);
149       case COPY:
150         return new CopyTask(argRule, delay,
151                             session.getRunner().getFileInformation(), session);
152       case COPYRENAME:
153         return new CopyRenameTask(argRule, delay,
154                                   session.getRunner().getFileInformation(),
155                                   session);
156       case EXEC:
157         return new ExecTask(argRule, delay,
158                             session.getRunner().getFileInformation(), session);
159       case EXECMOVE:
160         return new ExecMoveTask(argRule, delay,
161                                 session.getRunner().getFileInformation(),
162                                 session);
163       case LINKRENAME:
164         return new LinkRenameTask(argRule, delay,
165                                   session.getRunner().getFileInformation(),
166                                   session);
167       case TRANSFER:
168         return new TransferTask(argRule, delay,
169                                 session.getRunner().getFileInformation(),
170                                 session);
171       case VALIDFILEPATH:
172         return new ValidFilePathTask(argRule, delay,
173                                      session.getRunner().getFileInformation(),
174                                      session);
175       case DELETE:
176         return new DeleteTask(argRule, delay,
177                               session.getRunner().getFileInformation(),
178                               session);
179       case TAR:
180         return new TarTask(argRule, delay,
181                            session.getRunner().getFileInformation(), session);
182       case ZIP:
183         return new ZipTask(argRule, delay,
184                            session.getRunner().getFileInformation(), session);
185       case EXECOUTPUT:
186         return new ExecOutputTask(argRule, delay,
187                                   session.getRunner().getFileInformation(),
188                                   session);
189       case RESCHEDULE:
190         return new RescheduleTransferTask(argRule, delay, session.getRunner()
191                                                                  .getFileInformation(),
192                                           session);
193       case EXECJAVA:
194         return new ExecJavaTask(argRule, delay,
195                                 session.getRunner().getFileInformation(),
196                                 session);
197       case TRANSCODE:
198         return new TranscodeTask(argRule, delay,
199                                  session.getRunner().getFileInformation(),
200                                  session);
201       case SNMP:
202         return new SnmpTask(argRule, delay,
203                             session.getRunner().getFileInformation(), session);
204       case FTP:
205         return new FtpTransferTask(argRule, delay,
206                                    session.getRunner().getFileInformation(),
207                                    session);
208       case RENAME:
209         return new RenameTask(argRule, delay,
210                               session.getRunner().getFileInformation(),
211                               session);
212       case RESTART:
213         return new RestartServerTask(argRule, delay,
214                                      session.getRunner().getFileInformation(),
215                                      session);
216       case UNZEROED:
217         return new UnzeroedFileTask(argRule, delay,
218                                     session.getRunner().getFileInformation(),
219                                     session);
220       case CHMOD:
221         return new ChModTask(argRule, delay,
222                              session.getRunner().getFileInformation(), session);
223       case CHKFILE:
224         return new FileCheckTask(argRule, delay,
225                                  session.getRunner().getFileInformation(),
226                                  session);
227       case ICAP:
228         return new IcapTask(argRule, delay,
229                             session.getRunner().getFileInformation(), session);
230       case COMPRESS:
231         return new CompressTask(argRule, delay,
232                                 session.getRunner().getFileInformation(),
233                                 session);
234       case EXTENDED:
235         // Should not arrived here
236       default:
237         logger.error(NAME_UNKNOWN + type.name);
238         throw new OpenR66RunnerErrorException(UNVALID_TASK + type.name);
239     }
240   }

242   /**
243    * @param name
244    * @param argRule
245    * @param delay
246    * @param session
247    *
248    * @return the corresponding AbstractTask
249    *
250    * @throws OpenR66RunnerErrorException
251    */
252   public static AbstractTask getTaskFromId(final String name,
253                                            final String argRule,
254                                            final int delay,
255                                            final R66Session session)
256       throws OpenR66RunnerErrorException {
257     // EXTENDED first
258     final AbstractExtendedTaskFactory factory = extendedTaskTypeMap.get(name);
259     if (factory != null) {
260       return factory.getTaskFromId(name, argRule, delay, session);
261     }
262     final TaskType type;
263     try {
264       type = valueOf(name);
265     } catch (final NullPointerException e) {
266       logger.error("name empty " + name);
267       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
268     } catch (final IllegalArgumentException e) {
269       logger.error(NAME_UNKNOWN + name);
270       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
271     }
272     return getTaskFromId(type, argRule, delay, session);
273   }

275   /**
276    * For usage in ExecBusinessTask
277    *
278    * @param name
279    * @param argRule
280    * @param delay
281    * @param session
282    *
283    * @return the corresponding AbstractTask
284    *
285    * @throws OpenR66RunnerErrorException
286    */
287   public static AbstractTask getTaskFromIdForBusiness(final String name,
288                                                       final String argRule,
289                                                       final int delay,
290                                                       final R66Session session)
291       throws OpenR66RunnerErrorException {
292     // EXTENDED first
293     final AbstractExtendedTaskFactory factory = extendedTaskTypeMap.get(name);
294     if (factory != null) {
295       return factory.getTaskFromIdForBusiness(name, argRule, delay, session);
296     }
297     final TaskType type;
298     try {
299       type = valueOf(name);
300     } catch (final NullPointerException e) {
301       logger.error("name empty " + name);
302       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
303     } catch (final IllegalArgumentException e) {
304       // FIXME EXTENDED
305       logger.error(NAME_UNKNOWN + name);
306       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
307     }
308     switch (type) {
309       case LOG:
310         int newdelay = delay;
311         if (newdelay == 0) {
312           newdelay = 1;
313         }
314         return new LogTask(argRule, newdelay, "", session);
315       case EXEC:
316         return new ExecTask(argRule, delay, "", session);
317       case TRANSFER:
318         return new TransferTask(argRule, delay, "", session);
319       case TAR:
320         return new TarTask(argRule, delay, "", session);
321       case ZIP:
322         return new ZipTask(argRule, delay, "", session);
323       case EXECOUTPUT:
324         return new ExecOutputTask(argRule, delay, "", session);
325       case EXECJAVA:
326         return new ExecJavaTask(argRule, delay, "", session);
327       case SNMP:
328         return new SnmpTask(argRule, delay, "", session);
329       case FTP:
330         return new FtpTransferTask(argRule, delay, "", session);
331       case RESTART:
332         return new RestartServerTask(argRule, delay, "", session);
333       case ICAP:
334         return new IcapTask(argRule, delay, "", session);
335       case EXTENDED:
336         // Not allowed
337       case MOVE:
338       case MOVERENAME:
339       case COPY:
340       case COPYRENAME:
341       case EXECMOVE:
342       case LINKRENAME:
343       case VALIDFILEPATH:
344       case DELETE:
345       case RESCHEDULE:
346       case TRANSCODE:
347       case RENAME:
348       case UNZEROED:
349       case CHMOD:
350       case CHKFILE:
351       case COMPRESS:
352         throw new OpenR66RunnerErrorException(UNVALID_TASK + type.name);
353       default:
354         logger.error(NAME_UNKNOWN + type.name);
355         throw new OpenR66RunnerErrorException(UNVALID_TASK + type.name);
356     }
357   }
