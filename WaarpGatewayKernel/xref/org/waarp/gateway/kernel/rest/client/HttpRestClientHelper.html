83    /**
84     * @param baseUri base of all URI, in general simply "/" (default if
85     *     null)
86     * @param nbclient max number of client connected at once
87     * @param timeout timeout used in connection
88     * @param Initializer the associated client pipeline factory
89     */
90    public HttpRestClientHelper(String baseUri, int nbclient, long timeout,
91                                ChannelInitializer<SocketChannel> Initializer) {
92      if (logger == null) {
93        logger = WaarpLoggerFactory.getLogger(HttpRestClientHelper.class);
94      }
95      if (baseUri != null) {
96        this.baseUri = baseUri;
97      }
98      // Configure the client.
99      bootstrap = new Bootstrap();
100     workerGroup = new NioEventLoopGroup(nbclient, new WaarpThreadFactory(
101         "Rest_" + baseUri + '_'));
102     WaarpNettyUtil.setBootstrap(bootstrap, workerGroup, 30000);
103     // Configure the pipeline factory.
104     bootstrap.handler(Initializer);
105 
106     // will ignore real request
107     final HttpRequest request =
108         new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.GET, baseUri);
109     headers = request.headers();
110     headers.set(HttpHeaderNames.ACCEPT_ENCODING,
111                 HttpHeaderValues.GZIP + "," + HttpHeaderValues.DEFLATE);
112     headers.set(HttpHeaderNames.ACCEPT_CHARSET, "utf-8;q=0.7,*;q=0.7");
113     headers.set(HttpHeaderNames.ACCEPT_LANGUAGE, "fr,en");
114     headers
115         .set(HttpHeaderNames.USER_AGENT, "Netty Simple Http Rest Client side");
116     headers.set(HttpHeaderNames.ACCEPT,
117                 "text/html,text/plain,application/xhtml+xml,application/xml,application/json;q=0.9,*/*;q=0.8");
118     // connection will not close but needed
119     /*
120      * request.headers().set(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.KEEP_ALIVE)
121      */
122     // request.setHeader("Connection","keep-alive")
123     // request.setHeader("Keep-Alive","300")
124   }

135     // Start the connection attempt.
136     final ChannelFuture future =
137         bootstrap.connect(new InetSocketAddress(host, port));
138     // Wait until the connection attempt succeeds or fails.
139     final Channel channel = WaarpSslUtility.waitforChannelReady(future);
140     if (channel != null) {
141       final RestFutureent/RestFuture.html#RestFuture">RestFuture futureChannel = new RestFuture(true);
142       channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT)
143              .set(futureChannel);
144     }
145     return channel;
146   }

148   /**
149    * Send an HTTP query using the channel for target, using signature
150    *
151    * @param hmacSha256 SHA-256 key to create the signature
152    * @param channel target of the query
153    * @param method HttpMethod to use
154    * @param host target of the query (shall be the same as for the
155    *     channel)
156    * @param addedUri additional uri, added to baseUri (shall include
157    *     also
158    *     extra arguments) (might be null)
159    * @param user user to use in authenticated Rest procedure (might be
160    *     null)
161    * @param pwd password to use in authenticated Rest procedure (might
162    *     be
163    *     null)
164    * @param uriArgs arguments for Uri if any (might be null)
165    * @param json json to send as body in the request (might be null);
166    *     Useful
167    *     in PUT, POST but should not
168    *     in GET, DELETE, OPTIONS
169    *
170    * @return the RestFuture associated with this request
171    */
172   public RestFuture sendQuery(HmacSha256 hmacSha256, Channel channel,
173                               HttpMethod method, String host, String addedUri,
174                               String user, String pwd,
175                               Map<String, String> uriArgs, String json) {
176     // Prepare the HTTP request.
177     logger.debug("Prepare request: " + method + ':' + addedUri + ':' + json);
178     final RestFuture future =
179         channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT).get();
180     QueryStringEncoder encoder;
181     if (addedUri != null) {
182       encoder = new QueryStringEncoder(baseUri + addedUri);
183     } else {
184       encoder = new QueryStringEncoder(baseUri);
185     }
186     // add Form attribute
187     if (uriArgs != null) {
188       for (final Entry<String, String> elt : uriArgs.entrySet()) {
189         encoder.addParam(elt.getKey(), elt.getValue());
190       }
191     }
192     String[] result;
193     try {
194       result = RestArgument.getBaseAuthent(hmacSha256, encoder, user, pwd);
195       logger.debug("Authent encoded");
196     } catch (final HttpInvalidAuthenticationException e) {
197       logger.error(e.getMessage(), e);
198       future.setFailure(e);
199       return future;
200     }
201     URI uri;
202     try {
203       uri = encoder.toUri();
204     } catch (final URISyntaxException e) {
205       logger.error(e.getMessage());
206       future.setFailure(e);
207       return future;
208     }
209     logger.debug("Uri ready: " + uri.toASCIIString());
210 
211     FullHttpRequest request;
212     if (json != null) {
213       logger.debug("Add body");
214       final ByteBuf buffer =
215           Unpooled.wrappedBuffer(json.getBytes(WaarpStringUtils.UTF8));
216       request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method,
217                                            uri.toASCIIString(), buffer);
218       request.headers()
219              .set(HttpHeaderNames.CONTENT_LENGTH, buffer.readableBytes());
220     } else {
221       request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method,
222                                            uri.toASCIIString());
223     }
224     // it is legal to add directly header or cookie into the request until finalize
225     request.headers().add(headers);
226     request.headers().set(HttpHeaderNames.HOST, host);
227     if (user != null) {
228       request.headers().set(
229           (CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_USER.field,
230           user);
231     }
232     request.headers().set(
233         (CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_TIMESTAMP.field,
234         result[0]);
235     request.headers().set(
236         (CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_KEY.field,
237         result[1]);
238     // send request
239     logger.debug("Send request");
240     channel.writeAndFlush(request);
241     logger.debug("Request sent");
242     return future;
243   }

245   /**
246    * Send an HTTP query using the channel for target, but without any
247    * Signature
248    *
249    * @param channel target of the query
250    * @param method HttpMethod to use
251    * @param host target of the query (shall be the same as for the
252    *     channel)
253    * @param addedUri additional uri, added to baseUri (shall include
254    *     also
255    *     extra arguments) (might be null)
256    * @param user user to use in authenticated Rest procedure (might be
257    *     null)
258    * @param uriArgs arguments for Uri if any (might be null)
259    * @param json json to send as body in the request (might be null);
260    *     Useful
261    *     in PUT, POST but should not in
262    *     GET, DELETE, OPTIONS
263    *
264    * @return the RestFuture associated with this request
265    */
266   public RestFuture sendQuery(Channel channel, HttpMethod method, String host,
267                               String addedUri, String user,
268                               Map<String, String> uriArgs, String json) {
269     // Prepare the HTTP request.
270     logger.debug("Prepare request: " + method + ':' + addedUri + ':' + json);
271     final RestFuture future =
272         channel.attr(HttpRestClientSimpleResponseHandler.RESTARGUMENT).get();
273     QueryStringEncoder encoder;
274     if (addedUri != null) {
275       encoder = new QueryStringEncoder(baseUri + addedUri);
276     } else {
277       encoder = new QueryStringEncoder(baseUri);
278     }
279     // add Form attribute
280     if (uriArgs != null) {
281       for (final Entry<String, String> elt : uriArgs.entrySet()) {
282         encoder.addParam(elt.getKey(), elt.getValue());
283       }
284     }
285     URI uri;
286     try {
287       uri = encoder.toUri();
288     } catch (final URISyntaxException e) {
289       logger.error(e.getMessage());
290       future.setFailure(e);
291       return future;
292     }
293     logger.debug("Uri ready: " + uri.toASCIIString());
294 
295     FullHttpRequest request;
296     if (json != null) {
297       logger.debug("Add body");
298       final ByteBuf buffer =
299           Unpooled.wrappedBuffer(json.getBytes(WaarpStringUtils.UTF8));
300       request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method,
301                                            uri.toASCIIString(), buffer);
302       request.headers()
303              .set(HttpHeaderNames.CONTENT_LENGTH, buffer.readableBytes());
304     } else {
305       request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, method,
306                                            uri.toASCIIString());
307     }
308 
309     // it is legal to add directly header or cookie into the request until finalize
310     request.headers().add(headers);
311     request.headers().set(HttpHeaderNames.HOST, host);
312     if (user != null) {
313       request.headers().set(
314           (CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_USER.field,
315           user);
316     }
317     request.headers().set(
318         (CharSequence) RestArgument.REST_ROOT_FIELD.ARG_X_AUTH_TIMESTAMP.field,
319         new DateTime().toString());
320     // send request
321     logger.debug("Send request");
322     channel.writeAndFlush(request);
323     logger.debug("Request sent");
324     return future;
325   }

334   /**
335    * @param args as uri (http://host:port/uri method user pwd
336    *     sign=path|nosign [json])
337    */
338   public static void main(String[] args) {
339     WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
340     final WaarpLogger logger =
341         WaarpLoggerFactory.getLogger(HttpRestClientHelper.class);
342     if (args.length < 5) {
343       logger.error(
344           NEED_MORE_ARGUMENTS_HTTP_HOST_PORT_URI_METHOD_USER_SIGN_PATH_NOSIGN_JSON);
345       return;
346     }
347     final String uri = args[0];
348     final String meth = args[1];
349     final String user = args[2];
350     final String pwd = args[3];
351     final boolean sign = args[4].toLowerCase().contains("sign=");
352     HmacSha256 hmacSha256 = null;
353     if (sign) {
354       final String file = args[4].replace("sign=", "");
355       hmacSha256 = new HmacSha256();
356       try {
357         hmacSha256.setSecretKey(new File(file));
358       } catch (final CryptoException e) {
359         logger.error(
360             NEED_MORE_ARGUMENTS_HTTP_HOST_PORT_URI_METHOD_USER_SIGN_PATH_NOSIGN_JSON);
361         return;
362       } catch (final IOException e) {
363         logger.error(
364             NEED_MORE_ARGUMENTS_HTTP_HOST_PORT_URI_METHOD_USER_SIGN_PATH_NOSIGN_JSON);
365         return;
366       }
367     }
368     String json = null;
369     if (args.length > 5) {
370       json = args[5].replace("'", "\"");
371     }
372     final HttpMethod method = HttpMethod.valueOf(meth);
373     int port = -1;
374     String host;
375     String path;
376     try {
377       final URI realUri = new URI(uri);
378       port = realUri.getPort();
379       host = realUri.getHost();
380       path = realUri.getPath();
381     } catch (final URISyntaxException e) {
382       logger.error("Error", e);
383       return;
384     }
385     final HttpRestClientHelper/HttpRestClientHelper.html#HttpRestClientHelper">HttpRestClientHelper client = new HttpRestClientHelper(path, 1, 30000,
386                                                                  new HttpRestClientSimpleInitializer());
387     final Channel channel = client.getChannel(host, port);
388     if (channel == null) {
389       client.closeAll();
390       logger.error("Cannot connect to " + host + " on port " + port);
391       return;
392     }
393     RestFuture future;
394     if (sign) {
395       future = client
396           .sendQuery(hmacSha256, channel, method, host, null, user, pwd, null,
397                      json);
398     } else {
399       future = client.sendQuery(channel, method, host, null, user, null, json);
400     }
401     future.awaitOrInterruptible();
402     WaarpSslUtility.closingSslChannel(channel);
403     if (future.isSuccess()) {
404       logger.warn(future.getRestArgument().prettyPrint());
405     } else {
406       final RestArgument ra = future.getRestArgument();
407       if (ra != null) {
408         logger.error(ra.prettyPrint());
409       } else {
410         logger.error("Query in error", future.getCause());
411       }
412     }
413     client.closeAll();
414   }
