47        throws WaarpDatabaseNoConnectionException {
48      createTableMonitoring(session);
49    }

51    public static void createTableMonitoring(final DbSession session)
52        throws WaarpDatabaseNoConnectionException {
53      // Create tables: logs
54      final String createTableH2 = "CREATE TABLE ";
55      final String primaryKey = " PRIMARY KEY ";
56      final String notNull = " NOT NULL ";
57  
58      final DbRequest request = new DbRequest(session);
59      // TRANSLOG
60      StringBuilder action =
61          new StringBuilder(createTableH2 + DbTransferLog.table + '(');
62      final DbTransferLog.Columns[] acolumns = DbTransferLog.Columns.values();
63      for (int i = 0; i < acolumns.length; i++) {
64        action.append(acolumns[i].name())
65              .append(DBType.getType(DbTransferLog.dbTypes[i])).append(notNull)
66              .append(", ");
67      }
68      // Several columns for primary key
69      action.append(" CONSTRAINT TRANSLOG_PK " + primaryKey + '(');
70      for (int i = DbTransferLog.NBPRKEY; i > 1; i--) {
71        action.append(acolumns[acolumns.length - i].name()).append(',');
72      }
73      action.append(acolumns[acolumns.length - 1].name()).append("))");
74      SysErrLogger.FAKE_LOGGER.sysout(action);
75      try {
76        request.query(action.toString());
77      } catch (final WaarpDatabaseNoConnectionException e) {
78        SysErrLogger.FAKE_LOGGER.syserr(e);
79        return;
80      } catch (final WaarpDatabaseSqlException e) {
81        SysErrLogger.FAKE_LOGGER.syserr(e);
82        // XXX FIX No return
83      } finally {
84        request.close();
85      }
86      // Index TRANSLOG
87      action = new StringBuilder(
88          "CREATE INDEX IDX_TRANSLOG ON " + DbTransferLog.table + '(');
89      final DbTransferLog.Columns[] icolumns = DbTransferLog.indexes;
90      for (int i = 0; i < icolumns.length - 1; i++) {
91        action.append(icolumns[i].name()).append(", ");
92      }
93      action.append(icolumns[icolumns.length - 1].name()).append(')');
94      SysErrLogger.FAKE_LOGGER.sysout(action);
95      try {
96        request.query(action.toString());
97      } catch (final WaarpDatabaseNoConnectionException e) {
98        SysErrLogger.FAKE_LOGGER.syserr(e);
99        return;
100     } catch (final WaarpDatabaseSqlException e) {
101       SysErrLogger.FAKE_LOGGER.syserr(e);
102       // XXX FIX No return
103     } finally {
104       request.close();
105     }
106 
107     // cptrunner
108     action = new StringBuilder(
109         "CREATE SEQUENCE " + DbTransferLog.fieldseq + " MINVALUE " +
110         (DbConstant.ILLEGALVALUE + 1) + " RESTART WITH " +
111         (DbConstant.ILLEGALVALUE + 1));
112     SysErrLogger.FAKE_LOGGER.sysout(action);
113     try {
114       request.query(action.toString());
115     } catch (final WaarpDatabaseNoConnectionException e) {
116       SysErrLogger.FAKE_LOGGER.syserr(e);
117     } catch (final WaarpDatabaseSqlException e) {
118       SysErrLogger.FAKE_LOGGER.syserr(e);
119     } finally {
120       request.close();
121     }
122   }
