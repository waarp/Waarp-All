20  package org.waarp.gateway.kernel.rest;
21  
22  import io.netty.channel.ChannelInitializer;
23  import io.netty.channel.ChannelPipeline;
24  import io.netty.channel.nio.NioEventLoopGroup;
25  import io.netty.channel.socket.SocketChannel;
26  import io.netty.handler.codec.http.HttpServerCodec;
27  import io.netty.handler.stream.ChunkedWriteHandler;
28  import io.netty.util.concurrent.EventExecutorGroup;
29  import org.waarp.common.utility.WaarpThreadFactory;
30  
31  /**
32   *
33   */
34  public class HttpRestInitializer extends ChannelInitializer<SocketChannel> {
35    private final RestConfiguration restConfiguration;
36    private static final EventExecutorGroup executor =
37        new NioEventLoopGroup(10, new WaarpThreadFactory("HttpHandler"));
38  
39    /**
40     * @param restConfiguration
41     */
42    public HttpRestInitializer(RestConfiguration restConfiguration) {
43      this.restConfiguration = restConfiguration;
44    }
45  
46    @Override
47    protected void initChannel(SocketChannel ch) throws Exception {
48      // Create a default pipeline implementation.
49      final ChannelPipeline pipeline = ch.pipeline();
50  
51      pipeline.addLast("codec", new HttpServerCodec());
52      pipeline.addLast("chunkedWriter", new ChunkedWriteHandler());
53      final HttpRestTestHandler r66handler =
54          new HttpRestTestHandler(restConfiguration);
55      pipeline.addLast(executor, "handler", r66handler);
56    }
57  
58  }
